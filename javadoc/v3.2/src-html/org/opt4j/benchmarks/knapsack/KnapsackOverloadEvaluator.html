023package org.opt4j.benchmarks.knapsack;
024
025import static java.lang.Math.max;
026import static org.opt4j.core.Objective.Sign.MAX;
027import static org.opt4j.core.Objective.Sign.MIN;
028
029import java.util.Collection;
030import java.util.HashMap;
031import java.util.Map;
032
033import org.opt4j.core.Objective;
034import org.opt4j.core.Objectives;
035import org.opt4j.core.problem.Evaluator;
036import org.opt4j.core.problem.Priority;
037
038import com.google.inject.Inject;
039
040/**
041 * The {@link KnapsackOverloadEvaluator} evaluates a given {@link ItemSelection}
042 * using one {@link Objective} to sum up all knapsack overloads and one
043 * {@link Objective} for the profit of each knapsack.
044 * 
045 * @author reimann, lukasiewycz
046 * 
047 */
048@Priority(-10)
049public class KnapsackOverloadEvaluator implements Evaluator<ItemSelection> {
050        protected Objective objectiveOverload = new Objective("overload", MIN);
051        protected final Map<Knapsack, Objective> profitObjectives = new HashMap<Knapsack, Objective>();
052        protected KnapsackProblem problem;
053
054        /**
055         * Creates a new {@link KnapsackOverloadEvaluator}.
056         * 
057         * @param problem
058         *            the problem instance
059         */
060        @Inject
061        public KnapsackOverloadEvaluator(KnapsackProblem problem) {
062                this.problem = problem;
063                int i = 0;
064                for (Knapsack knapsack : problem.getKnapsacks()) {
065                        profitObjectives.put(knapsack, new Objective("profit" + i++, MAX));
066                }
067        }
068
069        /*
070         * (non-Javadoc)
071         * 
072         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
073         */
074        @Override
075        public Objectives evaluate(ItemSelection itemSelection) {
076                // evaluate overlead
077                double overload = 0;
078                for (Knapsack knapsack : problem.getKnapsacks()) {
079                        overload += getOverload(knapsack, itemSelection);
080                }
081
082                Objectives objectives = new Objectives();
083                objectives.add(objectiveOverload, overload);
084                itemSelection.setFeasible(overload == 0);
085                return objectives;
086        }
087
088        /**
089         * Computes how much the given item exceed the defined capacity of the
090         * knapsack.
091         * 
092         * @param knapsack
093         *            the knapsack
094         * @param items
095         *            the items
096         * @return the overload
097         */
098        protected double getOverload(Knapsack knapsack, Collection<Item> items) {
099                double weight = 0;
100                for (Item i : items) {
101                        weight += knapsack.getWeight(i);
102                }
103                return max(0, weight - knapsack.getCapacity());
104        }
105}
