024package org.opt4j.benchmarks.wfg;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.opt4j.benchmarks.K;
030import org.opt4j.benchmarks.M;
031
032import com.google.inject.Inject;
033
034/**
035 * The {@link WFG2} benchmark function.
036 * 
037 * @author lukasiewycz
038 * 
039 */
040public class WFG2 extends WFGEvaluator {
041
042        /**
043         * Constructs a {@link WFG2} benchmark function.
044         * 
045         * @param k
046         *            the position parameters
047         * @param M
048         *            the number of objectives
049         */
050        @Inject
051        public WFG2(@K int k, @M int M) {
052                super(k, M);
053        }
054
055        protected static List<Double> t2(final List<Double> y, final int k) {
056                final int n = y.size();
057                final int l = n - k;
058
059                assert (k >= 1);
060                assert (k < n);
061                assert (l % 2 == 0);
062
063                List<Double> t = new ArrayList<Double>();
064
065                for (int i = 0; i < k; i++) {
066                        t.add(y.get(i));
067                }
068
069                for (int i = k + 1; i <= k + l / 2; i++) {
070                        final int head = k + 2 * (i - k) - 2;
071                        final int tail = k + 2 * (i - k);
072
073                        t.add(WFGTransFunctions.r_nonsep(y.subList(head, tail), 2));
074                }
075
076                return t;
077        }
078
079        protected static List<Double> t3(final List<Double> y, final int k, final int M) {
080                final int n = y.size();
081
082                assert (k >= 1);
083                assert (k < n);
084                assert (M >= 2);
085                assert (k % (M - 1) == 0);
086
087                List<Double> w = new ArrayList<Double>();
088                for (int i = 0; i < n; i++) {
089                        w.add(1.0);
090                }
091
092                List<Double> t = new ArrayList<Double>();
093
094                for (int i = 1; i <= M - 1; i++) {
095                        final int head = (i - 1) * k / (M - 1);
096                        final int tail = i * k / (M - 1);
097
098                        final List<Double> y_sub = y.subList(head, tail);
099                        final List<Double> w_sub = w.subList(head, tail);
100
101                        t.add(WFGTransFunctions.r_sum(y_sub, w_sub));
102                }
103
104                final List<Double> y_sub = y.subList(k, n);
105                final List<Double> w_sub = w.subList(k, n);
106
107                t.add(WFGTransFunctions.r_sum(y_sub, w_sub));
108
109                return t;
110        }
111
112        protected static List<Double> shape(final List<Double> t_p) {
113
114                assert (t_p.size() >= 2);
115
116                final int M = t_p.size();
117
118                final List<Boolean> A = createA(M, false);
119                final List<Double> x = calculateX(t_p, A);
120
121                List<Double> h = new ArrayList<Double>();
122
123                for (int m = 1; m <= M - 1; m++) {
124                        h.add(WFGShapeFunctions.convex(x, m));
125                }
126                h.add(WFGShapeFunctions.disc(x, 5, 1.0, 1.0));
127
128                return calculateF(x, h);
129        }
130
131        /*
132         * (non-Javadoc)
133         * 
134         * @see org.opt4j.benchmark.wfg.WFGEvaluator#f(java.util.List)
135         */
136        @Override
137        public List<Double> f(List<Double> y) {
138                y = WFG1.t1(y, k);
139                y = WFG2.t2(y, k);
140                y = WFG2.t3(y, k, M);
141
142                return WFG2.shape(y);
143        }
144}
