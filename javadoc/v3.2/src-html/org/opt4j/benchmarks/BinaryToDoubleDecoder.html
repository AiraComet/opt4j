024package org.opt4j.benchmarks;
025
026import java.util.List;
027
028import org.opt4j.core.problem.Decoder;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link BinaryToDoubleDecoder}.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038public class BinaryToDoubleDecoder implements Decoder<BinaryString, DoubleString> {
039
040        protected final int n;
041
042        protected final int bits;
043
044        /**
045         * Constructs a {@link BinaryToDoubleDecoder}.
046         * 
047         * @param n
048         *            the n value
049         * @param bits
050         *            the number of bits per double value
051         */
052        @Inject
053        public BinaryToDoubleDecoder(@N int n, @Bits int bits) {
054                super();
055                this.n = n;
056                this.bits = bits;
057        }
058
059        /*
060         * (non-Javadoc)
061         * 
062         * @see org.opt4j.core.problem.Decoder#decode(org.opt4j.core.Genotype)
063         */
064        @Override
065        public DoubleString decode(BinaryString genotype) {
066                DoubleString phenotype = new DoubleString();
067                for (int i = 0; i < n; i++) {
068                        int begin = i * bits;
069                        int end = begin + bits;
070                        double x = toDouble(genotype, begin, end);
071                        phenotype.add(x);
072                }
073
074                return phenotype;
075        }
076
077        /**
078         * Converts a sublist of boolean values to an integer in [0;1).
079         * 
080         * @param list
081         *            the list of booleans
082         * @param begin
083         *            the begin of the sublist
084         * @param end
085         *            the end of the sublist
086         * @return the double value
087         */
088        private double toDouble(List<Boolean> list, int begin, int end) {
089                // this is very efficient since only integers and bit operations are
090                // used
091                int b = 1;
092                int sum = 0;
093                for (int i = end - 1; i >= begin; i--) {
094                        if (list.get(i)) {
095                                sum |= b;
096                        }
097                        b <<= 1;
098                }
099
100                return (double) sum / b;
101        }
102}
