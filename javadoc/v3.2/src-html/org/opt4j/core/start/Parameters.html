024package org.opt4j.core.start;
025
026import java.lang.reflect.ParameterizedType;
027import java.lang.reflect.Type;
028import java.lang.reflect.TypeVariable;
029import java.util.Collection;
030import java.util.HashMap;
031import java.util.HashSet;
032import java.util.Map;
033
034/**
035 * The {@link Parameters} is a class for the identification of types/classes of
036 * a generic parameter.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041public class Parameters {
042
043        /**
044         * Returns the class of a type.
045         * 
046         * @param type
047         *            the type
048         * @return the class
049         */
050        public static Class<?> getClass(Type type) {
051                if (type instanceof Class<?>) {
052                        return (Class<?>) type;
053                } else if (type instanceof ParameterizedType) {
054                        return (Class<?>) ((ParameterizedType) type).getRawType();
055                } else {
056                        return null;
057                }
058        }
059
060        /**
061         * Returns the type of the generic of the interface or class of an class for
062         * an object.
063         * 
064         * @param clazz
065         *            the interface with the undefined generic
066         * @param object
067         *            the current implementation of the interface
068         * @param variable
069         *            the identifier
070         * @return the implemented type of the generic
071         */
072        public static Type getType(Class<?> clazz, Object object, String variable) {
073                return getType(clazz, variable, getTypes(object.getClass()));
074        }
075
076        protected static Type getType(Class<?> clazz, String variable, Map<Type, ParameterizedType> map) {
077
078                // get first
079                ParameterizedType ptype = map.get(clazz);
080                if (ptype == null) {
081                        return null;
082                }
083
084                Type type = null;
085                for (int j = 0; j < ptype.getActualTypeArguments().length; j++) {
086                        TypeVariable<?> v = clazz.getTypeParameters()[j];
087                        if (v.getName().equals(variable)) {
088                                type = ptype.getActualTypeArguments()[j];
089                        }
090                }
091
092                while (type instanceof TypeVariable<?>) {
093                        TypeVariable<?> var = (TypeVariable<?>) type;
094                        ParameterizedType paramType = map.get(var.getGenericDeclaration());
095                        Class<?> freeType = (Class<?>) paramType.getRawType();
096
097                        int size = freeType.getTypeParameters().length;
098
099                        for (int j = 0; j < size; j++) {
100                                TypeVariable<?> f = freeType.getTypeParameters()[j];
101                                if (f.equals(var)) {
102                                        type = paramType.getActualTypeArguments()[j];
103                                        break;
104                                }
105                        }
106                }
107
108                return type;
109        }
110
111        protected static Map<Type, ParameterizedType> getTypes(Class<?> clazz) {
112                Collection<Class<?>> classes = getAllClasses(clazz);
113                Collection<ParameterizedType> paramTypes = getParameterizedTypes(classes);
114
115                Map<Type, ParameterizedType> map = new HashMap<Type, ParameterizedType>();
116
117                for (ParameterizedType p : paramTypes) {
118                        map.put(p.getRawType(), p);
119                }
120
121                return map;
122        }
123
124        protected static Collection<Class<?>> getAllClasses(final Class<?> clazz) {
125                Collection<Class<?>> set = new HashSet<Class<?>>();
126                if (clazz != null && !clazz.equals(Object.class)) {
127                        set.add(clazz);
128
129                        Class<?> superclass = clazz.getSuperclass();
130                        set.addAll(getAllClasses(superclass));
131
132                        for (Class<?> inf : clazz.getInterfaces()) {
133                                set.addAll(getAllClasses(inf));
134                        }
135                }
136                return set;
137        }
138
139        protected static Collection<ParameterizedType> getParameterizedTypes(Collection<Class<?>> classes) {
140                Collection<ParameterizedType> paramTypes = new HashSet<ParameterizedType>();
141
142                for (Class<?> clazz : classes) {
143
144                        Type type = clazz.getGenericSuperclass();
145
146                        if (type instanceof ParameterizedType) {
147                                ParameterizedType p = (ParameterizedType) type;
148                                paramTypes.add(p);
149                        }
150
151                        for (Type t : clazz.getGenericInterfaces()) {
152                                if (t instanceof ParameterizedType) {
153                                        ParameterizedType p = (ParameterizedType) t;
154                                        paramTypes.add(p);
155                                }
156                        }
157                }
158
159                return paramTypes;
160        }
161
162}
