018package org.opt4j.core.common.archive;
019
020import java.util.ArrayList;
021import java.util.Collection;
022import java.util.Collections;
023import java.util.Comparator;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027
028import org.opt4j.core.Individual;
029
030/**
031 * The {@link Crowding}, see "A fast and elitist multiobjective genetic algorithm : NSGA-II, K. Deb, A. Pratap, S.
032 * Agarwal, and T. Meyarivan Evolutionary Computation, IEEETransactions on, vol. 6, no. 2, pp. 182-197, August 2002.",
033 * calculates the crowding distance of {@link Individual}s as used for instance in the the {@code Nsga2} algorithm. A
034 * bounding box around each individual is defined and restricted by the neighboring individuals in each dimension. The
035 * crowding distance is the sum of the side-lengths of the bounding box. If an individual does not have neighbors on all
036 * sides, the crowding distance is maximal. A normalization in each dimension is applied.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041
042public class Crowding implements FrontDensityIndicator {
043
044        /**
045         * Calculates the crowding distance for a collection of {@link Individual}s.
046         * 
047         * @param individuals
048         *            the individuals
049         * @return a map of the individuals to the corresponding crowding distance
050         */
051        private Map<Individual, Double> getValues(Collection<Individual> individuals) {
052                Map<Individual, Double> map = new HashMap<Individual, Double>();
053                List<Individual> list = new ArrayList<Individual>(individuals);
054
055                for (Individual individual : list) {
056                        map.put(individual, 0.0);
057                }
058
059                int m = list.get(0).getObjectives().array().length;
060
061                for (int i = 0; i < m; i++) {
062                        final int dim = i;
063                        Collections.sort(list, new Comparator<Individual>() {
064                                @Override
065                                public int compare(Individual o1, Individual o2) {
066                                        double i1 = o1.getObjectives().array()[dim];
067                                        double i2 = o2.getObjectives().array()[dim];
068                                        return (int) Math.signum(i1 - i2);
069                                }
070                        });
071
072                        Individual minIndividual = list.get(0);
073                        Individual maxIndividual = list.get(list.size() - 1);
074
075                        double diff = maxIndividual.getObjectives().array()[dim] - minIndividual.getObjectives().array()[dim];
076                        if (diff > 0) {
077                                map.put(minIndividual, Double.MAX_VALUE);
078                                map.put(maxIndividual, Double.MAX_VALUE);
079
080                                for (int j = 1; j < list.size() - 1; j++) {
081                                        double p = list.get(j - 1).getObjectives().array()[dim];
082                                        double n = list.get(j + 1).getObjectives().array()[dim];
083                                        Individual ind = list.get(j);
084                                        map.put(ind, map.get(ind) + ((n - p) / diff));
085                                }
086                        }
087                }
088
089                return map;
090        }
091
092        /**
093         * Returns an ordered list of the {@link Individual}s corresponding to their crowding distance. {@link Individual}s
094         * with a high crowding distance are sorted to the front.
095         * 
096         * @param values
097         *            the map of individuals to their crowding distance values
098         * @return the sorted list of individuals based on their crowding distance
099         */
100        public List<Individual> order(final Map<Individual, Double> values) {
101                List<Individual> list = new ArrayList<Individual>(values.keySet());
102                Collections.sort(list, new Comparator<Individual>() {
103                        @Override
104                        public int compare(Individual o1, Individual o2) {
105                                return values.get(o2).compareTo(values.get(o1));
106                        }
107                });
108                return list;
109        }
110
111        /*
112         * (non-Javadoc)
113         * 
114         * @see org.opt4j.optimizer.ea.FrontDensityIndicator#getDensityValues(java.util .Collection)
115         */
116        @Override
117        public Map<Individual, Double> getDensityValues(Collection<Individual> individuals) {
118                return getValues(individuals);
119        }
120
121}
