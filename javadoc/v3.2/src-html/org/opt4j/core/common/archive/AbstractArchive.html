018package org.opt4j.core.common.archive;
019
020import java.util.ArrayList;
021import java.util.Collection;
022import java.util.Iterator;
023import java.util.List;
024import java.util.Set;
025
026import org.opt4j.core.Individual;
027import org.opt4j.core.Objectives;
028import org.opt4j.core.optimizer.Archive;
029
030/**
031 * This {@link AbstractArchive} provides some common methods for {@link Archive}
032 * s. If one or more new {@link Individual}s are added to this {@link Archive},
033 * it is ensured that all {@link Individual}s in this {@link Archive} are not
034 * Pareto-dominated. Actual implementations of this class may still refuse or
035 * drop some {@link Individual}s. An {@link Archive} can be a
036 * {@link BoundedArchive} if it has a bounded size or an
037 * {@link UnboundedArchive}, otherwise.
038 * 
039 * @author helwig, glass, lukasiewycz
040 * 
041 */
042public abstract class AbstractArchive extends Archive {
043
044        /*
045         * (non-Javadoc)
046         * 
047         * @see org.opt4j.core.optimizer.Archive#update(java.util.Set)
048         */
049        @Override
050        public boolean update(Set<? extends Individual> individuals) {
051                List<Individual> candidates = new ArrayList<Individual>(individuals);
052                /*
053                 * Remove all individuals that are already in the archive.
054                 */
055                candidates.removeAll(this);
056                removeDominatedCandidates(candidates);
057
058                removeArchiveDominated(candidates);
059
060                return updateWithNondominated(candidates);
061        }
062
063        /**
064         * Remove candidates, which are weakly dominated by another candidate.
065         * 
066         * Thus, the list of candidates which need to be tested against the whole
067         * archive is reduced.
068         * 
069         * @param candidates
070         *            the list of candidates to sanitize
071         */
072        void removeDominatedCandidates(List<Individual> candidates) {
073                for (int i = 0; i < candidates.size() - 1; i++) {
074                        Objectives o1 = candidates.get(i).getObjectives();
075                        for (int j = i + 1; j < candidates.size(); j++) {
076                                Objectives o2 = candidates.get(j).getObjectives();
077                                if (o2.weaklyDominates(o1)) {
078                                        candidates.remove(i);
079                                        i--;
080                                        break;
081                                } else if (o1.weaklyDominates(o2)) {
082                                        candidates.remove(j);
083                                        j--;
084                                }
085                        }
086                }
087        }
088
089        /**
090   * Remove those individuals from the candidates which are weakly
091   * dominated by the archive. Remove those individuals from the archive
092   * which are dominated by the candidates. In case of equal objectives,
093   * this gives priority to the individuals in the archive and avoids
094   * unnecessary archive updates.
095         * @param candidates
096         *            the list of candidates to sanitize
097         */
098        void removeArchiveDominated(List<Individual> candidates) {
099                Iterator<Individual> i1, i2;
100                for (i1 = candidates.iterator(); i1.hasNext();) {
101                        Objectives o1 = i1.next().getObjectives();
102                        for (i2 = this.iterator(); i2.hasNext();) {
103                                Objectives o2 = i2.next().getObjectives();
104                                if (o2.weaklyDominates(o1)) {
105                                        i1.remove();
106                                        break;
107                                } else if (o1.dominates(o2)) {
108                                        i2.remove();
109                                }
110                        }
111                }
112        }
113
114        /**
115         * Adds new {@code candidates} which are already checked to be not
116         * Pareto-dominated by any other individual in this {@link Archive}. All
117         * {@link Individual}s in the {@link Archive} which were dominated by the
118         * candidates have already been removed.
119         * 
120         * @param candidates
121         *            the non-dominated individuals which can be added
122         * @return true if one or more candidates are added to the archive
123         */
124        protected abstract boolean updateWithNondominated(Collection<Individual> candidates);
125
126}
