024package org.opt4j.core.common.logger;
025
026import static org.opt4j.core.Objective.INFEASIBLE;
027
028import java.io.BufferedWriter;
029import java.io.File;
030import java.io.FileWriter;
031import java.io.IOException;
032import java.io.PrintWriter;
033import java.util.Collection;
034
035import org.opt4j.core.Individual;
036import org.opt4j.core.Objective;
037import org.opt4j.core.Objectives;
038import org.opt4j.core.Value;
039import org.opt4j.core.optimizer.Archive;
040import org.opt4j.core.start.Constant;
041
042import com.google.inject.Inject;
043
044/**
045 * The {@link TsvLogger} writes all {@link Individual}s from the {@link Archive}
046 * to the specified file. It can be configured to write the data each
047 * {@code iterationStep} iterations or each {@code evaluationStep} evaluations.
048 * 
049 * The file format is TSV (tab separated values), according to the <a href=
050 * "http://www.iana.org/assignments/media-types/text/tab-separated-values"
051 * >Definition of tab-separated-values (tsv)</a> by IANA.
052 * 
053 * An infeasible Objective is printed as <it>INFEASIBLE</it>.
054 * 
055 * @see Objective
056 * @author reimann, lukasiewycz
057 */
058public class TsvLogger extends AbstractLogger implements Logger {
059
060        protected final Archive archive;
061
062        private final PrintWriter out;
063        private long startTime = -1;
064
065        /**
066         * Creates an {@link TsvLogger}.
067         * 
068         * @param archive
069         *            the archive
070         * @param filename
071         *            the filename (using namespace {@link TsvLogger})
072         * @param evaluationStep
073         *            the number of evaluations between two logging events (using
074         *            namespace {@link TsvLogger})
075         * @param iterationStep
076         *            the number of iterations between two logging events (using
077         *            namespace {@link TsvLogger})
078         */
079        @Inject
080        public TsvLogger(Archive archive, @Constant(value = "filename", namespace = TsvLogger.class) String filename,
081                        @Constant(value = "evaluationStep", namespace = TsvLogger.class) int evaluationStep,
082                        @Constant(value = "iterationStep", namespace = TsvLogger.class) int iterationStep) {
083                super(iterationStep, evaluationStep);
084                this.archive = archive;
085                this.out = initWriter(filename);
086        }
087
088        /**
089         * Initialize the print write from a filename
090         * 
091         * @param filename
092         *            the filename
093         * @return the print writer
094         */
095        protected PrintWriter initWriter(String filename) {
096                if (filename == null || filename.equals("")) {
097                        throw new IllegalArgumentException("File name must not be the empty string.");
098                }
099
100                File file = new File(filename);
101                File dir = file.getParentFile();
102                if (dir != null) {
103                        dir.mkdirs();
104                }
105                try {
106                        return new PrintWriter(new BufferedWriter(new FileWriter(file)));
107                } catch (IOException e) {
108                        throw new IllegalArgumentException(e);
109                }
110        }
111
112        /*
113         * (non-Javadoc)
114         * 
115         * @see org.opt4j.common.logger.AbstractLogger#optimizationStarted()
116         */
117        @Override
118        public void optimizationStarted() {
119                startTime = System.currentTimeMillis();
120        }
121
122        /*
123         * (non-Javadoc)
124         * 
125         * @see org.opt4j.common.logger.AbstractLogger#logEvent(int, int)
126         */
127        @Override
128        public void logEvent(int iteration, int evaluation) {
129                assert startTime != -1 : "not initialized";
130                double time = ((double) System.currentTimeMillis() - startTime) / 1000.0;
131                for (Individual individual : archive) {
132                        String statistics = getStatistics(iteration, evaluation, time);
133                        String objectives = getIndividual(individual);
134                        if (objectives != null) {
135                                out.println(statistics + objectives);
136                        }
137                }
138                out.flush();
139        }
140
141        /*
142         * (non-Javadoc)
143         * 
144         * @see org.opt4j.common.logger.AbstractLogger#optimizationStopped()
145         */
146        @Override
147        public void optimizationStopped() {
148                out.close();
149        }
150
151        /*
152         * (non-Javadoc)
153         * 
154         * @see
155         * org.opt4j.common.logger.AbstractLogger#logHeader(java.util.Collection)
156         */
157        @Override
158        public void logHeader(Collection<Objective> objectives) {
159                String header = getCommentDelimiter() + "iteration" + getColumnDelimiter() + "evaluations"
160                                + getColumnDelimiter() + "runtime[s]";
161                for (Objective objective : objectives) {
162                        String name = objective.getName().replaceAll("[ \n\t\r]", "_");
163                        header += getColumnDelimiter() + name + "[" + objective.getSign() + "]";
164                }
165                out.println(header);
166        }
167
168        /**
169         * The {@link String} separating two columns.
170         * 
171         * The tab character ("\t") is the default, which leads to a tab separated
172         * values file format (TSV).
173         * 
174         * @return the delimiter
175         */
176        protected String getColumnDelimiter() {
177                return "\t";
178        }
179
180        /**
181         * Creates a {@link String} representation of the given {@link Individual}.
182         * 
183         * Per default, the {@link Value}s of all {@link Objectives} of the
184         * individual, separated by {@link #getColumnDelimiter()}, are returned. The
185         * {@link String} representation of a {@link Value} must not contain the tab
186         * character.
187         * 
188         * If one of the values of the {@link Objectives} is INFEASIBLE,
189         * {@code null} is returned.
190         * 
191         * @see Objective
192         * @param individual
193         *            the individual
194         * @return the corresponding string or {@code null}
195         */
196        protected String getIndividual(Individual individual) {
197                String output = "";
198                Objectives objectives = individual.getObjectives();
199
200                for (Objective objective : objectives.getKeys()) {
201                        Value<?> value = objectives.get(objective);
202                        assert value != null : "Objective " + objective.getName() + " not set for individual " + individual;
203
204                        String valueString;
205                        if (value == INFEASIBLE || value == null || value.getValue() == null) {
206                                valueString = "INFEASIBLE";
207                        } else {
208                                String v = value.getValue().toString();
209                                if (v.contains("\t")) {
210                                        System.err.println(this + ": Value must not contain the tab character:" + v);
211                                        v.replace("\t", "_");
212                                }
213                                valueString = v;
214                        }
215                        output += getColumnDelimiter() + valueString;
216                }
217                return output;
218        }
219
220        /**
221         * Statistics for the given {@code iteration} and {@code evaluation}.
222         * 
223         * Per default, the iteration, the number of evaluations and the
224         * optimization run time in milliseconds, separated by
225         * {@link #getColumnDelimiter()}, are returned.
226         * 
227         * @param iteration
228         *            the current iteration
229         * @param evaluation
230         *            the current number of evaluations
231         * @param time
232         *            the current runtime of the optimization
233         * @return the corresponding string
234         */
235        protected String getStatistics(int iteration, int evaluation, double time) {
236                return iteration + getColumnDelimiter() + evaluation + getColumnDelimiter() + time;
237        }
238
239        /**
240         * The {@link String} starting a line comment.
241         * 
242         * Default is the empty String ("") according to the TSV specification.
243         * 
244         * @return the comment delimiter
245         */
246        protected String getCommentDelimiter() {
247                return "";
248        }
249
250}
