024package org.opt4j.core.config.visualization;
025
026import java.awt.BorderLayout;
027import java.awt.Dimension;
028
029import javax.swing.JPanel;
030import javax.swing.JSplitPane;
031import javax.swing.JTabbedPane;
032
033import org.opt4j.core.config.ExecutionEnvironment;
034import org.opt4j.core.config.Icons;
035
036import com.google.inject.Inject;
037
038/**
039 * The {@link DefaultContentPanel}.
040 * 
041 * @author lukasiewycz
042 * 
043 */
044@SuppressWarnings("serial")
045public class DefaultContentPanel extends ContentPanel {
046
047        protected final ExecutionEnvironment executionEnvironment;
048
049        protected final ModulesPanel modulesPanel;
050
051        protected final SelectedPanel selectedPanel;
052
053        protected final TasksPanel tasksPanel;
054
055        /**
056         * Constructs a {@link DefaultContentPanel}.
057         * 
058         * @param executionEnvironment
059         *            the execution environment
060         * @param modulesPanel
061         *            the modules panel
062         * @param selectedPanel
063         *            the selected panel
064         * @param tasksPanel
065         *            the tasks panel
066         */
067        @Inject
068        public DefaultContentPanel(final ExecutionEnvironment executionEnvironment, final ModulesPanel modulesPanel,
069                        final SelectedPanel selectedPanel, final TasksPanel tasksPanel) {
070                super();
071                this.executionEnvironment = executionEnvironment;
072                this.modulesPanel = modulesPanel;
073                this.selectedPanel = selectedPanel;
074                this.tasksPanel = tasksPanel;
075
076        }
077
078        /*
079         * (non-Javadoc)
080         * 
081         * @see org.opt4j.conf.gui.Startupable#startup()
082         */
083        @Override
084        public void startup() {
085
086                setLayout(new BorderLayout());
087                final JSplitPane splitPane = new JSplitPane(JSplitPane.VERTICAL_SPLIT);
088                splitPane.setOneTouchExpandable(true);
089                splitPane.setResizeWeight(1.0);
090                add(splitPane);
091
092                modulesPanel.setPreferredSize(new Dimension(220, 200));
093                selectedPanel.setPreferredSize(new Dimension(350, 200));
094
095                JPanel top = new JPanel(new BorderLayout());
096                top.add(modulesPanel, BorderLayout.WEST);
097                top.add(selectedPanel, BorderLayout.CENTER);
098
099                JTabbedPane bottom = new JTabbedPane();
100                bottom.addTab("Tasks", Icons.getIcon(Icons.CONSOLE), tasksPanel);
101
102                bottom.setPreferredSize(new Dimension(300, 160));
103                top.setMinimumSize(new Dimension(300, 80));
104
105                splitPane.add(top, JSplitPane.TOP);
106                splitPane.add(bottom, JSplitPane.BOTTOM);
107
108                modulesPanel.startup();
109                selectedPanel.startup();
110                tasksPanel.startup();
111
112                setVisible(true);
113
114        }
115
116}
