055/**
056 * The {@link PropertyModule} is a decorator for a {@link Module} that enables
057 * property methods.
058 * 
059 * @author lukasiewycz
060 * 
061 */
062@Ignore
063@SuppressWarnings("serial")
064public final class PropertyModule implements Module, Serializable, Comparable<PropertyModule> {
065
066        protected final Module module;
067
068        protected final List<Property> properties = new ArrayList<Property>();
069
070        protected static int c = 0;
071
072        public final int id;
073
074        /**
075         * Converts a String to an element of a given enumeration.
076         * 
077         * @param name
078         *            the name of the element
079         * @param type
080         *            the enumeration type
081         * @return the element of the enumeration
082         */
083        @SuppressWarnings("rawtypes")
084        public static Object toEnumElement(String name, Class<? extends Enum> type) {
085                for (Enum e : type.getEnumConstants()) {
086                        if (e.name().equalsIgnoreCase(name)) {
087                                return e;
088                        }
089                }
090                return null;
091        }
092
093        /**
094         * Constructs a {@link PropertyModule} that decorates a {@link Module}.
095         * 
096         * @param module
097         *            the decorated module
098         */
099        public PropertyModule(Module module) {
100                super();
101
102                synchronized (this) {
103                        id = c++;
104                }
105
106                this.module = module;
107
108                if (module instanceof PropertyModule) {
109                        throw new IllegalArgumentException("PropertyModule can not decorate a PropertyModule");
110                }
111
112                // search all bean methods and their annotations
113                Map<String, Method> getters = new HashMap<String, Method>();
114                Map<String, Method> setters = new HashMap<String, Method>();
115                Map<String, Class<?>> types = new HashMap<String, Class<?>>();
116                Map<String, Collection<Annotation>> annotations = new HashMap<String, Collection<Annotation>>();
117
118                for (Method method : module.getClass().getMethods()) {
119                        String name = method.getName();
120
121                        boolean get = name.startsWith("get");
122                        boolean is = name.startsWith("is");
123
124                        try {
125
126                                if (get || is) {
127
128                                        if (get) {
129                                                name = name.substring(3);
130                                        } else if (is) {
131                                                name = name.substring(2);
132                                        }
133
134                                        // this is a get method, search for a corresponding setter
135                                        Class<?> type = method.getReturnType();
136                                        Method setter = module.getClass().getMethod("set" + name, type);
137
138                                        // first letter to lower case
139                                        char[] letters = name.toCharArray();
140                                        letters[0] = Character.toLowerCase(letters[0]);
141                                        name = new String(letters);
142
143                                        Collection<Annotation> a = new HashSet<Annotation>();
144
145                                        Class<?> clazz = module.getClass();
146                                        while (!clazz.equals(Object.class)) {
147                                                try {
148                                                        Field field = clazz.getDeclaredField(name);
149                                                        for (Annotation annotation : field.getAnnotations()) {
150                                                                a.add(annotation);
151                                                        }
152                                                        break;
153                                                } catch (NoSuchFieldException e) {
154                                                }
155                                                clazz = clazz.getSuperclass();
156                                        }
157
158                                        for (Annotation annotation : method.getAnnotations()) {
159                                                a.add(annotation);
160                                        }
161                                        for (Annotation annotation : setter.getAnnotations()) {
162                                                a.add(annotation);
163                                        }
164
165                                        getters.put(name, method);
166                                        setters.put(name, setter);
167                                        types.put(name, type);
168                                        annotations.put(name, a);
169                                }
170
171                        } catch (SecurityException e) {
172                                // if the setter is not public
173                        } catch (NoSuchMethodException e) {
174                                // if the setter does not exist
175                        }
176                }
177
178                // construct the properties
179                for (String name : getters.keySet()) {
180                        Method getter = getters.get(name);
181                        Method setter = setters.get(name);
182                        Class<?> type = types.get(name);
183                        Collection<Annotation> anno = annotations.get(name);
184
185                        Property property = new Property(module, name, type, getter, setter, anno);
186                        properties.add(property);
187                }
188
189                // interpret the annotations of the properties
190                for (String name : getters.keySet()) {
191                        Collection<Annotation> a = annotations.get(name);
192                        Property property = getProperty(name);
193
194                        for (Annotation annotation : a) {
195
196                                if (annotation instanceof Info) {
197
198                                        Info info = (Info) annotation;
199                                        property.setInfo(info.value());
200
201                                } else if (annotation instanceof Order) {
202
203                                        Order order = (Order) annotation;
204                                        property.setOrder(order.value());
205
206                                } else if (annotation instanceof Required) {
207
208                                        Required required = (Required) annotation;
209
210                                        Property p = getProperty(required.property());
211                                        if (p == null) {
212                                                throw new NullPointerException("Unknown property " + required.property() + " in annotation "
213                                                                + required + " in module " + this.getClass());
214                                        }
215
216                                        Class<?> type = p.getType();
217
218                                        if (type.isEnum()) {
219
220                                                String[] values = required.elements();
221                                                Collection<Object> elements = new HashSet<Object>();
222
223                                                for (String value : values) {
224                                                        elements.add(toEnumElement(value, type.asSubclass(Enum.class)));
225                                                }
226
227                                                Requirement requirement = new EnumRequirement(p, elements);
228                                                property.addRequirement(requirement);
229
230                                        } else if (type.equals(Boolean.TYPE)) {
231
232                                                boolean value = required.value();
233
234                                                Requirement requirement = new BooleanRequirement(p, value);
235                                                property.addRequirement(requirement);
236
237                                        }
238
239                                }
240                        }
241                }
242
243                // sort the properties
244                final Map<Property, List<Property>> hierarchy = new HashMap<Property, List<Property>>();
245
246                for (Property property : properties) {
247                        Property parent = null;
248                        for (Requirement requirement : property.getRequirements()) {
249                                parent = requirement.getProperty();
250                        }
251                        assert (parent != property);
252
253                        List<Property> level = hierarchy.get(parent);
254                        if (level == null) {
255                                level = new ArrayList<Property>();
256                                hierarchy.put(parent, level);
257                        }
258                        level.add(property);
259                }
260
261                final class PropertyComparator implements Comparator<Property> {
262                        @Override
263                        public int compare(final Property o1, final Property o2) {
264                                Integer value1 = o1.getOrder();
265                                Integer value2 = o2.getOrder();
266
267                                if (!value1.equals(value2)) {
268                                        return value1.compareTo(value2);
269                                }
270
271                                String s1 = o1.getName();
272                                String s2 = o2.getName();
273                                return s1.compareTo(s2);
274                        }
275                }
276
277                for (List<Property> level : hierarchy.values()) {
278                        Collections.sort(level, new PropertyComparator());
279                }
280
281                properties.clear();
282                LinkedList<Property> added = new LinkedList<Property>();
283                added.add(null);
284
285                while (!added.isEmpty()) {
286                        Property property = added.pop();
287                        if (hierarchy.containsKey(property)) {
288                                List<Property> level = hierarchy.get(property);
289                                if (property == null) {
290                                        properties.addAll(level);
291                                } else {
292                                        int index = properties.indexOf(property);
293                                        properties.addAll(index + 1, level);
294                                }
295                                added.addAll(level);
296                        }
297                }
298
299                // remove ignored properties
300                for (Iterator<Property> it = properties.iterator(); it.hasNext();) {
301                        Property property = it.next();
302                        Ignore ignore = property.getAnnotation(Ignore.class);
303                        if (ignore != null) {
304                                it.remove();
305                        }
306                }
307
308        }

329        /**
330         * Returns the {@link Property} with the specified {@code name}.
331         * 
332         * @param name
333         *            the name of the property
334         * @return the property with the specified {@code name}
335         */
336        public Property getProperty(String name) {
337                for (Property property : properties) {
338                        if (property.getName().equals(name)) {
339                                return property;
340                        }
341                }
342                return null;
343        }

348         * @return the properties

351                return properties;
352        }

354        /*
355         * (non-Javadoc)
356         * 
357         * @see java.lang.Object#toString()
358         */
359        @Override
360        public String toString() {
361                String s = "Module " + module.getClass().getName() + " ";
362                for (Property property : properties) {
363                        s += property.getName() + "=\"" + property.getValue() + "\" ";
364                }
365
366                return s;
367        }

369        /**
370         * Configures a {@link PropertyModule} with an XML {@link Node}.
371         * 
372         * @see #getConfiguration
373         * @param node
374         *            the configuration as XML node
375         */
376        public void setConfiguration(Node node) {
377                JNode xode = new JNode(node);
378
379                Map<String, String> values = new HashMap<String, String>();
380
381                List<JNode> propertyNodes = xode.getChildren("property");
382                for (JNode propertyNode : propertyNodes) {
383                        String key = propertyNode.getAttribute("name");
384                        String value = null;
385                        if (propertyNode.hasText()) {
386                                value = propertyNode.getText();
387                        } else if (propertyNode.hasAttribute("value")) {
388                                value = propertyNode.getAttribute("value");
389                        }
390                        values.put(key, value);
391                }
392
393                for (Property property : properties) {
394                        String key = property.getName();
395                        if (values.containsKey(key)) {
396                                String value = values.get(key);
397                                try {
398                                        property.setValue(value);
399                                } catch (InvocationTargetException e) {
400                                        System.err.println(e.getMessage());
401                                }
402                        }
403                }
404        }

406        /**
407         * Returns the configuration of the {@link PropertyModule} as XML
408         * {@link Node}.
409         * 
410         * @see #setConfiguration
411         * @param document
412         *            the XML document
413         * @return the configuration as XML node
414         */
415        public Node getConfiguration(Document document) {
416                JNode node = new JNode(document, "module");
417                node.setAttribute("class", module.getClass().getName());
418
419                for (Property property : properties) {
420                        try {
421                                JNode p = node.appendChild("property");
422
423                                String name = property.getName();
424                                Object value = property.getValue();
425
426                                p.setAttribute("name", name);
427                                if (value != null) {
428                                        if (value instanceof Class<?>) {
429                                                p.setText(((Class<?>) value).getCanonicalName());
430                                        } else {
431                                                p.setText(value.toString());
432                                        }
433                                }
434
435                        } catch (IllegalArgumentException e) {
436                                e.printStackTrace();
437                        }
438
439                }
440                return node.getNode();
441        }

443        /*
444         * (non-Javadoc)
445         * 
446         * @see java.lang.Object#clone()
447         */
448        @Override
449        public PropertyModule clone() {
450                try {
451                        Module module = this.module.getClass().newInstance();
452                        PropertyModule propertyModule = new PropertyModule(module);
453                        for (Property p : getProperties()) {
454                                Property property = propertyModule.getProperty(p.getName());
455                                property.setValue(p.getValue());
456                        }
457                        return propertyModule;
458                } catch (IllegalAccessException e) {
459                        throw new IllegalArgumentException(e);
460                } catch (InstantiationException e) {
461                        throw new IllegalArgumentException(e);
462                } catch (InvocationTargetException e) {
463                        throw new IllegalArgumentException(e);
464                }
465        }

478        /*
479         * (non-Javadoc)
480         * 
481         * @see java.lang.Comparable#compareTo(java.lang.Object)
482         */
483        @Override
484        public int compareTo(PropertyModule other) {
485                Class<? extends Module> clazz1 = this.getModule().getClass();
486                Class<? extends Module> clazz2 = other.getModule().getClass();
487
488                String classname1 = clazz1.getName();
489                String classname2 = clazz2.getName();
490
491                if (!clazz1.equals(clazz2)) {
492                        return classname1.compareTo(classname2);
493                } else {
494                        Multi multi = clazz1.getAnnotation(Multi.class);
495                        if (multi == null) {
496                                return 0;
497                        } else {
498                                return this.id - other.id;
499                        }
500                }
501        }

513        /*
514         * (non-Javadoc)
515         * 
516         * @see java.lang.Object#equals(java.lang.Object)
517         */
518        @Override
519        public boolean equals(Object obj) {
520                if (this == obj) {
521                        return true;
522                }
523                if (obj == null) {
524                        return false;
525                }
526                if (getClass() != obj.getClass()) {
527                        return false;
528                }
529                PropertyModule other = (PropertyModule) obj;
530                return this.compareTo(other) == 0;
531        }
