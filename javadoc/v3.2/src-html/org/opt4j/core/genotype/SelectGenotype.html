023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Arrays;
027import java.util.List;
028import org.opt4j.core.Genotype;
029
030/**
031 * <p>
032 * The {@link SelectGenotype} selects for each index an element from the given
033 * list.
034 * </p>
035 * <p>
036 * Example problem: Draw five times from an urn which contains three balls and
037 * return the ball after each draw.<br/>
038 * Example usage: <blockquote>
039 * 
040 * <pre>
041 * SelectGenotype&lt;Ball&gt; genotype = new SelectGenotype&lt;Ball&gt;(Arrays.asList(ball1, ball2, ball3));
042 * genotype.init(new Random(), 5);
043 * </pre>
044 * 
045 * </blockquote> Example instance: [0=ball3, 1=ball2, 2=ball2, 3=ball1, 4=ball3]
046 * <br/>
047 * Example search space size: 3<sup>5</sup>
048 * </p>
049 * 
050 * @author lukasiewycz
051 * 
052 * @param <V>
053 *            the type of elements
054 */
055public class SelectGenotype<V> extends IntegerGenotype {
056
057        private static final long serialVersionUID = 1L;
058
059        protected final List<V> values;
060
061        /**
062         * Constructs a {@link SelectGenotype}.
063         * 
064         * @param values
065         *            the elements to be selected
066         */
067        public SelectGenotype(List<V> values) {
068                super(0, values.size() - 1);
069                this.values = values;
070        }
071
072        /**
073         * Constructs a {@link SelectGenotype}.
074         * 
075         * @param values
076         *            the elements to be selected
077         */
078        public SelectGenotype(V[] values) {
079                this(Arrays.asList(values));
080        }
081
082        /**
083         * Returns the element value of the index.
084         * 
085         * @param index
086         *            the index
087         * @return the element
088         */
089        public V getValue(int index) {
090                return values.get(get(index));
091        }
092
093        /*
094         * (non-Javadoc)
095         * 
096         * @see org.opt4j.genotype.IntegerGenotype#newInstance()
097         */
098        @Override
099        @SuppressWarnings({ "unchecked", "rawtypes" })
100        public <G extends Genotype> G newInstance() {
101                try {
102                        Constructor<? extends SelectGenotype> cstr = this.getClass().getConstructor(List.class);
103                        return (G) cstr.newInstance(values);
104                } catch (Exception e) {
105                        throw new RuntimeException(e);
106                }
107        }
108}
