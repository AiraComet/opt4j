023package org.opt4j.core.genotype;
024
025import java.util.Collection;
026import java.util.Map;
027
028/**
029 * The {@link MapGenotype} extends a {@link org.opt4j.core.Genotype} with
030 * {@link Map} functionalities.
031 * 
032 * @author lukasiewycz
033 * 
034 * @param <K>
035 *            the type of keys
036 * @param <V>
037 *            the type of values
038 */
039public interface MapGenotype<K, V> {
040
041        static final String ERROR_MESSAGE_NON_UNIQUE_KEYS = "The provided key objects have to be unique";
042        static final String ERROR_MESSAGE_INVALID_KEY = "Invalid key";
043        static final String ERROR_MESSAGE_OUT_OF_BOUNDS = "The provided value does not lie within the bounds for the provided key";
044        static final String ERROR_MESSAGE_UNSUPPORTED_INIT = "Use method init(Random) instead";
045
046        /**
047         * Return all keys.
048         * 
049         * @return all keys
050         */
051        public Collection<K> getKeys();
052
053        /**
054         * Returns the value for the specified key. Throws an exception if the key
055         * is not contained.
056         * 
057         * @see #setValue
058         * @param key
059         *            the key
060         * @return the value
061         */
062        public V getValue(K key);
063
064        /**
065         * Sets the value for the specified key. Throws an exception if the key is
066         * not contained.
067         * 
068         * @see #getValue
069         * @param key
070         *            the key
071         * @param value
072         *            the value
073         */
074        public void setValue(K key, V value);
075
076        /**
077         * Returns {@code true} if the key is contained.
078         * 
079         * @param key
080         *            the key
081         * @return {@code true} if the key is contained
082         */
083        public boolean containsKey(K key);
084
085        /**
086         * Returns the index of the key. Throws an exception if the key is not
087         * contained.
088         * 
089         * @param key
090         *            the key
091         * @return the index
092         */
093        public int getIndexOf(K key);
094
095}
