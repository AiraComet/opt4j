023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashSet;
029import java.util.List;
030import java.util.Random;
031import java.util.Set;
032
033import org.opt4j.core.Genotype;
034
035/**
036 * <p>
037 * The {@link BooleanMapGenotype} is a {@link BooleanGenotype} with the
038 * {@link MapGenotype} functionality.
039 * </p>
040 * <p>
041 * Example problem: Select on/off state of five switches switch<sub>1</sub>,
042 * switch<sub>2</sub>, switch<sub>3</sub>, switch<sub>4</sub>,
043 * switch<sub>5</sub><br/>
044 * Example usage: <blockquote>
045 * 
046 * <pre>
047 * BooleanMapGenotype&lt;Switch&gt; genotype = new BooleanMapGenotype&lt;Switch&gt;(
048 *              Arrays.asList(switch1, switch2, switch3, switch4, switch5));
049 * genotype.init(new Random());
050 * </pre>
051 * 
052 * </blockquote> Example instance:
053 * [switch1=true;switch2=true;switch3=false;switch4=false;switch5=true;] <br/>
054 * Example search space size: 2<sup>5</sup>
055 * </p>
056 * 
057 * @author lukasiewycz
058 * 
059 * @param <K>
060 *            the type of keys
061 */
062public class BooleanMapGenotype<K> extends BooleanGenotype implements MapGenotype<K, Boolean> {
063
064        protected final List<K> list;
065
066        /**
067         * Constructs a {@link BooleanMapGenotype}.
068         * 
069         * @param list
070         *            the list of keys
071         */
072        public BooleanMapGenotype(List<K> list) {
073                super();
074                Set<K> uniqueKeys = new HashSet<K>(list);
075                if (uniqueKeys.size() < list.size()) {
076                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
077                }
078                this.list = list;
079        }
080
081        /**
082         * Initialize this genotype with random values based on the number of keys.
083         * 
084         * @param random
085         *            the random number generator
086         */
087        public void init(Random random) {
088                super.init(random, list.size());
089        }
090
091        /**
092         * Not supported. Use {@link BooleanMapGenotype#init(Random)} instead.
093         * 
094         * @see org.opt4j.core.genotype.BooleanGenotype#init(java.util.Random, int)
095         */
096        @Override
097        public void init(Random random, int n) {
098                throw new UnsupportedOperationException(MapGenotype.ERROR_MESSAGE_UNSUPPORTED_INIT);
099        }
100
101        /*
102         * (non-Javadoc)
103         * 
104         * @see org.opt4j.genotype.MapGenotype#getValue(java.lang.Object)
105         */
106        @Override
107        public Boolean getValue(K key) {
108                if (!containsKey(key)) {
109                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
110                }
111                int i = list.indexOf(key);
112                return get(i);
113        }
114
115        /*
116         * (non-Javadoc)
117         * 
118         * @see org.opt4j.genotype.MapGenotype#setValue(java.lang.Object,
119         * java.lang.Object)
120         */
121        @Override
122        public void setValue(K key, Boolean value) {
123                if (!containsKey(key)) {
124                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
125                }
126                int i = list.indexOf(key);
127                while (size() <= i) {
128                        add(false);
129                }
130                set(i, value);
131        }
132
133        /*
134         * (non-Javadoc)
135         * 
136         * @see org.opt4j.genotype.MapGenotype#containsKey(java.lang.Object)
137         */
138        @Override
139        public boolean containsKey(K key) {
140                return list.contains(key);
141        }
142
143        /*
144         * (non-Javadoc)
145         * 
146         * @see org.opt4j.core.Genotype#newInstance()
147         */
148        @Override
149        @SuppressWarnings({ "unchecked", "rawtypes" })
150        public <G extends Genotype> G newInstance() {
151                try {
152                        Constructor<? extends BooleanMapGenotype> cstr = this.getClass().getConstructor(List.class);
153                        return (G) cstr.newInstance(list);
154                } catch (Exception e) {
155                        throw new RuntimeException(e);
156                }
157        }
158
159        /*
160         * (non-Javadoc)
161         * 
162         * @see java.util.AbstractCollection#toString()
163         */
164        @Override
165        public String toString() {
166                StringBuilder stringBuilder = new StringBuilder();
167                stringBuilder.append("[");
168                for (int i = 0; i < size(); i++) {
169                        K key = list.get(i);
170                        boolean value = this.get(i);
171                        stringBuilder.append(key + "=" + value + ";");
172                }
173                stringBuilder.append("]");
174                return stringBuilder.toString();
175        }
176
177        /*
178         * (non-Javadoc)
179         * 
180         * @see org.opt4j.genotype.MapGenotype#getIndexOf(java.lang.Object)
181         */
182        @Override
183        public int getIndexOf(K key) {
184                if (!containsKey(key)){
185                        throw new IllegalArgumentException(ERROR_MESSAGE_INVALID_KEY);
186                }
187                return list.indexOf(key);
188        }
189
190        private static final long serialVersionUID = 1L;
191
192        /*
193         * (non-Javadoc)
194         * 
195         * @see org.opt4j.genotype.MapGenotype#getKeys()
196         */
197        @Override
198        public Collection<K> getKeys() {
199                return Collections.unmodifiableList(list);
200        }
201
202}
