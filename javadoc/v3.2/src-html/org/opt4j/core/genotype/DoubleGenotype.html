018package org.opt4j.core.genotype;
019
020import java.lang.reflect.Constructor;
021import java.util.ArrayList;
022import java.util.Random;
023
024import org.opt4j.core.Genotype;
025
026/**
027 * <p>
028 * The {@link DoubleGenotype} consists of double values that can be used as a {@link Genotype}.
029 * </p>
030 * <p>
031 * Example problem: Select filling level of five bottles<br/>
032 * Example usage: <blockquote>
033 * 
034 * <pre>
035 * DoubleGenotype&lt;Switch&gt; genotype = new DoubleGenotype();
036 * genotype.init(new Random(), 5);
037 * </pre>
038 * 
039 * </blockquote> Example instance: [0.5035947840006195, 0.9693492473483428, 0.12786372316728167, 0.5299369900029843,
040 * 0.8055193291478467]<br/>
041 * Example search space size: [0;1]<sup>5</sup>
042 * </p>
043 * 
044 * @author lukasiewycz
045 * 
046 */
047@SuppressWarnings("serial")
048public class DoubleGenotype extends ArrayList<Double> implements ListGenotype<Double> {
049
050        protected final Bounds<Double> bounds;
051
052        /**
053         * Constructs a {@link DoubleGenotype} with lower bounds {@code 0.0} and upper bounds {@code 1.0}.
054         */
055        public DoubleGenotype() {
056                this(0, 1);
057        }
058
059        /**
060         * Constructs a {@link DoubleGenotype} with a specified lower and upper bound for all values.
061         * 
062         * @param lowerBound
063         *            the lower bound
064         * @param upperBound
065         *            the upper bound
066         */
067        public DoubleGenotype(double lowerBound, double upperBound) {
068                this(new FixedBounds<Double>(lowerBound, upperBound));
069        }
070
071        /**
072         * Constructs a {@link DoubleGenotype} with the given {@link Bounds}.
073         * 
074         * @param bounds
075         *            the bounds
076         */
077        public DoubleGenotype(Bounds<Double> bounds) {
078                this.bounds = bounds;
079        }
080
081        /**
082         * Returns the lower bound for the {@code i}-th element.
083         * 
084         * @param index
085         *            the {@code i}-th element
086         * @return the lower bound of the {@code i}-th element
087         */
088        public double getLowerBound(int index) {
089                return bounds.getLowerBound(index);
090        }
091
092        /**
093         * Returns the upper bound for the {@code i}-th element.
094         * 
095         * @param index
096         *            the {@code i}-th element
097         * @return the upper bound of the {@code i}-th element
098         */
099        public double getUpperBound(int index) {
100                return bounds.getUpperBound(index);
101        }
102
103        /*
104         * (non-Javadoc)
105         * 
106         * @see org.opt4j.core.Genotype#newInstance()
107         */
108        @Override
109        @SuppressWarnings("unchecked")
110        public <G extends Genotype> G newInstance() {
111                try {
112                        Constructor<? extends DoubleGenotype> cstr = this.getClass().getConstructor(Bounds.class);
113                        return (G) cstr.newInstance(bounds);
114                } catch (Exception e) {
115                        throw new RuntimeException(e);
116                }
117        }
118
119        /**
120         * Initialize this genotype with {@code n} random values.
121         * 
122         * @param random
123         *            the random number generator
124         * @param n
125         *            the number of elements in the resulting genotype
126         */
127        public void init(Random random, int n) {
128                try {
129                        getLowerBound(n - 1);
130                        getUpperBound(n - 1);
131                } catch (IndexOutOfBoundsException outOfBoundException) {
132                        throw new IllegalArgumentException(
133                                        "Cannot initialize a genotype with " + n + " entries with the specified bounds",
134                                        outOfBoundException);
135                }
136                for (int i = 0; i < n; i++) {
137                        double lo = getLowerBound(i);
138                        double hi = getUpperBound(i);
139                        double value = lo + random.nextDouble() * (hi - lo);
140                        if (i >= size()) {
141                                add(value);
142                        } else {
143                                set(i, value);
144                        }
145                }
146        }
147}
