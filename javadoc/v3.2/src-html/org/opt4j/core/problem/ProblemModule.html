024package org.opt4j.core.problem;
025
026import java.util.HashSet;
027import java.util.Set;
028
029import org.opt4j.core.Genotype;
030import org.opt4j.core.config.Icons;
031import org.opt4j.core.config.annotations.Category;
032import org.opt4j.core.config.annotations.Icon;
033import org.opt4j.core.start.Opt4JModule;
034
035import com.google.inject.Binder;
036import com.google.inject.TypeLiteral;
037import com.google.inject.multibindings.Multibinder;
038
039/**
040 * The {@link ProblemModule} is an abstract module class for the binding of the
041 * {@link Creator}, {@link Decoder}, and {@link Evaluator}.
042 * 
043 * @see Creator
044 * @see Decoder
045 * @see Evaluator
046 * @author lukasiewycz
047 * 
048 */
049@Icon(Icons.PROBLEM)
050@Category
051public abstract class ProblemModule extends Opt4JModule {
052
053        /**
054         * <p>
055         * Binds a problem. A value {@code null} is allowed. In this case, the
056         * corresponding interface is not bound. Therefore, the binding for the
057         * omitted interfaces has to be done in other modules.
058         * </p>
059         * <p>
060         * Additional {@link Evaluator}s can be bound using
061         * {@link #addEvaluator(Class)}.
062         * </p>
063         * 
064         * 
065         * @param creator
066         *            the creator
067         * @param decoder
068         *            the decoder
069         * @param evaluator
070         *            the evaluator
071         */
072        protected void bindProblem(final Class<? extends Creator<? extends Genotype>> creator,
073                        final Class<? extends Decoder<? extends Genotype, ? extends Object>> decoder,
074                        final Class<? extends Evaluator<? extends Object>> evaluator) {
075                bindProblem(binder(), creator, decoder, evaluator);
076        }
077
078        /**
079         * Binds an additional {@link Evaluator}.
080         * 
081         * @param evaluator
082         *            the evaluator to use
083         */
084        protected void addEvaluator(final Class<? extends Evaluator<? extends Object>> evaluator) {
085                addEvaluator(binder(), evaluator);
086        }
087
088        /**
089         * Binds an additional {@link Evaluator} to a given {@link Binder}.
090         * 
091         * @param binder
092         *            the guice binder to use
093         * @param evaluator
094         *            the evaluator to use
095         * @see #binder()
096         */
097        @SuppressWarnings({ "unchecked" })
098        public static void addEvaluator(Binder binder, final Class<? extends Evaluator<? extends Object>> evaluator) {
099                Multibinder<Evaluator<Object>> multibinder = Multibinder.newSetBinder(binder,
100                                new TypeLiteral<Evaluator<Object>>() {
101                                });
102                multibinder.addBinding().to((Class<Evaluator<Object>>) evaluator);
103        }
104
105        /**
106         * <p>
107         * Binds a problem. A value {@code null} is allowed. In this case, the
108         * corresponding interface is not bound. Therefore, the binding for the
109         * omitted interfaces has to be done in other modules.
110         * </p>
111         * <p>
112         * Additional {@link Evaluator}s can be bound using
113         * {@link #addEvaluator(Class)}.
114         * </p>
115         * 
116         * @param binder
117         *            the guice binder
118         * @param creator
119         *            the creator
120         * @param decoder
121         *            the decoder
122         * @param evaluator
123         *            the evaluator
124         * @see #binder()
125         */
126        @SuppressWarnings("unchecked")
127        public static void bindProblem(Binder binder, final Class<? extends Creator<? extends Genotype>> creator,
128                        final Class<? extends Decoder<? extends Genotype, ? extends Object>> decoder,
129                        final Class<? extends Evaluator<? extends Object>> evaluator) {
130
131                Set<Class<?>> classes = new HashSet<Class<?>>();
132                if (creator != null) {
133                        classes.add(creator);
134                }
135                if (decoder != null) {
136                        classes.add(decoder);
137                }
138
139                for (Class<?> clazz : classes) {
140                        binder.bind(clazz).in(SINGLETON);
141                }
142
143                if (creator != null) {
144                        binder.bind(new TypeLiteral<Creator<Genotype>>() {
145                        }).to((Class<? extends Creator<Genotype>>) creator);
146                }
147                if (decoder != null) {
148                        binder.bind(new TypeLiteral<Decoder<Genotype, Object>>() {
149                        }).to((Class<? extends Decoder<Genotype, Object>>) decoder);
150                }
151                if (evaluator != null) {
152                        addEvaluator(binder, evaluator);
153                }
154        }
155}
