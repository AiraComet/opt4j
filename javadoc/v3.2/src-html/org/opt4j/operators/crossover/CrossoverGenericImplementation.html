024package org.opt4j.operators.crossover;
025
026import org.opt4j.core.Genotype;
027import org.opt4j.core.genotype.CompositeGenotype;
028import org.opt4j.operators.AbstractGenericOperator;
029
030import com.google.inject.Inject;
031import com.google.inject.Singleton;
032
033/**
034 * Implementation of the {@link Crossover} interface.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039@Singleton
040public class CrossoverGenericImplementation extends AbstractGenericOperator<Crossover<Genotype>, Crossover<?>>
041                implements Crossover<Genotype> {
042
043        /**
044         * Constructs the {@link CrossoverGenericImplementation}.
045         */
046        @SuppressWarnings("unchecked")
047        @Inject
048        protected CrossoverGenericImplementation() {
049                super(CrossoverBoolean.class, CrossoverDouble.class, CrossoverInteger.class, CrossoverPermutation.class);
050        }
051
052        /*
053         * (non-Javadoc)
054         * 
055         * @see
056         * org.opt4j.operator.crossover.Crossover#crossover(org.opt4j.core.Genotype,
057         * org.opt4j.core.Genotype)
058         */
059        @Override
060        public Pair<Genotype> crossover(Genotype parent1, Genotype parent2) {
061                Crossover<Genotype> crossover = getOperator(parent1);
062
063                if (crossover == null) {
064                        return crossoverComposite((CompositeGenotype<?, ?>) parent1, (CompositeGenotype<?, ?>) parent2);
065                } else {
066                        return crossover.crossover(parent1, parent2);
067                }
068        }
069
070        protected Pair<Genotype> crossoverComposite(CompositeGenotype<?, ?> p1, CompositeGenotype<?, ?> p2) {
071                CompositeGenotype<Object, Genotype> o1 = p1.newInstance();
072                CompositeGenotype<Object, Genotype> o2 = p2.newInstance();
073
074                for (Object key : p1.keySet()) {
075                        final Genotype g1 = p1.get(key);
076                        final Genotype g2 = p2.get(key);
077                        Pair<Genotype> genotype = crossover(g1, g2);
078                        o1.put(key, genotype.getFirst());
079                        o2.put(key, genotype.getSecond());
080                }
081
082                Pair<Genotype> offspring = new Pair<Genotype>(o1, o2);
083                return offspring;
084        }
085}
