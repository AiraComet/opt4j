024package org.opt4j.operators.crossover;
025
026import org.opt4j.core.common.random.Rand;
027import org.opt4j.core.start.Constant;
028import org.opt4j.operators.normalize.NormalizeDouble;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link CrossoverDoubleBLX} is an implementation of the blend crossover
034 * operator proposed by Eshelman and Schaffer, 1993.
035 * 
036 * <p>
037 * The BLX crossover is applied element-wise. For two double values {@code y}
038 * and {@code x} (assuming {@code y > x} one offspring value is created in the
039 * uniform interval {@code [x-(y-x)*alpha;y+(y-x)*alpha]}.
040 * </p>
041 * 
042 * <p>
043 * If {@code alpha} is set to {@code 0}, the operator creates a random solution
044 * between {@code x} and {@code y}. Values greater zero allow offspring that is
045 * apart from the interval between {@code x} and {@code y}. The authors report
046 * best performance with {@code alpha = 0.5} which is then called the BLX-0.5
047 * operator.
048 * </p>
049 * 
050 * @author glass, lukasiewycz
051 * 
052 */
053public class CrossoverDoubleBLX extends CrossoverDoubleElementwise {
054
055        protected final double alpha;
056
057        /**
058         * Constructs a {@link CrossoverDoubleBLX} with an alpha value, an
059         * {@link NormalizeDouble} operator, and a random number generator.
060         * 
061         * @param alpha
062         *            the alpha value
063         * @param normalize
064         *            the normalize operator
065         * @param random
066         *            the random number generator
067         */
068        @Inject
069        public CrossoverDoubleBLX(@Constant(value = "alpha", namespace = CrossoverDoubleBLX.class) double alpha,
070                        NormalizeDouble normalize, Rand random) {
071                super(normalize, random);
072                this.alpha = alpha;
073        }
074
075        /*
076         * (non-Javadoc)
077         * 
078         * @see
079         * org.opt4j.operator.crossover.CrossoverDoubleElementwise#crossover(double,
080         * double)
081         */
082        @Override
083        public Pair<Double> crossover(double x, double y) {
084                double lo = Math.min(x, y);
085                double hi = Math.max(x, y);
086
087                double o1 = next(lo, hi);
088                double o2 = next(lo, hi);
089
090                return new Pair<Double>(o1, o2);
091        }
092
093        /**
094         * Calculates a random value in the interval
095         * {@code [lo-(hi-lo)*alpha;hi+(hi-lo)*alpha]}.
096         * 
097         * @param lo
098         *            the smaller double value
099         * @param hi
100         *            the bigger double value
101         * @return a random value in the interval
102         */
103        protected double next(double lo, double hi) {
104                double u = random.nextDouble();
105                double gamma = (1 + 2 * alpha) * u - alpha;
106
107                double o = lo + gamma * (hi - lo);
108                return o;
109        }
110
111}
