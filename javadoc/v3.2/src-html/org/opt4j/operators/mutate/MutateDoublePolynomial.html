024package org.opt4j.operators.mutate;
025
026import org.opt4j.core.common.random.Rand;
027import org.opt4j.core.start.Constant;
028import org.opt4j.operators.normalize.NormalizeDouble;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link MutateDoublePolynomial}.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038public class MutateDoublePolynomial extends MutateDoubleElementwise {
039
040        protected final double eta;
041
042        /**
043         * Constructs a {@link MutateDoubleGauss} with a {@link Rand} random number
044         * generator, a {@link NormalizeDouble}, and an eta value.
045         * 
046         * @param random
047         *            the random number generator
048         * @param normalize
049         *            the normalizer
050         * @param eta
051         *            the eta value
052         */
053        @Inject
054        public MutateDoublePolynomial(Rand random, NormalizeDouble normalize,
055                        @Constant(value = "eta", namespace = MutateDoublePolynomial.class) double eta) {
056                super(random, normalize);
057                this.eta = eta;
058        }
059
060        /*
061         * (non-Javadoc)
062         * 
063         * @see
064         * org.opt4j.operator.mutate.MutateDoubleElementwise#mutateElement(double,
065         * double, double, double)
066         */
067        @Override
068        protected double mutateElement(double x, double lb, double ub, double p) {
069                if (random.nextDouble() < p) {
070                        double u = random.nextDouble();
071                        double delta = 0;
072                        double diff = ub - lb;
073
074                        if (u < 0.5) {
075                                double m = 1.0 - ((x - lb) / diff);
076                                double b = 2 * u + (1 - 2 * u) * (Math.pow(m, (eta + 1)));
077                                delta = Math.pow(b, (1.0 / (eta + 1))) - 1.0;
078                        } else {
079                                double m = 1.0 - ((ub - x) / diff);
080                                double b = 2 * (1 - u) + 2 * (u - 0.5) * (Math.pow(m, (eta + 1)));
081                                delta = 1.0 - Math.pow(b, (1.0 / (eta + 1)));
082                        }
083
084                        return x + delta * diff;
085                }
086                return x;
087        }
088}
