024package org.opt4j.operators.copy;
025
026import org.opt4j.core.genotype.BooleanGenotype;
027import org.opt4j.core.genotype.DoubleGenotype;
028import org.opt4j.core.genotype.ListGenotype;
029import org.opt4j.core.genotype.PermutationGenotype;
030
031/**
032 * Copy operator for plain lists like {@link BooleanGenotype},
033 * {@link DoubleGenotype}, and {@link PermutationGenotype}.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038public class CopyList implements Copy<ListGenotype<?>> {
039
040        /*
041         * (non-Javadoc)
042         * 
043         * @see org.opt4j.operator.copy.Copy#copy(org.opt4j.core.Genotype)
044         */
045        @Override
046        public ListGenotype<?> copy(ListGenotype<?> genotype) {
047                ListGenotype<Object> copy = genotype.newInstance();
048
049                for (Object element : genotype) {
050                        copy.add(element);
051                }
052
053                return copy;
054        }
055
056}
