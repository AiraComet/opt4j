018package org.opt4j.optimizers.mopso;
019
020import static org.opt4j.core.config.annotations.Citation.PublicationMonth.UNKNOWN;
021
022import org.opt4j.core.IndividualFactory;
023import org.opt4j.core.config.annotations.Citation;
024import org.opt4j.core.config.annotations.Info;
025import org.opt4j.core.config.annotations.Name;
026import org.opt4j.core.config.annotations.Order;
027import org.opt4j.core.optimizer.MaxIterations;
028import org.opt4j.core.optimizer.OptimizerModule;
029import org.opt4j.core.start.Constant;
030
031/**
032 * The {@link MOPSOModule} for the {@link MOPSO} optimizer.
033 * 
034 * @author lukasiewycz
035 */
036@Info("Multi-objective particle swarm optimizer. Mostly based on the OMOPSO. Works only with real-valued problems.")
037@Citation(title = "Improving PSO-based Multi-Objective Optimization using Crowding, Mutation and Ïµ-Dominance", authors = "Margarita Reyes Sierra and Carlos A. Coello Coello", journal = "Proceedings of Evolutionary Multi-Criterion Optimization", pageFirst = 505, pageLast = 519, month = UNKNOWN, year = 2005, doi = "10.1007/978-3-540-31880-4_35")
038public class MOPSOModule extends OptimizerModule {
039
040        @Info("The number of particles.")
041        @Order(1)
042        @Constant(value = "size", namespace = MOPSO.class)
043        protected int particles = 100;
044
045        @Info("The archive size of the global leaders.")
046        @Name("archive size")
047        @Order(2)
048        @Constant(value = "archiveSize", namespace = MOPSO.class)
049        protected int archiveSize = 100;
050
051        @Info("The number of iterations.")
052        @Order(0)
053        @MaxIterations
054        protected int iterations = 1000;
055
056        @Info("The perturbation index for the mutation.")
057        @Order(3)
058        protected double perturbation = 0.5;
059
060        /**
061         * Returns the perturbation.
062         * 
063         * @see #setPerturbation
064         * @return the perturbation
065         */
066        public double getPerturbation() {
067                return perturbation;
068        }
069
070        /**
071         * Sets the perturbation.
072         * 
073         * @see #getPerturbation
074         * @param perturbation
075         *            the perturbation to set
076         */
077        public void setPerturbation(double perturbation) {
078                this.perturbation = perturbation;
079        }
080
081        /**
082         * Returns the number of iterations.
083         * 
084         * @see #setIterations
085         * @return the iterations
086         */
087        public int getIterations() {
088                return iterations;
089        }
090
091        /**
092         * Sets the the number of iterations.
093         * 
094         * @see #getIterations
095         * @param iterations
096         *            the iterations to set
097         */
098        public void setIterations(int iterations) {
099                this.iterations = iterations;
100        }
101
102        /**
103         * Returns the number of particles.
104         * 
105         * @see #setParticles
106         * @return the particles
107         */
108        public int getParticles() {
109                return particles;
110        }
111
112        /**
113         * Sets the number of particles.
114         * 
115         * @see #getParticles
116         * @param particles
117         *            the particles to set
118         */
119        public void setParticles(int particles) {
120                this.particles = particles;
121        }
122
123        /**
124         * Returns the leader archive size.
125         * 
126         * @see #setArchiveSize
127         * @return the archiveSize
128         */
129        public int getArchiveSize() {
130                return archiveSize;
131        }
132
133        /**
134         * Sets the leader archive size.
135         * 
136         * @see #getArchiveSize
137         * @param archiveSize
138         *            the archiveSize to set
139         */
140        public void setArchiveSize(int archiveSize) {
141                this.archiveSize = archiveSize;
142        }
143
144        /*
145         * (non-Javadoc)
146         * 
147         * @see org.opt4j.start.Opt4JModule#config()
148         */
149        @Override
150        protected void config() {
151                bindIterativeOptimizer(MOPSO.class);
152                bind(IndividualFactory.class).to(ParticleFactory.class);
153
154                bindConstant("perturbation", MutateDoubleNonUniform.class).to(perturbation);
155                bindConstant("perturbation", MutateDoubleUniform.class).to(perturbation);
156        }
157
158}
