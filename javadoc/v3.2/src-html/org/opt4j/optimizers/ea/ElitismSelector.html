024package org.opt4j.optimizers.ea;
025
026import java.util.ArrayList;
027import java.util.Collection;
028import java.util.Collections;
029import java.util.Comparator;
030import java.util.HashMap;
031import java.util.List;
032import java.util.Map;
033import java.util.Random;
034
035import org.opt4j.core.Individual;
036import org.opt4j.core.common.random.Rand;
037
038import com.google.inject.Inject;
039
040/**
041 * The {@link ElitismSelector} is a single objective elitism select. If a
042 * multi-objective problem is optimized, the objectives are summed up to a
043 * single value.
044 * 
045 * @author lukasiewycz
046 * 
047 */
048public class ElitismSelector implements Selector {
049
050        protected final Random random;
051
052        protected final Map<Individual, Double> fitness = new HashMap<Individual, Double>();
053
054        /**
055         * Comparator that sorts the {@link Individual}s based on their fitness
056         * values.
057         * 
058         * @author lukasiewycz
059         * 
060         */
061        protected class FitnessComparator implements Comparator<Individual> {
062                @Override
063                public int compare(Individual o1, Individual o2) {
064                        final double f1 = fitness.get(o1);
065                        final double f2 = fitness.get(o2);
066                        if (f1 < f2) {
067                                return -1;
068                        } else if (f2 < f1) {
069                                return 1;
070                        } else {
071                                return 0;
072                        }
073                }
074        }
075
076        /**
077         * Constructs an {@link ElitismSelector}.
078         * 
079         * @param random
080         *            the random number generator
081         */
082        @Inject
083        public ElitismSelector(Rand random) {
084                this.random = random;
085        }
086
087        /*
088         * (non-Javadoc)
089         * 
090         * @see org.opt4j.optimizer.ea.Selector#getLames(int, java.util.Collection)
091         */
092        @Override
093        public Collection<Individual> getLames(int lambda, Collection<Individual> population) {
094                List<Individual> list = new ArrayList<Individual>(population);
095                calculateFitness(list);
096                Collections.sort(list, new FitnessComparator());
097                Collections.reverse(list);
098
099                List<Individual> lames = new ArrayList<Individual>();
100                for (int i = 0; i < lambda; i++) {
101                        lames.add(list.get(i));
102                }
103                return lames;
104        }
105
106        /*
107         * (non-Javadoc)
108         * 
109         * @see org.opt4j.optimizer.ea.Selector#getParents(int,
110         * java.util.Collection)
111         */
112        @Override
113        public Collection<Individual> getParents(int mu, Collection<Individual> population) {
114                List<Individual> parents = new ArrayList<Individual>();
115                List<Individual> individuals = new ArrayList<Individual>(population);
116
117                for (int i = 0; i < mu; i++) {
118                        int r = random.nextInt(individuals.size());
119                        parents.add(individuals.get(r));
120                }
121                return parents;
122        }
123
124        /**
125         * Calculates the fitness of the {@link Individual}s: the sum of all double
126         * values (these always have to be minimized) of the objectives.
127         * 
128         * @param individuals
129         *            the individuals to process
130         */
131        protected void calculateFitness(Collection<Individual> individuals) {
132                fitness.clear();
133                for (Individual individual : individuals) {
134                        double f = 0;
135                        for (double v : individual.getObjectives().array()) {
136                                f += v;
137                        }
138                        fitness.put(individual, f);
139                }
140        }
141
142        /*
143         * (non-Javadoc)
144         * 
145         * @see org.opt4j.optimizer.ea.Selector#init(int)
146         */
147        @Override
148        public void init(int maxsize) {
149                // do nothing
150        }
151
152}
