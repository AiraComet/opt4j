023package org.opt4j.optimizers.ea;
024
025import java.util.Collection;
026
027import org.opt4j.core.Individual;
028import org.opt4j.core.IndividualFactory;
029import org.opt4j.core.optimizer.IndividualCompleter;
030import org.opt4j.core.optimizer.IterativeOptimizer;
031import org.opt4j.core.optimizer.Population;
032import org.opt4j.core.optimizer.TerminationException;
033import org.opt4j.core.start.Constant;
034import org.opt4j.operators.crossover.Crossover;
035import org.opt4j.operators.mutate.Mutate;
036
037import com.google.inject.Inject;
038
039/**
040 * The {@link EvolutionaryAlgorithm} is an implementation of an Evolutionary
041 * Algorithm based on the operators {@link Crossover} and {@link Mutate}. It
042 * uses a {@link Selector} for the mating and environmental selection.
043 * 
044 * @author lukasiewycz, glass
045 * 
046 */
047public class EvolutionaryAlgorithm implements IterativeOptimizer {
048
049        protected final int alpha;
050
051        protected final int lambda;
052
053        protected final int mu;
054
055        protected final Selector selector;
056
057        protected final Mating mating;
058
059        private final IndividualFactory individualFactory;
060
061        private final Population population;
062
063        private final IndividualCompleter completer;
064
065        /**
066         * Constructs an {@link EvolutionaryAlgorithm} with a {@link Population}, an
067         * {@link IndividualFactory}, a {@link IndividualCompleter}, a
068         * {@link Selector}, a {@link Mating}, the number of generations, the
069         * population size, the number of parents, the number of offspring, and a
070         * random number generator.
071         * 
072         * @param population
073         *            the population
074         * @param individualFactory
075         *            the individual factory
076         * @param completer
077         *            the completer
078         * @param selector
079         *            the selector
080         * @param mating
081         *            the mating
082         * @param alpha
083         *            the population size
084         * @param mu
085         *            the number of parents
086         * @param lambda
087         *            the number of offspring
088         */
089        @Inject
090        public EvolutionaryAlgorithm(
091                        Population population,
092                        IndividualFactory individualFactory,
093                        IndividualCompleter completer,
094                        Selector selector,
095                        Mating mating,
096                        @Constant(value = "alpha", namespace = EvolutionaryAlgorithm.class) int alpha,
097                        @Constant(value = "mu", namespace = EvolutionaryAlgorithm.class) int mu,
098                        @Constant(value = "lambda", namespace = EvolutionaryAlgorithm.class) int lambda) {
099                this.selector = selector;
100                this.mating = mating;
101                this.alpha = alpha;
102                this.mu = mu;
103                this.lambda = lambda;
104                this.individualFactory = individualFactory;
105                this.population = population;
106                this.completer = completer;
107
108                if (alpha <= 0) {
109                        throw new IllegalArgumentException("Invalid alpha: " + alpha);
110                }
111
112                if (mu <= 0) {
113                        throw new IllegalArgumentException("Invalid mu: " + mu);
114                }
115
116                if (lambda <= 0) {
117                        throw new IllegalArgumentException("Invalid lambda: " + lambda);
118                }
119        }
120
121        /*
122         * (non-Javadoc)
123         * 
124         * @see org.opt4j.core.optimizer.IterativeOptimizer#initialize()
125         */
126        @Override
127        public void initialize() {
128                selector.init(alpha + lambda);
129        }
130
131        /*
132         * (non-Javadoc)
133         * 
134         * @see org.opt4j.core.optimizer.IterativeOptimizer#next()
135         */
136        @Override
137        public void next() throws TerminationException {
138                // remove lames
139                if (population.size() > alpha) {
140                        Collection<Individual> lames = selector.getLames(population.size()
141                                        - alpha, population);
142                        population.removeAll(lames);
143                }
144                // set the number of new offspring individuals per generation
145                int offspringCount = lambda;
146
147                // 1) fill the population until it reaches the size alpha
148                while (population.size() < alpha && offspringCount > 0) {
149                        population.add(individualFactory.create());
150                        offspringCount--;
151                }
152
153                // 2) generate offspring by mating
154                if (offspringCount > 0) {
155                        if (offspringCount < lambda) { // evaluate new individuals first
156                                completer.complete(population);
157                        }
158
159                        Collection<Individual> parents = selector
160                                        .getParents(mu, population);
161                        Collection<Individual> offspring = mating.getOffspring(
162                                        offspringCount, parents);
163                        population.addAll(offspring);
164                }
165        }
166}
