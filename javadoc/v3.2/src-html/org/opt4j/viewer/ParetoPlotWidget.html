060/**
061 * The {@link ParetoPlotWidget} is a widget that displays the {@link Population}
062 * and {@link Archive} in two dimensional plot.
063 * 
064 * @author lukasiewycz
065 * 
066 */
067@WidgetParameters(title = "Pareto Plot", icon = Icons.PARETO)
068public class ParetoPlotWidget implements OptimizerIterationListener, Widget, ObjectivesListener {
069
070        protected final SynchronizedIndividualSet population;
071
072        protected final SynchronizedIndividualSet archive;
073
074        protected final DelayTask task = new DelayTask(40);
075
076        protected final Plot plot;
077
078        protected final Selection selection;
079
080        protected final JPanel panel = new JPanel();
081
082        /**
083         * The selection box for the current two objectives.
084         * 
085         * @author lukasiewycz
086         * 
087         */
088        protected class Selection extends JToolBar implements ActionListener {
089
090                private static final long serialVersionUID = 1L;
091
092                protected ObjectiveDropDown firstComboBox;
093
094                protected ObjectiveDropDown secondComboBox;
095
096                public Selection() {
097                        firstComboBox = new ObjectiveDropDown();
098                        secondComboBox = new ObjectiveDropDown();
099
100                        firstComboBox.addActionListener(this);
101                        secondComboBox.addActionListener(this);
102
103                        add(new JLabel("x-Axis: "));
104                        add(firstComboBox);
105                        addSeparator();
106                        add(new JLabel("y-Axis: "));
107                        add(secondComboBox);
108                        setFloatable(false);
109
110                        Border border = BorderFactory.createMatteBorder(0, 0, 1, 0, getBackground().darker());
111                        setBorder(border);
112
113                }
114
115                public void init(Collection<Objective> objectives) {
116                        for (Objective objective : objectives) {
117                                firstComboBox.addItem(objective);
118                                secondComboBox.addItem(objective);
119                        }
120
121                        initSelection(new ArrayList<Objective>(objectives));
122                }
123
124                private void initSelection(List<Objective> objectives) {
125                        if (objectives.size() <= 0) {
126                                return;
127                        }
128
129                        final Objective objective1 = objectives.get(0);
130                        final Objective objective2;
131                        if (objectives.size() > 1) {
132                                objective2 = objectives.get(1);
133                        } else {
134                                objective2 = objectives.get(0);
135                        }
136                        firstComboBox.setSelected(objective1);
137                        secondComboBox.setSelected(objective2);
138                }
139
140                public Objective getFirst() {
141                        return firstComboBox.getSelected();
142                }
143
144                public Objective getSecond() {
145                        return secondComboBox.getSelected();
146                }
147
148                @Override
149                public void actionPerformed(ActionEvent e) {
150                        ParetoPlotWidget.this.doPaint();
151                }
152        }
153
154        /**
155         * Constructs a {@link ParetoPlotWidget}.
156         * 
157         * @param population
158         *            the population
159         * @param archive
160         *            the archive
161         * @param autoZoomButton
162         *            the button to reset zooming
163         */
164        @Inject
165        public ParetoPlotWidget(Population population, Archive archive, AutoZoomButton autoZoomButton) {
166                this.population = new SynchronizedIndividualSet(population);
167                this.archive = new SynchronizedIndividualSet(archive);
168
169                selection = new Selection();
170                selection.addSeparator();
171                selection.add(autoZoomButton);
172
173                plot = new Plot();
174                autoZoomButton.setPlotBox(plot);
175
176                plot.addLegend(0, "Archive");
177                plot.addLegend(1, "Population");
178                plot.setMarksStyle("dots");
179
180                Color[] colors = new Color[3];
181                colors[0] = Color.RED;
182                colors[1] = Color.LIGHT_GRAY;
183                colors[2] = Color.BLUE;
184                plot.setColors(colors);
185
186                panel.setLayout(new BorderLayout());
187                panel.add(selection, BorderLayout.NORTH);
188                panel.add(plot, BorderLayout.CENTER);
189        }
190

199                optimizer.addOptimizerIterationListener(this);
200        }

204         * 
205         * @param objectivesMonitor
206         *            the objectives monitor
207         */
208        @Inject(optional = true)
209        public void registerAtObjectivesMonitor(ObjectivesMonitor objectivesMonitor) {
210                objectivesMonitor.addListener(this);
211        }

220
221        }

223        /*
224         * (non-Javadoc)
225         * 
226         * @see
227         * org.opt4j.core.optimizer.OptimizerIterationListener#iterationComplete
228         * (org.opt4j.core.optimizer.Optimizer, int)
229         */
230        @Override
231        public void iterationComplete(int iteration) {
232                doPaint();
233        }
234
235        /**
236         * Returns the two dimensional representation of the given
237         * {@code IndividualCollection}.
238         * 
239         * @param indivualCollection
240         *            the {IndividualCollection
241         * @return a set of points
242         */
243        protected Set<Point2D.Double> getPoints(Collection<Individual> indivualCollection, Objective one, Objective two) {
244
245                Set<Point2D.Double> points = new HashSet<Point2D.Double>();
246                for (Individual individual : indivualCollection) {
247                        if (individual != null && individual.isEvaluated()) {
248                                Objectives objectives = individual.getObjectives();
249
250                                if (one == null || two == null) {
251                                        continue;
252                                }
253
254                                Value<?> onev = objectives.get(one);
255                                Value<?> twov = objectives.get(two);
256
257                                if (onev == null || twov == null || onev.getValue() == Objective.INFEASIBLE
258                                                || twov.getValue() == Objective.INFEASIBLE) {
259                                        continue;
260                                }
261
262                                double x = objectives.get(one).getDouble();
263                                double y = objectives.get(two).getDouble();
264                                Point2D.Double point = new Point2D.Double(x, y);
265                                points.add(point);
266                        }
267                }
268
269                return points;
270        }
271
272        /**
273         * Repaints the diagram.
274         */
275        protected void doPaint() {
276                task.execute(new Runnable() {
277                        @Override
278                        public void run() {
279                                final Collection<Individual> a = new HashSet<Individual>();
280                                synchronized (archive) {
281                                        a.addAll(archive);
282                                }
283                                final Collection<Individual> p = new HashSet<Individual>();
284                                synchronized (population) {
285                                        p.addAll(population);
286                                }
287
288                                paint(a, p);
289                        }
290                });
291        }
292
293        /**
294         * Repaints the diagram.
295         */
296        protected void paint(Collection<Individual> archive, Collection<Individual> population) {
297                final Objective one = selection.getFirst();
298                final Objective two = selection.getSecond();
299
300                population.removeAll(archive);
301                final Set<Point2D.Double> archivePoints = getPoints(archive, one, two);
302                final Set<Point2D.Double> populationPoints = getPoints(population, one, two);
303
304                SwingUtilities.invokeLater(new Runnable() {
305                        @Override
306                        public void run() {
307                                plot.clear(false);
308                                plot.setXLabel(one != null ? one.getName() : "");
309                                plot.setYLabel(two != null ? two.getName() : "");
310
311                                for (Point2D.Double point : archivePoints) {
312                                        plot.addPoint(0, point.getX(), point.getY(), false);
313                                }
314
315                                for (Point2D.Double point : populationPoints) {
316                                        plot.addPoint(1, point.getX(), point.getY(), false);
317                                }
318
319                                plot.revalidate();
320                                plot.repaint();
321                        }
322                });
323        }
324

326         * (non-Javadoc)
327         * 
328         * @see org.opt4j.viewer.Widget#getPanel()
329         */
330        @Override
331        public JPanel getPanel() {
332                return panel;
333        }

340         * .util.Collection)
341         */
342        @Override
343        public void objectives(Collection<Objective> objectives) {
344                selection.init(objectives);
345        }
