024package org.opt4j.viewer;
025
026import java.awt.event.ActionEvent;
027import java.awt.event.ActionListener;
028
029import javax.swing.JButton;
030
031import ptolemy.plot.PlotBox;
032import ptolemy.plot.zoomBox.ZoomIcons;
033
034/**
035 * The {@link AutoZoomButton} is a specialized {@link JButton} which reactivates
036 * the auto zoom feature of a {@link PlotBox}. If the auto zoom feature is
037 * enabled, the button is disabled.
038 * 
039 * @author reimann
040 * 
041 */
042public class AutoZoomButton extends JButton implements ActionListener {
043        private static final long serialVersionUID = 1L;
044
045        private PlotBox plotBox = null;
046
047        /**
048         * Creates a new {@link AutoZoomButton}.
049         */
050        public AutoZoomButton() {
051                super(ZoomIcons.getIcon(ZoomIcons.ZOOMaUTO));
052                addActionListener(this);
053                setEnabled(false);
054                setToolTipText("Auto Zoom");
055                setPreferredSize(getMinimumSize());
056        }
057
058        /**
059         * Returns the {@link PlotBox} this {@link AutoZoomButton} is listening to.
060         * 
061         * @return the plot box
062         */
063        public PlotBox getPlotBox() {
064                return plotBox;
065        }
066
067        /**
068         * Set the {@link PlotBox} to listen for.
069         * 
070         * @param plotBox
071         *            the plot box
072         */
073        public void setPlotBox(PlotBox plotBox) {
074                if (this.plotBox != null) {
075                        this.plotBox.removeListener(this);
076                }
077                this.plotBox = plotBox;
078                plotBox.registerListener(this);
079        }
080
081        /*
082         * (non-Javadoc)
083         * 
084         * @see
085         * java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
086         */
087        @Override
088        public void actionPerformed(ActionEvent e) {
089                if (plotBox != null) {
090                        if (e.getSource() == this) {
091                                // this button was pressed
092                                plotBox.reactivateAutoScale();
093                        } else if (e.getSource() == plotBox
094                                        && e.getActionCommand() == PlotBox.Action.MANUAL_ZOOM_ACTIVATED.toString()) {
095                                // the plot box has activated manual zoom
096                                setEnabled(true);
097                        } else if (e.getSource() == plotBox
098                                        && e.getActionCommand() == PlotBox.Action.AUTOMATIC_ZOOM_ACTIVATED.toString()) {
099                                // the plot box has activated automatic zoom
100                                setEnabled(false);
101                        }
102                }
103        }
104}
