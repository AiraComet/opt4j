068/**
069 * The {@link SAT4JSolver} implements a {@link Solver}. It is based on the Java
070 * SAT/PB-Solver from <a href="http://www.sat4j.org">Sat4J.org</a>.
071 * 
072 * @author lukasiewycz
073 * 
074 */
075@Singleton
076public class SAT4JSolver implements Solver {
077
078        public enum Learning {
079                @Info("Learning only constraints with less or equal number of variables.")
080                FIXEDLENGTH, @Info("Use the MiniSAT learning scheme.")
081                MINISAT, @Info("Learn only clauses.")
082                CLAUSEONLY;
083        }
084
085        public enum Restarts {
086                @Info("Use MiniSAT restarts.")
087                MINISAT, @Info("Use Luby restarts.")
088                LUBY, @Info("Use Rapid restarts.")
089                RAPID;
090        }
091
092        protected final Queue<Constraint> constraints = new ConcurrentLinkedQueue<Constraint>();
093
094        protected final PBSolverResolution solver;
095
096        protected final Map<Object, Integer> variables = new HashMap<Object, Integer>();
097
098        protected int nextVariable = 1;
099
100        private boolean solverValid = false;
101
102        /**
103         * Constructs a new {@link SAT4JSolver} with a timeout and the number of
104         * kept learning clauses. Additionally, this constructor allows the
105         * specification of the learning and restart strategy.
106         * 
107         * @param timeout
108         *            timeout in seconds
109         * @param clauseLearningLength
110         *            clauses are learned if they have a smaller or equal number of
111         *            literals per clause
112         * @param learning
113         *            the learning strategy
114         * @param restarts
115         *            the restart strategy
116         */
117        @Inject
118        public SAT4JSolver(@Constant(value = "timeout", namespace = SAT4JSolver.class) int timeout,
119                        @Constant(value = "clauseLearningLength", namespace = SAT4JSolver.class) int clauseLearningLength,
120                        @Constant(value = "learning", namespace = SAT4JSolver.class) Learning learning,
121                        @Constant(value = "restarts", namespace = SAT4JSolver.class) Restarts restarts) {
122                LearningStrategy<PBDataStructureFactory> l = null;
123                switch (learning) {
124                case FIXEDLENGTH:
125                        if (clauseLearningLength < 0) {
126                                throw new IllegalArgumentException("learning length must not be less than 0.");
127                        }
128                        l = new FixedLengthLearning<PBDataStructureFactory>(clauseLearningLength);
129                        break;
130                case MINISAT:
131                        l = new MiniSATLearning<PBDataStructureFactory>();
132                        break;
133                case CLAUSEONLY:
134                        l = new ClauseOnlyLearning<PBDataStructureFactory>();
135                        break;
136                default:
137                        throw new IllegalArgumentException("Learning strategy not supported: " + learning);
138                }
139
140                RestartStrategy r = null;
141                switch (restarts) {
142                case MINISAT:
143                        r = new MiniSATRestarts();
144                        break;
145                case LUBY:
146                        r = new LubyRestarts();
147                        break;
148                case RAPID:
149                        r = new ArminRestarts();
150                        break;
151                default:
152                        throw new IllegalArgumentException("Restart strategy not supported: " + restarts);
153                }
154                solver = new PBSolverResolution(l, new CompetResolutionPBMixedHTClauseCardConstrDataStructure(),
155                                new VarOrderHeap(new PositiveLiteralSelectionStrategy()), r);
156                l.setSolver(solver);
157                l.setVarActivityListener(solver);
158                if (l instanceof MiniSATLearning) {
159                        ((MiniSATLearning<PBDataStructureFactory>) l).setDataStructureFactory(solver.getDSFactory());
160                }
161
162                if (timeout <= 0) {
163                        throw new IllegalArgumentException("Invalid timeout: " + timeout);
164                }
165                solver.setTimeout(timeout);
166                solver.setVerbose(false);
167
168                setNVars(100);
169                solverValid = true;
170        }
171
172        /*
173         * (non-Javadoc)
174         * 
175         * @see org.opt4j.sat.Solver#addConstraint(org.opt4j.sat.Constraint)
176         */
177        @Override
178        public void addConstraint(Constraint constraint) {
179                constraints.add(constraint);
180        }
181
182        /*
183         * (non-Javadoc)
184         * 
185         * @see org.opt4j.sat.Solver#solve(org.opt4j.sat.Order)
186         */
187        @Override
188        public synchronized Model solve(Order order) throws TimeoutException, ContradictionException {
189                if (!solverValid) {
190                        throw new ContradictionException();
191                }
192
193                Constraint c = null;
194                while ((c = constraints.poll()) != null) {
195                        addConstraintToSolver(c);
196                }
197
198                if (order instanceof VarOrder) {
199
200                        VarOrder varOrder = (VarOrder) order;
201                        VariableOrder o = new VariableOrder();
202                        solver.setOrder(o);
203
204                        for (Entry<Object, Double> entry : varOrder.getActivityEntrySet()) {
205                                if (variables.containsKey(entry.getKey())) {
206                                        int var = variables.get(entry.getKey());
207                                        o.setVarActivity(var, entry.getValue());
208                                }
209                        }
210
211                        for (Entry<Object, Boolean> entry : varOrder.getPhaseEntrySet()) {
212                                if (variables.containsKey(entry.getKey())) {
213                                        int var = variables.get(entry.getKey());
214                                        o.setVarPhase(var, entry.getValue());
215                                }
216                        }
217
218                        o.setVarInc(varOrder.getVarInc());
219                        o.setVarDecay(varOrder.getVarDecay());
220
221                }
222
223                try {
224                        boolean success = solver.isSatisfiable();
225
226                        if (success) {
227                                Model model = new Model();
228
229                                for (Entry<Object, Integer> entry : variables.entrySet()) {
230                                        model.set(entry.getKey(), solver.model(entry.getValue()));
231                                }
232                                return model;
233                        } else {
234                                throw new ContradictionException();
235                        }
236                } catch (org.sat4j.specs.TimeoutException e) {
237                        throw new TimeoutException();
238                }
239        }
240
241        protected void addConstraintToSolver(Constraint constraint) {
242
243                VecInt lits = toVecInt(constraint.getLiterals());
244
245                IVec<BigInteger> coeffs = new Vec<BigInteger>();
246
247                for (Integer value : constraint.getCoefficients()) {
248                        coeffs.push(BigInteger.valueOf(value));
249                }
250
251                BigInteger d = BigInteger.valueOf(constraint.getRhs());
252                Operator operator = constraint.getOperator();
253
254                try {
255                        if (operator == LE || operator == EQ) {
256                                solver.addPseudoBoolean(lits, coeffs, false, d);
257                        }
258                        if (operator == GE || operator == EQ) {
259                                solver.addPseudoBoolean(lits, coeffs, true, d);
260                        }
261                } catch (org.sat4j.specs.ContradictionException e) {
262                        solverValid = false;
263                        throw new ContradictionException(e);
264                }
265        }
266
267        protected VecInt toVecInt(Iterable<Literal> list) {
268                VecInt vector = new VecInt();
269
270                for (Literal literal : list) {
271                        Object var = literal.variable();
272                        if (!variables.containsKey(var)) {
273                                variables.put(var, nextVariable++);
274                                if (variables.size() > solver.nVars()) {
275                                        setNVars(variables.size() * 2);
276                                }
277                        }
278                        boolean phase = literal.phase();
279
280                        vector.push(variables.get(var) * (phase ? 1 : -1));
281                }
282
283                return vector;
284        }
285
286        /**
287         * Sets the instance to n vars.
288         * 
289         * @param n
290         *            the number of vars.
291         */
292        protected void setNVars(int n) {
293                solver.newVar(n);
294        }
