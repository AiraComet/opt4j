024package org.opt4j.satdecoding;
025
026import java.util.List;
027import java.util.Map;
028
029import org.opt4j.core.Genotype;
030import org.opt4j.core.genotype.BooleanGenotype;
031import org.opt4j.core.genotype.BooleanMapGenotype;
032import org.opt4j.core.genotype.DoubleBounds;
033import org.opt4j.core.genotype.DoubleGenotype;
034import org.opt4j.core.genotype.DoubleMapGenotype;
035
036import com.google.inject.Inject;
037import com.google.inject.Singleton;
038
039/**
040 * The {@link MixedSATManager} encodes the decision strategy into two vectors:
041 * One binary vector for the phase and one double vector for the priority.
042 * 
043 * @author lukasiewycz
044 * 
045 */
046@Singleton
047public class MixedSATManager implements SATManager {
048
049        protected final Solver solver;
050
051        /**
052         * Constructs a {@link MixedSATManager}.
053         * 
054         * @param solver
055         *            the solver
056         */
057        @Inject
058        public MixedSATManager(Solver solver) {
059                this.solver = solver;
060        }
061
062        /*
063         * (non-Javadoc)
064         * 
065         * @see org.opt4j.sat.SATManager#createSATGenotype(java.util.List,
066         * java.util.Map, java.util.Map)
067         */
068        @Override
069        public Genotype createSATGenotype(List<Object> variables, Map<Object, Double> lowerBounds,
070                        Map<Object, Double> upperBounds, Map<Object, Double> priorities, Map<Object, Boolean> phases) {
071
072                int size = variables.size();
073
074                double[] lower = new double[size];
075                double[] upper = new double[size];
076
077                for (int i = 0; i < size; i++) {
078                        Object variable = variables.get(i);
079                        Double lb = lowerBounds.get(variable);
080                        Double ub = upperBounds.get(variable);
081
082                        if (lb == null) {
083                                lb = 0.0;
084                        }
085                        if (ub == null) {
086                                ub = 1.0;
087                        }
088                        lower[i] = lb;
089                        upper[i] = ub;
090                }
091
092                DoubleBounds doubleBounds = new DoubleBounds(lower, upper);
093                DoubleGenotype doubleVector = new DoubleMapGenotype<Object>(variables, doubleBounds);
094                BooleanGenotype booleanVector = new BooleanMapGenotype<Object>(variables);
095
096                SATGenotype satGenotype = new SATGenotype(booleanVector, doubleVector);
097
098                for (Object variable : variables) {
099                        double priority = priorities.get(variable);
100                        boolean phase = phases.get(variable);
101
102                        doubleVector.add(priority);
103                        booleanVector.add(phase);
104                }
105
106                return satGenotype;
107        }
108
109        /*
110         * (non-Javadoc)
111         * 
112         * @see org.opt4j.sat.SATManager#decodeSATGenotype(java.util.List,
113         * org.opt4j.core.Genotype)
114         */
115        @Override
116        public Model decodeSATGenotype(List<Object> variables, Genotype genotype) {
117                SATGenotype satGenotype = (SATGenotype) genotype;
118
119                BooleanGenotype booleanVector = satGenotype.getBooleanVector();
120                DoubleGenotype doubleVector = satGenotype.getDoubleVector();
121
122                VarOrder varorder = new VarOrder();
123
124                for (int i = 0; i < variables.size(); i++) {
125                        Object var = variables.get(i);
126                        varorder.setActivity(var, doubleVector.get(i));
127                        varorder.setPhase(var, booleanVector.get(i));
128                }
129                varorder.setVarInc(1.0 / (2.0 * variables.size()));
130                varorder.setVarDecay(1.0 / 0.95);
131
132                Model model = null;
133                try {
134                        model = solver.solve(varorder);
135                        if (model == null) {
136                                throw new ContradictionException("no satisfying solution left");
137                        }
138                } catch (TimeoutException e) {
139                        System.err.println("timeout");
140                }
141                return model;
142        }
143
144        /*
145         * (non-Javadoc)
146         * 
147         * @see org.opt4j.sat.SATManager#getSolver()
148         */
149        @Override
150        public Solver getSolver() {
151                return solver;
152        }
153
154}
