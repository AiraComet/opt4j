024package org.opt4j.satdecoding;
025
026import java.util.HashMap;
027import java.util.Map;
028import java.util.Map.Entry;
029import java.util.Set;
030
031/**
032 * The {@link Model} is a data structure that represents a solution of the given
033 * problem, i.e., all {@link Literal}s are set such that all {@link Constraint}s
034 * specified by the {@link AbstractSATDecoder} are feasible.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039public class Model {
040
041        protected final Map<Object, Boolean> map = new HashMap<Object, Boolean>();
042
043        /**
044         * Constructs a {@link Model}.
045         */
046        public Model() {
047                super();
048        }
049
050        /**
051         * Returns the value of the variable, or null if this variable is not
052         * defined.
053         * 
054         * @param var
055         *            the variable
056         * @return {@code true} if the variable in this model is true or
057         *         {@code null} if it is not defined
058         */
059        public Boolean get(Object var) {
060                return map.get(var);
061        }
062
063        /**
064         * Sets the variable to a phase.
065         * 
066         * @param var
067         *            the variable
068         * @param phase
069         *            the phase
070         */
071        public void set(Object var, boolean phase) {
072                map.put(var, phase);
073        }
074
075        /**
076         * Returns all variables of the model.
077         * 
078         * @return all variables
079         */
080        public Set<Object> getVars() {
081                return map.keySet();
082        }
083
084        /**
085         * Returns the entry set: The entry pairs of all variables to the boolean
086         * values.
087         * 
088         * @return the entry set
089         */
090        public Set<Entry<Object, Boolean>> pairs() {
091                return map.entrySet();
092        }
093
094}
