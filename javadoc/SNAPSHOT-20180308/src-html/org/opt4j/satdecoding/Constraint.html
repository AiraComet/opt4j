024package org.opt4j.satdecoding;
025
026import java.util.ArrayList;
027import java.util.Iterator;
028
029/**
030 * A linear pseudo-Boolean {@link Constraint} as used in an Integer Linear
031 * Program (ILP). A set of {@link Constraint}s specify a search space.
032 * <p>
033 * 
034 * This {@link Constraint} consists of a list of {@link Term}s, an
035 * {@link Operator} and a right hand side. E.g., {@code 4a - 3b + not(c) <= 7}
036 * is a valid {@link Constraint} consisting of three {@link Literal}s
037 * {@code a, b, not(c)} with coefficients {@code 4, 3, 1} , the less or equal
038 * operator from {@link Operator}, and the right hand side {@code 7}.
039 * <p>
040 * 
041 * This implementation is a plain data structure, i.e., {@code 4a - 3a + c <= 1}
042 * is not simplified to {@code a + c <= 1}. Moreover, two {@link Constraint}s
043 * are equal if they have an equal {@link Operator}, right hand side, and equal
044 * {@link Term}s in the same order.
045 * 
046 * @author lukasiewycz
047 * 
048 */
049public class Constraint extends ArrayList<Term> {
050
051        protected Operator operator = Operator.GE;
052        protected int rhs = 1;
053
054        /**
055         * The {@link Operator}s correspond to the Boolean operators &lt;=,=,&gt;=.
056         */
057        public enum Operator {
058                /**
059                 * Boolean operation &lt;=.
060                 */
061                LE,
062                /**
063                 * Boolean operation =.
064                 */
065                EQ,
066                /**
067                 * Boolean operation &gt;=.
068                 */
069                GE;
070
071                /**
072                 * Returns a {@link String} representation of this {@link Operator}.
073                 * 
074                 * @return a string representation of this operator
075                 */
076                @Override
077                public String toString() {
078                        switch (this) {
079                        case LE:
080                                return "<=";
081                        case EQ:
082                                return "=";
083                        default: // GE
084                                return ">=";
085                        }
086                }
087
088                /**
089                 * Returns {@code true} if {@code lhs} {@link Operator} {@code rhs} is
090                 * true.
091                 * 
092                 * @param lhs
093                 *            the left hand side value
094                 * @param rhs
095                 *            the right hand side value
096                 * @return {@code true} if the operation is true for the two values
097                 */
098                public boolean isTrue(int lhs, int rhs) {
099                        switch (this) {
100                        case LE:
101                                return (lhs <= rhs);
102                        case EQ:
103                                return (lhs == rhs);
104                        default: // GE
105                                return (lhs >= rhs);
106                        }
107                }
108
109                /**
110                 * Converts a {@link String} to an operator.
111                 * 
112                 * @param string
113                 *            the string (&lt;=,=,&gt;=).
114                 * @return the operator
115                 */
116                public static Operator getOperator(String string) {
117                        if ("<=".equals(string)) {
118                                return LE;
119                        } else if ("=".equals(string)) {
120                                return EQ;
121                        } else if (">=".equals(string)) {
122                                return GE;
123                        } else {
124                                throw new IllegalArgumentException("Unknown operator " + string + ". Allowed operators are: <=,=,>=");
125                        }
126                }
127        }
128
129        /**
130         * Constructs a linear {@link Constraint} with {@code >=1}.
131         */
132        public Constraint() {
133                this(Operator.GE, 1);
134        }
135
136        /**
137         * Constructs a linear {@link Constraint}.
138         * 
139         * @param operator
140         *            the operator (&lt;=,=,&gt;=) as a String
141         * @param rhs
142         *            the right-hand-side value
143         */
144        public Constraint(String operator, int rhs) {
145                this(Operator.getOperator(operator), rhs);
146        }
147
148        /**
149         * Constructs a linear {@link Constraint}.
150         * 
151         * @param operator
152         *            the operator
153         * @param rhs
154         *            the right-hand-side value
155         */
156        public Constraint(Operator operator, int rhs) {
157                this.operator = operator;
158                this.rhs = rhs;
159        }
160
161        /**
162         * Returns an iterable of the coefficients.
163         * 
164         * @return the coefficients
165         */
166        public Iterable<Integer> getCoefficients() {
167                return new Iterable<Integer>() {
168                        @Override
169                        public Iterator<Integer> iterator() {
170                                return new Iterator<Integer>() {
171
172                                        int i = 0;
173
174                                        @Override
175                                        public boolean hasNext() {
176                                                return (i < size());
177                                        }
178
179                                        @Override
180                                        public Integer next() {
181                                                return get(i++).getCoefficient();
182                                        }
183
184                                        @Override
185                                        public void remove() {
186                                                throw new RuntimeException("Operation not supported");
187                                        }
188                                };
189                        }
190                };
191        }
192
193        /**
194         * Returns an iterable of the {@link Literal}s.
195         * 
196         * @return the literals.
197         */
198        public Iterable<Literal> getLiterals() {
199                return new Iterable<Literal>() {
200                        @Override
201                        public Iterator<Literal> iterator() {
202                                return new Iterator<Literal>() {
203
204                                        int i = 0;
205
206                                        @Override
207                                        public boolean hasNext() {
208                                                return (i < size());
209                                        }
210
211                                        @Override
212                                        public Literal next() {
213                                                return get(i++).getLiteral();
214                                        }
215
216                                        @Override
217                                        public void remove() {
218                                                throw new RuntimeException("Operation not supported");
219                                        }
220                                };
221                        }
222                };
223        }
224
225        /**
226         * Returns the {@link Operator}.
227         * 
228         * @see #setOperator
229         * @return the operator
230         */
231        public Operator getOperator() {
232                return operator;
233        }
234
235        /**
236         * Sets the {@link Operator}.
237         * 
238         * @see #getOperator
239         * @param operator
240         *            the operator to be set
241         */
242        public void setOperator(Operator operator) {
243                this.operator = operator;
244        }
245
246        /**
247         * Return the right hand side value.
248         * 
249         * @see #setRhs
250         * @return the right hand side value
251         */
252        public int getRhs() {
253                return rhs;
254        }
255
256        /**
257         * Sets the right hand side value.
258         * 
259         * @see #getRhs
260         * @param rhs
261         *            the right hand side value to set
262         */
263        public void setRhs(int rhs) {
264                this.rhs = rhs;
265        }
266
267        /*
268         * (non-Javadoc)
269         * 
270         * @see java.util.AbstractCollection#toString()
271         */
272        @Override
273        public String toString() {
274                String s = "";
275
276                for (int i = 0; i < size(); i++) {
277                        s += get(i);
278                        if (i < size() - 1) {
279                                s += " + ";
280                        }
281                }
282
283                s += " " + operator + " ";
284                s += rhs;
285
286                return s;
287        }
288
289        /**
290         * Checks, if this constraint is satisfied for a given model.
291         * 
292         * @param model
293         *            the model
294         * @return true, iff this constraint is satisfied
295         */
296        public boolean isSatisfied(Model model) {
297                int count = getCount(model);
298                return operator.isTrue(count, rhs);
299
300        }
301
302        /**
303         * Specifies if the {@link Constraint} is satisfied for a given
304         * {@link Model}.
305         * 
306         * @param model
307         *            the model
308         * @return {@code true} if this constraint is satisfied for the model
309         */
310        public int getViolationCount(Model model) {
311                int count = getCount(model);
312
313                switch (operator) {
314                case LE:
315                        return (count <= rhs) ? 0 : (count - rhs);
316                case EQ:
317                        return Math.abs(count - rhs);
318                default: // GE
319                        return (count >= rhs) ? 0 : (rhs - count);
320                }
321        }
322
323        /**
324         * Add a {@link Literal} with the coefficient {@code 1}.
325         * 
326         * @param lit
327         *            the literal
328         */
329        public void add(Literal lit) {
330                add(1, lit);
331        }
332
333        /**
334         * Add a {@link Literal} with specified coefficient.
335         * 
336         * @param coeff
337         *            the coefficient
338         * @param lit
339         *            the literal
340         */
341        public void add(int coeff, Literal lit) {
342                Term term = new Term(coeff, lit);
343                add(term);
344        }
345
346        /**
347         * Returns the sum of the left hand side of the constraint for a given
348         * model.
349         * 
350         * @param model
351         *            the model
352         * @return the sum
353         */
354        private int getCount(Model model) {
355                int count = 0;
356                for (Term term : this) {
357                        Literal literal = term.getLiteral();
358                        int coefficient = term.getCoefficient();
359
360                        Object var = literal.variable();
361                        Boolean phase = literal.phase();
362
363                        if (phase.equals(model.get(var))) {
364                                count += coefficient;
365                        }
366                }
367                return count;
368        }
369
370        /**
371         * Copies the {@link Constraint}.
372         * 
373         * @return a copy of the constraint
374         */
375        public Constraint copy() {
376                Constraint pb = new Constraint(operator, rhs);
377                for (Term term : this) {
378                        pb.add(term.copy());
379                }
380                return pb;
381        }
382
383        /**
384         * Returns {@code true} if the {@link Constraint} contains the
385         * {@link Literal}.
386         * 
387         * @param literal
388         *            the literal
389         * 
390         * @return {@code true} if the constrain} contains the literal
391         */
392        public boolean contains(Literal literal) {
393                for (Term term : this) {
394                        if (literal.equals(term.getLiteral())) {
395                                return true;
396                        }
397                }
398                return false;
399        }
400
401        /*
402         * (non-Javadoc)
403         * 
404         * @see java.util.AbstractList#hashCode()
405         */
406        @Override
407        public int hashCode() {
408                final int prime = 31;
409                int result = super.hashCode();
410                result = prime * result + ((operator == null) ? 0 : operator.hashCode());
411                result = prime * result + rhs;
412                return result;
413        }
414
415        /*
416         * (non-Javadoc)
417         * 
418         * @see java.util.AbstractList#equals(java.lang.Object)
419         */
420        @Override
421        public boolean equals(Object obj) {
422                if (this == obj) {
423                        return true;
424                }
425                if (getClass() != obj.getClass()) {
426                        return false;
427                }
428                Constraint other = (Constraint) obj;
429                return (super.equals(other) && operator == other.operator && rhs == other.rhs);
430        }
431
432        private static final long serialVersionUID = 1L;
