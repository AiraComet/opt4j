024package org.opt4j.satdecoding;
025
026import java.util.ArrayList;
027import java.util.Collection;
028import java.util.HashMap;
029import java.util.HashSet;
030import java.util.List;
031import java.util.Map;
032import java.util.Random;
033import java.util.Set;
034
035import org.opt4j.core.Genotype;
036import org.opt4j.core.common.random.Rand;
037import org.opt4j.core.optimizer.Optimizer;
038import org.opt4j.core.problem.Creator;
039import org.opt4j.core.problem.Decoder;
040
041import com.google.inject.Inject;
042
043/**
044 * The {@link AbstractSATDecoder} can be used for the construction of custom SAT
045 * {@link Decoder}s.
046 * <p>
047 * The {@link Creator} part takes the problem dependent {@link Constraint}s and
048 * creates a {@link SATGenotype} according to the variables which is the input
049 * for the specified {@link Optimizer}.
050 * <p>
051 * The {@link Decoder} takes a {@link SATGenotype}, converts it to an
052 * {@link Order} of the variables, and solves the problem in the {@link Order}
053 * using a SAT {@link Solver} . The resulting {@link Model}, which respects the
054 * {@link Constraint}s, must then be converted to the problem specific
055 * phenotype.
056 * 
057 * @author lukasiewycz
058 * 
059 * @param <G>
060 *            The genotype (use {@link Genotype} if you do not override the
061 *            decode method)
062 * @param <P>
063 *            The phenotype
064 */
065public abstract class AbstractSATDecoder<G extends Genotype, P extends Object> implements Decoder<G, P>, Creator<G> {
066
067        private final List<Constraint> constraints = new ArrayList<Constraint>();
068        private final List<Object> variables = new ArrayList<Object>();
069
070        protected Map<Object, Double> lowerBounds;
071
072        protected Map<Object, Double> upperBounds;
073
074        protected final Random random;
075
076        protected final SATManager manager;
077
078        protected boolean isInit = false;
079
080        /**
081         * Constructs an {@link AbstractSATDecoder} with the given
082         * {@link SATManager}.
083         * 
084         * @param manager
085         *            the specified sat manager
086         * @param random
087         *            the random number generator
088         */
089        @Inject
090        public AbstractSATDecoder(SATManager manager, Rand random) {
091                this.random = random;
092                this.manager = manager;
093        }
094
095        /**
096         * Initializes the {@link Constraint}s and variables.
097         */
098        protected synchronized void init() {
099
100                if (!isInit) {
101                        try {
102                                Solver solver = manager.getSolver();
103
104                                Set<Constraint> constraints = createConstraints();
105
106                                Set<Object> variables = new HashSet<Object>();
107
108                                for (Constraint constraint : constraints) {
109                                        solver.addConstraint(constraint);
110                                        this.constraints.add(constraint);
111
112                                        for (Literal literal : constraint.getLiterals()) {
113                                                variables.add(literal.variable());
114                                        }
115                                }
116
117                                Set<Object> ignores = ignoreVariables(variables);
118
119                                variables.removeAll(ignores);
120                                this.variables.addAll(variables);
121
122                                lowerBounds = getLowerBounds(variables);
123                                upperBounds = getUpperBounds(variables);
124
125                        } catch (Throwable e) {
126                                System.err.println("Failed initialization of " + getClass().getName() + " with " + e);
127                                throw new RuntimeException(e);
128                        } finally {
129                                isInit = true;
130                        }
131
132                }
133        }
134
135        /**
136         * Creates all {@link Constraint}s.
137         * <p>
138         * Override this method to return the problem specific {@link Constraint}s
139         * that have to be respected by the resulting {@link Model}.
140         * 
141         * @return the set of all constraints of the given problem
142         * 
143         */
144        public abstract Set<Constraint> createConstraints();
145
146        /**
147         * Creates a random initialization of the priorities and phases for the
148         * variables. Override this method for a custom initialization.
149         * 
150         * @param variables
151         *            all variables
152         * @param lowerBounds
153         *            the lower bounds for the priority for the variables
154         * @param upperBounds
155         *            the upper bounds for the priority for the variables
156         * @param priorities
157         *            the priorities of the variables
158         * @param phases
159         *            the phases of the variables
160         */
161        public void randomize(Collection<Object> variables, Map<Object, Double> lowerBounds,
162                        Map<Object, Double> upperBounds, Map<Object, Double> priorities, Map<Object, Boolean> phases) {
163                for (Object variable : variables) {
164                        Double lb = lowerBounds.get(variable);
165                        Double ub = upperBounds.get(variable);
166                        if (lb == null) {
167                                lb = 0.0;
168                        }
169                        if (ub == null) {
170                                ub = 1.0;
171                        }
172
173                        double priority = lb + (ub - lb) * random.nextDouble();
174
175                        priorities.put(variable, priority);
176                        phases.put(variable, random.nextBoolean());
177                }
178        }
179
180        /**
181         * Set the variables to be ignored (not part of the search process and
182         * genotype). Override this method if variables are ignored.
183         * 
184         * @param variables
185         *            all variables of the problem
186         * @return the variables to be ignored
187         */
188        public Set<Object> ignoreVariables(Set<Object> variables) {
189                return new HashSet<Object>();
190        }
191
192        /**
193         * Returns the lower bounds for the variables.
194         * 
195         * @param variables
196         *            all variables of the problem
197         * @return the lower bounds
198         */
199        public Map<Object, Double> getLowerBounds(Set<Object> variables) {
200                return new HashMap<Object, Double>();
201        }
202
203        /**
204         * Returns the upper bounds for the variables.
205         * 
206         * @param variables
207         *            all variables of the problem
208         * @return the upper bounds
209         */
210        public Map<Object, Double> getUpperBounds(Set<Object> variables) {
211                return new HashMap<Object, Double>();
212        }
213
214        /**
215         * Converts a {@link Model} into a phenotype. Override this with your custom
216         * method.
217         * 
218         * @param model
219         *            the found model
220         * @return the converted phenotype
221         */
222        public abstract P convertModel(Model model);
223
224        /*
225         * (non-Javadoc)
226         * 
227         * @see org.opt4j.core.problem.Creator#create()
228         */
229        @Override
230        @SuppressWarnings("unchecked")
231        public G create() {
232                return (G) createSATGenotype();
233        }
234
235        /**
236         * Returns a random SAT genotype of the constrained problem.
237         * 
238         * @return a random SAT genotype
239         */
240        protected Genotype createSATGenotype() {
241                if (!isInit) {
242                        init();
243                }
244
245                Map<Object, Double> priorities = new HashMap<Object, Double>();
246                Map<Object, Boolean> phases = new HashMap<Object, Boolean>();
247
248                randomize(variables, lowerBounds, upperBounds, priorities, phases);
249
250                return manager.createSATGenotype(variables, lowerBounds, upperBounds, priorities, phases);
251        }
252
253        /*
254         * (non-Javadoc)
255         * 
256         * @see org.opt4j.core.problem.Decoder#decode(org.opt4j.core.Genotype)
257         */
258        @Override
259        public P decode(G genotype) {
260                Model model = decodeSATGenotype(genotype);
261                assert model != null;
262
263                return convertModel(model);
264        }
265
266        /**
267         * Decodes the {@link Genotype} to a phenotype by using a SAT/PB solver.
268         * 
269         * @param genotype
270         *            the genotype
271         * @return the phenotype
272         */
273        protected Model decodeSATGenotype(Genotype genotype) {
274                if (!isInit) {
275                        init();
276                }
277
278                return manager.decodeSATGenotype(variables, genotype);
279        }
280
281        /**
282         * Returns the constraints.
283         * 
284         * @return the constraints
285         */
286        public List<Constraint> getConstraints() {
287                if (!isInit) {
288                        init();
289                }
290                return constraints;
291        }
292
293        /**
294         * Returns the variables.
295         * 
296         * @return the variables
297         */
298        public List<Object> getVariables() {
299                if (!isInit) {
300                        init();
301                }
302                return variables;
303        }
304
