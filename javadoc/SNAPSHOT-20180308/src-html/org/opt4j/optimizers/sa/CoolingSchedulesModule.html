024package org.opt4j.optimizers.sa;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Order;
028import org.opt4j.core.config.annotations.Parent;
029import org.opt4j.core.config.annotations.Required;
030import org.opt4j.core.start.Constant;
031
032/**
033 * This module provides several common {@link CoolingSchedule}s for the
034 * {@link SimulatedAnnealing}.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039@Info("Common Cooling Schedules")
040@Parent(SimulatedAnnealingModule.class)
041public class CoolingSchedulesModule extends CoolingScheduleModule {
042
043        @Constant(value = "t0", namespace = CoolingSchedule.class)
044        protected double initialTemperature = 10.0;
045
046        @Constant(value = "tn", namespace = CoolingSchedule.class)
047        protected double finalTemperature = 1.0;
048
049        @Required(property = "type", elements = { "EXPONENTIAL" })
050        @Constant(value = "alpha", namespace = CoolingScheduleExponential.class)
051        protected double alpha = 0.995;
052
053        @Order(0)
054        protected Type type = Type.LINEAR;
055
056        /**
057         * Constructs a {@link CoolingSchedulesModule}.
058         * 
059         */
060        public CoolingSchedulesModule() {
061                super();
062        }
063
064        /**
065         * Type of {@link CoolingSchedule} to use.
066         * 
067         * @author lukasiewycz
068         * 
069         */
070        public enum Type {
071
072                /**
073                 * Use the {@link CoolingScheduleLinear}.
074                 */
075                LINEAR,
076                /**
077                 * Use the {@link CoolingScheduleHyperbolic}.
078                 */
079                HYPERBOLIC,
080                /**
081                 * Use the {@link CoolingScheduleExponential}.
082                 */
083                EXPONENTIAL;
084        }
085
086        /**
087         * Returns the final temperature.
088         * 
089         * @see #setFinalTemperature
090         * @return the final temperature
091         */
092        @Info("The temperature at the end of the optimization")
093        public double getFinalTemperature() {
094                return finalTemperature;
095        }
096
097        /**
098         * Sets the final temperature.
099         * 
100         * @see #getFinalTemperature
101         * @param finalTemperature
102         *            the final temperature
103         */
104        public void setFinalTemperature(double finalTemperature) {
105                this.finalTemperature = finalTemperature;
106        }
107
108        /**
109         * Returns the initial temperature.
110         * 
111         * @see #setInitialTemperature
112         * @return the initial temperature
113         */
114        @Info("The temperature at the beginning of the optimization")
115        public double getInitialTemperature() {
116                return initialTemperature;
117        }
118
119        /**
120         * Sets the initial temperature.
121         * 
122         * @see #getInitialTemperature
123         * @param initialTemperature
124         *            the initial temperature
125         */
126        public void setInitialTemperature(double initialTemperature) {
127                this.initialTemperature = initialTemperature;
128        }
129
130        /**
131         * Returns the type of cooling schedule.
132         * 
133         * @see #setType
134         * @return the type of cooling schedule
135         */
136        @Info("The type of the cooling schedule")
137        public Type getType() {
138                return type;
139        }
140
141        /**
142         * Sets the type of cooling schedule.
143         * 
144         * @see #getType
145         * @param type
146         *            the type of cooling schedule
147         */
148        public void setType(Type type) {
149                this.type = type;
150        }
151
152        /**
153         * Returns the alpha value.
154         * 
155         * @see #setAlpha
156         * @return the alpha value
157         */
158        public double getAlpha() {
159                return alpha;
160        }
161
162        /**
163         * Sets the alpha value.
164         * 
165         * @see #getAlpha
166         * @param alpha
167         *            the alpha value
168         */
169        public void setAlpha(double alpha) {
170                this.alpha = alpha;
171        }
172
173        /*
174         * (non-Javadoc)
175         * 
176         * @see org.opt4j.start.Opt4JModule#config()
177         */
178        @Override
179        public void config() {
180
181                switch (type) {
182                case LINEAR:
183                        bindCoolingSchedule(CoolingScheduleLinear.class);
184                        break;
185                case HYPERBOLIC:
186                        bindCoolingSchedule(CoolingScheduleHyperbolic.class);
187                        break;
188                default: // EXPONENTIAL
189                        bindCoolingSchedule(CoolingScheduleExponential.class);
190                        break;
191                }
192        }
193
194}
