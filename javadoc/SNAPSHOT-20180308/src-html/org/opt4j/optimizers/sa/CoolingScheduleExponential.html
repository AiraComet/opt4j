024package org.opt4j.optimizers.sa;
025
026import org.opt4j.core.start.Constant;
027
028import com.google.inject.Inject;
029
030/**
031 * <p>
032 * The {@link CoolingScheduleExponential} is a {@link CoolingSchedule} for the
033 * {@link SimulatedAnnealing}.
034 * </p>
035 * <p>
036 * {@code tn - final temperature}<br />
037 * {@code t0 - initial temperature}<br />
038 * {@code i - current iteration}<br />
039 * {@code n - maximal number of iterations} <br />
040 * {@code a - alpha value}<br />
041 * </p>
042 * <p>
043 * The current temperature is calculated by {@code tn + t0 * a^i}.
044 * </p>
045 * 
046 * 
047 * 
048 * @author lukasiewycz
049 * 
050 */
051public class CoolingScheduleExponential implements CoolingSchedule {
052
053        protected final double t0;
054
055        protected final double tn;
056
057        protected final double alpha;
058
059        /**
060         * Constructs a new {@link CoolingScheduleExponential}.
061         * 
062         * @param t0
063         *            the initial temperature (using namespace
064         *            {@link CoolingSchedule})
065         * @param tn
066         *            the final temperature (using namespace {@link CoolingSchedule}
067         *            ) )
068         * @param alpha
069         *            the alpha value
070         */
071        @Inject
072        public CoolingScheduleExponential(@Constant(value = "t0", namespace = CoolingSchedule.class) double t0,
073                        @Constant(value = "tn", namespace = CoolingSchedule.class) double tn,
074                        @Constant(value = "alpha", namespace = CoolingScheduleExponential.class) double alpha) {
075                this.t0 = t0;
076                this.tn = tn;
077                this.alpha = alpha;
078        }
079
080        /*
081         * (non-Javadoc)
082         * 
083         * @see org.opt4j.optimizer.sa.CoolingSchedule#getTemperature(int, int)
084         */
085        @Override
086        public double getTemperature(int i, int n) {
087                return tn + t0 * Math.pow(alpha, i);
088        }
089
090}
