023package org.opt4j.optimizers.mopso;
024
025import org.opt4j.core.IndividualFactory;
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Name;
028import org.opt4j.core.config.annotations.Order;
029import org.opt4j.core.optimizer.MaxIterations;
030import org.opt4j.core.optimizer.OptimizerModule;
031import org.opt4j.core.start.Constant;
032
033/**
034 * The {@link MOPSOModule} for the {@link MOPSO} optimizer.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039@Info("Multi-objective particle swarm optimizer. Mostly based on the OMOPSO. Works only with real-valued problems.")
040public class MOPSOModule extends OptimizerModule {
041
042        @Info("The number of particles.")
043        @Order(1)
044        @Constant(value = "size", namespace = MOPSO.class)
045        protected int particles = 100;
046
047        @Info("The archive size of the global leaders.")
048        @Name("archive size")
049        @Order(2)
050        @Constant(value = "archiveSize", namespace = MOPSO.class)
051        protected int archiveSize = 100;
052
053        @Info("The number of iterations.")
054        @Order(0)
055        @MaxIterations
056        protected int iterations = 1000;
057
058        @Info("The perturbation index for the mutation.")
059        @Order(3)
060        protected double perturbation = 0.5;
061
062        /**
063         * Returns the perturbation.
064         * 
065         * @see #setPerturbation
066         * @return the perturbation
067         */
068        public double getPerturbation() {
069                return perturbation;
070        }
071
072        /**
073         * Sets the perturbation.
074         * 
075         * @see #getPerturbation
076         * @param perturbation
077         *            the perturbation to set
078         */
079        public void setPerturbation(double perturbation) {
080                this.perturbation = perturbation;
081        }
082
083        /**
084         * Returns the number of iterations.
085         * 
086         * @see #setIterations
087         * @return the iterations
088         */
089        public int getIterations() {
090                return iterations;
091        }
092
093        /**
094         * Sets the the number of iterations.
095         * 
096         * @see #getIterations
097         * @param iterations
098         *            the iterations to set
099         */
100        public void setIterations(int iterations) {
101                this.iterations = iterations;
102        }
103
104        /**
105         * Returns the number of particles.
106         * 
107         * @see #setParticles
108         * @return the particles
109         */
110        public int getParticles() {
111                return particles;
112        }
113
114        /**
115         * Sets the number of particles.
116         * 
117         * @see #getParticles
118         * @param particles
119         *            the particles to set
120         */
121        public void setParticles(int particles) {
122                this.particles = particles;
123        }
124
125        /**
126         * Returns the leader archive size.
127         * 
128         * @see #setArchiveSize
129         * @return the archiveSize
130         */
131        public int getArchiveSize() {
132                return archiveSize;
133        }
134
135        /**
136         * Sets the leader archive size.
137         * 
138         * @see #getArchiveSize
139         * @param archiveSize
140         *            the archiveSize to set
141         */
142        public void setArchiveSize(int archiveSize) {
143                this.archiveSize = archiveSize;
144        }
145
146        /*
147         * (non-Javadoc)
148         * 
149         * @see org.opt4j.start.Opt4JModule#config()
150         */
151        @Override
152        protected void config() {
153                bindIterativeOptimizer(MOPSO.class);
154                bind(IndividualFactory.class).to(ParticleFactory.class);
155
156                bindConstant("perturbation", MutateDoubleNonUniform.class).to(perturbation);
157                bindConstant("perturbation", MutateDoubleUniform.class).to(perturbation);
158        }
159
160}
