056/**
057 * The {@link MOPSO} is an implementation of a multi-objective particle swarm
058 * optimizer, see "Improving PSO-based Multi-Objective Optimization using
059 * Crowding, Mutation and e-Dominance, M. Reyes Sierra and C. A. Coello Coello,
060 * In Proceedings of Evolutionary Multi-Criterion Optimization, 2005". This
061 * implementation is based on the OMOPSO. Thus, this MOPSO is restricted to
062 * problems that are based on the {@link DoubleGenotype}.
063 * 
064 * @author lukasiewycz
065 * 
066 */
067public class MOPSO implements IterativeOptimizer {
068
069        protected final int size;
070
071        protected final Archive leaders;
072
073        protected final ParticleFactory particleFactory;
074
075        protected final AlgebraDouble algebra;
076
077        protected final Random random;
078
079        protected final VelocityTerm velocityTerm;
080
081        protected final Term positionTerm;
082
083        protected final MutateDoubleUniform uniform;
084
085        protected final MutateDoubleUniform nonUniform;
086
087        protected final MutationRate mutationRate;
088
089        private final Population population;
090
091        private final IndividualCompleter completer;
092
093        /**
094         * Constructs a {@link MOPSO}.
095         * 
096         * @param population
097         *            the population
098         * @param individualFactory
099         *            the individual (particle) factory
100         * @param completer
101         *            the completer
102         * @param random
103         *            the random number generator
104         * @param uniform
105         *            the uniform mutation
106         * @param nonUniform
107         *            the non-uniform mutation
108         * @param mutationRate
109         *            the mutation rate
110         * @param size
111         *            the number of particles
112         * @param archiveSize
113         *            the size of the archive for the global leaders
114         */
115        @Inject
116        public MOPSO(
117                        Population population,
118                        IndividualFactory individualFactory,
119                        IndividualCompleter completer,
120                        Rand random,
121                        MutateDoubleUniform uniform,
122                        MutateDoubleNonUniform nonUniform,
123                        MutationRate mutationRate,
124                        @Constant(value = "size", namespace = MOPSO.class) int size,
125                        @Constant(value = "archiveSize", namespace = MOPSO.class) int archiveSize) {
126                this.particleFactory = (ParticleFactory) individualFactory;
127                this.leaders = new CrowdingArchive(archiveSize);
128                this.random = random;
129                this.uniform = uniform;
130                this.nonUniform = nonUniform;
131                this.mutationRate = mutationRate;
132                this.population = population;
133                this.completer = completer;
134
135                this.size = size;
136
137                this.algebra = new AlgebraDouble(new NormalizeDouble() {
138                        @Override
139                        public void normalize(DoubleGenotype genotype) {
140                                // do nothing
141                        }
142                });
143
144                Index x = new Index(0);
145                Index v = new Index(1);
146                positionTerm = new Add(x, v);
147                velocityTerm = new VelocityTerm(random);
148        }
149
150        /*
151         * (non-Javadoc)
152         * 
153         * @see org.opt4j.core.optimizer.IterativeOptimizer#initialize()
154         */
155        @Override
156        public void initialize() {
157
158        }
159
160        /*
161         * (non-Javadoc)
162         * 
163         * @see org.opt4j.core.optimizer.IterativeOptimizer#next()
164         */
165        @Override
166        public void next() throws TerminationException {
167                if (population.isEmpty()) {
168                        // the first iteration
169                        int id = 0;
170                        while (population.size() < size) {
171                                Particle particle = particleFactory.create();
172                                particle.setId(id++);
173                                Genotype genotype = particle.getGenotype();
174                                if (!(genotype instanceof DoubleGenotype)) {
175                                        throw new IncompatibilityException(
176                                                        "MOPSO is restricted to " + DoubleGenotype.class
177                                                                        + ", current Genotype is: "
178                                                                        + genotype.getClass());
179                                }
180                                population.add(particle);
181                        }
182                } else {
183                        // all iterations > 1
184                        // determine the leaders
185                        updateLeaders(leaders, population);
186
187                        // determine one leader for each particle
188                        Map<Particle, Particle> lead = getLeaders(leaders, population);
189                        // determine the next position of each particle
190                        Map<Particle, Particle> next = move(population, lead);
191
192                        population.addAll(next.values());
193                        completer.complete(population);
194
195                        // update the personal best of each particle
196                        updatePersonalBest(next);
197
198                        // remove the old positions
199                        population.removeAll(next.keySet());
200                }
201        }
202
203        /**
204         * Determine the new positions for the {@link Particle}s in the
205         * {@link Population}.
206         * 
207         * @param population
208         *            the population
209         * @param leaders
210         *            the map for the global leaders
211         * @return the map of the old to the new particle
212         */
213        protected Map<Particle, Particle> move(Population population,
214                        Map<Particle, Particle> leaders) {
215                Map<Particle, Particle> map = new HashMap<Particle, Particle>();
216
217                for (Individual individual : population) {
218
219                        Particle particle = (Particle) individual;
220
221                        DoubleGenotype position = (DoubleGenotype) particle.getGenotype();
222                        DoubleGenotype velocity = (DoubleGenotype) particle.getVelocity();
223                        DoubleGenotype best = (DoubleGenotype) particle.getBest();
224                        int id = particle.getId();
225
226                        DoubleGenotype leader = (DoubleGenotype) leaders.get(particle)
227                                        .getGenotype();
228
229                        velocityTerm.randomize();
230
231                        DoubleGenotype nextVelocity = algebra.algebra(velocityTerm,
232                                        position, velocity, best, leader);
233
234                        DoubleGenotype nextPosition = algebra.algebra(positionTerm,
235                                        position, nextVelocity);
236
237                        for (int k = 0; k < nextPosition.size(); k++) {
238                                double value = nextPosition.get(k);
239                                double lb = nextPosition.getLowerBound(k);
240                                double ub = nextPosition.getUpperBound(k);
241                                if (value < lb) {
242                                        nextPosition.set(k, lb);
243                                        nextVelocity.set(k, -nextVelocity.get(k));
244                                } else if (value > ub) {
245                                        nextPosition.set(k, ub);
246                                        nextVelocity.set(k, -nextVelocity.get(k));
247                                }
248                        }
249
250                        if (id % 3 == 0) {
251                                uniform.mutate(nextPosition, mutationRate.get());
252                        } else if (id % 3 == 1) {
253                                nonUniform.mutate(nextPosition, mutationRate.get());
254                        } // else do nothing
255
256                        Particle p = particleFactory.create(id, nextPosition, nextVelocity);
257
258                        map.put(particle, p);
259                }
260
261                return map;
262        }
263
264        /**
265         * Update the global leaders {@link Archive}.
266         * 
267         * @param leaders
268         *            the archive
269         * @param population
270         *            the population
271         */
272        protected void updateLeaders(Archive leaders, Population population) {
273                leaders.update(population);
274        }
275
276        /**
277         * Update the personal best of each {@link Particle}.
278         * 
279         * @param next
280         *            the old and new positions
281         */
282        protected void updatePersonalBest(Map<Particle, Particle> next) {
283                for (Entry<Particle, Particle> entry : next.entrySet()) {
284
285                        Particle old = entry.getKey();
286                        Particle current = entry.getValue();
287
288                        if (dominates(old, current)) {
289                                current.setBest(old.getBest(), old.getBestObjectives());
290                        } else {
291                                current.setBest(current.getGenotype(), current.getObjectives());
292                        }
293                }
294        }
295
296        /**
297         * Returns {@code true} if the old particle position dominates the new one.
298         * 
299         * @param old
300         *            the old particle
301         * @param current
302         *            the new particle
303         * @return {@code true} if the old particle position dominates the new one
304         */
305        protected boolean dominates(Particle old, Particle current) {
306                if (old.getBest() == null) {
307                        return false;
308                }
309
310                Objectives oldObjectives = old.getBestObjectives();
311                Objectives currentObjectives = current.getObjectives();
312                return oldObjectives.dominates(currentObjectives);
313        }
314
315        /**
316         * Assigns each {@link Particle} a leader.
317         * 
318         * @param leaders
319         *            the archive of available leader
320         * @param population
321         *            the population
322         * @return the map of each particle to its leader
323         */
324        protected Map<Particle, Particle> getLeaders(Archive leaders,
325                        Population population) {
326                Map<Particle, Particle> map = new HashMap<Particle, Particle>();
327
328                Crowding crowding = new Crowding();
329                Map<Individual, Double> values = crowding.getDensityValues(leaders);
330
331                List<Individual> l = new ArrayList<Individual>(leaders);
332                List<Individual> best = new ArrayList<Individual>();
333
334                for (int i = 0; i < population.size(); i++) {
335                        Individual i1 = l.get(random.nextInt(l.size()));
336                        Individual i2 = l.get(random.nextInt(l.size()));
337
338                        if (values.get(i1) >= values.get(i2)) {
339                                best.add(i1);
340                        } else {
341                                best.add(i2);
342                        }
343                }
344
345                int i = 0;
346
347                for (Individual individual : population) {
348                        Particle particle = (Particle) individual;
349                        Particle leader = (Particle) best.get(i++);
350                        map.put(particle, leader);
351                }
352
353                return map;
354        }
