044/**
045 * <p>
046 * The {@link Spea2}-Selector is a Java implementation of the <a
047 * href="http://e-collection.ethbib.ethz.ch/ecol-pool/incoll/incoll_324.pdf"
048 * >SPEA2-MOEA</a>, see "SPEA2: Improving the Strength Pareto Evolutionary
049 * Algorithm For Multiobjective Optimization, Eckart Zitzler, Marco Laumanns,
050 * and Lothar Thiele, In Evolutionary Methods for Design, Optimisation, and
051 * Control, pages 19&ndash;26, 2002.".
052 * </p>
053 * <p>
054 * The {@link Spea2}-Selector will not work properly if the {@link Objectives}
055 * are not fixed, i.e., the {@link Objectives} of an {@link Individual} change
056 * during the optimization process. This caused by the internal caching of
057 * distances, etc. In this case, it is recommended to use the {@link Nsga2}.
058 * </p>
059 * 
060 * @author lukasiewycz
061 * 
062 */
063public class Spea2 implements Selector {
064
065        protected final Rand random;
066        protected final int tournament;
067
068        protected final Map<Individual, Spea2IndividualSet> map = new LinkedHashMap<Individual, Spea2IndividualSet>();
069        protected final Set<Spea2IndividualSet> individualSets = new LinkedHashSet<Spea2IndividualSet>();
070
071        protected final LinkedList<Integer> freeIDs = new LinkedList<Integer>();
072        protected double[][] distance;
073        protected boolean fitnessDirty = true;
074
075        /**
076         * A wrapper for multiple equal (based on their objectives)
077         * {@link Individual}s.
078         * 
079         * @author lukasiewycz
080         * 
081         */
082        private static class Spea2IndividualSet extends LinkedHashSet<Individual> implements Comparable<Spea2IndividualSet> {
083                private static final long serialVersionUID = 1L;
084                protected final int id;
085                protected int fitness;
086                protected int strength;
087                protected final Objectives objectives;
088                protected double nextDistance = 0;
089
090                Spea2IndividualSet(Individual individual, int id) {
091                        this.id = id;
092                        this.add(individual);
093                        this.objectives = individual.getObjectives();
094                        this.fitness = 0;
095                        this.strength = 0;
096                }
097
098                public Objectives getObjectives() {
099                        return objectives;
100                }
101
102                public int getFitness() {
103                        return fitness;
104                }
105
106                public void setFitness(int fitness) {
107                        this.fitness = fitness;
108                }
109
110                public int getStrength() {
111                        return strength;
112                }
113
114                public void setStrength(int strength) {
115                        this.strength = strength;
116                }
117
118                public Individual first() {
119                        assert (size() != 0);
120                        return iterator().next();
121                }
122
123                public int getId() {
124                        return id;
125                }
126
127                public double getNextDistance() {
128                        return nextDistance;
129                }
130
131                public void setNextDistance(double nextDistance) {
132                        this.nextDistance = nextDistance;
133                }
134
135                @Override
136                public int hashCode() {
137                        return id;
138                }
139
140                @Override
141                public boolean equals(Object obj) {
142                        Spea2IndividualSet other = (Spea2IndividualSet) obj;
143                        return (id == other.id);
144                }
145
146                public boolean dominates(Spea2IndividualSet individualSet) {
147                        return this.getObjectives().dominates(individualSet.getObjectives());
148                }
149
150                @Override
151                public int compareTo(Spea2IndividualSet o) {
152                        return o.fitness - this.fitness;
153                }
154
155        }

157        /**
158         * Constructs a {@link Spea2}-{@link Selector}.
159         * 
160         * @param tournament
161         *            the number of individuals that fight against each other to
162         *            become a parent
163         * @param random
164         *            the random number generator
165         */
166        @Inject
167        public Spea2(@Constant(value = "tournament", namespace = Spea2.class) int tournament, Rand random) {
168                this.tournament = tournament;
169                this.random = random;
170        }

172        /*
173         * (non-Javadoc)
174         * 
175         * @see org.opt4j.optimizer.ea.Selector#init(int)
176         */
177        @Override
178        public void init(int maxsize) {
179                for (int i = 0; i < maxsize; i++) {
180                        freeIDs.add(i);
181                }
182
183                distance = new double[maxsize][maxsize];
184        }

186        /*
187         * (non-Javadoc)
188         * 
189         * @see org.opt4j.optimizer.ea.Selector#getParents(int,
190         * java.util.Collection)
191         */
192        @Override
193        public Collection<Individual> getParents(int mu, Collection<Individual> population) {
194                update(population);
195
196                List<Individual> parents = new ArrayList<Individual>();
197
198                List<Individual> candidates = new ArrayList<Individual>(population);
199                int size = candidates.size();
200
201                for (int i = 0; i < mu; i++) {
202                        Individual winner = candidates.get(random.nextInt(size));
203
204                        for (int j = 0; j < tournament; j++) {
205                                Individual opponent = candidates.get(random.nextInt(size));
206
207                                Spea2IndividualSet wWinner = map.get(winner);
208                                Spea2IndividualSet wOpponent = map.get(opponent);
209
210                                double oFitness = wWinner.getFitness();
211                                double wFitness = wOpponent.getFitness();
212
213                                if (oFitness > wFitness || (winner == opponent)) {
214                                        winner = opponent;
215                                } else if (oFitness == wFitness) {
216
217                                        double oDist = getMinDistance(wOpponent);
218                                        double wDist = getMinDistance(wWinner);
219
220                                        if (oDist > wDist) {
221                                                winner = opponent;
222                                        }
223                                }
224                        }
225                        parents.add(winner);
226                }
227
228                assert (parents.size() == mu);
229
230                return parents;
231        }

233        /*
234         * (non-Javadoc)
235         * 
236         * @see org.opt4j.optimizer.ea.Selector#getLames(int, java.util.Collection)
237         */
238        @Override
239        public Collection<Individual> getLames(int lambda, Collection<Individual> population) {
240                update(population);
241
242                assert (lambda <= population.size());
243
244                Set<Individual> lames = new LinkedHashSet<Individual>();
245
246                if (lambda > 0) {
247                        List<Spea2IndividualSet> dominated = getDominated();
248
249                        if (countIndividuals(dominated) >= lambda) {
250
251                                Collections.sort(dominated);
252                                List<Individual> lameCandidates = new ArrayList<Individual>();
253
254                                int i = 0;
255                                while (lameCandidates.size() < lambda) {
256                                        lameCandidates.addAll(dominated.get(i));
257                                        i++;
258                                }
259                                lames.addAll(lameCandidates.subList(0, lambda));
260
261                        } else {
262
263                                for (Spea2IndividualSet w0 : dominated) {
264                                        lames.addAll(w0);
265                                }
266
267                                for (Individual individual : getLamesFromNonDominated(lambda - lames.size())) {
268                                        lames.add(individual);
269                                }
270                        }
271
272                }
273
274                assert (lames.size() == lambda);
275
276                return lames;
277        }

279        /**
280         * Returns a specific number of lames from the non-dominated
281         * {@link Individual}s.
282         * 
283         * @param count
284         *            the specified number
285         * @return a collection of the lame individuals
286         */
287        public Collection<Individual> getLamesFromNonDominated(int count) {
288                Set<Individual> set = new LinkedHashSet<Individual>();
289
290                while (set.size() < count) {
291                        int maxsize = 0;
292
293                        List<Spea2IndividualSet> candidates = new ArrayList<Spea2IndividualSet>();
294                        for (Spea2IndividualSet individualSet : getNonDominated()) {
295                                if (individualSet.size() > maxsize) {
296                                        maxsize = individualSet.size();
297                                        candidates.clear();
298                                }
299                                if (individualSet.size() == maxsize) {
300                                        candidates.add(individualSet);
301                                }
302                        }
303
304                        if (candidates.size() <= (count - set.size())) {
305                                for (Spea2IndividualSet individualSet : candidates) {
306                                        Individual individual = individualSet.first();
307                                        remove(individual);
308                                        set.add(individual);
309                                }
310                        } else { // candidates.size() > (count - set.size())
311                                for (Spea2IndividualSet individualSet : getNearest(count - set.size(), candidates)) {
312                                        Individual individual = individualSet.first();
313                                        remove(individual);
314                                        set.add(individual);
315                                }
316
317                        }
318
319                }
320
321                assert (set.size() == count);
322
323                return set;
324        }

326        /**
327         * Returns n with nearest neighbor based on distances.
328         * 
329         * @param n
330         *            the number of required IndividualSets
331         * @param candidates
332         *            the candidate IndividualSets
333         * @return the n nearest neighbors
334         */
335        protected List<Spea2IndividualSet> getNearest(int n, Collection<Spea2IndividualSet> candidates) {
336
337                assert (candidates.size() > n);
338
339                List<Spea2IndividualSet> lames = new ArrayList<Spea2IndividualSet>();
340
341                Map<Spea2IndividualSet, List<Spea2IndividualSet>> orderedLists = new LinkedHashMap<Spea2IndividualSet, List<Spea2IndividualSet>>();
342
343                for (Spea2IndividualSet w0 : candidates) {
344                        List<Spea2IndividualSet> list = new ArrayList<Spea2IndividualSet>();
345                        for (Spea2IndividualSet w1 : candidates) {
346                                if (w0 != w1) {
347                                        w1.setNextDistance(distance(w0, w1));
348                                        list.add(w1);
349                                }
350                        }
351                        Collections.sort(list, new Comparator<Spea2IndividualSet>() {
352                                @Override
353                                public int compare(Spea2IndividualSet o1, Spea2IndividualSet o2) {
354                                        double v = o1.getNextDistance() - o2.getNextDistance();
355                                        if (v < 0) {
356                                                return -1;
357                                        } else if (v > 0) {
358                                                return 1;
359                                        } else {
360                                                return 0;
361                                        }
362                                }
363
364                        });
365                        orderedLists.put(w0, list);
366                }
367
368                while (lames.size() < n) {
369                        List<Spea2IndividualSet> lcandidates = new ArrayList<Spea2IndividualSet>(orderedLists.keySet());
370
371                        int size = lcandidates.size();
372
373                        for (int k = 0; k < size - 1; k++) {
374                                double min = Double.MAX_VALUE;
375
376                                for (Spea2IndividualSet candidate : lcandidates) {
377                                        double value = distance(candidate, orderedLists.get(candidate).get(k));
378                                        min = Math.min(min, value);
379                                }
380
381                                for (Iterator<Spea2IndividualSet> it = lcandidates.iterator(); it.hasNext();) {
382                                        Spea2IndividualSet candidate = it.next();
383                                        double value = distance(candidate, orderedLists.get(candidate).get(k));
384                                        if (value > min) {
385                                                it.remove();
386                                        }
387                                }
388
389                                if (lcandidates.size() == 1) {
390                                        break;
391                                }
392                        }
393
394                        Spea2IndividualSet lame = lcandidates.get(0);
395                        lames.add(lame);
396                        orderedLists.remove(lame);
397
398                        for (List<Spea2IndividualSet> list : orderedLists.values()) {
399                                list.remove(lame);
400                        }
401
402                }
403
404                assert (lames.size() == n);
405
406                return lames;
407        }

409        protected double getMinDistance(Spea2IndividualSet w0) {
410                double min = Double.MAX_VALUE;
411                for (Spea2IndividualSet w1 : individualSets) {
412                        if (w0 != w1) {
413                                min = Math.min(min, distance(w0, w1));
414                        }
415                }
416                return min;
417        }

419        /**
420         * Update with current population.
421         * 
422         * @param population
423         *            the current population
424         */
425        protected void update(Collection<Individual> population) {
426                Set<Individual> popSet = new LinkedHashSet<Individual>(population);
427
428                if (!popSet.equals(map.keySet())) {
429
430                        Set<Individual> adds = new LinkedHashSet<Individual>(popSet);
431                        adds.removeAll(map.keySet());
432
433                        Set<Individual> removes = new LinkedHashSet<Individual>(map.keySet());
434                        removes.removeAll(popSet);
435
436                        for (Individual individual : removes) {
437                                remove(individual);
438                        }
439
440                        for (Individual individual : adds) {
441                                add(individual);
442                        }
443                }
444
445                assert (population.size() == map.size());
446
447                if (fitnessDirty) {
448                        calculateFitness();
449                        fitnessDirty = false;
450                }
451
452        }

454        /**
455         * Return the distance of two {@code Spea2IndividualSet}s.
456         * 
457         * @param w0
458         *            first set
459         * @param w1
460         *            second set
461         * @return the distance
462         */
463        protected double distance(Spea2IndividualSet w0, Spea2IndividualSet w1) {
464                return distance[w0.getId()][w1.getId()];
465        }
466
467        /**
468         * Add a new {@link Individual}.
469         * 
470         * @param individual
471         *            the individual to add
472         */
473        protected void add(Individual individual) {
474                int id0 = freeIDs.removeFirst();
475                Spea2IndividualSet w0 = new Spea2IndividualSet(individual, id0);
476
477                Spea2IndividualSet eq = null;
478
479                for (Spea2IndividualSet w1 : individualSets) {
480                        int id1 = w1.getId();
481                        double dist = calculateDistance(w0, w1);
482                        if (dist == 0.0) {
483                                eq = w1;
484                                break;
485                        }
486                        distance[id0][id1] = dist;
487                        distance[id1][id0] = dist;
488                }
489                if (eq != null) {
490                        freeIDs.add(id0);
491                        w0 = eq;
492                        w0.add(individual);
493                } else {
494                        distance[id0][id0] = 0.0;
495                        individualSets.add(w0);
496                }
497
498                map.put(individual, w0);
499                fitnessDirty = true;
500        }
501
502        /**
503         * Remove an {@link Individual}.
504         * 
505         * @param individual
506         *            the individual to remove
507         */
508        protected void remove(Individual individual) {
509                Spea2IndividualSet individualSet = map.remove(individual);
510
511                if (individualSet.size() == 1) {
512                        individualSets.remove(individualSet);
513                        freeIDs.add(individualSet.getId());
514                } else {
515                        individualSet.remove(individual);
516                }
517
518                fitnessDirty = true;
519        }

521        /**
522         * Calculate the distance between two {@code Spea2IndividualSet}s.
523         * 
524         * @param w0
525         *            the first set
526         * @param w1
527         *            the second set
528         * @return the distance
529         */
530        protected double calculateDistance(Spea2IndividualSet w0, Spea2IndividualSet w1) {
531                return w0.getObjectives().distance(w1.getObjectives());
532        }
533
534        /**
535         * Calculate the fitness.
536         */
537        protected void calculateFitness() {
538                for (Spea2IndividualSet individual : individualSets) {
539                        int s = 0;
540                        for (Spea2IndividualSet other : individualSets) {
541                                if (individual != other && individual.dominates(other)) {
542                                        s += other.size();
543                                }
544                        }
545                        individual.setStrength(s);
546                }
547
548                for (Spea2IndividualSet individual : individualSets) {
549                        int f = 0;
550                        for (Spea2IndividualSet other : individualSets) {
551                                if (individual != other && other.dominates(individual)) {
552                                        f += other.getStrength() * other.size();
553                                }
554                        }
555                        individual.setFitness(f);
556                }
557        }
558
559        /**
560         * Returns all dominated {@code Spea2IndividualSet}s (fitness &gt; 0).
561         * 
562         * @return all dominated IndividualSets
563         */
564        protected List<Spea2IndividualSet> getDominated() {
565                List<Spea2IndividualSet> dominated = new ArrayList<Spea2IndividualSet>();
566
567                for (Spea2IndividualSet w0 : individualSets) {
568                        if (w0.getFitness() > 0.0) {
569                                dominated.add(w0);
570                        }
571                }
572
573                return dominated;
574        }
575
576        /**
577         * Returns all non-dominated {@code Spea2IndividualSet}s (fitness == 0).
578         * 
579         * @return all non-dominated individualSets
580         */
581        protected List<Spea2IndividualSet> getNonDominated() {
582                List<Spea2IndividualSet> dominated = new ArrayList<Spea2IndividualSet>();
583
584                for (Spea2IndividualSet w0 : individualSets) {
585                        if (w0.getFitness() == 0.0) {
586                                dominated.add(w0);
587                        }
588                }
589
590                return dominated;
591        }

593        /**
594         * Returns the total number of {@link Individual}s in a collection of
595         * {@code Spea2IndividualSet}s.
596         * 
597         * @param collection
598         *            the collection of IndividualSets
599         * @return the total number of Individuals
600         */
601        private int countIndividuals(Collection<Spea2IndividualSet> collection) {
602                int c = 0;
603                for (Spea2IndividualSet w : collection) {
604                        c += w.size();
605                }
606                return c;
607        }
