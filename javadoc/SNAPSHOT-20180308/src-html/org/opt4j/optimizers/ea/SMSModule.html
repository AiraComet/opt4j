024package org.opt4j.optimizers.ea;
025
026import org.opt4j.core.common.archive.FrontDensityIndicator;
027import org.opt4j.core.config.annotations.Info;
028import org.opt4j.core.start.Constant;
029
030/**
031 * Module for the S-Metric Selection ({@link Selector}) based on the
032 * {@link Hypervolume} contribution, see "M. Emmerich, N. Beume, and B. Naujoks.
033 * An EMO Algorithm Using the Hypervolume Measure as Selection Criterion. EMO
034 * 2005.".
035 * 
036 * @see Hypervolume
037 * @author lukasiewycz
038 * @author Ramin Etemaadi
039 * 
040 */
041@Info("SMS-EMOA: Multiobjective selection based on dominated hypervolume")
042public class SMSModule extends SelectorModule {
043
044        @Info("The offset value")
045        @Constant(value = "offset", namespace = Hypervolume.class)
046        protected double offset = 1.0;
047
048        @Info("The tournament value")
049        @Constant(value = "tournament", namespace = Nsga2.class)
050        protected int tournament = 0;
051
052        /**
053         * Returns the tournament value.
054         * 
055         * @see #setTournament
056         * @return the tournament value
057         */
058        public int getTournament() {
059                return tournament;
060        }
061
062        /**
063         * Sets the tournament value.
064         * 
065         * @see #getTournament
066         * @param tournament
067         *            the tournament to set
068         */
069        public void setTournament(int tournament) {
070                this.tournament = tournament;
071        }
072
073        /**
074         * Returns the offset value.
075         * 
076         * @return the offset value
077         */
078        public double getOffset() {
079                return offset;
080        }
081
082        /**
083         * Sets the offset value.
084         * 
085         * @param offset
086         *            the offset to set
087         */
088        public void setOffset(double offset) {
089                this.offset = offset;
090        }
091
092        /*
093         * (non-Javadoc)
094         * 
095         * @see org.opt4j.start.Opt4JModule#config()
096         */
097        @Override
098        public void config() {
099                bindSelector(Nsga2.class);
100                bind(FrontDensityIndicator.class).to(Hypervolume.class);
101        }
102
103}
