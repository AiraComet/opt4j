024package org.opt4j.optimizers.ea;
025
026import java.util.ArrayList;
027import java.util.Collection;
028import java.util.Collections;
029import java.util.Comparator;
030import java.util.HashMap;
031import java.util.List;
032import java.util.Map;
033import java.util.Random;
034
035import org.opt4j.core.Individual;
036import org.opt4j.core.Objectives;
037import org.opt4j.core.common.archive.FrontDensityIndicator;
038import org.opt4j.core.common.random.Rand;
039import org.opt4j.core.start.Constant;
040
041import com.google.inject.Inject;
042
043/**
044 * The {@link Nsga2} {@link Selector}, see "A Fast Elitist Non-Dominated Sorting
045 * Genetic Algorithm for Multi-Objective Optimization: NSGA-II, K. Deb, Samir
046 * Agrawal, Amrit Pratap, and T. Meyarivan, Parallel MockProblem Solving from
047 * Nature, 2000".
048 * 
049 * @see Nsga2Module
050 * @author lukasiewycz, noorshams
051 * 
052 */
053public class Nsga2 implements Selector {
054
055        protected final Random random;
056        protected final int tournament;
057        protected final FrontDensityIndicator indicator;
058
059        /**
060         * Constructs a {@link Nsga2} {@link Selector}.
061         * 
062         * @param random
063         *            the random number generator
064         * @param tournament
065         *            the tournament value
066         */
067        @Inject
068        public Nsga2(Rand random, @Constant(value = "tournament", namespace = Nsga2.class) int tournament,
069                        FrontDensityIndicator indicator) {
070                this.random = random;
071                this.tournament = tournament;
072                this.indicator = indicator;
073        }
074
075        /*
076         * (non-Javadoc)
077         * 
078         * @see org.opt4j.optimizer.ea.Selector#init(int)
079         */
080        @Override
081        public void init(int maxsize) {
082                // do nothing
083        }
084
085        /*
086         * (non-Javadoc)
087         * 
088         * @see org.opt4j.optimizer.ea.Selector#getParents(int,
089         * java.util.Collection)
090         */
091        @Override
092        public Collection<Individual> getParents(int mu, Collection<Individual> population) {
093                List<Individual> all = new ArrayList<Individual>(population);
094                List<Individual> parents = new ArrayList<Individual>();
095
096                List<List<Individual>> fronts = fronts(all);
097                Map<Individual, Integer> rank = getRank(fronts);
098                Map<Individual, Double> distance = new HashMap<Individual, Double>();
099
100                final int size = all.size();
101
102                for (int i = 0; i < mu; i++) {
103                        Individual winner = all.get(random.nextInt(size));
104
105                        for (int t = 0; t < tournament; t++) {
106                                Individual opponent = all.get(random.nextInt(size));
107                                if (rank.get(opponent) < rank.get(winner) || opponent == winner) {
108                                        winner = opponent;
109                                } else if (rank.get(opponent) == rank.get(winner)) {
110                                        // The winner is determined considering the crowding
111                                        // distance
112
113                                        if (!distance.containsKey(winner)) {
114                                                List<Individual> front = fronts.get(rank.get(winner));
115                                                distance.putAll(indicator.getDensityValues(front));
116                                        }
117
118                                        // Opponent wins, if it has a better crowding distance
119                                        if (distance.get(opponent) > distance.get(winner)) {
120                                                winner = opponent;
121                                        }
122
123                                }
124                        }
125
126                        parents.add(winner);
127                }
128
129                return parents;
130        }
131
132        /*
133         * (non-Javadoc)
134         * 
135         * @see org.opt4j.optimizer.ea.Selector#getLames(int, java.util.Collection)
136         */
137        @Override
138        public Collection<Individual> getLames(int size, Collection<Individual> population) {
139                List<Individual> lames = new ArrayList<Individual>();
140
141                List<List<Individual>> fronts = fronts(population);
142                Collections.reverse(fronts);
143
144                for (List<Individual> front : fronts) {
145                        if (lames.size() + front.size() < size) {
146                                lames.addAll(front);
147                        } else {
148                                final Map<Individual, Double> density = indicator.getDensityValues(front);
149                                Collections.sort(front, new Comparator<Individual>() {
150                                        @Override
151                                        public int compare(Individual o1, Individual o2) {
152                                                return density.get(o1).compareTo(density.get(o2));
153                                        }
154                                });
155                                lames.addAll(front.subList(0, size - lames.size()));
156                        }
157                }
158                return lames;
159        }
160
161        /**
162         * Determine the ranks of fronts.
163         * 
164         * @param fronts
165         *            the fronts
166         * @return the ranks
167         */
168        protected Map<Individual, Integer> getRank(List<List<Individual>> fronts) {
169                Map<Individual, Integer> ranks = new HashMap<Individual, Integer>();
170                for (int i = 0; i < fronts.size(); i++) {
171                        for (Individual p : fronts.get(i)) {
172                                ranks.put(p, i);
173                        }
174                }
175                return ranks;
176        }
177
178        /**
179         * Evaluate the fronts and set the correspondent rank values.
180         * 
181         * @param individuals
182         *            the individuals
183         * @return the fronts
184         */
185        public List<List<Individual>> fronts(Collection<Individual> individuals) {
186
187                List<Individual> pop = new ArrayList<Individual>(individuals);
188                Map<Individual, Integer> id = new HashMap<Individual, Integer>();
189                for (int i = 0; i < pop.size(); i++) {
190                        id.put(pop.get(i), i);
191                }
192
193                List<List<Individual>> fronts = new ArrayList<List<Individual>>();
194
195                Map<Individual, List<Individual>> S = new HashMap<Individual, List<Individual>>();
196                int[] n = new int[pop.size()];
197
198                for (Individual e : pop) {
199                        S.put(e, new ArrayList<Individual>());
200                        n[id.get(e)] = 0;
201                }
202
203                for (int i = 0; i < pop.size(); i++) {
204                        for (int j = i + 1; j < pop.size(); j++) {
205                                Individual p = pop.get(i);
206                                Individual q = pop.get(j);
207
208                                Objectives po = p.getObjectives();
209                                Objectives qo = q.getObjectives();
210
211                                if (po.dominates(qo)) {
212                                        S.get(p).add(q);
213                                        n[id.get(q)]++;
214                                } else if (qo.dominates(po)) {
215                                        S.get(q).add(p);
216                                        n[id.get(p)]++;
217                                }
218                        }
219                }
220
221                List<Individual> f1 = new ArrayList<Individual>();
222                for (Individual i : pop) {
223                        if (n[id.get(i)] == 0) {
224                                f1.add(i);
225                        }
226                }
227                fronts.add(f1);
228                List<Individual> fi = f1;
229                while (!fi.isEmpty()) {
230                        List<Individual> h = new ArrayList<Individual>();
231                        for (Individual p : fi) {
232                                for (Individual q : S.get(p)) {
233                                        n[id.get(q)]--;
234                                        if (n[id.get(q)] == 0) {
235                                                h.add(q);
236                                        }
237                                }
238                        }
239                        fronts.add(h);
240                        fi = h;
241                }
242                return fronts;
243        }
