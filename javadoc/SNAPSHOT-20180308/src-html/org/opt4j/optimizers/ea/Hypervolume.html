027import java.util.Arrays;
028import java.util.Collection;
029import java.util.Collections;

031import java.util.HashMap;
032import java.util.Iterator;
033import java.util.List;
034import java.util.Map;

036import org.opt4j.core.Individual;

042/**
043 * The {@link Hypervolume}, see "Zitzler, E., and Thiele, L. (1998):
044 * Multiobjective Optimization Using Evolutionary Algorithms - A Comparative
045 * Case Study. Parallel Problem Solving from Nature (PPSN-V), 292-301." is a
046 * {@link FrontDensityIndicator} based on determination of the hypervolume
047 * contribution. The calculation is based on a normalization between 0 and 1 in
048 * each dimension and a transformation to a maximization problem. Additionally
049 * an offset value (default 1) is added to each dimension.
050 * 
051 * 
052 * @see SMSModule
053 * @author Ramin Etemaadi
054 * @author Johannes Kruisselbrink
055 * @author Rui Li
056 * @author lukasiewycz
057 * 
058 */
059public class Hypervolume implements FrontDensityIndicator {
060
061        protected final double offset;
062
063        /**
064         * Constructs a {@link Hypervolume}.
065         * 
066         * @param offset
067         *            the offset that is added to each dimension before the
068         *            hypervolume is calculated
069         */
070        @Inject
071        public Hypervolume(@Constant(value = "offset", namespace = Hypervolume.class) double offset) {
072                this.offset = offset;
073        }
074
075        /*
076         * (non-Javadoc)
077         * 
078         * @see
079         * org.opt4j.optimizer.ea.FrontDensityIndicator#getDensityValues(java.util
080         * .Collection)
081         */
082        @Override
083        public Map<Individual, Double> getDensityValues(Collection<Individual> individuals) {
084                return getDensityValues(individuals, this.offset);
085        }
086
087        /**
088         * Calculates the density values for a front of non-dominated individuals
089         * based on the contribution of the {@link Hypervolume}.
090         * 
091         * A special approach for two dimension exists as well as a general approach
092         * for n dimensions.
093         * 
094         * @param individuals
095         *            the individuals
096         * @param offset
097         *            the offset
098         * @return the map of density values
099         */
100        protected Map<Individual, Double> getDensityValues(Collection<Individual> individuals, double offset) {
101                if (individuals.isEmpty()) {
102                        throw new IllegalArgumentException("Individuals is empty.");
103                }
104
105                List<Individual> orderIndividuals = new ArrayList<Individual>(individuals);
106                int m = individuals.iterator().next().getObjectives().size();
107
108                if (m < 2) {
109                        Map<Individual, Double> result = new HashMap<Individual, Double>();
110                        for (Individual individual : individuals) {
111                                result.put(individual, 0.0);
112                        }
113                        return result;
114                } else if (m == 2) {
115                        return calculateHypervolumeContribution2D(orderIndividuals, offset);
116                } else {
117                        return calculateHypervolumeContributionN(orderIndividuals, offset);
118                }
119        }
120
121        /**
122         * Calculates the {@link Hypervolume} contribution for n dimensions.
123         * 
124         * @param individuals
125         *            the individuals
126         * @param offset
127         *            the offset
128         * @return the map of density values
129         */
130        protected Map<Individual, Double> calculateHypervolumeContributionN(List<Individual> individuals, double offset) {
131                Map<Individual, Double> result = new HashMap<Individual, Double>();
132                List<double[]> front = invert(normalize(getMinValues(individuals)), offset);
133
134                int m = front.get(0).length;
135
136                double hvAll = calculateHypervolume(front, m);
137
138                for (int i = 0; i < front.size(); i++) {
139                        List<double[]> iFront = new ArrayList<double[]>(front);
140                        iFront.remove(i);
141                        double iHv = calculateHypervolume(iFront, m);
142                        result.put(individuals.get(i), hvAll - iHv);
143                }
144
145                return result;
146        }
147
148        /**
149         * Calculates the {@link Hypervolume} contribution for two dimensions.
150         * 
151         * @param individuals
152         *            the individuals
153         * @param offset
154         *            the offset
155         * @return the map of density values
156         */
157        protected Map<Individual, Double> calculateHypervolumeContribution2D(List<Individual> individuals, double offset) {
158                Map<Individual, Double> result = new HashMap<Individual, Double>();
159                List<double[]> front = invert(normalize(getMinValues(individuals)), offset);
160                List<double[]> sorted = new ArrayList<double[]>(front);
161
162                Collections.sort(sorted, new Comparator<double[]>() {
163                        @Override
164                        public int compare(double[] o1, double[] o2) {
165                                Double v1 = o1[0];
166                                Double v2 = o2[0];
167                                return v1.compareTo(v2);
168                        }
169                });
170
171                final int size = sorted.size();
172
173                for (int i = 0; i < size; i++) {
174                        double diffX = sorted.get(i)[0] - (i > 0 ? sorted.get(i - 1)[0] : 0);
175                        double diffY = sorted.get(i)[1] - (i < size - 1 ? sorted.get(i + 1)[1] : 0);
176                        double contribution = diffX * diffY;
177
178                        result.put(individuals.get(front.indexOf(sorted.get(i))), contribution);
179                }
180
181                return result;
182        }
183
184        /**
185         * Transforms the non-dominated {@link Individual}s to a front where each
186         * objective is to be minimized.
187         * 
188         * @param individuals
189         *            the individuals
190         * @return the front of vectors that is minimized
191         */
192        protected List<double[]> getMinValues(List<Individual> individuals) {
193                List<double[]> minValues = new ArrayList<double[]>();
194                for (Individual individual : individuals) {
195                        minValues.add(individual.getObjectives().array());
196                }
197                return minValues;
198        }
199
200        /**
201         * Normalizes a front of non-dominated solutions to values between 0 and 1.
202         * 
203         * @param front
204         *            the front of non-dominated solutions
205         * @return the normalized front
206         */
207        protected List<double[]> normalize(List<double[]> front) {
208                int m = front.get(0).length;
209
210                double[] min = new double[m];
211                double[] max = new double[m];
212
213                Arrays.fill(min, +Double.MAX_VALUE);
214                Arrays.fill(max, -Double.MAX_VALUE);
215
216                for (double[] p : front) {
217                        for (int i = 0; i < m; i++) {
218                                min[i] = Math.min(min[i], p[i]);
219                                max[i] = Math.max(max[i], p[i]);
220                        }
221                }
222
223                for (int i = 0; i < m; i++) {
224                        if (min[i] == max[i]) {
225                                max[i]++;
226                        }
227                }
228
229                List<double[]> normalized = new ArrayList<double[]>();
230                for (double[] p : front) {
231                        double[] pn = new double[m];
232                        for (int i = 0; i < m; i++) {
233                                pn[i] = (p[i] - min[i]) / (max[i] - min[i]);
234                        }
235                        normalized.add(pn);
236                }
237                return normalized;
238        }
239
240        /**
241         * Inverts (from a minimization to a maximization problem) a front of
242         * solutions and adds an offset value to each dimension.
243         * 
244         * @param front
245         *            the front of non-dominated solutions
246         * @param offset
247         *            the offset
248         * @return the inverted front
249         */
250        protected List<double[]> invert(List<double[]> front, double offset) {
251                int m = front.get(0).length;
252
253                double[] nadir = new double[m];
254                Arrays.fill(nadir, 1.0 + offset);
255
256                List<double[]> inverted = new ArrayList<double[]>();
257                for (double[] element : front) {
258                        double[] in = new double[element.length];
259                        for (int i = 0; i < element.length; i++) {
260                                in[i] = nadir[i] - element[i];
261                        }
262                        inverted.add(in);
263                }
264
265                return inverted;
266        }
267
268        /**
269         * Implements the {@link Hypervolume} calculations as proposed by Zitzler,
270         * E., and Thiele, L. (1998). All points have positive values in all
271         * dimensions and the hypervolume is calculated from 0.
272         * 
273         * @param front
274         *            the front of non-dominated solutions
275         * @param nObjectives
276         *            the number of objectives
277         * @return the hypervolume
278         */
279        protected double calculateHypervolume(List<double[]> front, int nObjectives) {
280                double volume = 0.0;
281                double distance = 0.0;
282
283                while (!front.isEmpty()) {
284                        List<double[]> nondominatedPoints = filterNondominatedSet(front, nObjectives - 1);
285                        double tempVolume;
286                        if (nObjectives < 3) {
287                                assert nondominatedPoints.size() > 0;
288                                tempVolume = nondominatedPoints.get(0)[0];
289                        } else {
290                                tempVolume = calculateHypervolume(nondominatedPoints, nObjectives - 1);
291                        }
292                        double tempDistance = surfaceUnchangedTo(front, nObjectives - 1);
293                        volume += tempVolume * (tempDistance - distance);
294                        distance = tempDistance;
295                        front = reduceNondominatedSet(front, nObjectives - 1, distance);
296                }
297
298                return volume;
299        }
300
301        protected List<double[]> filterNondominatedSet(List<double[]> front, int nObjectives) {
302                List<double[]> nondominated = new ArrayList<double[]>();
303
304                for (double[] p1 : front) {
305                        boolean dominated = false;
306                        for (double[] p2 : nondominated) {
307                                if (dominates(p2, p1, nObjectives)) {
308                                        dominated = true;
309                                        break;
310                                }
311                        }
312
313                        if (!dominated) {
314                                for (Iterator<double[]> it = nondominated.iterator(); it.hasNext();) {
315                                        double[] p2 = it.next();
316                                        if (dominates(p1, p2, nObjectives)) {
317                                                it.remove();
318                                        }
319                                }
320                                nondominated.add(p1);
321                        }
322                }
323
324                return nondominated;
325        }
326
327        protected boolean dominates(double[] p1, double[] p2, int nObjectives) {
328                boolean strong = false;
329                for (int i = 0; i < nObjectives; i++) {
330                        if (p1[i] > p2[i]) {
331                                strong = true;
332                        } else if (p1[i] < p2[i]) {
333                                return false;
334                        }
335                }
336                return strong;
337        }
338
339        protected double surfaceUnchangedTo(List<double[]> front, int objective) {
340                assert front.size() > 0;
341                double value = Double.MAX_VALUE;
342                for (double[] p : front) {
343                        value = Math.min(value, p[objective]);
344                }
345                return value;
346        }
347
348        protected List<double[]> reduceNondominatedSet(List<double[]> front, int objective, double threshold) {
349                List<double[]> result = new ArrayList<double[]>();
350
351                for (double[] p : front) {
352                        if (p[objective] > threshold) {
353                                result.add(p);
354                        }
355                }
356                return result;
357        }
