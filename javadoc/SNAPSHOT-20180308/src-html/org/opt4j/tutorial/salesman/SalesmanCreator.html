001package org.opt4j.tutorial.salesman;
002
003import java.util.Collections;
004
005import org.opt4j.core.genotype.PermutationGenotype;
006import org.opt4j.core.problem.Creator;
007import org.opt4j.tutorial.salesman.SalesmanProblem.City;
008
009import com.google.inject.Inject;
010
011public class SalesmanCreator implements Creator<PermutationGenotype<City>> {
012
013        protected final SalesmanProblem problem;
014
015        @Inject
016        public SalesmanCreator(SalesmanProblem problem) {
017                this.problem = problem;
018        }
019
020        public PermutationGenotype<City> create() {
021                PermutationGenotype<City> genotype = new PermutationGenotype<City>();
022                for (City city : problem.getCities()) {
023                        genotype.add(city);
024                }
025
026                Collections.shuffle(genotype);
027
028                return genotype;
029        }
030}
