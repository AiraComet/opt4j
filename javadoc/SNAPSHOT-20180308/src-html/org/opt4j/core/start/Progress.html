023package org.opt4j.core.start;
024
025import org.opt4j.core.optimizer.MaxIterations;
026import org.opt4j.core.optimizer.Optimizer;
027import org.opt4j.core.optimizer.OptimizerIterationListener;
028
029import com.google.inject.Inject;
030import com.google.inject.Singleton;
031
032/**
033 * The {@link Progress} offers some methods for the measurement of the progress
034 * of an {@link Optimizer}.
035 * 
036 * @see Optimizer
037 * @author lukasiewycz
038 * 
039 */
040@Singleton
041public class Progress implements OptimizerIterationListener {
042
043        @Inject(optional = true)
044        @MaxIterations
045        protected Integer maxIterations = null;
046
047        protected int iteration = 0;
048
049        /**
050         * Returns the progress value between {@code 0} and {@code 1}.
051         * 
052         * @return the progress value
053         */
054        public Double get() {
055                if (maxIterations == null) {
056                        return null;
057                }
058                return (double) getCurrentIteration() / (double) getMaxIterations();
059        }
060
061        /**
062         * Returns the number of maximal iterations.
063         * 
064         * @return the number of maximal iteration
065         */
066        public Integer getMaxIterations() {
067                return maxIterations == null ? getCurrentIteration() : maxIterations;
068        }
069
070        /**
071         * Returns the current iteration.
072         * 
073         * @return the current iteration
074         */
075        public int getCurrentIteration() {
076                return Math.min(iteration, maxIterations == null ? iteration : maxIterations);
077        }
078
079        /*
080         * (non-Javadoc)
081         * 
082         * @see
083         * org.opt4j.core.optimizer.OptimizerIterationListener#iterationComplete
084         * (org.opt4j.core.optimizer.Optimizer, int)
085         */
086        @Override
087        public void iterationComplete(int iteration) {
088                this.iteration = iteration;
089        }
090
091}
