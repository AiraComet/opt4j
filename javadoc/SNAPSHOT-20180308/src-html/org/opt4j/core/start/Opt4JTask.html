023package org.opt4j.core.start;
024
025import java.lang.reflect.Type;
026import java.util.Set;
027
028import org.opt4j.core.Genotype;
029import org.opt4j.core.config.Task;
030import org.opt4j.core.config.TaskStateListener;
031import org.opt4j.core.optimizer.Control;
032import org.opt4j.core.optimizer.ControlListener;
033import org.opt4j.core.optimizer.Optimizer;
034import org.opt4j.core.optimizer.OptimizerIterationListener;
035import org.opt4j.core.problem.Creator;
036import org.opt4j.core.problem.Decoder;
037import org.opt4j.core.problem.Evaluator;
038
039import com.google.inject.Guice;
040import com.google.inject.Inject;
041import com.google.inject.Injector;
042import com.google.inject.Key;
043import com.google.inject.TypeLiteral;
044
045/**
046 * The {@link Opt4JTask} executes one optimization process.
047 * 
048 * @author lukasiewycz
049 * @see Task
050 * 
051 */
052public class Opt4JTask extends Task implements ControlListener, OptimizerIterationListener {
053
054        protected Optimizer optimizer = null;
055
056        protected Injector injector = null;
057
058        protected Injector parentInjector = null;
059
060        protected final boolean closeOnStop;
061
062        protected boolean isClosed = false;
063
064        /**
065         * Constructs a {@link Opt4JTask}.
066         * 
067         */
068        @Inject
069        public Opt4JTask() {
070                this(true);
071        }
072
073        /**
074         * Constructs a {@link Opt4JTask}.
075         * 
076         * @param closeOnStop
077         *            close automatically after optimization
078         */
079        public Opt4JTask(boolean closeOnStop) {
080                this.closeOnStop = closeOnStop;
081        }
082
083        /*
084         * (non-Javadoc)
085         * 
086         * @see org.opt4j.config.Task#execute()
087         */
088        @Override
089        public void execute() throws Exception {
090                open();
091                check(injector);
092
093                Control control = injector.getInstance(Control.class);
094                optimizer = injector.getInstance(Optimizer.class);
095                stateChanged();
096                control.addListener(this);
097                optimizer.addOptimizerIterationListener(this);
098
099                Optimizer optimizer = injector.getInstance(Optimizer.class);
100                optimizer.startOptimization();
101
102                if (closeOnStop) {
103                        close();
104                }
105        }
106
107        /**
108         * Initialize with a parent {@link Injector}.
109         * 
110         * @param injector
111         *            the parent injector
112         */
113        public void init(Injector injector) {
114                this.parentInjector = injector;
115        }
116
117        /**
118         * Close the task.
119         */
120        public synchronized void close() {
121                optimizer = null;
122                injector = null;
123                isClosed = true;
124        }
125
126        /**
127         * Initialize a task manually before executing it. This enables to get
128         * instances of classes before the optimization starts.
129         */
130        public synchronized void open() {
131                if (injector == null && !isClosed) {
132                        if (!isInit) {
133                                throw new RuntimeException("Task is not initialized. Call method init(modules) first.");
134                        }
135                        if (parentInjector == null) {
136                                injector = Guice.createInjector(modules);
137                        } else {
138                                injector = parentInjector.createChildInjector(modules);
139                        }
140                }
141        }
142
143        /**
144         * Returns the current iteration.
145         * 
146         * @return the current iteration
147         */
148        public int getIteration() {
149                Optimizer optimizer = getOptimizer();
150                if (optimizer == null) {
151                        return 0;
152                }
153                return optimizer.getIteration();
154        }
155
156        /**
157         * Returns the instance of the given class.
158         * 
159         * @param <O>
160         *            the type of class
161         * @param type
162         *            the class
163         * @return the instance of the class
164         */
165        public <O> O getInstance(Class<O> type) {
166                Injector injector = getInjector();
167                if (injector == null) {
168                        return null;
169                }
170                return injector.getInstance(type);
171        }
172
173        /**
174         * Returns the {@link Injector} of the task.
175         * 
176         * @return the injector
177         */
178        protected Injector getInjector() {
179                return injector;
180        }
181
182        /**
183         * Returns the {@link Optimizer} of the task.
184         * 
185         * @return the optimizer
186         */
187        protected Optimizer getOptimizer() {
188                return optimizer;
189        }
190
191        /**
192         * Checks for configuration errors in the {@link Injector}.
193         * 
194         * @param injector
195         *            the injector
196         */
197        private void check(Injector injector) {
198
199                Creator<Genotype> creator = null;
200                Decoder<Genotype, Object> decoder = null;
201                Set<Evaluator<Object>> evaluators = null;
202
203                try {
204                        creator = injector.getInstance(Key.get(new TypeLiteral<Creator<Genotype>>() {
205                        }));
206                        decoder = injector.getInstance(Key.get(new TypeLiteral<Decoder<Genotype, Object>>() {
207                        }));
208                        evaluators = injector.getInstance(Key.get(new TypeLiteral<Set<Evaluator<Object>>>() {
209                        }));
210                } catch (Exception e) {
211                        throw new RuntimeException("Problem configuration Exception: \n" + e.getLocalizedMessage(), e);
212                }
213                try {
214                        injector.getInstance(Optimizer.class);
215                } catch (Exception e) {
216                        throw new RuntimeException("Optimizer configuration Exception: \n" + e.getLocalizedMessage(), e);
217                }
218
219                try {
220
221                        Type creatorType0 = Parameters.getType(Creator.class, creator, "G");
222                        Type decoderType0 = Parameters.getType(Decoder.class, decoder, "G");
223                        Type decoderType1 = Parameters.getType(Decoder.class, decoder, "P");
224                        Type evaluatorType0 = Parameters.getType(Evaluator.class, evaluators, "P");
225
226                        Class<?> creatorGenotype = Parameters.getClass(creatorType0);
227                        Class<?> decoderGenotype = Parameters.getClass(decoderType0);
228                        Class<?> decoderPhenotype = Parameters.getClass(decoderType1);
229                        Class<?> evaluatorPhenotype = Parameters.getClass(evaluatorType0);
230
231                        if (creatorGenotype == null) {
232                                creatorGenotype = Genotype.class;
233                        }
234                        if (decoderGenotype == null) {
235                                decoderGenotype = Genotype.class;
236                        }
237                        if (decoderPhenotype == null) {
238                                decoderPhenotype = Object.class;
239                        }
240                        if (evaluatorPhenotype == null) {
241                                evaluatorPhenotype = Object.class;
242                        }
243
244                        boolean a = decoderGenotype.isAssignableFrom(creatorGenotype);
245                        boolean b = evaluatorPhenotype.isAssignableFrom(decoderPhenotype);
246
247                        if (!a) {
248                                throw new IllegalArgumentException("Creator(" + creator.getClass() + ") and Decoder("
249                                                + decoder.getClass() + ") are not compatible. Creator creates \"" + creatorGenotype
250                                                + "\", Decoder decodes \"" + decoderGenotype + "\".");
251                        }
252                        if (!b) {
253                                throw new IllegalArgumentException("Decoder(" + decoder.getClass() + ") and Evaluator(" + evaluators
254                                                + ") are not compatible. Decoder decodes \"" + decoderPhenotype + "\", Evaluator evaluates \""
255                                                + evaluatorPhenotype + "\".");
256                        }
257                } catch (NullPointerException e) {
258
259                }
260        }
261
262        /*
263         * (non-Javadoc)
264         * 
265         * @see
266         * org.opt4j.core.optimizer.ControlListener#stateChanged(org.opt4j.core.
267         * optimizer.Control.State)
268         */
269        @Override
270        public void stateChanged(org.opt4j.core.optimizer.Control.State state) {
271                stateChanged();
272        }
273
274        /*
275         * (non-Javadoc)
276         * 
277         * @see
278         * org.opt4j.core.optimizer.OptimizerIterationListener#iterationComplete
279         * (org.opt4j.core.optimizer.Optimizer, int)
280         */
281        @Override
282        public void iterationComplete(int iteration) {
283                stateChanged();
284        }
285
286        /**
287         * Calls the listeners that listen to the state of this task.
288         */
289        private void stateChanged() {
290                for (TaskStateListener listener : listeners) {
291                        listener.stateChanged(this);
292                }
293        }
