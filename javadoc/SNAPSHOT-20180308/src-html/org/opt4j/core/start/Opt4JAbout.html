080        /*
081         * (non-Javadoc)
082         * 
083         * @see org.opt4j.config.gui.Startupable#startup()
084         */
085        @Override
086        public void startup() {
087                Container content = this;
088
089                content.setLayout(new BorderLayout());
090
091                JLabel logoLabel = new JLabel(Icons.getIcon("img/top_logo.png"));
092                logoLabel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
093                JPanel logo = new JPanel(new BorderLayout());
094                logo.setBackground(Color.WHITE);
095                logo.add(logoLabel);
096
097                content.add(logo, BorderLayout.PAGE_START);
098
099                JTextPane license = new JTextPane();
100                license.setEditable(false);
101                final JScrollPane licenseScroll = new JScrollPane(license);
102                licenseScroll.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS);
103                StyledDocument doc = license.getStyledDocument();
104
105                Style regular = StyleContext.getDefaultStyleContext().getStyle(StyleContext.DEFAULT_STYLE);
106
107                try {
108                        doc.insertString(doc.getLength(), LICENSE_TEXT, regular);
109                } catch (BadLocationException e) {
110                        e.printStackTrace();
111                }
112                license.setPreferredSize(new Dimension(360, 100));
113                content.add(licenseScroll, BorderLayout.CENTER);
114
115                SwingUtilities.invokeLater(new Runnable() {
116                        @Override
117                        public void run() {
118                                licenseScroll.getVerticalScrollBar().setValue(0);
119                        }
120                });
121                JPanel footer = new JPanel(new BorderLayout());
122                footer.setBackground(Color.WHITE);
123
124                // Add Copyright & Credits
125                String copyright = "<html>Build " + Opt4J.getDateISO() + " <br /> Version " + Opt4J.getVersion()
126                                + "   \u00a9 Opt4J.org 2007</html>";
127                JLabel copyrightLabel = new JLabel(copyright);
128                copyrightLabel.setHorizontalAlignment(SwingConstants.CENTER);
129                copyrightLabel.setVerticalAlignment(SwingConstants.BOTTOM);
130                copyrightLabel.setBorder(BorderFactory.createEmptyBorder(3, 3, 3, 3));
131                footer.add(copyrightLabel, BorderLayout.EAST);
132
133                String credits = "<html><p>Credits:<br />";
134                for (String author : AUTHORS) {
135                        credits += author + "<br/>";
136                }
137                credits += "</p></html>";
138
139                JLabel creditsLabel = new JLabel(credits);
140                creditsLabel.setHorizontalAlignment(SwingConstants.CENTER);
141                creditsLabel.setBorder(BorderFactory.createEmptyBorder(3, 3, 3, 3));
142                footer.add(creditsLabel, BorderLayout.WEST);
143
144                content.add(footer, BorderLayout.PAGE_END);
145        }
146
147        /*
148         * (non-Javadoc)
149         * 
150         * @see
151         * org.opt4j.config.visualization.About#getDialog(org.opt4j.config.visualization
152         * .ApplicationFrame)
153         */
154        @Override
155        public JDialog getDialog(ApplicationFrame frame) {
156                JDialog dialog = new JDialog(frame, "About Opt4J", true);
157                dialog.setBackground(Color.WHITE);
158                dialog.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
159                dialog.setResizable(false);
160
161                Opt4JAbout content = new Opt4JAbout();
162                content.startup();
163                dialog.add(content);
164
165                Dimension screen = Toolkit.getDefaultToolkit().getScreenSize();
166                Dimension window = dialog.getPreferredSize();
167                dialog.setLocation((screen.width - window.width) / 2, (screen.height - window.height) / 2);
168
169                return dialog;
170        }
