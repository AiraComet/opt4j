057/**
058 * The {@link Opt4JModule} is the superclass for all modules. The class and its
059 * fields might be extended with certain annotations. See {@link Constant} and the
060 * {@code org.opt4j.core.config.annotations} package.
061 * 
062 * @see Constant
063 * @see Category
064 * @see File
065 * @see Icon
066 * @see Ignore
067 * @see Info
068 * @see Multi
069 * @see Name
070 * @see Order
071 * @see Panel
072 * @see Parent
073 * @see Required
074 * @author lukasiewycz
075 * 
076 * 
077 */
078public abstract class Opt4JModule extends AbstractModule {
079
080        /**
081         * The singleton scope.
082         */
083        public static Scope SINGLETON = Scopes.SINGLETON;
084
085        /**
086         * Bind a value.
087         * 
088         * @param annotation
089         *            the type of annotation of the value
090         * @return the constant binding builder that allows a binding
091         */
092        public ConstantBindingBuilder bindConstant(Class<? extends Annotation> annotation) {
093                return bindConstant().annotatedWith(annotation);
094        }
095
096        /**
097         * Bind a value.
098         * 
099         * @param annotation
100         *            the annotation of the value
101         * @return the constant binding builder that allows a binding
102         */
103        public ConstantBindingBuilder bindConstant(Annotation annotation) {
104                return bindConstant().annotatedWith(annotation);
105        }
106
107        /*
108         * (non-Javadoc)
109         * 
110         * @see com.google.inject.AbstractModule#configure()
111         */
112        @SuppressWarnings({ "unchecked", "rawtypes" })
113        @Override
114        protected void configure() {
115
116                /**
117                 * Configure injected constants.
118                 */
119                PropertyModule module = new PropertyModule(this);
120                for (Property property : module.getProperties()) {
121                        for (Annotation annotation : property.getAnnotations()) {
122                                if (annotation.annotationType().getAnnotation(BindingAnnotation.class) != null) {
123                                        Class<?> type = property.getType();
124                                        Object value = property.getValue();
125
126                                        ConstantBindingBuilder builder = bindConstant(annotation);
127                                        if (type.equals(Integer.TYPE)) {
128                                                builder.to((Integer) value);
129                                        } else if (type.equals(Long.TYPE)) {
130                                                builder.to((Long) value);
131                                        } else if (type.equals(Double.TYPE)) {
132                                                builder.to((Double) value);
133                                        } else if (type.equals(Float.TYPE)) {
134                                                builder.to((Float) value);
135                                        } else if (type.equals(Byte.TYPE)) {
136                                                builder.to((Byte) value);
137                                        } else if (type.equals(Short.TYPE)) {
138                                                builder.to((Short) value);
139                                        } else if (type.equals(Boolean.TYPE)) {
140                                                builder.to((Boolean) value);
141                                        } else if (type.equals(Character.TYPE)) {
142                                                builder.to((Character) value);
143                                        } else if (type.equals(String.class)) {
144                                                builder.to((String) value);
145                                        } else if (type.equals(Class.class)) {
146                                                builder.to((Class<?>) value);
147                                        } else if (value instanceof Enum<?>) {
148                                                builder.to((Enum) value);
149                                        } else {
150                                                String message = "Constant type not bindable: " + type + " of field " + property.getName() + " in module "
151                                                                + this.getClass().getName();
152                                                throw new ConfigurationException(Arrays.asList(new Message(message)));
153                                        }
154                                }
155                        }
156                }
157
158                multi(OptimizerStateListener.class);
159                multi(OptimizerIterationListener.class);
160                multi(IndividualStateListener.class);
161
162                config();
163        }
164
165        protected void multi(Class<?> clazz) {
166                Multibinder.newSetBinder(binder(), clazz);
167        }
168
169        /**
170         * Configure the module.
171         * 
172         * Bind constants, listeners, and bind arbitrary classes.
173         * 
174         * @see Binder
175         */
176        protected abstract void config();
177
178        /**
179         * Returns the implementation of the {@link Constant} annotation.
180         * 
181         * @param value
182         *            the value
183         * @param namespace
184         *            the namespace
185         * @return the constant annotation
186         */
187        public static ConstantImpl constant(String value, Class<?> namespace) {
188                return new ConstantImpl(value, namespace);
189        }
190
191        /**
192         * The {@link ConstantImpl} is the implementation of the {@link Constant}
193         * annotation.
194         * 
195         * @author lukasiewycz
196         * 
197         */
198        @SuppressWarnings("all")
199        static class ConstantImpl implements Serializable, Constant {
200
201                private final String value;
202                private final Class<?> namespace;
203
204                /**
205                 * Constructs a {@link ConstantImpl}.
206                 * 
207                 * @param value
208                 *            the value
209                 * @param namespace
210                 *            the namespace
211                 */
212                ConstantImpl(String value, Class<?> namespace) {
213                        this.value = value;
214                        this.namespace = namespace;
215                }
216
217                /*
218                 * (non-Javadoc)
219                 * 
220                 * @see org.opt4j.start.Constant#value()
221                 */
222                @Override
223                public String value() {
224                        return value;
225                }
226
227                /*
228                 * (non-Javadoc)
229                 * 
230                 * @see org.opt4j.start.Constant#namespace()
231                 */
232                @Override
233                public Class<?> namespace() {
234                        return namespace;
235                }
236
237                /*
238                 * (non-Javadoc)
239                 * 
240                 * @see java.lang.Object#hashCode()
241                 */
242                @Override
243                public int hashCode() {
244                        return ((127 * "value".hashCode()) ^ value.hashCode()) + ((127 * "namespace".hashCode()) ^ namespace.hashCode());
245                }
246
247                /*
248                 * (non-Javadoc)
249                 * 
250                 * @see java.lang.Object#equals(java.lang.Object)
251                 */
252                @Override
253                public boolean equals(Object o) {
254                        if (o == null || !(o instanceof Constant)) {
255                                return false;
256                        }
257
258                        Constant other = (Constant) o;
259                        return (value.equals(other.value()) && namespace.equals(other.namespace()));
260                }
261
262                /*
263                 * (non-Javadoc)
264                 * 
265                 * @see java.lang.annotation.Annotation#annotationType()
266                 */
267                @Override
268                public Class<? extends Annotation> annotationType() {
269                        return Constant.class;
270                }
271
272                /*
273                 * (non-Javadoc)
274                 * 
275                 * @see java.lang.Object#toString()
276                 */
277                @Override
278                public String toString() {
279                        return "@" + Constant.class.getName() + "(value=" + value + ", namespace=" + namespace.getName() + ")";
280                }
281
282                private static final long serialVersionUID = 0;
283        }
284
285        /**
286         * Bind a {@link Constant}.
287         * 
288         * @param value
289         *            the value from the {@link Constant} annotation with an empty
290         *            namespace
291         * @return the constant binding builder that allows a binding
292         */
293        protected ConstantBindingBuilder bindConstant(String value) {
294                return bindConstant(value, Object.class);
295        }
296
297        /**
298         * Bind a constant.
299         * 
300         * @param value
301         *            the value from the {@link Constant}
302         * @param namespace
303         *            the namespace from the {@link Constant}
304         * @return the constant binding builder that allows a binding
305         */
306        public ConstantBindingBuilder bindConstant(String value, Class<?> namespace) {
307                assert (namespace != null);
308                Constant c = new ConstantImpl(value, namespace);
309                return bindConstant(c);
310        }
