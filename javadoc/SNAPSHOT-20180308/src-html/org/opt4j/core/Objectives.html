024package org.opt4j.core;
025
026import static org.opt4j.core.Objective.Sign.MAX;
027
028import java.util.Collection;
029import java.util.Iterator;
030import java.util.Map.Entry;
031import java.util.SortedMap;
032import java.util.TreeMap;
033
034import org.opt4j.core.Objective.Sign;
035
036/**
037 * The {@link Objectives} contains the {@link Objective}-{@link Value}s pairs of
038 * an {@link Individual}.
039 * 
040 * @see Value
041 * @see Objective
042 * @author lukasiewycz
043 * 
044 */
045public class Objectives implements Iterable<Entry<Objective, Value<?>>> {
046
047        protected final SortedMap<Objective, Value<?>> map = new TreeMap<Objective, Value<?>>();
048
049        protected double[] array = null;
050
051        /*
052         * (non-Javadoc)
053         * 
054         * @see java.lang.Iterable#iterator()
055         */
056        @Override
057        public Iterator<Entry<Objective, Value<?>>> iterator() {
058                return map.entrySet().iterator();
059        }
060
061        /**
062         * Returns an array of all values which all have to be minimized. Do not
063         * call this method before all objectives were added!
064         * 
065         * @see Value#getDouble()
066         * @return an array containing values which have to be minimized
067         */
068        public double[] array() {
069                if (array == null) {
070                        submit();
071                }
072
073                return array;
074        }
075
076        /**
077         * Calculates the array.
078         */
079        protected synchronized void submit() {
080                if (array == null) {
081                        array = new double[size()];
082                        int i = 0;
083                        for (Entry<Objective, Value<?>> entry : this) {
084                                Objective objective = entry.getKey();
085                                Value<?> value = entry.getValue();
086
087                                Double v = value.getDouble();
088
089                                if (v == null) {
090                                        array[i] = Double.MAX_VALUE;
091                                } else if (objective.getSign() == MAX) {
092                                        array[i] = -v;
093                                } else {
094                                        array[i] = v;
095                                }
096
097                                i++;
098                        }
099                }
100        }
101
102        /**
103         * Returns the number of set {@link Objective}s.
104         * 
105         * @return the number of set objectives
106         */
107        public int size() {
108                return map.size();
109        }
110
111        /**
112         * Returns all objectives.
113         * 
114         * @return all objectives
115         */
116        public Collection<Objective> getKeys() {
117                return map.keySet();
118        }
119
120        /**
121         * Returns all values.
122         * 
123         * @return all values
124         */
125        public Collection<Value<?>> getValues() {
126                return map.values();
127        }
128
129        /**
130         * Returns the value that is assigned to the given objective. Returns
131         * {@code null} if the objective does not exist.
132         * 
133         * @param objective
134         *            the given objective
135         * @return the value
136         */
137        public Value<?> get(Objective objective) {
138                return map.get(objective);
139        }
140
141        /**
142         * Returns the objective that is assigned to the given value. Returns
143         * {@code null} if the value does not exist.
144         * 
145         * @param value
146         *            the given value
147         * @return the objective
148         */
149        public Objective get(Value<?> value) {
150                Objective objective = null;
151                for (Entry<Objective, Value<?>> entry : this) {
152                        Value<?> v = entry.getValue();
153                        if (value.equals(v)) {
154                                Objective o = entry.getKey();
155                                objective = o;
156                                break;
157                        }
158                }
159                return objective;
160        }
161
162        /**
163         * Adds the objective with the specified value.
164         * 
165         * @param objective
166         *            the objective
167         * @param value
168         *            the value
169         */
170        public void add(Objective objective, Value<?> value) {
171                if (value == null) {
172                        map.put(objective, new DoubleValue(null));
173                } else {
174                        map.put(objective, value);
175                }
176                array = null;
177        }
178
179        /**
180         * Adds the objective with the specified value.
181         * 
182         * @param name
183         *            the name of the objective
184         * @param sign
185         *            the optimization direction
186         * @param value
187         *            the value
188         */
189        public void add(String name, Sign sign, Value<?> value) {
190                add(new Objective(name, sign), value);
191        }
192
193        /**
194         * Adds the objective with the specified double value.
195         * 
196         * @param objective
197         *            the objective
198         * @param value
199         *            the value
200         */
201        public void add(Objective objective, double value) {
202                add(objective, new DoubleValue(value));
203        }
204        
205        /**
206         * Adds the objective with the specified value.
207         * 
208         * @param name
209         *            the name of the objective
210         * @param sign
211         *            the optimization direction
212         * @param value
213         *            the value
214         */
215        public void add(String name, Sign sign, double value) {
216                add(new Objective(name, sign), value);
217        }
218
219        /**
220         * Adds the objective with the specified integer value.
221         * 
222         * @param objective
223         *            the objective
224         * @param value
225         *            the value
226         */
227        public void add(Objective objective, int value) {
228                add(objective, new IntegerValue(value));
229        }
230        
231        /**
232         * Adds the objective with the specified value.
233         * 
234         * @param name
235         *            the name of the objective
236         * @param sign
237         *            the optimization direction
238         * @param value
239         *            the value
240         */
241        public void add(String name, Sign sign, int value) {
242                add(new Objective(name, sign), value);
243        }
244
245        /**
246         * Adds all objective with the specified value specified in
247         * {@link Objectives}.
248         * 
249         * @param objectives
250         *            the objectives
251         */
252        public void addAll(Objectives objectives) {
253                map.putAll(objectives.map);
254                array = null;
255        }
256
257        /**
258         * Returns {@code true} if this objectives weakly dominates the specified
259         * objectives. This comparison is based on the {@link #array()} values.
260         * 
261         * @param opponent
262         *            other objectives
263         * @return {@code true} if this objectives weakly dominate the
264         *         {@code opponent}
265         */
266        public boolean weaklyDominates(Objectives opponent) {
267                double[] va = this.array();
268                double[] vb = opponent.array();
269                for (int i = 0; i < va.length; i++) {
270                        if (vb[i] < va[i]) {
271                                return false;
272                        }
273                }
274                return true;
275        }
276
277        /**
278         * Returns {@code true} if this objectives dominate the specified
279         * objectives. This comparison is based on the {@link #array()} values.
280         * 
281         * @param opponent
282         *            other objectives
283         * @return {@code true} if these objectives dominate the {@code opponent}
284         */
285        public boolean dominates(Objectives opponent) {
286                double[] va = this.array();
287                double[] vb = opponent.array();
288
289                boolean equal = true;
290                for (int i = 0; i < va.length; i++) {
291                        if (va[i] > vb[i]) {
292                                return false;
293                        } else if (va[i] < vb[i]) {
294                                equal = false;
295                        }
296                }
297                return !equal;
298        }
299
300        /**
301         * Returns {@code true} if this objectives are equal to the specified
302         * objectives. This comparison is based on the {@link #array()} values.
303         * 
304         * @param opponent
305         *            other objectives
306         * @return {@code true} if these objectives dominate the {@code opponent}
307         */
308        public boolean isEqual(Objectives opponent) {
309                double[] va = this.array();
310                double[] vb = opponent.array();
311
312                for (int i = 0; i < va.length; i++) {
313                        if (va[i] != vb[i]) {
314                                return false;
315                        }
316                }
317                return true;
318        }
319
320        /**
321         * Calculates the euclidean distance of two {@link Objectives}. This
322         * calculation is based on the {@link #array()} values.
323         * 
324         * @param other
325         *            the second objectives
326         * @return the euclidean distance
327         */
328        public double distance(Objectives other) {
329                double[] va = this.array();
330                double[] vb = other.array();
331
332                double s = 0;
333                for (int i = 0; i < va.length; i++) {
334                        s += (va[i] - vb[i]) * (va[i] - vb[i]);
335                }
336
337                return Math.sqrt(s);
338        }
339
340        /*
341         * (non-Javadoc)
342         * 
343         * @see java.lang.Object#toString()
344         */
345        @Override
346        public String toString() {
347                String s = "";
348                for (Entry<Objective, Value<?>> entry : map.entrySet()) {
349                        Objective o = entry.getKey();
350                        Value<?> v = entry.getValue();
351                        s += o + "=" + v + " ";
352                }
353                return s;
354        }
