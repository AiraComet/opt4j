024package org.opt4j.core.common.archive;
025
026import java.util.ArrayList;
027import java.util.Collection;
028import java.util.Iterator;
029import java.util.List;
030import java.util.Set;
031
032import org.opt4j.core.Individual;
033import org.opt4j.core.Objectives;
034import org.opt4j.core.optimizer.Archive;
035
036/**
037 * This {@link AbstractArchive} provides some common methods for {@link Archive}
038 * s. If one or more new {@link Individual}s are added to this {@link Archive},
039 * it is ensured that all {@link Individual}s in this {@link Archive} are not
040 * Pareto-dominated. Actual implementations of this class may still refuse or
041 * drop some {@link Individual}s. An {@link Archive} can be a
042 * {@link BoundedArchive} if it has a bounded size or an
043 * {@link UnboundedArchive}, otherwise.
044 * 
045 * @author helwig, glass, lukasiewycz
046 * 
047 */
048public abstract class AbstractArchive extends Archive {
049
050        /*
051         * (non-Javadoc)
052         * 
053         * @see org.opt4j.core.optimizer.Archive#update(java.util.Set)
054         */
055        @Override
056        public boolean update(Set<? extends Individual> individuals) {
057                List<Individual> candidates = new ArrayList<Individual>(individuals);
058                Objectives o1, o2;
059
060                /*
061                 * Remove all individuals that are already in the archive and those that
062                 * are dominated among the candidates
063                 */
064                candidates.removeAll(this);
065                for (int i = 0; i < candidates.size() - 1; i++) {
066                        o1 = candidates.get(i).getObjectives();
067                        for (int j = i + 1; j < candidates.size(); j++) {
068                                o2 = candidates.get(j).getObjectives();
069                                if (o2.weaklyDominates(o1)) {
070                                        candidates.remove(i);
071                                        i--;
072                                        break;
073                                } else if (o1.weaklyDominates(o2)) {
074                                        candidates.remove(j);
075                                        j--;
076                                }
077                        }
078                }
079
080                /*
081                 * Remove those individuals from the archive which are dominated by the
082                 * candidates. Remove those individuals from the candidates which are
083                 * dominated by the archive.
084                 */
085                Iterator<Individual> i1, i2;
086                for (i1 = candidates.iterator(); i1.hasNext();) {
087                        o1 = i1.next().getObjectives();
088                        for (i2 = this.iterator(); i2.hasNext();) {
089                                o2 = i2.next().getObjectives();
090                                if (o1.weaklyDominates(o2)) {
091                                        i2.remove();
092                                } else if (o2.dominates(o1)) {
093                                        i1.remove();
094                                        break;
095                                }
096                        }
097                }
098
099                return updateWithNondominated(candidates);
100        }
101
102        /**
103         * Adds new {@code candidates} which are already checked to be not
104         * Pareto-dominated by any other individual in this {@link Archive}. All
105         * {@link Individual}s in the {@link Archive} which were dominated by the
106         * candidates have already been removed.
107         * 
108         * @param candidates
109         *            the non-dominated individuals which can be added
110         * @return true if one or more candidates are added to the archive
111         */
112        protected abstract boolean updateWithNondominated(Collection<Individual> candidates);
113
114}
