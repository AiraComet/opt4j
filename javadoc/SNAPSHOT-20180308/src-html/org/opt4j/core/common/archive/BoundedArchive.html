024package org.opt4j.core.common.archive;
025
026import java.util.Collection;
027import java.util.LinkedHashSet;
028import java.util.Set;
029
030import org.opt4j.core.Individual;
031
032import com.google.inject.Inject;
033
034/**
035 * An {@link org.opt4j.core.optimizer.Archive} with bounded size.
036 * 
037 * @author helwig, lukasiewycz
038 */
039public abstract class BoundedArchive extends AbstractArchive {
040
041        /** The capacity of this {@link org.opt4j.core.optimizer.Archive} */
042        protected int capacity;
043
044        /**
045         * Constructs a bounded archive with the specified capacity.
046         * 
047         * @param capacity
048         *            Capacity of this archive
049         */
050        @Inject
051        public BoundedArchive(int capacity) {
052                super();
053                this.capacity = capacity;
054
055                if (capacity < 0) {
056                        throw new IllegalArgumentException("Invalid capacity: " + capacity);
057                }
058        }
059
060        /**
061         * Sets the capacity of this {@link BoundedArchive}.
062         * 
063         * @see #getCapacity
064         * @param capacity
065         *            new capacity of this bounded archive
066         */
067        public void setCapacity(int capacity) {
068                if (capacity < 0) {
069                        throw new IllegalArgumentException("Invalid capacity: " + capacity);
070                }
071                this.capacity = capacity;
072        }
073
074        /**
075         * Returns the capacity of this {@link BoundedArchive}.
076         * 
077         * @see #setCapacity
078         * @return capacity of this bounded archive
079         */
080        public int getCapacity() {
081                return capacity;
082        }
083
084        /**
085         * @throws IndexOutOfBoundsException
086         *             if the capacity is reached.
087         */
088        /*
089         * (non-Javadoc)
090         * 
091         * @see org.opt4j.core.IndividualCollection#add(org.opt4j.core.Individual)
092         */
093        @Override
094        public boolean addCheckedIndividual(Individual individual) {
095                if (!contains(individual) && individuals.size() >= capacity) {
096                        throw new IndexOutOfBoundsException();
097                }
098                return super.addCheckedIndividual(individual);
099        }
100
101        /**
102         * @throws IndexOutOfBoundsException
103         *             if the capacity is reached.
104         */
105        /*
106         * (non-Javadoc)
107         * 
108         * @see org.opt4j.core.IndividualCollection#addAll(java.util.Collection)
109         */
110        @Override
111        public boolean addCheckedIndividuals(Collection<? extends Individual> c) {
112                Set<Individual> set = new LinkedHashSet<Individual>(c);
113                set.removeAll(this);
114                if (individuals.size() + set.size() > capacity) {
115                        throw new IndexOutOfBoundsException();
116                }
117
118                return super.addCheckedIndividuals(c);
119        }
120}
