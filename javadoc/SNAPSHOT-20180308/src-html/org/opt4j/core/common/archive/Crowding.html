023package org.opt4j.core.common.archive;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.Comparator;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032
033import org.opt4j.core.Individual;
034
035/**
036 * The {@link Crowding}, see "A fast and elitist multiobjective genetic
037 * algorithm : NSGA-II, K. Deb, A. Pratap, S. Agarwal, and T. Meyarivan
038 * Evolutionary Computation, IEEETransactions on, vol. 6, no. 2, pp. 182-197,
039 * August 2002.", calculates the crowding distance of {@link Individual}s as
040 * used for instance in the the {@code Nsga2} algorithm. A bounding box around
041 * each individual is defined and restricted by the neighboring individuals in
042 * each dimension. The crowding distance is the sum of the side-lengths of the
043 * bounding box. If an individual does not have neighbors on all sides, the
044 * crowding distance is maximal. A normalization in each dimension is applied.
045 * 
046 * @author lukasiewycz
047 * 
048 */
049public class Crowding implements FrontDensityIndicator {
050
051        /**
052         * Calculates the crowding distance for a collection of {@link Individual}s.
053         * 
054         * @param individuals
055         *            the individuals
056         * @return a map of the individuals to the corresponding crowding distance
057         */
058        private Map<Individual, Double> getValues(Collection<Individual> individuals) {
059                Map<Individual, Double> map = new HashMap<Individual, Double>();
060                List<Individual> list = new ArrayList<Individual>(individuals);
061
062                for (Individual individual : list) {
063                        map.put(individual, 0.0);
064                }
065
066                int m = list.get(0).getObjectives().array().length;
067
068                for (int i = 0; i < m; i++) {
069                        final int dim = i;
070                        Collections.sort(list, new Comparator<Individual>() {
071                                @Override
072                                public int compare(Individual o1, Individual o2) {
073                                        double i1 = o1.getObjectives().array()[dim];
074                                        double i2 = o2.getObjectives().array()[dim];
075                                        return (int) Math.signum(i1 - i2);
076                                }
077                        });
078
079                        Individual minIndividual = list.get(0);
080                        Individual maxIndividual = list.get(list.size() - 1);
081
082                        double diff = maxIndividual.getObjectives().array()[dim] - minIndividual.getObjectives().array()[dim];
083                        if (diff > 0) {
084                                map.put(minIndividual, Double.MAX_VALUE);
085                                map.put(maxIndividual, Double.MAX_VALUE);
086
087                                for (int j = 1; j < list.size() - 1; j++) {
088                                        double p = list.get(j - 1).getObjectives().array()[dim];
089                                        double n = list.get(j + 1).getObjectives().array()[dim];
090                                        Individual ind = list.get(j);
091                                        map.put(ind, map.get(ind) + ((n - p) / diff));
092                                }
093                        }
094                }
095
096                return map;
097        }
098
099        /**
100         * Returns an ordered list of the {@link Individual}s corresponding to their
101         * crowding distance. {@link Individual}s with a high crowding distance are
102         * sorted to the front.
103         * 
104         * @param values
105         *            the map of individuals to their crowding distance values
106         * @return the sorted list of individuals based on their crowding distance
107         */
108        public List<Individual> order(final Map<Individual, Double> values) {
109                List<Individual> list = new ArrayList<Individual>(values.keySet());
110                Collections.sort(list, new Comparator<Individual>() {
111                        @Override
112                        public int compare(Individual o1, Individual o2) {
113                                return values.get(o2).compareTo(values.get(o1));
114                        }
115                });
116                return list;
117        }
118
119        /*
120         * (non-Javadoc)
121         * 
122         * @see
123         * org.opt4j.optimizer.ea.FrontDensityIndicator#getDensityValues(java.util
124         * .Collection)
125         */
126        @Override
127        public Map<Individual, Double> getDensityValues(Collection<Individual> individuals) {
128                return getValues(individuals);
129        }
130
131}
