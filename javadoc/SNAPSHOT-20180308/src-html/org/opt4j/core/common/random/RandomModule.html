024package org.opt4j.core.common.random;
025
026import java.util.Random;
027
028import org.opt4j.core.config.Icons;
029import org.opt4j.core.config.annotations.Icon;
030import org.opt4j.core.config.annotations.Info;
031import org.opt4j.core.config.annotations.Required;
032import org.opt4j.core.start.Opt4JModule;
033
034/**
035 * The {@link RandomModule} is used to configure the used random number
036 * generator.
037 * 
038 * @author helwig, lukasiewycz
039 */
040@Icon(Icons.PUZZLE_BLUE)
041@Info("Global random number generator for the optimization process.")
042public class RandomModule extends Opt4JModule {
043
044        @Info("Seed of the random number generator.")
045        @Required(property = "usingSeed", elements = { "true" })
046        protected long seed = 0;
047
048        @Info("Use a specific seed.")
049        protected boolean usingSeed = true;
050
051        @Info("Type of the random number generator.")
052        protected RandType type = RandType.MERSENNE_TWISTER;
053
054        /**
055         * The {@link RandType}.
056         * 
057         * @author lukasiewycz
058         * 
059         */
060        public enum RandType {
061                /**
062                 * Use the standard Java random number generator.
063                 * 
064                 * @see RandomJava
065                 */
066                @Info("Use the standard java Random")
067                JAVA,
068
069                /**
070                 * Use the mersenne twister random number generator.
071                 * 
072                 * @see RandomMersenneTwister
073                 */
074                @Info("Use the mersenne twister random number generator")
075                MERSENNE_TWISTER;
076        }
077
078        /*
079         * (non-Javadoc)
080         * 
081         * @see org.opt4j.start.Opt4JModule#config()
082         */
083        @Override
084        public void config() {
085                Class<? extends Rand> randomClass;
086                switch (type) {
087                case JAVA:
088                        randomClass = RandomJava.class;
089                        break;
090                default: // MERSENNE TWISTER
091                        randomClass = RandomMersenneTwister.class;
092                        break;
093                }
094
095                long seed = this.seed;
096                if (!usingSeed) {
097                        seed = System.currentTimeMillis();
098                }
099                bindConstant("seed", Random.class).to(seed);
100
101                bind(Rand.class).to(randomClass).in(SINGLETON);
102        }
103
104        /**
105         * Sets the seed that is used by the random number generator.
106         * 
107         * @see #getSeed
108         * @param seed
109         *            the seed that is used by the random number generator
110         */
111        public void setSeed(long seed) {
112                this.seed = seed;
113        }
114
115        /**
116         * Returns the seed the is used by the random number generator.
117         * 
118         * @see #setSeed
119         * @return the seed the is used by the random number generator
120         */
121        public long getSeed() {
122                return seed;
123        }
124
125        /**
126         * Sets the type of the random number generator to the specified value.
127         * 
128         * @see #getType
129         * @param type
130         *            the type of the random number generator
131         */
132        public void setType(RandType type) {
133                this.type = type;
134        }
135
136        /**
137         * Returns the type of the random number generator.
138         * 
139         * @see #setType
140         * @return the type of the random number generator
141         */
142        public RandType getType() {
143                return type;
144        }
145
146        /**
147         * Returns {@code true} if a specific seed is given.
148         * 
149         * @see #setUsingSeed
150         * @return the useSeed
151         */
152        public boolean isUsingSeed() {
153                return usingSeed;
154        }
155
156        /**
157         * Select if a specific seed should be used.
158         * 
159         * @see #isUsingSeed
160         * @param value
161         *            {@code true} if a seed shall be used, {@code false} otherwise
162         */
163        public void setUsingSeed(boolean value) {
164                this.usingSeed = value;
165        }
166
167}
