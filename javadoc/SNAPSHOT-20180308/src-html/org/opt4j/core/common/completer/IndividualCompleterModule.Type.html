024package org.opt4j.core.common.completer;
025
026import org.opt4j.core.config.Icons;
027import org.opt4j.core.config.annotations.Icon;
028import org.opt4j.core.config.annotations.Info;
029import org.opt4j.core.config.annotations.Required;
030import org.opt4j.core.optimizer.IndividualCompleter;
031import org.opt4j.core.start.Constant;
032import org.opt4j.core.start.Opt4JModule;
033
034/**
035 * The {@link IndividualCompleterModule} is used to choose and configure a
036 * {@link IndividualCompleter}.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041@Icon(Icons.PUZZLE_BLUE)
042@Info("The IndividualCompleter decodes and evaluates the individuals in the optimization process.")
043public class IndividualCompleterModule extends Opt4JModule {
044
045        @Info("Sets the type of the individual completer.")
046        protected Type type = Type.SEQUENTIAL;
047
048        @Info("Sets the number of parallel processes.")
049        @Required(property = "type", elements = { "PARALLEL" })
050        @Constant(value = "maxThreads", namespace = ParallelIndividualCompleter.class)
051        protected int threads = 4;
052
053        /**
054         * The {@link Type} of {@link IndividualCompleter} to use.
055         * 
056         * @author glass
057         * 
058         */
059        public enum Type {
060                /**
061                 * Use a sequential completer.
062                 * 
063                 * @see SequentialIndividualCompleter
064                 */
065                @Info("Use a SequentialCompleter")
066                SEQUENTIAL,
067
068                /**
069                 * Use a parallel completer.
070                 * 
071                 * @see ParallelIndividualCompleter
072                 */
073                @Info("Use a ParallelCompleter")
074                PARALLEL;
075        }
076
077        /**
078         * Returns the type of the completer.
079         * 
080         * @see #setType
081         * @return type of the completer
082         */
083        public Type getType() {
084                return type;
085        }
086
087        /**
088         * Sets the type of the completer.
089         * 
090         * @see #getType
091         * @param type
092         *            the type of the completer
093         */
094        public void setType(Type type) {
095                this.type = type;
096        }
097
098        /**
099         * Returns the maximal number of parallel threads.
100         * 
101         * @see #setThreads
102         * @return the maximal number of parallel threads
103         */
104        public int getThreads() {
105                return threads;
106        }
107
108        /**
109         * Sets the maximal number of parallel threads.
110         * 
111         * @see #getThreads
112         * @param threads
113         *            the maximal number of parallel threads
114         */
115        public void setThreads(int threads) {
116                if (threads <= 0) {
117                        throw new IllegalArgumentException("The number of threads must be positive: " + threads);
118                }
119                this.threads = threads;
120        }
121
122        /*
123         * (non-Javadoc)
124         * 
125         * @see org.opt4j.start.Opt4JModule#config()
126         */
127        @Override
128        public void config() {
129                switch (type) {
130                case SEQUENTIAL:
131                        bind(IndividualCompleter.class).to(SequentialIndividualCompleter.class).in(SINGLETON);
132                        break;
133                default: // PARALLEL
134                        bind(ParallelIndividualCompleter.class).in(SINGLETON);
135                        bind(IndividualCompleter.class).to(ParallelIndividualCompleter.class);
136                        addOptimizerStateListener(ParallelIndividualCompleter.class);
137                        break;
138                }
139        }
140}
