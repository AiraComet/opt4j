018package org.opt4j.core.common.completer;
019
020import java.util.Arrays;
021import java.util.Collection;
022import java.util.HashSet;
023import java.util.List;
024import java.util.Set;
025
026import org.opt4j.core.Genotype;
027import org.opt4j.core.Individual;
028import org.opt4j.core.Individual.State;
029import org.opt4j.core.Objective;
030import org.opt4j.core.Objectives;
031import org.opt4j.core.optimizer.Control;
032import org.opt4j.core.optimizer.IndividualCompleter;
033import org.opt4j.core.optimizer.TerminationException;
034import org.opt4j.core.problem.Decoder;
035import org.opt4j.core.problem.Evaluator;
036
037import com.google.inject.Inject;
038
039/**
040 * <p>
041 * The {@link SequentialIndividualCompleter} completes the {@link Individual}s sequentially.
042 * </p>
043 * <p>
044 * It updates the {@link State} of the {@link Individual} according to the state of the completion process. It uses
045 * {@link Control} between the different (possibly time consuming) completion steps to allow the user to control the
046 * completion process.
047 * </p>
048 * 
049 * @author lukasiewycz
050 * 
051 */
052public class SequentialIndividualCompleter implements IndividualCompleter {
053
054        protected final Decoder<Genotype, Object> decoder;
055        protected final Evaluator<Object> evaluator;
056        protected final Control control;
057
058        /**
059         * Constructs a {@link SequentialIndividualCompleter}.
060         * 
061         * @param control
062         *            the optimization control
063         * @param decoder
064         *            the decoder
065         * @param evaluator
066         *            the evaluator
067         */
068        @Inject
069        public SequentialIndividualCompleter(Control control, Decoder<Genotype, Object> decoder,
070                        Evaluator<Object> evaluator) {
071                super();
072                this.control = control;
073                this.decoder = decoder;
074                this.evaluator = evaluator;
075        }
076
077        /*
078         * (non-Javadoc)
079         * 
080         * @see org.opt4j.core.optimizer.Completer#complete(java.lang.Iterable)
081         */
082        @Override
083        public void complete(Iterable<? extends Individual> iterable) throws TerminationException {
084                for (Individual individual : iterable) {
085                        if (!individual.isEvaluated()) {
086                                control.checkpoint();
087                                decode(individual);
088                                control.checkpoint();
089                                evaluate(individual);
090                                control.checkpoint();
091                        }
092                }
093        }
094
095        /*
096         * (non-Javadoc)
097         * 
098         * @see org.opt4j.core.optimizer.Completer#complete(org.opt4j.core.Individual[])
099         */
100        @Override
101        public void complete(Individual... individuals) throws TerminationException {
102                List<Individual> list = Arrays.asList(individuals);
103                complete(list);
104        }
105
106        /**
107         * Evaluates the phenotype of the {@link Individual}. After this operation, the {@link Individual} is in
108         * {@link State} {@link State#EVALUATED} and {@link Individual#getObjectives()} returns the {@link Objectives}.
109         * 
110         * @param individual
111         */
112        protected void evaluate(Individual individual) {
113                State state = individual.getState();
114
115                if (state == State.PHENOTYPED) {
116                        individual.setState(State.EVALUATING);
117                        Object phenotype = individual.getPhenotype();
118
119                        Objectives objectives = evaluator.evaluate(phenotype);
120                        assert isSameLength(objectives.getKeys()) : "Objectives changed: " + objectives.getKeys();
121
122                        individual.setObjectives(objectives);
123                } else {
124                        throw new IllegalStateException("Cannot evaluate Individual, current state: " + state);
125                }
126        }
127
128        /**
129         * Decodes the {@link Genotype} of the {@link Individual}. After this operation, the {@link Individual} is in
130         * {@link State} {@link State#PHENOTYPED} and {@link Individual#getPhenotype()} returns the phenotype.
131         * 
132         * @param individual
133         */
134        protected void decode(Individual individual) {
135                State state = individual.getState();
136
137                if (state == State.GENOTYPED) {
138                        individual.setState(State.DECODING);
139                        Genotype genotype = individual.getGenotype();
140                        Object phenotype = decoder.decode(genotype);
141                        individual.setPhenotype(phenotype);
142                } else {
143                        throw new IllegalStateException("Cannot decode Individual, current state: " + state);
144                }
145        }
146
147        private Set<Objective> objectives = null;
148
149        /**
150         * Check if the given {@link Objectives} have the same length as prior evaluated ones.
151         * 
152         * @param objectives
153         *            the objectives to check
154         * @return true if the number of objectives is constant
155         */
156        private boolean isSameLength(Collection<Objective> objectives) {
157                Set<Objective> set = new HashSet<Objective>(objectives);
158                if (this.objectives == null) {
159                        this.objectives = set;
160                        return true;
161                }
162                return this.objectives.equals(set);
163        }
164}
