024package org.opt4j.core.common.completer;
025
026import java.util.ArrayList;
027import java.util.List;
028import java.util.concurrent.Callable;
029import java.util.concurrent.ExecutionException;
030import java.util.concurrent.ExecutorService;
031import java.util.concurrent.Executors;
032import java.util.concurrent.Future;
033
034import org.opt4j.core.Genotype;
035import org.opt4j.core.Individual;
036import org.opt4j.core.optimizer.Control;
037import org.opt4j.core.optimizer.Optimizer;
038import org.opt4j.core.optimizer.OptimizerStateListener;
039import org.opt4j.core.optimizer.TerminationException;
040import org.opt4j.core.problem.Decoder;
041import org.opt4j.core.problem.Evaluator;
042import org.opt4j.core.start.Constant;
043
044import com.google.inject.Inject;
045
046/**
047 * The {@link ParallelIndividualCompleter} completes {@link Individual}s with
048 * multiple threads.
049 * 
050 * @author lukasiewycz
051 * 
052 */
053public class ParallelIndividualCompleter extends SequentialIndividualCompleter implements OptimizerStateListener {
054        protected final ExecutorService executor;
055
056        /**
057         * The {@link Complete} class completes a single {@link Individual}.
058         * 
059         * @author lukasiewycz
060         * 
061         */
062        protected class Complete implements Callable<Void> {
063
064                protected final Individual individual;
065
066                protected final Control control;
067
068                /**
069                 * Constructs {@link Complete} with an {@link Individual}.
070                 * 
071                 * @param individual
072                 *            the individual to complete
073                 * @param control
074                 *            the control
075                 */
076                public Complete(final Individual individual, final Control control) {
077                        this.individual = individual;
078                        this.control = control;
079                }
080
081                /*
082                 * (non-Javadoc)
083                 * 
084                 * @see java.util.concurrent.Callable#call()
085                 */
086                @Override
087                public Void call() throws TerminationException {
088                        if (!individual.isEvaluated()) {
089                                control.checkpoint();
090                                ParallelIndividualCompleter.this.decode(individual);
091                                control.checkpoint();
092                                ParallelIndividualCompleter.this.evaluate(individual);
093                                control.checkpoint();
094                        }
095                        return null;
096                }
097        }
098
099        /**
100         * Constructs a {@link ParallelIndividualCompleter} with a specified maximal
101         * number of concurrent threads.
102         * 
103         * @param control
104         *            the control
105         * @param decoder
106         *            the decoder
107         * @param evaluator
108         *            the evaluator
109         * @param maxThreads
110         *            the maximal number of parallel threads (using namespace
111         *            {@link ParallelIndividualCompleter})
112         */
113        @Inject
114        public ParallelIndividualCompleter(Control control, Decoder<Genotype, Object> decoder, Evaluator<Object> evaluator,
115                        @Constant(value = "maxThreads", namespace = ParallelIndividualCompleter.class) int maxThreads) {
116                super(control, decoder, evaluator);
117
118                if (maxThreads < 1) {
119                        throw new IllegalArgumentException("Invalid number of threads: " + maxThreads);
120                }
121                this.executor = Executors.newFixedThreadPool(maxThreads);
122        }
123
124        /*
125         * (non-Javadoc)
126         * 
127         * @see
128         * org.opt4j.common.completer.SequentialCompleter#complete(java.lang.Iterable
129         * )
130         */
131        @Override
132        public void complete(Iterable<? extends Individual> iterable) throws TerminationException {
133
134                try {
135
136                        List<Future<Void>> returns = new ArrayList<Future<Void>>();
137
138                        for (Individual individual : iterable) {
139                                if (individual.getState() != Individual.State.EVALUATED) {
140                                        returns.add(executor.submit(new Complete(individual, control)));
141                                }
142                        }
143
144                        for (Future<Void> future : returns) {
145                                try {
146                                        future.get();
147                                } catch (InterruptedException e) {
148                                        e.printStackTrace();
149                                }
150                        }
151                } catch (ExecutionException ex) {
152                        executor.shutdownNow();
153                        if (ex.getCause() instanceof TerminationException) {
154                                throw (TerminationException) ex.getCause();
155                        }
156                        throw new RuntimeException(ex);
157                }
158        }
159
160        /*
161         * (non-Javadoc)
162         * 
163         * @see java.lang.Object#finalize()
164         */
165        @Override
166        protected void finalize() throws Throwable {
167                shutdownExecutorService();
168                super.finalize();
169        }
170
171        /**
172         * Shutdown the {@link ExecutorService}.
173         */
174        protected synchronized void shutdownExecutorService() {
175                if (!executor.isShutdown()) {
176                        executor.shutdown();
177                }
178        }
179
180        /*
181         * (non-Javadoc)
182         * 
183         * @see
184         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStarted(org
185         * .opt4j.core.optimizer.Optimizer)
186         */
187        @Override
188        public void optimizationStarted(Optimizer optimizer) {
189                // do nothing
190        }
191
192        /*
193         * (non-Javadoc)
194         * 
195         * @see
196         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStopped(org
197         * .opt4j.core.optimizer.Optimizer)
198         */
199        @Override
200        public void optimizationStopped(Optimizer optimizer) {
201                shutdownExecutorService();
202        }
203}
