018package org.opt4j.core;
019
020/**
021 * The {@link IntegerValue} is a {@link Value} with an {@link Integer} as value. The {@link Objectives} contains a
022 * convenience method that allows to add an {@link IntegerValue} without creating a new instance:
023 * {@link Objectives#add(Objective, int)}.
024 * 
025 * @see Objectives#add(Objective, int)
026 * @author lukasiewycz
027 * 
028 */
029public class IntegerValue implements Value<Integer> {
030
031        protected Integer value = null;
032
033        /**
034         * Constructs a {@link IntegerValue}.
035         * 
036         * @param value
037         *            the specified integer
038         */
039        public IntegerValue(Integer value) {
040                this.value = value;
041        }
042
043        /*
044         * (non-Javadoc)
045         * 
046         * @see org.opt4j.core.Value#getDouble()
047         */
048        @Override
049        public Double getDouble() {
050                if (value == null) {
051                        return null;
052                }
053                double v = value;
054                return v;
055        }
056
057        /*
058         * (non-Javadoc)
059         * 
060         * @see org.opt4j.core.Value#getValue()
061         */
062        @Override
063        public Integer getValue() {
064                return value;
065        }
066
067        /*
068         * (non-Javadoc)
069         * 
070         * @see org.opt4j.core.Value#setValue(java.lang.Object)
071         */
072        @Override
073        public void setValue(Integer value) {
074                this.value = value;
075        }
076
077        /*
078         * (non-Javadoc)
079         * 
080         * @see java.lang.Comparable#compareTo(java.lang.Object)
081         */
082        @Override
083        public int compareTo(Value<Integer> arg) {
084                if (arg == null) {
085                        return -1;
086                }
087                Double other = arg.getDouble();
088                if (value == null) {
089                        if (other == null) {
090                                return 0;
091                        }
092                        return 1;
093                } else if (other == null) {
094                        return -1;
095                }
096                return getDouble().compareTo(other);
097        }
098
099        /*
100         * (non-Javadoc)
101         * 
102         * @see java.lang.Object#hashCode()
103         */
104        @Override
105        public int hashCode() {
106                final int prime = 31;
107                int result = 1;
108                result = prime * result + ((value == null) ? 0 : value.hashCode());
109                return result;
110        }
111
112        /*
113         * (non-Javadoc)
114         * 
115         * @see java.lang.Object#equals(java.lang.Object)
116         */
117        @Override
118        public boolean equals(Object obj) {
119                if (this == obj) {
120                        return true;
121                }
122                if (obj == null) {
123                        return false;
124                }
125                if (!(obj instanceof IntegerValue)) {
126                        return false;
127                }
128                IntegerValue other = (IntegerValue) obj;
129                if (value == null) {
130                        if (other.value != null) {
131                                return false;
132                        }
133                } else if (!value.equals(other.value)) {
134                        return false;
135                }
136                return true;
137        }
138
139        /*
140         * (non-Javadoc)
141         * 
142         * @see java.lang.Object#toString()
143         */
144        @Override
145        public String toString() {
146                return value == null ? "null" : value.toString();
147        }
148
149}
