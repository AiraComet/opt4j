018package org.opt4j.core;
019
020import java.util.Arrays;
021import java.util.Collection;
022import java.util.Iterator;
023import java.util.LinkedHashSet;
024import java.util.Set;
025import java.util.concurrent.CopyOnWriteArraySet;
026
027/**
028 * The {@link IndividualSet} is a {@link Set} of {@link Individual}s. It allows to add and remove listeners, see
029 * {@link IndividualSetListener}.
030 * 
031 * @see org.opt4j.core.optimizer.Archive
032 * @see org.opt4j.core.optimizer.Population
033 * @author lukasiewycz
034 * 
035 */
036public class IndividualSet implements Set<Individual> {
037
038        protected final Set<Individual> individuals = new LinkedHashSet<Individual>();
039
040        protected final Set<IndividualSetListener> listeners = new CopyOnWriteArraySet<IndividualSetListener>();
041
042        /*
043         * (non-Javadoc)
044         * 
045         * @see java.util.Collection#add(java.lang.Object)
046         */
047        @Override
048        public boolean add(Individual individual) {
049                boolean b = false;
050                if (!individuals.contains(individual)) {
051                        b = individuals.add(individual);
052                }
053                if (b) {
054                        for (IndividualSetListener listener : listeners) {
055                                listener.individualAdded(this, individual);
056                        }
057                }
058                return b;
059        }
060
061        /*
062         * (non-Javadoc)
063         * 
064         * @see java.util.Collection#iterator()
065         */
066        @Override
067        public Iterator<Individual> iterator() {
068                final Iterator<Individual> iterator = individuals.iterator();
069
070                return new Iterator<Individual>() {
071
072                        Individual current = null;
073
074                        @Override
075                        public boolean hasNext() {
076                                return iterator.hasNext();
077                        }
078
079                        @Override
080                        public Individual next() {
081                                current = iterator.next();
082                                return current;
083                        }
084
085                        @Override
086                        public void remove() {
087                                iterator.remove();
088                                if (current != null) {
089                                        for (IndividualSetListener listener : listeners) {
090                                                listener.individualRemoved(IndividualSet.this, current);
091                                        }
092                                }
093                        }
094                };
095
096        }
097
098        /*
099         * (non-Javadoc)
100         * 
101         * @see java.util.Collection#size()
102         */
103        @Override
104        public int size() {
105                return individuals.size();
106        }
107
108        /**
109         * Adds a listener.
110         * 
111         * @see #removeListener
112         * @param listener
113         *            the added listener
114         */
115        public void addListener(IndividualSetListener listener) {
116                listeners.add(listener);
117        }
118
119        /**
120         * Removes a listener.
121         * 
122         * @see #addListener
123         * @param listener
124         *            the removed listener
125         */
126        public void removeListener(IndividualSetListener listener) {
127                listeners.remove(listener);
128        }
129
130        /*
131         * (non-Javadoc)
132         * 
133         * @see java.util.Collection#clear()
134         */
135        @Override
136        public void clear() {
137                Iterator<Individual> it = iterator();
138                while (it.hasNext()) {
139                        it.next();
140                        it.remove();
141                }
142        }
143
144        /*
145         * (non-Javadoc)
146         * 
147         * @see java.util.Collection#addAll(java.util.Collection)
148         */
149        @Override
150        public boolean addAll(Collection<? extends Individual> c) {
151                boolean res = false;
152                for (Individual individual : c) {
153                        res |= add(individual);
154                }
155                return res;
156        }
157
158        /**
159         * Add all {@link Individual}s.
160         * 
161         * @param c
162         *            the individuals to be added
163         * @return true if at least one individual was added
164         */
165        public boolean addAll(Individual... c) {
166                return addAll(Arrays.asList(c));
167        }
168
169        /*
170         * (non-Javadoc)
171         * 
172         * @see java.util.Collection#contains(java.lang.Object)
173         */
174        @Override
175        public boolean contains(Object o) {
176                return individuals.contains(o);
177        }
178
179        /*
180         * (non-Javadoc)
181         * 
182         * @see java.util.Collection#containsAll(java.util.Collection)
183         */
184        @Override
185        public boolean containsAll(Collection<?> c) {
186                return individuals.containsAll(c);
187        }
188
189        /*
190         * (non-Javadoc)
191         * 
192         * @see java.util.Collection#isEmpty()
193         */
194        @Override
195        public boolean isEmpty() {
196                return individuals.isEmpty();
197        }
198
199        /*
200         * (non-Javadoc)
201         * 
202         * @see java.util.Collection#remove(java.lang.Object)
203         */
204        @Override
205        public boolean remove(Object o) {
206                boolean value = individuals.remove(o);
207                if (value) {
208                        for (IndividualSetListener listener : listeners) {
209                                listener.individualRemoved(this, (Individual) o);
210                        }
211                }
212                return value;
213        }
214
215        /*
216         * (non-Javadoc)
217         * 
218         * @see java.util.Collection#removeAll(java.util.Collection)
219         */
220        @Override
221        public boolean removeAll(Collection<?> c) {
222                boolean res = false;
223                Iterator<Individual> it = iterator();
224                while (it.hasNext()) {
225                        if (c.contains(it.next())) {
226                                it.remove();
227                                res = true;
228                        }
229                }
230                return res;
231        }
232
233        /*
234         * (non-Javadoc)
235         * 
236         * @see java.util.Collection#retainAll(java.util.Collection)
237         */
238        @Override
239        public boolean retainAll(Collection<?> c) {
240                boolean res = false;
241                Iterator<Individual> it = iterator();
242                while (it.hasNext()) {
243                        if (!c.contains(it.next())) {
244                                it.remove();
245                                res = true;
246                        }
247                }
248                return res;
249        }
250
251        /*
252         * (non-Javadoc)
253         * 
254         * @see java.util.Collection#toArray()
255         */
256        @Override
257        public Object[] toArray() {
258                return individuals.toArray();
259        }
260
261        /*
262         * (non-Javadoc)
263         * 
264         * @see java.util.Collection#toArray(T[])
265         */
266        @Override
267        public <T> T[] toArray(T[] a) {
268                return individuals.toArray(a);
269        }
270
271}
