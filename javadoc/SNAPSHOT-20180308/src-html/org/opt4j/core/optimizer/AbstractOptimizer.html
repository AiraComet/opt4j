024package org.opt4j.core.optimizer;
025
026import java.util.Set;
027import java.util.concurrent.CopyOnWriteArraySet;
028
029import org.opt4j.core.IndividualFactory;
030
031import com.google.inject.Inject;
032
033/**
034 * The {@link AbstractOptimizer} is an abstract implementation of a
035 * population-based {@link Optimizer}. It provides the necessary coupling to the
036 * framework by informing all {@link OptimizerStateListener}s and
037 * {@link OptimizerIterationListener}s, calling the {@link IndividualCompleter}
038 * to decode and evaluate new {@link org.opt4j.core.Individual}s and adding them
039 * to the {@link Archive}. The {@link Control} is checked regularly to allow
040 * user interaction.
041 * 
042 * To implement an iteration-based {@link Optimizer}, the
043 * {@link IterativeOptimizer} interface is available.
044 * 
045 * @author glass, lukasiewycz
046 * 
047 */
048public abstract class AbstractOptimizer implements Optimizer {
049
050        protected final Iteration iteration;
051
052        protected final Population population;
053
054        protected final Archive archive;
055
056        protected final IndividualCompleter completer;
057
058        protected final Control control;
059
060        protected final Set<OptimizerStateListener> stateListeners = new CopyOnWriteArraySet<OptimizerStateListener>();
061
062        protected final Set<OptimizerIterationListener> iterationListeners = new CopyOnWriteArraySet<OptimizerIterationListener>();
063
064        protected boolean optimizing = false;
065
066        /**
067         * Constructs an {@link AbstractOptimizer} with a {@link Population}, an
068         * {@link Archive}, an {@link IndividualFactory}, and a
069         * {@link IndividualCompleter}.
070         * 
071         * @param population
072         *            the specified population
073         * @param archive
074         *            the specified archive
075         * @param completer
076         *            the specified completer
077         * @param control
078         *            the control
079         * @param iteration
080         *            the iteration counter
081         */
082        public AbstractOptimizer(Population population, Archive archive, IndividualCompleter completer, Control control,
083                        Iteration iteration) {
084                this.population = population;
085                this.archive = archive;
086                this.completer = completer;
087                this.control = control;
088                this.iteration = iteration;
089        }
090
091        @Inject
092        protected void injectListeners(Set<OptimizerStateListener> stateListeners,
093                        Set<OptimizerIterationListener> iterationListeners) {
094                this.stateListeners.addAll(stateListeners);
095                this.iterationListeners.addAll(iterationListeners);
096        }
097
098        /*
099         * (non-Javadoc)
100         * 
101         * @see org.opt4j.core.optimizer.Optimizer#getIteration()
102         */
103        @Override
104        public int getIteration() {
105                return iteration.value;
106        }
107
108        /*
109         * (non-Javadoc)
110         * 
111         * @see org.opt4j.core.optimizer.Optimizer#isRunning()
112         */
113        @Override
114        public boolean isRunning() {
115                return optimizing;
116        }
117
118        /**
119         * Call this method if a new iteration started.
120         * 
121         * @throws StopException
122         *             if the optimization is stopped
123         * @throws TerminationException
124         *             if the optimization is terminated
125         */
126        protected void nextIteration() throws TerminationException, StopException {
127                completer.complete(population);
128                archive.update(population);
129                iteration.next();
130                for (OptimizerIterationListener listener : iterationListeners) {
131                        listener.iterationComplete(iteration.value());
132                }
133                control.checkpointStop();
134        }
135
136        /*
137         * (non-Javadoc)
138         * 
139         * @see org.opt4j.core.optimizer.Optimizer#startOptimization()
140         */
141        @Override
142        public void startOptimization() {
143                optimizing = true;
144                for (OptimizerStateListener listener : stateListeners) {
145                        listener.optimizationStarted(this);
146                }
147                try {
148                        optimize();
149                } catch (StopException e) {
150                        System.out.println("Optimization stopped.");
151                } catch (TerminationException e) {
152                        System.err.println("Optimization terminated.");
153                } finally {
154                        stopOptimization();
155                }
156        }
157
158        /*
159         * (non-Javadoc)
160         * 
161         * @see org.opt4j.core.optimizer.Optimizer#stopOptimization()
162         */
163        @Override
164        public void stopOptimization() {
165                optimizing = false;
166                for (OptimizerStateListener listener : stateListeners) {
167                        listener.optimizationStopped(this);
168                }
169        }
170
171        /*
172         * (non-Javadoc)
173         * 
174         * @see
175         * org.opt4j.core.optimizer.Optimizer#addOptimizerIterationListener(org.
176         * opt4j.core.optimizer.OptimizerIterationListener)
177         */
178        @Override
179        public void addOptimizerIterationListener(OptimizerIterationListener listener) {
180                iterationListeners.add(listener);
181        }
182
183        /*
184         * (non-Javadoc)
185         * 
186         * @see
187         * org.opt4j.core.optimizer.Optimizer#addOptimizerStateListener(org.opt4j
188         * .core.optimizer.OptimizerStateListener)
189         */
190        @Override
191        public void addOptimizerStateListener(OptimizerStateListener listener) {
192                stateListeners.add(listener);
193        }
194
195        /*
196         * (non-Javadoc)
197         * 
198         * @see
199         * org.opt4j.core.optimizer.Optimizer#removeOptimizerIterationListener(org
200         * .opt4j.core.optimizer.OptimizerIterationListener)
201         */
202        @Override
203        public void removeOptimizerIterationListener(OptimizerIterationListener listener) {
204                iterationListeners.remove(listener);
205        }
206
207        /*
208         * (non-Javadoc)
209         * 
210         * @see
211         * org.opt4j.core.optimizer.Optimizer#removeOptimizerStateListener(org.opt4j
212         * .core.optimizer.OptimizerStateListener)
213         */
214        @Override
215        public void removeOptimizerStateListener(OptimizerStateListener listener) {
216                stateListeners.remove(listener);
217        }
218
219}
