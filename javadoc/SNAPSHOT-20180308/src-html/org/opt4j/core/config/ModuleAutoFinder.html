023package org.opt4j.core.config;
024
025import java.io.File;
026import java.io.IOException;
027import java.lang.reflect.Modifier;
028import java.util.ArrayList;
029import java.util.Collection;
030import java.util.Collections;
031import java.util.HashSet;
032import java.util.List;
033import java.util.Set;
034import java.util.StringTokenizer;
035import java.util.concurrent.CopyOnWriteArraySet;
036import java.util.zip.ZipEntry;
037import java.util.zip.ZipException;
038import java.util.zip.ZipFile;
039
040import org.opt4j.core.config.annotations.Ignore;
041import org.opt4j.core.start.Opt4JModule;
042
043import com.google.inject.Inject;
044import com.google.inject.Module;
045
046/**
047 * The {@link ModuleAutoFinder} searches the classpath for all
048 * {@link PropertyModule}s that are not annotated with {@link Ignore}.
049 * 
050 * @author lukasiewycz
051 * 
052 */
053public class ModuleAutoFinder implements ModuleList {
054
055        protected final Transformer<Class<? extends Module>, Boolean> accept;
056
057        protected final Transformer<Class<? extends Module>, Boolean> ignore;
058
059        protected final Set<ModuleAutoFinderListener> listeners = new CopyOnWriteArraySet<ModuleAutoFinderListener>();
060
061        protected ClassLoader classLoader;
062
063        /**
064         * The {@link AllTrue} is a transformer that always returns {@code true}.
065         * 
066         * @author lukasiewycz
067         * 
068         */
069        private static class AllTrue implements Transformer<Class<? extends Module>, Boolean> {
070                @Override
071                public Boolean transform(Class<? extends Module> arg) {
072                        return true;
073                }
074        }
075
076        /**
077         * The {@link AllFalse} is a transformer that always returns {@code false}.
078         * 
079         * @author lukasiewycz
080         * 
081         */
082        private static class AllFalse implements Transformer<Class<? extends Module>, Boolean> {
083                @Override
084                public Boolean transform(Class<? extends Module> arg) {
085                        return false;
086                }
087        }
088
089        /**
090         * Constructs a {@link ModuleAutoFinder}.
091         */
092        @Inject
093        public ModuleAutoFinder() {
094                this(null, null);
095        }
096
097        /**
098         * Constructs a {@link ModuleAutoFinder}.
099         * 
100         * @param accept
101         *            the accept transformer
102         * @param ignore
103         *            the ignore transformer
104         */
105        public ModuleAutoFinder(Transformer<Class<? extends Module>, Boolean> accept,
106                        Transformer<Class<? extends Module>, Boolean> ignore) {
107                super();
108                this.accept = (accept != null) ? accept : new AllTrue();
109                this.ignore = (ignore != null) ? ignore : new AllFalse();
110        }
111
112        /*
113         * (non-Javadoc)
114         * 
115         * @see org.opt4j.conf.ModuleFinder#getModules()
116         */
117        @Override
118        public Collection<Class<? extends Module>> getModules() {
119                return getAll();
120        }
121
122        /**
123         * Returns all not abstract classes that implement {@link PropertyModule}.
124         * 
125         * @return all property modules
126         */
127        protected Collection<Class<? extends Module>> getAll() {
128
129                Starter starter = new Starter();
130                Collection<File> files = starter.addPlugins();
131
132                classLoader = ClassLoader.getSystemClassLoader();
133
134                files.addAll(getFilesFromClasspath());
135
136                List<Class<?>> classes = new ArrayList<Class<?>>();
137
138                for (File file : files) {
139
140                        if (isJar(file)) {
141
142                                try {
143                                        classes.addAll(getAllClasses(new ZipFile(file)));
144                                } catch (ZipException e) {
145                                        e.printStackTrace();
146                                } catch (IOException e) {
147                                        e.printStackTrace();
148                                } catch (UnsupportedClassVersionError e) {
149                                        System.err.println(file + " not supported: bad version number");
150                                }
151                        } else {
152                                classes.addAll(getAllClasses(file));
153                        }
154                }
155
156                List<Class<? extends Module>> modules = new ArrayList<Class<? extends Module>>();
157
158                for (Class<?> clazz : classes) {
159                        if (Opt4JModule.class.isAssignableFrom(clazz) && !Modifier.isAbstract(clazz.getModifiers())) {
160                                Class<? extends Module> module = clazz.asSubclass(Module.class);
161                                Ignore i = module.getAnnotation(Ignore.class);
162
163                                if (i == null && !module.isAnonymousClass() && accept.transform(module) && !ignore.transform(module)) {
164                                        modules.add(module);
165                                        invokeOut("Add module: " + module.toString());
166                                }
167                        }
168                }
169
170                return modules;
171
172        }
173
174        /**
175         * Collects all {@link File}s specified in the classpath.
176         * 
177         * @return set of directories and jar archives given in the classpath
178         */
179        protected Set<File> getFilesFromClasspath() {
180                Set<File> files = new HashSet<File>();
181                String paths = System.getProperty("java.class.path");
182
183                // split classpathes like a.jar;b.jar
184                // pathSeparator is ";" on Windows and ":" on Linux and MAC
185                StringTokenizer st = new StringTokenizer(paths, File.pathSeparator + "\n");
186
187                while (st.hasMoreTokens()) {
188                        String path = st.nextToken();
189                        File f = new File(path);
190
191                        if (f.exists()) {
192                                try {
193                                        f = f.getCanonicalFile();
194                                        files.add(f);
195                                } catch (IOException e) {
196                                        e.printStackTrace();
197                                }
198                        }
199                }
200                return files;
201        }
202
203        /**
204         * Returns {@code true} if the file is a Jar archive.
205         * 
206         * @param file
207         *            the tested file
208         * @return {@code true} if the file is a Jar archive
209         */
210        protected boolean isJar(File file) {
211                if (file.isDirectory()) {
212                        return false;
213                }
214
215                try {
216                        ZipFile zf = new ZipFile(file);
217                        zf.close();
218                        return true;
219                } catch (ZipException e) {
220
221                } catch (IOException e) {
222                        e.printStackTrace();
223                }
224                return false;
225        }
226
227        /**
228         * Retrieves all Classes from one {@code directory}.
229         * 
230         * @param directory
231         *            the directory
232         * @return all classes in a list
233         */
234        protected List<Class<?>> getAllClasses(File directory) {
235                return getAllClasses(directory, directory);
236        }
237
238        /**
239         * Recursive methods searching for classes in a root directory.
240         * 
241         * @param root
242         *            the root directory
243         * @param file
244         *            the current file
245         * @return the list of all found classes
246         */
247        protected List<Class<?>> getAllClasses(File root, File file) {
248                List<Class<?>> classes = new ArrayList<Class<?>>();
249                if (file.isDirectory()) {
250                        for (File f : file.listFiles()) {
251                                classes.addAll(getAllClasses(root, f));
252                        }
253                } else {
254                        int rootLength = root.getAbsolutePath().length();
255                        String s = file.getAbsolutePath().substring(rootLength + 1);
256                        s = s.replace("\\", ".");
257                        s = s.replace("/", ".");
258                        if (s.endsWith(".class")) {
259                                s = s.substring(0, s.length() - 6);
260                                try {
261                                        Class<?> clazz = classLoader.loadClass(s);
262                                        classes.add(clazz);
263                                        invokeOut("Check: " + clazz.getName());
264                                } catch (ClassNotFoundException e) {
265                                        e.printStackTrace();
266                                } catch (UnsupportedClassVersionError e) {
267                                        System.err.println(s + " not supported: bad version number");
268                                        invokeErr(s + " not supported");
269                                }
270                        }
271                }
272
273                return classes;
274
275        }
276
277        /**
278         * Retrieves all Classes from a {@link ZipFile} (Jar archive).
279         * 
280         * @param zipFile
281         *            the Jar archive
282         * @return the list of all classes
283         */
284        protected List<Class<?>> getAllClasses(ZipFile zipFile) {
285                invokeOut(zipFile.toString());
286                List<Class<?>> classes = new ArrayList<Class<?>>();
287
288                List<? extends ZipEntry> entries = Collections.list(zipFile.entries());
289                for (int i = 0; i < entries.size(); i++) {
290
291                        ZipEntry entry = entries.get(i);
292
293                        String s = entry.getName();
294
295                        if (s.endsWith(".class")) {
296
297                                s = s.replace("/", ".");
298                                s = s.substring(0, s.length() - 6);
299
300                                try {
301                                        Class<?> clazz = classLoader.loadClass(s);
302                                        classes.add(clazz);
303                                        invokeOut("Check: " + clazz.getName());
304                                } catch (ClassNotFoundException e) {
305                                } catch (NoClassDefFoundError e) {
306                                } catch (UnsupportedClassVersionError e) {
307                                        System.err.println(s + " not supported: bad version number");
308                                        invokeErr(s + " not supported");
309                                }
310
311                        }
312                }
313
314                return classes;
315        }
316
317        /**
318         * Invoke an out message for the {@link ModuleAutoFinderListener}.
319         * 
320         * @param message
321         *            the message
322         */
323        protected void invokeOut(String message) {
324                for (ModuleAutoFinderListener listener : listeners) {
325                        listener.out(message);
326                }
327        }
328
329        /**
330         * Invoke an err message for the {@link ModuleAutoFinderListener}.
331         * 
332         * @param message
333         *            the message
334         */
335        protected void invokeErr(String message) {
336                for (ModuleAutoFinderListener listener : listeners) {
337                        listener.out(message);
338                }
339        }
340
341        /**
342         * Add a {@link ModuleAutoFinderListener}.
343         * 
344         * @param listener
345         *            the listener to be added
346         */
347        public void addListener(ModuleAutoFinderListener listener) {
348                listeners.add(listener);
349        }
350
351        /**
352         * Remove a {@link ModuleAutoFinderListener}.
353         * 
354         * @see #addListener
355         * @param listener
356         *            the listener to be removed
357         */
358        public void removeListener(ModuleAutoFinderListener listener) {
359                listeners.remove(listener);
360        }
