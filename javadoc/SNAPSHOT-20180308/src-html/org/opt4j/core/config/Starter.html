024package org.opt4j.core.config;
025
026import java.io.File;
027import java.io.IOException;
028import java.net.URI;
029import java.net.URISyntaxException;
030import java.net.URL;
031import java.util.ArrayList;
032import java.util.Collection;
033import java.util.HashSet;
034import java.util.List;
035import java.util.Set;
036
037import com.google.inject.Module;
038
039/**
040 * The {@link Starter} executes configuration files.
041 * 
042 * @author lukasiewycz.
043 * 
044 */
045public class Starter {
046
047        /**
048         * Start configuration files. Using multiple configuration files: Execute
049         * the {@link Task}s successively.
050         * 
051         * @param args
052         *            the configuration files
053         * @throws Exception
054         */
055        public static void main(String[] args) throws Exception {
056                Starter starter = new Starter();
057                starter.execute(args);
058        }
059
060        /**
061         * Executor method.
062         * 
063         * @param args
064         *            the configuration files
065         * @throws Exception
066         */
067        public void execute(String[] args) throws Exception {
068                addPlugins();
069
070                String taskname = args[0];
071                Class<? extends Task> taskClass = Class.forName(taskname).asSubclass(Task.class);
072                String[] filenames = new String[args.length - 1];
073                for (int i = 1; i < args.length; i++) {
074                        filenames[i - 1] = args[i];
075                }
076
077                execute(taskClass, filenames);
078        }
079
080        /**
081         * Executor method.
082         * 
083         * @param taskClass
084         *            the class of the task to use
085         * @param filenames
086         *            the configuration files
087         * @throws Exception
088         */
089        public void execute(Class<? extends Task> taskClass, String[] filenames) throws Exception {
090                ModuleLoader loader = new ModuleLoader(new ModuleRegister(new ModuleAutoFinder()));
091
092                for (String filename : filenames) {
093                        Set<Module> modules = new HashSet<Module>();
094                        modules.addAll(loader.load(filename));
095                        Task task = taskClass.newInstance();
096                        task.init(modules);
097                        task.call();
098                }
099
100        }
101
102        /**
103         * Adds {@code jar} files to the classpath. Searches in folder
104         * {@code plugins}.
105         * 
106         * @return all found and added {@code jar} files
107         */
108        public Collection<File> addPlugins() {
109
110                List<File> files = new ArrayList<File>();
111
112                try {
113
114                        List<String> dirs = new ArrayList<String>();
115                        dirs.add("plugins");
116                        dirs.add("../plugins");
117                        URL url = getClass().getProtectionDomain().getCodeSource().getLocation();
118                        URI uri = new URI(url.toString());
119                        File parent = new File(uri);
120                        while (parent.isFile()) {
121                                parent = parent.getParentFile();
122                        }
123
124                        for (String filename : dirs) {
125                                File plugins = new File(parent, filename);
126                                try {
127                                        if (plugins.exists()) {
128                                                for (File jar : plugins.listFiles()) {
129                                                        if (jar.getName().endsWith(".jar")) {
130                                                                ClassPath.addFile(jar);
131                                                                files.add(jar);
132                                                        }
133                                                }
134                                        }
135                                } catch (IOException e1) {
136                                        e1.printStackTrace();
137                                }
138                        }
139
140                } catch (URISyntaxException e) {
141                        e.printStackTrace();
142                }
143
144                return files;
145
146        }
147
148}
