034/**
035 * The {@link Property} contains information about a single property of a {@link PropertyModule}.
036 * 
037 * @author lukasiewycz
038 * 
039 */
040public class Property {
041
042        protected final Module module;
043
044        protected final String name;
045
046        protected final Class<?> type;
047
048        protected final Method getter;
049
050        protected final Method setter;
051
052        protected String info = "";
053
054        protected int order = 10000;
055
056        protected final Collection<Requirement> requirements = new ArrayList<Requirement>();
057
058        protected final Map<Class<? extends Annotation>, Annotation> annotations = new HashMap<Class<? extends Annotation>, Annotation>();
059
060        /**
061         * Constructs a {@link Property} .
062         * 
063         * @param module
064         *            the module
065         * @param name
066         *            the name
067         * @param type
068         *            the type
069         * @param getter
070         *            the getter method
071         * @param setter
072         *            the setter method
073         * 
074         * @param annotations
075         *            the annotations
076         */
077        public Property(Module module, String name, Class<?> type, Method getter, Method setter,
078                        Iterable<Annotation> annotations) {
079                super();
080                this.module = module;
081                this.name = name;
082                this.type = type;
083                this.getter = getter;
084                this.setter = setter;
085
086                for (Annotation annotation : annotations) {
087                        Class<? extends Annotation> clazz = annotation.annotationType();
088                        this.annotations.put(clazz, annotation);
089                }
090        }

182         * 
183         * @return {@code true} if the property is active
184         */
185        public boolean isActive() {
186
187                for (Requirement requirement : requirements) {
188                        if (!requirement.isFulfilled()) {
189                                return false;
190                        }
191                }
192                return true;

205        /**
206         * Returns the value of the property.
207         * 
208         * @see #setValue
209         * @return the value of the property
210         */
211        public Object getValue() {
212                try {
213                        return getter.invoke(module);
214                } catch (IllegalAccessException e) {
215                } catch (IllegalArgumentException e) {
216                } catch (InvocationTargetException e) {
217                }
218                return null;
219        }

221        /**
222         * Sets the value of the property.
223         * 
224         * @see #getValue
225         * @param value
226         *            the value to set
227         * @throws InvocationTargetException
228         *             thrown if the value cannot be assigned
229         */
230        public void setValue(Object value) throws InvocationTargetException {
231                setValueObject(value);
232        }
233
234        /**
235         * Sets the value of the property. The property has first to be converted to a the corresponding type.
236         * 
237         * @param value
238         *            the value to set
239         * @throws InvocationTargetException
240         *             thrown if the value cannot be assigned
241         */
242        public void setValue(String value) throws InvocationTargetException {
243                Class<?> type = this.type;
244
245                if (type.equals(Integer.TYPE)) {
246                        type = Integer.class;
247                } else if (type.equals(Long.TYPE)) {
248                        type = Long.class;
249                } else if (type.equals(Double.TYPE)) {
250                        type = Double.class;
251                } else if (type.equals(Float.TYPE)) {
252                        type = Float.class;
253                } else if (type.equals(Boolean.TYPE)) {
254                        type = Boolean.class;
255                } else if (type.equals(Byte.TYPE)) {
256                        type = Byte.class;
257                } else if (type.equals(Short.TYPE)) {
258                        type = Short.class;
259                } else if (type.equals(String.class) && value == null) {
260                        value = "";
261                }
262
263                Object object = null;
264                try {
265                        if (Character.TYPE.equals(type) || Character.class.equals(type)) {
266                                object = value.toCharArray()[0];
267                        } else if (type.isEnum()) {
268                                object = PropertyModule.toEnumElement(value, type.asSubclass(Enum.class));
269                        } else if (type.equals(Class.class)) {
270                                if (value != null && !value.equals("")) {
271                                        Class<?> c = Class.forName(value);
272
273                                        Type gtype = getter.getGenericReturnType();
274                                        Type actual = getEnclosingType(gtype);
275
276                                        if (isAssignable(actual, c)) {
277                                                object = c;
278                                        } else {
279                                                System.err.println("Can not assign " + c + " to Class<" + actual + ">");
280                                        }
281                                }
282                        } else {
283                                // for Byte/Integer/Long/Float/Double/String/Boolean
284                                // and all objects that have a constructor that accepts one
285                                // string
286                                Constructor<?> constructor = type.getConstructor(String.class);
287                                object = constructor.newInstance(value.trim());
288                        }
289                } catch (Exception e) {
290                        Throwable t = e;
291                        while (t.getCause() != null) {
292                                t = t.getCause();
293                        }
294                        String message = t.getLocalizedMessage();
295
296                        throw new InvocationTargetException(e,
297                                        "Failed assignment: module=" + module.getClass().getName() + " property=\"" + getName()
298                                                        + "\" value=\"" + value + "\" (" + t.getClass().getName()
299                                                        + (message != null ? ": " + message : "") + ")");
300                }
301                setValueObject(object);
302        }

304        /**
305         * Returns {@code true} if the property represents a number.
306         * 
307         * @return {@code true} if the property represents a number
308         */
309        public boolean isNumber() {
310                boolean isNumber = false;
311                if (type.equals(Integer.TYPE)) {
312                        isNumber = true;
313                } else if (type.equals(Long.TYPE)) {
314                        isNumber = true;
315                } else if (type.equals(Double.TYPE)) {
316                        isNumber = true;
317                } else if (type.equals(Float.TYPE)) {
318                        isNumber = true;
319                } else if (type.equals(Byte.TYPE)) {
320                        isNumber = true;
321                } else if (type.equals(Short.TYPE)) {
322                        isNumber = true;
323                }
324                return isNumber;
325        }

327        /**
328         * Returns the enclosing actual type of a class.
329         * 
330         * @param gtype
331         *            the type of the class
332         * @return the enclosing actual type
333         */
334        private Type getEnclosingType(Type gtype) {
335                if (gtype instanceof ParameterizedType) {
336                        ParameterizedType ptype = (ParameterizedType) gtype;
337                        Type rtype = ptype.getRawType();
338                        if (rtype.equals(Class.class)) {
339                                Type actual = ptype.getActualTypeArguments()[0];
340                                return actual;
341                        }
342                }
343                return null;
344        }
345
346        /**
347         * Verifies if a class is assignable to a type.
348         * 
349         * @param type
350         *            the type
351         * @param clazz
352         *            the class
353         * @return {@code true} if assignable
354         */
355        private boolean isAssignable(Type type, Class<?> clazz) {
356                if (type instanceof WildcardType) {
357                        WildcardType wildcard = (WildcardType) type;
358                        for (Type lower : wildcard.getLowerBounds()) {
359                                if (lower instanceof Class<?>) {
360                                        Class<?> lowerClass = (Class<?>) lower;
361                                        if (!clazz.isAssignableFrom(lowerClass)) {
362                                                return false;
363                                        }
364                                }
365                        }
366                        for (Type upper : wildcard.getUpperBounds()) {
367                                if (upper instanceof Class<?>) {
368                                        Class<?> upperClass = (Class<?>) upper;
369                                        if (!upperClass.isAssignableFrom(clazz)) {
370                                                return false;
371                                        }
372                                }
373                        }
374                }
375                return true;
376        }

378        /**
379         * Sets the value of the property.
380         * 
381         * @param value
382         *            the value to set
383         * @throws InvocationTargetException
384         *             thrown if the value cannot be assigned
385         */
386        protected void setValueObject(Object value) throws InvocationTargetException {
387                try {
388                        setter.invoke(module, value);
389                } catch (Exception e) {
390                        throw new InvocationTargetException(e);
391                }
392        }

403        /**
404         * Returns the annotation of the specified class or null if not existent.
