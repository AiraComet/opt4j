024package org.opt4j.core.config;
025
026import java.util.ArrayList;
027import java.util.Arrays;
028import java.util.Collection;
029import java.util.List;
030import java.util.Set;
031import java.util.concurrent.Callable;
032import java.util.concurrent.CopyOnWriteArraySet;
033
034import com.google.inject.Module;
035
036/**
037 * The {@link Task} is an abstract class for optimization tasks.
038 * 
039 * @author lukasiewycz
040 * @see ExecutionEnvironment
041 */
042public abstract class Task implements Callable<Void> {
043
044        protected State state = State.WAITING;
045
046        protected Exception e = null;
047
048        protected final List<Module> modules = new ArrayList<Module>();
049
050        protected boolean isInit = false;
051
052        protected final Set<TaskStateListener> listeners = new CopyOnWriteArraySet<TaskStateListener>();
053
054        /**
055         * The {@link State} of a task.
056         * 
057         * @author lukasiewycz
058         * 
059         */
060        public enum State {
061                /**
062                 * The task is waiting for its execution.
063                 */
064                WAITING,
065                /**
066                 * The task is currently executed.
067                 */
068                EXECUTING,
069                /**
070                 * The task is done.
071                 */
072                DONE;
073        }
074
075        /**
076         * Initializes the {@link Task} with modules.
077         * 
078         * @see Task#init(Collection)
079         * @param modules
080         *            the modules for the task
081         */
082        public void init(Module... modules) {
083                init(Arrays.asList(modules));
084        }
085
086        /**
087         * Initializes the {@link Task} with modules. Note that if a module is a
088         * {@link PropertyModule}, a copy is made with the
089         * {@link PropertyModule#clone()} method. If the {@link Module} is not a
090         * {@link PropertyModule}, the module is copied as a reference, i.e., it can
091         * be changed until the {@link Task#execute()} is called.
092         * 
093         * @param modules
094         *            the modules for the task
095         */
096        public void init(Collection<Module> modules) {
097                for (Module module : modules) {
098                        if (module instanceof PropertyModule) {
099                                PropertyModule pModule = (PropertyModule) module;
100                                this.modules.add(pModule.clone());
101                        } else {
102                                this.modules.add(module);
103                        }
104                }
105                isInit = true;
106        }
107
108        /**
109         * Executes the {@link Task}.
110         * 
111         * @throws Exception
112         */
113        public abstract void execute() throws Exception;
114
115        /*
116         * (non-Javadoc)
117         * 
118         * @see java.util.concurrent.Callable#call()
119         */
120        @Override
121        public Void call() throws Exception {
122                setState(State.EXECUTING);
123                try {
124                        execute();
125                } catch (Error e) {
126                        this.e = new RuntimeException(e);
127                        e.printStackTrace();
128                        throw e;
129                } catch (Exception e) {
130                        this.e = e;
131                        e.printStackTrace();
132                        throw e;
133                } finally {
134                        setState(State.DONE);
135                }
136
137                return null;
138        }
139
140        /**
141         * Returns the {@link Exception} that was thrown during the {@link Task}
142         * execution if an error occurred.
143         * 
144         * @return the occurred exception
145         */
146        public Exception getException() {
147                return e;
148        }
149
150        /**
151         * Returns the state of the {@link Task}.
152         * 
153         * @return the state of the task
154         */
155        public State getState() {
156                return state;
157        }
158
159        /**
160         * Sets the state of the {@link Task}.
161         * 
162         * @param state
163         *            the state of the task to be set
164         */
165        protected void setState(State state) {
166                this.state = state;
167                for (TaskStateListener listener : listeners) {
168                        listener.stateChanged(this);
169                }
170        }
171
172        /*
173         * (non-Javadoc)
174         * 
175         * @see java.lang.Object#toString()
176         */
177        @Override
178        public String toString() {
179                return "";
180        }
181
182        /**
183         * Adds a {@link TaskStateListener}.
184         * 
185         * @see #removeStateListener
186         * @param listener
187         *            the listener to be added
188         */
189        public void addStateListener(TaskStateListener listener) {
190                listeners.add(listener);
191        }
192
193        /**
194         * Removes a {@link TaskStateListener}.
195         * 
196         * @see #addStateListener
197         * @param listener
198         *            the listener to be removed
199         */
200        public void removeStateListener(TaskStateListener listener) {
201                listeners.remove(listener);
202        }
203
204        /**
205         * Returns a copy of the {@link Module}s of this {@link Task}.
206         * 
207         * @return a set of modules
208         */
209        public Collection<Module> getModules() {
210                List<Module> clones = new ArrayList<Module>();
211                for (Module module : modules) {
212                        PropertyModule pModule;
213                        if (module instanceof PropertyModule) {
214                                pModule = (PropertyModule) module;
215                        } else {
216                                pModule = new PropertyModule(module);
217                        }
218
219                        clones.add(pModule.clone());
220                }
221                return clones;
222        }
223}
