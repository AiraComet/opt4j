024package org.opt4j.core.config.visualization;
025
026import java.util.ArrayList;
027import java.util.Collection;
028import java.util.Set;
029import java.util.TreeSet;
030import java.util.concurrent.CopyOnWriteArraySet;
031
032import org.opt4j.core.config.PropertyModule;
033
034import com.google.inject.Singleton;
035
036/**
037 * The {@link SelectedModules}. A set of all selected modules that is extended
038 * by a listener functionality.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043@SuppressWarnings("serial")
044@Singleton
045public class SelectedModules extends TreeSet<PropertyModule> {
046
047        protected Set<SetListener> listeners = new CopyOnWriteArraySet<SetListener>();
048
049        /**
050         * Constructs a new, empty {@link SelectedModules} set.
051         */
052        public SelectedModules() {
053                super();
054        }
055
056        /*
057         * (non-Javadoc)
058         * 
059         * @see java.util.HashSet#add(java.lang.Object)
060         */
061        @Override
062        public boolean add(PropertyModule module) {
063                boolean b = super.add(module);
064                for (SetListener listener : listeners) {
065                        listener.moduleAdded(this, module);
066                }
067                return b;
068        }
069
070        /*
071         * (non-Javadoc)
072         * 
073         * @see java.util.AbstractCollection#addAll(java.util.Collection)
074         */
075        @Override
076        public boolean addAll(Collection<? extends PropertyModule> modules) {
077                boolean r = false;
078                for (PropertyModule module : modules) {
079                        r = add(module) || r;
080                }
081                return r;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see java.util.HashSet#remove(java.lang.Object)
088         */
089        @Override
090        public boolean remove(Object o) {
091                boolean b = super.remove(o);
092                if (b) {
093                        for (SetListener listener : listeners) {
094                                listener.moduleRemoved(this, (PropertyModule) o);
095                        }
096                }
097                return b;
098        }
099
100        /*
101         * (non-Javadoc)
102         * 
103         * @see java.util.HashSet#clear()
104         */
105        @Override
106        public void clear() {
107                Collection<PropertyModule> modules = new ArrayList<PropertyModule>();
108                modules.addAll(this);
109
110                for (PropertyModule module : modules) {
111                        remove(module);
112                }
113        }
114
115        /**
116         * Adds a {@link SetListener}.
117         * 
118         * @see #removeListener
119         * @param listener
120         *            the listener to be added
121         */
122        public void addListener(SetListener listener) {
123                listeners.add(listener);
124        }
125
126        /**
127         * Removes a {@link SetListener}.
128         * 
129         * @see #addListener
130         * @param listener
131         *            the listener to be removed
132         */
133        public void removeListener(SetListener listener) {
134                listeners.remove(listener);
135        }
136
137}
