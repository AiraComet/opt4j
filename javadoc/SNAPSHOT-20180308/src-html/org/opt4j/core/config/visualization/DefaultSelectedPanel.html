095                        return panel;

100        /**
101         * The {@link TabbedPane} is a custom {@link JTabbedPane}.
102         */
103        protected class TabbedPane extends JTabbedPane {
104                @Override
105                public void remove(int index) {
106                        Component component = tabs.getComponentAt(index);
107                        super.remove(index);
108
109                        if (component instanceof MyScrollPane) {
110                                MyScrollPane scroll = (MyScrollPane) component;
111                                map.remove(scroll.getPanel().getModule());
112                        }
113                }
114
115                protected int getIndex(PropertyModule module) {
116                        if (map.keySet().contains(module)) {
117                                for (int i = 0; i < tabs.getTabCount(); i++) {
118                                        Component component = tabs.getComponentAt(i);
119                                        if (component instanceof MyScrollPane) {
120                                                MyScrollPane scroll = (MyScrollPane) component;
121                                                if (scroll.getPanel().getModule().equals(module)) {
122                                                        return i;
123                                                }
124                                        }
125                                }
126                        }
127                        return -1;
128                }
129        }

131        /**
132         * The custom tab of the {@link JTabbedPane}.
133         */
134        protected class TabComponent extends JPanel {
135
136                private JLabel label;
137                private final JButton xmlButton;
138                private final JButton closeButton;
139
140                private final PropertyModule module;
141
142                public TabComponent(PropertyModule module) {
143                        super(new FlowLayout(FlowLayout.LEFT, 0, 0));
144                        this.module = module;
145                        setOpaque(false);
146
147                        String name = format.getName(module);
148                        ImageIcon icon = format.getIcon(module);
149
150                        label = new JLabel(name, icon, SwingConstants.LEFT);
151                        xmlButton = new XMLButton();
152                        xmlButton.setToolTipText("Show module xml");
153                        closeButton = new CloseButton();
154                        closeButton.setToolTipText("Remove module");
155
156                        setLabelName(name);
157                }
158
159                protected void setLabelName(String name) {
160                        if (name.length() > 22) {
161                                name = name.substring(0, 18) + "...";
162                        }
163                        label.setName(name);
164                        removeAll();
165
166                        label = new JLabel(label.getName(), label.getIcon(), SwingConstants.LEFT);
167
168                        add(label);
169                        add(new JLabel(" "));
170                        label.setBorder(BorderFactory.createEmptyBorder(0, label.getIcon() == null ? 5 : 0, 0, 5));
171
172                        add(xmlButton);
173                        add(closeButton);
174                        setBorder(BorderFactory.createEmptyBorder(2, 0, 0, 0));
175                }
176
177                protected PropertyModule getModule() {
178                        return module;
179                }
180
181                private abstract class TabButton extends JButton implements ActionListener {
182
183                        protected final static int SIZE = 16;
184
185                        TabButton() {
186                                setPreferredSize(new Dimension(SIZE, SIZE));
187                                setContentAreaFilled(false);
188                                setFocusable(false);
189                                setBorder(BorderFactory.createEtchedBorder());
190                                setBorderPainted(false);
191                                addMouseListener(buttonMouseListener);
192                                setRolloverEnabled(true);
193                                addActionListener(this);
194                        }
195
196                        // paint the cross
197                        @Override
198                        protected void paintComponent(Graphics g) {
199                                super.paintComponent(g);
200                                Graphics2D g2 = (Graphics2D) g.create();
201                                g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
202                                // shift the image for pressed buttons
203                                if (getModel().isPressed()) {
204                                        g2.translate(1, 1);
205                                }
206                                drawIcon(g2);
207                                g2.dispose();
208                        }
209
210                        protected abstract void drawIcon(Graphics2D g);
211
212                }
213
214                private class CloseButton extends TabButton {
215
216                        @Override
217                        public void actionPerformed(ActionEvent e) {
218                                int i = tabs.indexOfTabComponent(TabComponent.this);
219                                if (i != -1) {
220                                        selectedModules.remove(module);
221                                }
222                        }
223
224                        @Override
225                        protected void drawIcon(Graphics2D g) {
226                                g.setStroke(new BasicStroke(2));
227                                g.setColor(Color.RED);
228
229                                int delta = 5;
230                                g.drawLine(delta, delta, getWidth() - delta - 2, getHeight() - delta - 2);
231                                g.drawLine(getWidth() - delta - 2, delta, delta, getHeight() - delta - 2);
232                        }
233                }
234
235                private class XMLButton extends TabButton {
236                        @Override
237                        public void actionPerformed(ActionEvent e) {
238                                ModuleSaver saver = new ModuleSaver();
239                                final ClipboardFrame frame = new ClipboardFrame(saver.toXMLString(module));
240                                frame.pack();
241                                frame.setVisible(true);
242                        }
243
244                        @Override
245                        protected void drawIcon(Graphics2D g) {
246                                g.setStroke(new BasicStroke(1));
247                                Image image = Icons.getIcon(Icons.XMLTAG).getImage();
248                                g.drawImage(image, 0, -1, 15, 16, this);
249                        }
250                }
251
252                private final MouseListener buttonMouseListener = new MouseAdapter() {
253                        @Override
254                        public void mouseEntered(MouseEvent e) {
255                                Component component = e.getComponent();
256                                if (component instanceof AbstractButton) {
257                                        AbstractButton button = (AbstractButton) component;
258                                        button.setBorderPainted(true);
259                                }
260                        }
261
262                        @Override
263                        public void mouseExited(MouseEvent e) {
264                                Component component = e.getComponent();
265                                if (component instanceof AbstractButton) {
266                                        AbstractButton button = (AbstractButton) component;
267                                        button.setBorderPainted(false);
268                                }
269                        }
270                };
271        }

285        /**
286         * The drop listener for the drag-and-drop functionality.
287         */
288        protected final DropTargetListener dropListener = new DropTargetAdapter() {
289                @Override
290                public void drop(DropTargetDropEvent dtde) {
291                        Transferable transferable = dtde.getTransferable();
292                        DataFlavor flaver = ModuleTransferable.localModuleFlavor;
293                        try {
294
295                                if (transferable.isDataFlavorSupported(flaver)) {
296                                        Object o = transferable.getTransferData(flaver);
297                                        PropertyModule module = (PropertyModule) o;
298                                        selectedModules.add(module);
299
300                                        dtde.acceptDrop(DnDConstants.ACTION_MOVE);
301                                        dtde.getDropTargetContext().dropComplete(true);
302                                } else {
303                                        dtde.rejectDrop();
304                                }
305
306                        } catch (UnsupportedFlavorException e) {
307                                e.printStackTrace();
308                        } catch (IOException e) {
309                                e.printStackTrace();
310                        }
311                }
312        };

352        /**
353         * Adds a module to the panel.
354         * 
355         * @param module
356         *            the module to add
357         */
358        protected void addModule(PropertyModule module) {
359                if (!map.keySet().contains(module)) {
360
361                        PropertyPanel panel = null;
362                        Panel p = module.getModule().getClass().getAnnotation(Panel.class);
363                        if (p != null) {
364                                Class<? extends PropertyPanel> clazz = p.value();
365
366                                try {
367                                        Constructor<? extends PropertyPanel> cstr = clazz.getConstructor(PropertyModule.class,
368                                                        FileChooser.class, Format.class);
369                                        panel = cstr.newInstance(module, fileChooser, format);
370
371                                } catch (Exception e) {
372                                        throw new RuntimeException(e);
373                                }
374
375                        } else {
376                                panel = new PropertyPanel(module, fileChooser, format);
377                        }
378
379                        MyScrollPane scroll = new MyScrollPane(panel);
380                        scroll.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED);
381                        scroll.setPreferredSize(new Dimension(300, 300));
382
383                        String name = format.getName(module);
384                        String tooltip = format.getTooltip(module);
385
386                        int count = selectedModules.subSet(selectedModules.first(), module).size();
387                        tabs.insertTab(name, null, scroll, tooltip, count);
388                        tabs.setSelectedIndex(count);
389                        tabs.setTabComponentAt(count, new TabComponent(module));
390
391                        map.put(module, panel);
392
393                } else {
394                        int index = tabs.getIndex(module);
395                        if (index != -1) {
396                                tabs.setSelectedIndex(index);
397                        }
398                }
399                updateTabNames();
400        }
401
402        /**
403         * Removes a module from the panel.
404         * 
405         * @param module
406         *            the module to remove
407         */
408        protected void removeModule(PropertyModule module) {
409                int index = tabs.getIndex(module);
410                if (index != -1) {
411                        map.remove(module);
412                        tabs.remove(index);
413                }
414                updateTabNames();
415        }
416
417        /**
418         * Update the tab names. Consider multiple tabs of the same module.
419         */
420        protected void updateTabNames() {
421                Set<Class<? extends Module>> exist = new HashSet<Class<? extends Module>>();
422                Set<Class<? extends Module>> multi = new HashSet<Class<? extends Module>>();
423                for (PropertyModule pm : map.keySet()) {
424                        Class<? extends Module> module = pm.getModule().getClass();
425                        if (exist.contains(module)) {
426                                multi.add(module);
427                        } else {
428                                exist.add(module);
429                        }
430                }
431
432                Map<Class<? extends Module>, Integer> counter = new HashMap<Class<? extends Module>, Integer>();
433                for (Class<? extends Module> module : multi) {
434                        counter.put(module, 1);
435                }
436
437                for (int i = 0; i < tabs.getTabCount(); i++) {
438                        TabComponent comp = (TabComponent) tabs.getTabComponentAt(i);
439
440                        Class<? extends Module> module = comp.getModule().getModule().getClass();
441                        if (counter.containsKey(module)) {
442                                int c = counter.get(module);
443                                String name = "(" + c + ") " + format.getName(comp.getModule());
444                                comp.setLabelName(name);
445                                counter.put(module, c + 1);
446                        }
447                }
448                tabs.repaint();
449        }
