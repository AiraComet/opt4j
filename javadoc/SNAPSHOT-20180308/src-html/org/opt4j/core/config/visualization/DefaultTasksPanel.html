023package org.opt4j.core.config.visualization;
024
025import java.awt.BorderLayout;
026import java.awt.Color;
027import java.awt.Component;
028
029import javax.swing.BorderFactory;
030import javax.swing.JComponent;
031import javax.swing.JScrollPane;
032import javax.swing.JTable;
033import javax.swing.ScrollPaneConstants;
034import javax.swing.SwingUtilities;
035import javax.swing.table.AbstractTableModel;
036import javax.swing.table.TableCellRenderer;
037
038import org.opt4j.core.config.ExecutionEnvironment;
039import org.opt4j.core.config.Task;
040import org.opt4j.core.config.TaskListener;
041
042import com.google.inject.Inject;
043
044/**
045 * The {@link DefaultTasksPanel} extends the {@link TasksPanel} and shows all
046 * {@link Task}s in a table.
047 * 
048 * @author lukasiewycz
049 * 
050 */
051@SuppressWarnings("serial")
052public class DefaultTasksPanel extends TasksPanel implements TaskListener {
053
054        protected final ExecutionEnvironment executionEnvironment;
055
056        protected final Format format;
057
058        protected JTable table;
059
060        protected JScrollPane scroll;
061
062        protected final DelayTask delay = new DelayTask(40);
063
064        /**
065         * The {@link Table}.
066         */
067        protected class Table extends JTable {
068
069                /*
070                 * (non-Javadoc)
071                 * 
072                 * @see
073                 * javax.swing.JTable#prepareRenderer(javax.swing.table.TableCellRenderer
074                 * , int, int)
075                 */
076                @Override
077                public Component prepareRenderer(TableCellRenderer renderer, int row, int column) {
078                        Component c = super.prepareRenderer(renderer, row, column);
079                        Task task = executionEnvironment.getTasks().get(row);
080                        if (task.getException() == null) {
081                                c.setForeground(Color.BLACK);
082                        } else {
083                                c.setForeground(Color.RED);
084                        }
085
086                        if (c instanceof JComponent) {
087                                JComponent jc = (JComponent) c;
088                                jc.setBorder(BorderFactory.createEmptyBorder(0, 3, 0, 0));
089
090                                char[] chars = this.getValueAt(row, column).toString().toCharArray();
091                                int length = jc.getFontMetrics(jc.getFont()).charsWidth(chars, 0, chars.length);
092                                if (this.getColumnModel().getColumn(column).getWidth() < length) {
093                                        jc.setToolTipText(format.formatTooltip(this.getValueAt(row, column).toString()));
094                                } else {
095                                        jc.setToolTipText(null);
096                                }
097                        }
098
099                        return c;
100                }
101
102        }
103
104        /**
105         * The {@link Model}.
106         */
107        protected class Model extends AbstractTableModel {
108
109                @Override
110                public int getColumnCount() {
111                        return 2;
112                }
113
114                @Override
115                public int getRowCount() {
116                        return executionEnvironment.getTasks().size();
117                }
118
119                @Override
120                public Object getValueAt(int row, int column) {
121                        Task task = executionEnvironment.getTasks().get(row);
122                        if (column == 0) {
123                                return row + " " + task.toString();
124                        } else if (column == 1) {
125                                if (task.getException() != null) {
126                                        return "EXCEPTION: \n" + task.getException();
127                                }
128                                return task.getState();
129                        }
130                        return null;
131                }
132
133                protected String[] columnNames = { "Task", "State" };
134
135                @Override
136                public String getColumnName(int i) {
137                        return columnNames[i];
138                }
139        }
140
141        /**
142         * Constructs a {@link DefaultTasksPanel}.
143         * 
144         * @param executionEnvironment
145         *            the execution environment
146         * @param format
147         *            the format informations
148         */
149        @Inject
150        public DefaultTasksPanel(ExecutionEnvironment executionEnvironment, Format format) {
151                super();
152                this.executionEnvironment = executionEnvironment;
153                this.format = format;
154        }
155
156        /**
157         * Registers a listener at the execution environment.
158         */
159        @Inject
160        public void init() {
161                executionEnvironment.addListener(this);
162        }
163
164        /*
165         * (non-Javadoc)
166         * 
167         * @see org.opt4j.conf.gui.Startupable#startup()
168         */
169        @Override
170        public void startup() {
171                this.setLayout(new BorderLayout());
172
173                AbstractTableModel model = getModel();
174                table = getTable();
175                table.setModel(model);
176                table.getColumnModel().getColumn(0).setPreferredWidth(50);
177                table.getColumnModel().getColumn(1).setPreferredWidth(500);
178
179                scroll = new JScrollPane(table, ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED,
180                                ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER);
181
182                this.add(scroll, BorderLayout.CENTER);
183        }
184
185        /**
186         * Returns the instance of the table.
187         * 
188         * @return the table
189         */
190        protected JTable getTable() {
191                return new Table();
192        }
193
194        /**
195         * Returns the instance of the model.
196         * 
197         * @return the model
198         */
199        protected AbstractTableModel getModel() {
200                return new Model();
201        }
202
203        /*
204         * (non-Javadoc)
205         * 
206         * @see org.opt4j.config.TaskListener#added(org.opt4j.config.Task)
207         */
208        @Override
209        public void added(Task task) {
210                table.revalidate();
211                table.repaint();
212                SwingUtilities.invokeLater(new Runnable() {
213                        /*
214                         * (non-Javadoc)
215                         * 
216                         * @see java.lang.Runnable#run()
217                         */
218                        @Override
219                        public void run() {
220                                scroll.getVerticalScrollBar().setValue(table.getRowCount() * 120);
221                        }
222                });
223        }
224
225        /*
226         * (non-Javadoc)
227         * 
228         * @see
229         * org.opt4j.config.TaskStateListener#stateChanged(org.opt4j.config.Task)
230         */
231        @Override
232        public void stateChanged(Task task) {
233                delay.execute(new Runnable() {
234                        /*
235                         * (non-Javadoc)
236                         * 
237                         * @see java.lang.Runnable#run()
238                         */
239                        @Override
240                        public void run() {
241                                table.revalidate();
242                                table.repaint();
243                        }
244                });
245        }
246}
