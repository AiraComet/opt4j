024package org.opt4j.core.config.visualization;
025
026import java.awt.event.ActionEvent;
027import java.awt.event.ActionListener;
028import java.awt.event.KeyEvent;
029import java.io.File;
030
031import javax.swing.JDialog;
032import javax.swing.JMenu;
033import javax.swing.JMenuBar;
034import javax.swing.JMenuItem;
035
036import org.opt4j.core.config.Icons;
037import org.opt4j.core.config.visualization.FileOperations.FileOperationsListener;
038
039import com.google.inject.Inject;
040import com.google.inject.Provider;
041
042/**
043 * The {@link Menu}.
044 * 
045 * @author lukasiewycz
046 * 
047 */
048@SuppressWarnings("serial")
049public class Menu extends JMenuBar implements FileOperationsListener, Startupable {
050
051        final JMenuItem load = new JMenuItem("Load ...", Icons.getIcon(Icons.FOLDER));
052
053        final JMenuItem save = new JMenuItem("Save", Icons.getIcon(Icons.DISK));
054
055        final JMenuItem saveAs = new JMenuItem("Save As ...", Icons.getIcon(Icons.DISK));
056
057        final JMenuItem about = new JMenuItem("About");
058
059        protected final FileOperations fileOperations;
060
061        protected final Provider<ApplicationFrame> frame;
062
063        protected final About aboutInfo;
064
065        /**
066         * Constructs a {@link Menu}.
067         * 
068         * @param fileOperations
069         *            the file operations model
070         * @param about
071         *            the about information
072         * @param frame
073         *            the provider of the application frame
074         */
075        @Inject
076        public Menu(final FileOperations fileOperations, final About about, final Provider<ApplicationFrame> frame) {
077                super();
078                this.fileOperations = fileOperations;
079                this.aboutInfo = about;
080                this.frame = frame;
081        }
082
083        /*
084         * (non-Javadoc)
085         * 
086         * @see org.opt4j.config.visualization.Startupable#startup()
087         */
088        @Override
089        public void startup() {
090
091                JMenu file = new JMenu("File");
092                JMenu etc = new JMenu("?");
093
094                file.setMnemonic(KeyEvent.VK_F);
095
096                add(file);
097                add(etc);
098
099                load.setMnemonic(KeyEvent.VK_L);
100                load.addActionListener(new ActionListener() {
101                        @Override
102                        public void actionPerformed(ActionEvent e) {
103                                fileOperations.load();
104                        }
105                });
106
107                save.setMnemonic(KeyEvent.VK_S);
108                save.addActionListener(new ActionListener() {
109                        @Override
110                        public void actionPerformed(ActionEvent e) {
111                                fileOperations.save();
112                        }
113                });
114                save.setEnabled(false);
115
116                saveAs.setMnemonic(KeyEvent.VK_A);
117                saveAs.addActionListener(new ActionListener() {
118                        @Override
119                        public void actionPerformed(ActionEvent e) {
120                                fileOperations.saveAs();
121                        }
122                });
123
124                about.setMnemonic(KeyEvent.VK_A);
125                about.addActionListener(new ActionListener() {
126                        @Override
127                        public void actionPerformed(ActionEvent e) {
128                                JDialog dialog = aboutInfo.getDialog(frame.get());
129                                dialog.pack();
130                                dialog.setVisible(true);
131                        }
132                });
133
134                file.add(load);
135                file.add(save);
136                file.add(saveAs);
137
138                etc.add(about);
139        }
140
141        /**
142         * Initializes the listeners.
143         */
144        @Inject
145        public void init() {
146                fileOperations.addListener(this);
147        }
148
149        /*
150         * (non-Javadoc)
151         * 
152         * @see
153         * org.opt4j.config.visualization.FileOperations.FileOperationsListener#
154         * setCurrentFile(java.io.File)
155         */
156        @Override
157        public void setCurrentFile(File file) {
158                if (file != null) {
159                        save.setEnabled(true);
160                } else {
161                        save.setEnabled(false);
162                }
163        }
164
165}
