024package org.opt4j.core.config.visualization;
025
026import java.awt.event.ActionEvent;
027import java.awt.event.ActionListener;
028import java.awt.event.KeyEvent;
029import java.io.File;
030import java.util.HashSet;
031import java.util.Set;
032
033import javax.swing.JButton;
034import javax.swing.JToolBar;
035
036import org.opt4j.core.config.ExecutionEnvironment;
037import org.opt4j.core.config.Icons;
038import org.opt4j.core.config.ModuleSaver;
039import org.opt4j.core.config.visualization.FileOperations.FileOperationsListener;
040
041import com.google.inject.Inject;
042import com.google.inject.Module;
043import com.google.inject.Singleton;
044
045/**
046 * The {@link ToolBar}.
047 * 
048 * @author lukasiewycz
049 * 
050 */
051@SuppressWarnings("serial")
052@Singleton
053public class ToolBar extends JToolBar implements FileOperationsListener, Startupable {
054
055        protected JButton execute;
056        protected JButton save;
057        protected JButton saveAs;
058        protected JButton load;
059        protected JButton showXML;
060
061        protected final ExecutionEnvironment executionEnvironment;
062        protected final FileOperations fileOperations;
063        protected final SelectedModules selectedModules;
064
065        /**
066         * Constructs a {@link ToolBar}.
067         * 
068         * @param executionEnvironment
069         *            the execution environment
070         * @param selectedModules
071         *            the selected modules
072         * @param fileOperations
073         *            the file operations
074         */
075        @Inject
076        public ToolBar(final ExecutionEnvironment executionEnvironment, final SelectedModules selectedModules,
077                        FileOperations fileOperations) {
078                super();
079                this.executionEnvironment = executionEnvironment;
080                this.selectedModules = selectedModules;
081                this.fileOperations = fileOperations;
082        }
083
084        /**
085         * Initializes the listeners.
086         */
087        @Inject
088        public void init() {
089                fileOperations.addListener(this);
090                setFloatable(false);
091        }
092
093        /*
094         * (non-Javadoc)
095         * 
096         * @see org.opt4j.conf.gui.Startupable#startup()
097         */
098        @Override
099        public void startup() {
100                execute = new JButton("Run ", Icons.getIcon(Icons.PLAY)) {
101                        {
102                                setFocusable(false);
103                                setMnemonic(KeyEvent.VK_R);
104                                addActionListener(new ActionListener() {
105                                        @Override
106                                        public void actionPerformed(ActionEvent e) {
107                                                Set<Module> modules = new HashSet<Module>();
108                                                modules.addAll(selectedModules);
109                                                executionEnvironment.execute(modules);
110                                        }
111                                });
112                        }
113                };
114
115                load = new JButton("Load ... ", Icons.getIcon(Icons.FOLDER)) {
116                        {
117                                setFocusable(false);
118                                setMnemonic(KeyEvent.VK_L);
119                                addActionListener(new ActionListener() {
120                                        @Override
121                                        public void actionPerformed(ActionEvent e) {
122                                                fileOperations.load();
123                                        }
124                                });
125                        }
126                };
127                save = new JButton("Save ", Icons.getIcon(Icons.DISK)) {
128                        {
129                                setFocusable(false);
130                                setMnemonic(KeyEvent.VK_S);
131                                setEnabled(false);
132                                addActionListener(new ActionListener() {
133                                        @Override
134                                        public void actionPerformed(ActionEvent e) {
135                                                fileOperations.save();
136                                        }
137                                });
138                        }
139                };
140                saveAs = new JButton("Save As ... ", Icons.getIcon(Icons.DISK)) {
141                        {
142                                setFocusable(false);
143                                setMnemonic(KeyEvent.VK_A);
144                                addActionListener(new ActionListener() {
145                                        @Override
146                                        public void actionPerformed(ActionEvent e) {
147                                                fileOperations.saveAs();
148                                        }
149                                });
150                        }
151                };
152
153                showXML = new JButton("Show configuration", Icons.getIcon(Icons.XML)) {
154                        {
155                                setFocusable(false);
156                                setMnemonic(KeyEvent.VK_X);
157                                addActionListener(new ActionListener() {
158                                        @Override
159                                        public void actionPerformed(ActionEvent e) {
160                                                ModuleSaver saver = new ModuleSaver();
161                                                ClipboardFrame clipboardFrame = new ClipboardFrame(saver.toXMLString(selectedModules));
162                                                clipboardFrame.pack();
163                                                clipboardFrame.setVisible(true);
164                                        }
165                                });
166                        }
167                };
168
169                add(execute);
170                addSeparator();
171                add(load);
172                add(save);
173                add(saveAs);
174                addSeparator();
175                add(showXML);
176        }
177
178        /*
179         * (non-Javadoc)
180         * 
181         * @see
182         * org.opt4j.conf.gui.FileOperations.FileOperationsListener#setCurrentFile
183         * (java.io.File)
184         */
185        @Override
186        public void setCurrentFile(File file) {
187                save.setEnabled(file == null ? false : true);
188        }
189}
