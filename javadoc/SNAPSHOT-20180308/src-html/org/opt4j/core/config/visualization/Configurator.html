024package org.opt4j.core.config.visualization;
025
026import java.io.File;
027
028import javax.swing.SwingUtilities;
029import javax.swing.UIManager;
030
031import org.opt4j.core.config.Task;
032
033import com.google.inject.Binder;
034import com.google.inject.Guice;
035import com.google.inject.Injector;
036import com.google.inject.Module;
037import com.google.inject.Singleton;
038
039/**
040 * The {@link Configurator} configures the modules and starts the specific
041 * {@link ApplicationFrame}.
042 * 
043 * @author lukasiewycz
044 * @see ApplicationFrame
045 * 
046 */
047@Singleton
048public class Configurator {
049
050        /**
051         * Starts the {@link Configurator}.
052         * 
053         * @param args
054         *            specify one or more input configuration files to be loaded
055         * @throws Exception
056         */
057        public static void main(String[] args) throws Exception {
058                Configurator configurator = new Configurator();
059                configurator.start(args);
060        }
061
062        /**
063         * Starts the {@link Configurator}.
064         * 
065         * @param args
066         *            the input configuration files to be loaded
067         * @throws ClassNotFoundException
068         */
069        public void start(String[] args) throws ClassNotFoundException {
070                String filename = null;
071                Class<? extends Task> taskClass = null;
072                if (args.length < 1) {
073                        throw new RuntimeException("Specify the task class");
074                }
075
076                String taskname = args[0];
077                taskClass = Class.forName(taskname).asSubclass(Task.class);
078                if (args.length > 1) {
079                        filename = args[1];
080                }
081
082                main(taskClass, filename);
083        }
084
085        /**
086         * Returns the module for the {@link Configurator}.
087         * 
088         * @param taskClass
089         *            the task class
090         * @return the module
091         */
092        public Module getModule(final Class<? extends Task> taskClass) {
093                Module module = new Module() {
094                        @Override
095                        public void configure(Binder b) {
096                                b.bind(Task.class).to(taskClass);
097                        }
098                };
099                return module;
100        }
101
102        /**
103         * Starts the {@link Configurator} with the specified task class and the
104         * file to be loaded.
105         * 
106         * @param taskClass
107         *            the task class
108         * @param filename
109         *            the filename of the configuration file to be loaded
110         */
111        public void main(final Class<? extends Task> taskClass, final String filename) {
112
113                try {
114                        UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
115                } catch (Exception e) {
116                        e.printStackTrace();
117                }
118
119                Module module = getModule(taskClass);
120
121                final Injector injector = Guice.createInjector(module);
122
123                SwingUtilities.invokeLater(new Runnable() {
124                        @Override
125                        public void run() {
126                                ApplicationFrame frame = injector.getInstance(ApplicationFrame.class);
127                                frame.startup();
128                                if (filename != null) {
129                                        File file = new File(filename);
130                                        FileOperations fo = injector.getInstance(FileOperations.class);
131                                        fo.load(file);
132                                }
133                        }
134                });
135        }
136
137}
