024package org.opt4j.core.config.visualization;
025
026import java.io.File;
027import java.util.concurrent.RejectedExecutionException;
028
029import javax.swing.JFileChooser;
030
031import com.google.inject.Inject;
032import com.google.inject.Singleton;
033
034/**
035 * The {@link FileChooser}.
036 * 
037 * @author lukasiewycz
038 * 
039 */
040@Singleton
041public class FileChooser {
042
043        protected JFileChooser fileChooser = null;
044
045        /**
046         * Constructs a {@link FileChooser}.
047         * 
048         */
049        @Inject
050        public FileChooser() {
051                // Initializing custom file chooser in new thread to prevent blocking
052                // the main thread
053                Thread thread = new Thread() {
054                        @Override
055                        public void run() {
056                                try {
057                                        fileChooser = new MyFileChooser();
058                                        fileChooser.setVisible(false);
059                                        ready();
060                                } catch (RejectedExecutionException e) {
061                                        // if configurator is closed before the file chooser is
062                                        // ready
063                                }
064                        }
065                };
066                thread.start();
067        }
068
069        /**
070         * Returns the file chooser.
071         * 
072         * @return the file chooser
073         */
074        public synchronized JFileChooser get() {
075                while (fileChooser == null) {
076                        try {
077                                wait();
078                        } catch (InterruptedException e) {
079                        }
080                }
081                return fileChooser;
082        }
083
084        /**
085         * Notify all.
086         */
087        public synchronized void ready() {
088                notifyAll();
089        }
090
091        /**
092         * The {@link MyFileChooser} is a custom file chooser.
093         * 
094         * @author lukasiewycz
095         * 
096         */
097        @SuppressWarnings("serial")
098        private static class MyFileChooser extends JFileChooser {
099
100                public MyFileChooser() {
101                        super(new File(System.getProperty("user.home")));
102                }
103
104                @Override
105                public void setCurrentDirectory(final File file) {
106                        if (file == null) {
107                                return;
108                        }
109
110                        File dir;
111                        if (!file.isDirectory()) {
112                                dir = file.getParentFile();
113                                super.setSelectedFile(file);
114                        } else {
115                                dir = file;
116                        }
117
118                        if (dir == null || !dir.equals(getCurrentDirectory())) {
119                                super.setCurrentDirectory(dir);
120                        }
121                }
122
123        }
124
125}
