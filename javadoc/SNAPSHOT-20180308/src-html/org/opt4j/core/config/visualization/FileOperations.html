024package org.opt4j.core.config.visualization;
025
026import java.io.File;
027import java.util.Collection;
028import java.util.Set;
029import java.util.concurrent.CopyOnWriteArraySet;
030
031import javax.swing.JFileChooser;
032
033import org.opt4j.core.config.ModuleLoader;
034import org.opt4j.core.config.ModuleRegister;
035import org.opt4j.core.config.ModuleSaver;
036import org.opt4j.core.config.PropertyModule;
037
038import com.google.inject.Inject;
039import com.google.inject.Module;
040import com.google.inject.Singleton;
041
042/**
043 * The {@link FileOperations}.
044 * 
045 * @author lukasiewycz
046 * 
047 */
048@Singleton
049public class FileOperations {
050
051        /**
052         * The {@link FileOperationsListener}.
053         * 
054         * @author lukasiewycz
055         * 
056         */
057        public interface FileOperationsListener {
058
059                /**
060                 * Invoked if the current file is changed (on load or save).
061                 * 
062                 * @param file
063                 *            the current file
064                 */
065                public void setCurrentFile(File file);
066        }
067
068        protected File file = null;
069
070        protected final FileChooser fileChooser;
071
072        protected final ModuleRegister moduleRegister;
073
074        protected final SelectedModules selectedModules;
075
076        protected final Set<FileOperationsListener> listeners = new CopyOnWriteArraySet<FileOperationsListener>();
077
078        /**
079         * Constructs a {@link FileOperations}.
080         * 
081         * @param moduleRegister
082         *            all available modules
083         * @param selectedModules
084         *            all selected modules
085         * @param fileChooser
086         *            the FileChooser
087         */
088        @Inject
089        public FileOperations(ModuleRegister moduleRegister, SelectedModules selectedModules, FileChooser fileChooser) {
090                super();
091                this.moduleRegister = moduleRegister;
092                this.selectedModules = selectedModules;
093                this.fileChooser = fileChooser;
094        }
095
096        /**
097         * Sets the current file
098         * 
099         * @see #getFile
100         * @param file
101         *            the current file
102         */
103        protected void setFile(File file) {
104                this.file = file;
105                for (FileOperationsListener listener : listeners) {
106                        listener.setCurrentFile(file);
107                }
108        }
109
110        /**
111         * Returns the current file.
112         * 
113         * @see #setFile
114         * @return the current file
115         */
116        public File getFile() {
117                return file;
118        }
119
120        /**
121         * Opens the file chooser for loading the current configuration file.
122         */
123        public void load() {
124                JFileChooser fc = fileChooser.get();
125                fc.setFileFilter(null);
126                fc.setVisible(true);
127                fc.setCurrentDirectory(getFile());
128
129                int returnVal = fc.showDialog(null, "Load");
130
131                if (returnVal == JFileChooser.APPROVE_OPTION) {
132                        load(fc.getSelectedFile());
133                }
134        }
135
136        /**
137         * Opens the file chooser for saving the current configuration file.
138         */
139        public void saveAs() {
140                JFileChooser fc = fileChooser.get();
141                fc.setFileFilter(null);
142                fc.setVisible(true);
143                fc.setCurrentDirectory(getFile());
144
145                int returnVal = fc.showDialog(null, "Save");
146
147                if (returnVal == JFileChooser.APPROVE_OPTION) {
148                        save(fc.getSelectedFile());
149                }
150        }
151
152        /**
153         * Loads the selected modules from the specified file.
154         * 
155         * @param file
156         *            the configuration file
157         */
158        public void load(File file) {
159                setFile(file);
160
161                selectedModules.clear();
162
163                ModuleLoader loader = new ModuleLoader(moduleRegister);
164
165                Collection<? extends Module> modules = loader.load(file);
166                for (Module module : modules) {
167                        PropertyModule m = new PropertyModule(module);
168                        selectedModules.add(m);
169                }
170                System.out.println("Loaded configuration from " + file);
171        }
172
173        /**
174         * Saves the current selected modules to the current file.
175         */
176        public void save() {
177                save(file);
178        }
179
180        /**
181         * Saves the current selected modules to the specified file.
182         * 
183         * @param file
184         *            the configuration file
185         */
186        public void save(File file) {
187                setFile(file);
188                ModuleSaver saver = new ModuleSaver();
189
190                Collection<PropertyModule> modules = selectedModules;
191
192                System.out.println(modules);
193
194                saver.save(file, modules);
195                System.out.println("Saved configuration to " + file);
196        }
197
198        /**
199         * Adds a listener.
200         * 
201         * @see #removeListener
202         * @param listener
203         *            the listener to add
204         */
205        public synchronized void addListener(FileOperationsListener listener) {
206                listeners.add(listener);
207        }
208
209        /**
210         * Removes a listener.
211         * 
212         * @see #addListener
213         * @param listener
214         *            the listener to remove
215         */
216        public synchronized void removeListener(FileOperationsListener listener) {
217                listeners.remove(listener);
218        }
219
220}
