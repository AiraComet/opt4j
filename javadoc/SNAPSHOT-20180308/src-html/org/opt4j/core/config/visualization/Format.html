024package org.opt4j.core.config.visualization;
025
026import java.lang.reflect.Field;
027
028import javax.swing.ImageIcon;
029
030import org.opt4j.core.config.Icons;
031import org.opt4j.core.config.Property;
032import org.opt4j.core.config.PropertyModule;
033import org.opt4j.core.config.annotations.Category;
034import org.opt4j.core.config.annotations.Icon;
035import org.opt4j.core.config.annotations.Info;
036import org.opt4j.core.config.annotations.Name;
037
038/**
039 * The {@link Format} contains several format rules.
040 * 
041 * @author lukasiewycz
042 * 
043 */
044public class Format {
045        protected static final String xmlBreak = "<br/>";
046
047        /**
048         * Returns the name of a {@link Class}.
049         * 
050         * @param clazz
051         *            the class
052         * @return the name
053         */
054        public String getName(Class<?> clazz) {
055                Name name = clazz.getAnnotation(Name.class);
056                Category category = clazz.getAnnotation(Category.class);
057                if (name != null && !"".equals(name.value())) {
058                        return name.value();
059                } else if (category != null && !"".equals(category.value())) {
060                        return category.value();
061                } else {
062                        String n = clazz.getName();
063                        int c;
064                        while ((c = n.indexOf('.')) != -1) {
065                                n = n.substring(c + 1);
066                        }
067
068                        // Trim suffix "Module".
069                        if (n.endsWith("Module")) {
070                                n = n.substring(0, n.length() - "Module".length());
071                        }
072                        return n;
073                }
074        }
075
076        /**
077         * Returns the info of a {@link Class}.
078         * 
079         * @param c
080         *            the class
081         * @return the info
082         */
083        public String getInfo(Class<?> c) {
084                Info info = c.getAnnotation(Info.class);
085                String text = null;
086                if (info != null && !"".equals(info.value())) {
087                        text = info.value();
088                }
089                return text;
090        }
091
092        /**
093         * Returns the tooltip of a {@link Class}.
094         * 
095         * @param c
096         *            the class
097         * @return the tooltip
098         */
099        public String getTooltip(Class<?> c) {
100
101                String name = getName(c);
102                String info = getInfo(c);
103
104                String text = "<html><b>" + name + "</b>";
105                if (info != null) {
106                        text += xmlBreak + info;
107                }
108                text += "</html>";
109
110                return text;
111        }
112
113        /**
114         * Returns the icon of a {@link Class}.
115         * 
116         * @param clazz
117         *            the class
118         * @return the icon
119         */
120        public ImageIcon getIcon(Class<?> clazz) {
121                Icon icon = clazz.getAnnotation(Icon.class);
122
123                if (icon == null) {
124                        return null;
125                }
126
127                return Icons.getIcon(icon.value());
128        }
129
130        /**
131         * Returns the name of a {@link PropertyModule}.
132         * 
133         * @param module
134         *            the module
135         * @return the name
136         */
137        public String getName(PropertyModule module) {
138                Class<?> c = module.getModule().getClass();
139
140                return getName(c);
141        }
142
143        /**
144         * Returns the tooltip of a {@link PropertyModule}.
145         * 
146         * @param module
147         *            the module
148         * @return the tooltip
149         */
150        public String getTooltip(PropertyModule module) {
151                Class<?> c = module.getModule().getClass();
152
153                return getTooltip(c);
154        }
155
156        /**
157         * Returns the {@link ImageIcon} of a {@link PropertyModule}.
158         * 
159         * @param module
160         *            the module
161         * @return the icon
162         */
163        public ImageIcon getIcon(PropertyModule module) {
164                Class<?> c = module.getModule().getClass();
165
166                return getIcon(c);
167        }
168
169        /**
170         * Converts the given {@link ImageIcon} into a folder icon.
171         * 
172         * @param icon
173         *            the icon
174         * @return the converted folder icon
175         */
176        public ImageIcon asFolder(ImageIcon icon) {
177                ImageIcon i = Icons.getIcon(Icons.FOLDER);
178                if (icon != null) {
179                        ImageIcon background = icon;
180                        ImageIcon foreground = Icons.getIcon(Icons.SFOLDER);
181                        i = Icons.merge(background, foreground);
182                }
183                return i;
184        }
185
186        /**
187         * Returns the formatted name of the property.
188         * 
189         * @param property
190         *            the property
191         * @return the formatted name
192         */
193        public String getName(Property property) {
194                String name;
195                Name n = property.getAnnotation(Name.class);
196                if (n == null) {
197                        name = property.getName();
198                } else {
199                        name = n.value();
200                }
201                return name;
202        }
203
204        /**
205         * Returns the formatted (html) tooltip of a {@link Property}.
206         * 
207         * @param property
208         *            the property
209         * @return the tooltip
210         */
211        public String getTooltip(Property property) {
212                String text = "<html><b>" + getName(property) + "</b>";
213                String info = property.getInfo();
214                if (info != null) {
215                        text += xmlBreak + info;
216                }
217                if (property.getType().isEnum()) {
218                        Class<?> type = property.getType();
219                        text += xmlBreak;
220
221                        Field[] fields = type.getDeclaredFields();
222                        for (Field field : fields) {
223                                if (field.isEnumConstant()) {
224                                        String name = field.getName();
225                                        Icon icon = field.getAnnotation(Icon.class);
226                                        Info i = field.getAnnotation(Info.class);
227                                        text += "&nbsp;" + name;
228                                        if (icon != null || i != null) {
229                                                text += " - ";
230                                        }
231                                        if (icon != null) {
232                                                text += "<img src=\"" + Icons.getURL(icon.value()) + "\">";
233
234                                                System.out.println(text + " " + icon.value());
235                                        }
236                                        if (i != null) {
237                                                text += i.value();
238                                        }
239                                        text += xmlBreak;
240                                }
241                        }
242                }
243
244                text += "</html>";
245
246                return text;
247        }
248
249        /**
250         * Formats a text for tooltip (inserts break-lines).
251         * 
252         * @param text
253         *            the input text
254         * @return the formatted tooltip text
255         */
256        public String formatTooltip(String text) {
257                return "<html>" + text.replaceAll("\n", xmlBreak) + "</html>";
258        }
259}
