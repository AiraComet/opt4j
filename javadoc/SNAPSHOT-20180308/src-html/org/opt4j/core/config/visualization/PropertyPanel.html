051/**
052 * The {@link PropertyPanel} is a panel for the configuration of one module. Properties and values are arranged in a
053 * table.
054 * 
055 * @author lukasiewycz
056 * 
057 */
058@SuppressWarnings("serial")
059public class PropertyPanel extends JPanel {
060
061        protected final PropertyModule module;
062
063        protected final FileChooser fileChooser;
064
065        protected final Format format;
066
067        protected final JPanel panel;
068
069        protected final Map<Property, Component> components = new HashMap<Property, Component>();
070
071        /**
072         * Constructs a {@link PropertyPanel} for one {@link PropertyModule} instance.
073         * 
074         * @param module
075         *            the instance of the {@link PropertyModule}
076         * @param fileChooser
077         *            the FileChooser
078         * @param format
079         *            the format
080         */
081        public PropertyPanel(PropertyModule module, FileChooser fileChooser, Format format) {
082                this.module = module;
083                this.fileChooser = fileChooser;
084                this.format = format;
085
086                fillComponentsMap();
087
088                panel = new JPanel(new DialogLayout(20, 2));
089                panel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
090
091                updatePropertyPanel();
092                update();
093
094                setLayout(new BorderLayout());
095                add(BorderLayout.NORTH, panel);
096        }
097
098        protected Component createComponent(final Property property) {
099                Component component = null;
100                Object value = property.getValue();
101                Class<?> type = property.getType();
102
103                if (type.isEnum()) {
104                        Object[] obj = new Object[type.getEnumConstants().length];
105                        for (int k = 0; k < obj.length; k++) {
106                                obj[k] = type.getEnumConstants()[k];
107                        }
108                        final JComboBox<Object> box = new JComboBox<Object>(obj);
109                        box.setSelectedItem(value);
110
111                        box.addActionListener(new ActionListener() {
112                                @Override
113                                public void actionPerformed(ActionEvent e) {
114                                        Object selected = box.getSelectedItem();
115                                        try {
116                                                property.setValue(selected);
117                                                update();
118                                        } catch (InvocationTargetException ex) {
119                                                System.err.println(ex.getMessage());
120                                        }
121                                }
122                        });
123
124                        component = box;
125                } else if (type.equals(Boolean.TYPE)) {
126                        final JCheckBox checkbox = new JCheckBox();
127
128                        boolean b = (Boolean) property.getValue();
129                        checkbox.setSelected(b);
130                        checkbox.addActionListener(new ActionListener() {
131                                @Override
132                                public void actionPerformed(ActionEvent e) {
133                                        boolean b = checkbox.isSelected();
134                                        try {
135                                                property.setValue(b);
136                                                update();
137                                        } catch (InvocationTargetException ex) {
138                                                System.err.println(ex.getMessage());
139                                        }
140
141                                }
142                        });
143
144                        component = checkbox;
145                } else {
146                        final JTextField field = new JTextField();
147
148                        if (property.isNumber()) {
149                                field.setHorizontalAlignment(SwingConstants.RIGHT);
150                        }
151                        if (value == null) {
152                                field.setText("");
153                        } else {
154                                field.setText(value.toString());
155                        }
156
157                        field.addFocusListener(new FocusAdapter() {
158                                @Override
159                                public void focusLost(FocusEvent e) {
160                                        String value = format(property, field.getText());
161                                        try {
162                                                property.setValue(value);
163                                        } catch (InvocationTargetException ex) {
164                                                System.err.println(ex.getMessage());
165                                        } finally {
166                                                field.setText(property.getValue().toString());
167                                                update();
168                                        }
169                                }
170                        });
171
172                        field.addKeyListener(new KeyAdapter() {
173
174                                @Override
175                                public void keyReleased(KeyEvent e) {
176                                        String value = format(property, field.getText());
177
178                                        try {
179                                                property.setValue(value);
180                                        } catch (InvocationTargetException ex) {
181                                                System.err.println(ex.getMessage());
182                                                field.setText(property.getValue().toString());
183                                        }
184                                }
185                        });
186
187                        component = field;
188                }
189                return component;
190        }
191
192        protected String format(Property property, String value) {
193                if (property.isNumber()) {
194                        value = value.trim();
195                        if (value.equals("")) {
196                                value = "0";
197                        } else if (value.equals("-")) {
198                                value = "0";
199                        } else {
200                                if (value.startsWith(".")) {
201                                        value = "0" + value;
202                                }
203                                if (value.endsWith(".")) {
204                                        value = value + "0";
205                                }
206                        }
207                        return value;
208                }
209                return value;
210        }
211
212        protected void fillComponentsMap() {
213                for (final Property property : module.getProperties()) {
214                        Component component = createComponent(property);
215                        components.put(property, component);
216                }
217        }
218
219        protected void updatePropertyPanel() {
220                panel.removeAll();
221
222                for (final Property property : module.getProperties()) {
223                        if (property.isActive()) {
224                                String name = property.getName();
225
226                                int i = getIndent(property);
227                                String s = "";
228                                for (int j = 0; j < i; j++) {
229                                        s += "     ";
230                                }
231                                if (i > 0) {
232                                        s = s.substring(2) + "\u21aa ";
233                                }
234
235                                JPanel labelPanel = new JPanel(new FlowLayout(FlowLayout.LEFT, 0, 0));
236                                JLabel label = new JLabel(s + name);
237                                label.setFocusable(false);
238                                String tooltip = format.getTooltip(property);
239                                if (tooltip != null) {
240                                        label.setToolTipText(tooltip);
241                                }
242                                labelPanel.add(label);
243
244                                File file = property.getAnnotation(File.class);
245                                if (file != null) {
246                                        JButton browse = new JButton(Icons.getIcon(Icons.FOLDER));
247                                        browse.setFocusable(false);
248                                        browse.setBorderPainted(false);
249                                        browse.setContentAreaFilled(false);
250                                        browse.setBorder(BorderFactory.createEmptyBorder(0, 4, 0, 0));
251
252                                        browse.addActionListener(new ActionListener() {
253                                                @Override
254                                                public void actionPerformed(ActionEvent e) {
255                                                        selectFile(property);
256                                                }
257                                        });
258                                        browse.setCursor(new Cursor(Cursor.HAND_CURSOR));
259                                        browse.setToolTipText("Browse ...");
260                                        labelPanel.add(browse);
261                                }
262                                panel.add(labelPanel);
263
264                                Component component = components.get(property);
265                                panel.add(component);
266                        }
267                }
268        }
269
270        protected void update() {
271                updatePropertyPanel();
272                revalidate();
273                repaint();
274        }
275
276        /**
277         * Returns the instance of the {@link PropertyModule}.
278         * 
279         * @return the instance of the property module
280         */
281        public PropertyModule getModule() {
282                return module;
283        }
284
285        protected int getIndent(Property property) {
286                int i = 0;
287                for (Requirement requirement : property.getRequirements()) {
288                        i = Math.max(i, getIndent(requirement.getProperty()));
289                }
290                if (!property.getRequirements().isEmpty()) {
291                        i++;
292                }
293                return i;
294        }
295
296        private void selectFile(Property property) {
297                final File file = property.getAnnotation(File.class);
298                final JFileChooser fileChooser = PropertyPanel.this.fileChooser.get();
299
300                java.io.File dir = null;
301                try {
302                        dir = new java.io.File(property.getValue().toString());
303                } catch (Exception ex) {
304                }
305
306                fileChooser.setCurrentDirectory(dir);
307
308                if (file != null && !file.value().equals("")) {
309                        FileFilter filter = new FileFilter() {
310
311                                @Override
312                                public boolean accept(java.io.File pathname) {
313                                        if (pathname.isDirectory()) {
314                                                return true;
315                                        }
316                                        String f = pathname.getName().toLowerCase();
317                                        String ext = file.value().toLowerCase();
318
319                                        if (f.endsWith(ext)) {
320                                                return true;
321                                        }
322                                        return false;
323                                }
324
325                                @Override
326                                public String getDescription() {
327                                        String ext = file.value().toLowerCase();
328                                        return "(*" + ext + ")";
329                                }
330                        };
331
332                        fileChooser.setFileFilter(filter);
333                } else {
334                        fileChooser.setFileFilter(null);
335                }
336
337                fileChooser.setVisible(true);
338
339                int status = fileChooser.showOpenDialog(null);
340                if (status == JFileChooser.APPROVE_OPTION) {
341                        java.io.File f = fileChooser.getSelectedFile();
342                        try {
343                                property.setValue(f.getAbsolutePath());
344                                JTextField field = (JTextField) components.get(property);
345                                field.setText("" + property.getValue());
346                                update();
347                        } catch (InvocationTargetException ex) {
348                                System.err.println(ex.getMessage());
349                        }
350                }
351        }
