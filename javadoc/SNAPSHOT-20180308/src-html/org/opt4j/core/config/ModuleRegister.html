024package org.opt4j.core.config;
025
026import java.util.Collections;
027import java.util.HashMap;
028import java.util.HashSet;
029import java.util.Iterator;
030import java.util.Map;
031import java.util.Set;
032
033import org.opt4j.core.config.annotations.Multi;
034import org.opt4j.core.config.visualization.Configurator;
035
036import com.google.inject.Inject;
037import com.google.inject.Module;
038import com.google.inject.Singleton;
039
040/**
041 * The {@link ModuleRegister} contains all found modules for the
042 * {@link Configurator}.
043 * 
044 * @author lukasiewycz.
045 * 
046 */
047@Singleton
048public class ModuleRegister implements Iterable<Class<? extends Module>> {
049
050        protected final Map<Class<? extends Module>, PropertyModule> map = Collections
051                        .synchronizedMap(new HashMap<Class<? extends Module>, PropertyModule>());
052        protected final Set<Class<? extends Module>> set = Collections
053                        .synchronizedSet(new HashSet<Class<? extends Module>>());
054
055        protected final ModuleList finder;
056
057        protected boolean isInit = false;
058
059        /**
060         * Constructs and populates the {@link ModuleRegister}.
061         * 
062         * @param finder
063         *            the module finder to be used
064         */
065        @Inject
066        public ModuleRegister(final ModuleList finder) {
067                super();
068                this.finder = finder;
069        }
070
071        /*
072         * (non-Javadoc)
073         * 
074         * @see java.lang.Iterable#iterator()
075         */
076        @Override
077        public synchronized Iterator<Class<? extends Module>> iterator() {
078                checkInit();
079
080                return set.iterator();
081        }
082
083        /**
084         * Returns the number of found modules.
085         * 
086         * @see java.util.HashSet#size()
087         * @return the number of found modules
088         */
089        public synchronized int size() {
090                checkInit();
091                return set.size();
092        }
093
094        /**
095         * Returns the {@link PropertyModule} of the specified class and creates a
096         * new instance if necessary.
097         * 
098         * If the property module can not be created, {@code null} is returned.
099         * 
100         * @param clazz
101         *            the class of the module
102         * @return the property module or null
103         */
104        public PropertyModule get(Class<? extends Module> clazz) {
105                PropertyModule propertyModule = null;
106
107                if (clazz.getAnnotation(Multi.class) == null && map.containsKey(clazz)) {
108                        propertyModule = map.get(clazz);
109                } else {
110
111                        try {
112                                Module module = clazz.newInstance();
113                                propertyModule = new PropertyModule(module);
114                                map.put(clazz, propertyModule);
115                        } catch (InstantiationException e) {
116                                System.err.println("Failed to create an instance of module " + clazz + ".");
117                        } catch (IllegalAccessException e) {
118                                System.err.println("Failed to create an instance of module " + clazz + ".");
119                        }
120                }
121
122                return propertyModule;
123        }
124
125        /**
126         * Initialize the modules once.
127         */
128        private synchronized void checkInit() {
129                while (!isInit) {
130                        System.out.print("Searching Modules ... ");
131                        for (Class<? extends Module> clazz : finder.getModules()) {
132                                set.add(clazz);
133                                try {
134                                        // try to initialize static code of the module classes while
135                                        // searching modules and not later
136                                        clazz.newInstance();
137                                } catch (InstantiationException e) {
138                                } catch (IllegalAccessException e) {
139                                }
140                        }
141                        System.out.println("Done");
142
143                        isInit = true;
144                }
145        }
146
147}
