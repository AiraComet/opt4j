196        /**
197         * Returns the {@link URL} of a file.
198         * 
199         * @param filename
200         *            the filename
201         * @return the url
202         */
203        public static URL getURL(String filename) {
204                ClassLoader classLoader = ClassLoader.getSystemClassLoader();
205                URL url = classLoader.getResource(filename);
206                return url;
207        }
208
209        /**
210         * Returns an {@link ImageIcon} from a filename.
211         * 
212         * @param filename
213         *            the filename
214         * @return the corresponding image icon
215         */
216        public static ImageIcon getIcon(String filename) {
217                URL url = getURL(filename);
218                try {
219                        return new ImageIcon(url);
220                } catch (NullPointerException e) {
221                        System.err.println("Image " + filename + " not found.");
222                        e.printStackTrace();
223                        return null;
224                }
225        }
226
227        /**
228         * Returns a transparent 16x16px {@link ImageIcon}.
229         * 
230         * @return a transparent 16x16px {@link ImageIcon}
231         */
232        public static ImageIcon getDefault() {
233                Image image = new BufferedImage(16, 16, BufferedImage.TYPE_INT_ARGB);
234                ImageIcon icon = new ImageIcon(image);
235
236                return icon;
237        }

239        /**
240         * This methods merges multiple {@link ImageIcon}s into a single
241         * {@link ImageIcon}.
242         * 
243         * @param icons
244         *            the icons to merge
245         * @return the merged icons
246         */
247        public static ImageIcon merge(ImageIcon... icons) {
248
249                /**
250                 * The {@link MergedIcon} merges two icons into one.
251                 * 
252                 * @author lukasiewycz
253                 * 
254                 */
255                @SuppressWarnings("serial")
256                class MergedIcon extends ImageIcon {
257
258                        private final ImageIcon[] icons;
259
260                        public MergedIcon(ImageIcon[] icons) {
261                                super();
262                                this.icons = icons;
263                        }
264
265                        @Override
266                        public void paintIcon(Component c, Graphics g, int x, int y) {
267                                for (ImageIcon icon : icons) {
268                                        icon.paintIcon(c, g, x, y);
269                                }
270                        }
271
272                        @Override
273                        public int getIconWidth() {
274                                return icons[0].getIconWidth();
275                        }
276
277                        @Override
278                        public int getIconHeight() {
279                                return icons[0].getIconHeight();
280                        }
281                }
282
283                return new MergedIcon(icons);
284        }
