024package org.opt4j.core.config;
025
026import java.util.HashMap;
027import java.util.LinkedList;
028import java.util.List;
029import java.util.Map;
030
031import org.w3c.dom.Document;
032import org.w3c.dom.Element;
033import org.w3c.dom.NamedNodeMap;
034import org.w3c.dom.Node;
035import org.w3c.dom.NodeList;
036
037/**
038 * Wrapper for {@link org.w3c.dom.Node} elements. This class integrates Java 5
039 * functionality with generics.
040 * 
041 * @author lukasiewycz
042 * 
043 */
044public class JNode {
045
046        protected final Node node;
047
048        /**
049         * Creates a Wrapper {@link JNode} for a {@link Node} element.
050         * 
051         * @param node
052         *            the node to wrap
053         */
054        public JNode(Node node) {
055                this.node = node;
056        }
057
058        /**
059         * Constructs a new {@link JNode} for a {@link Document}.
060         * 
061         * @param document
062         *            the XML document
063         * @param name
064         *            the specified name of the node
065         */
066        public JNode(Document document, String name) {
067                node = document.createElement(name);
068        }
069
070        /**
071         * Retrieves the {@link Node} element.
072         * 
073         * @return the XML node
074         */
075        public Node getNode() {
076                return node;
077        }
078
079        /**
080         * Returns the name.
081         * 
082         * @return the name
083         */
084        public String getName() {
085                return node.getNodeName();
086        }
087
088        /**
089         * Returns a {@link Map} of all attributes.
090         * 
091         * @return a map of all attributes
092         */
093        public Map<String, String> getAttributes() {
094
095                Map<String, String> map = new HashMap<String, String>();
096
097                try {
098
099                        NamedNodeMap attributes = node.getAttributes();
100                        for (int i = 0; i < attributes.getLength(); i++) {
101                                Node attribute = attributes.item(i);
102                                map.put(attribute.getNodeName(), attribute.getNodeValue());
103                        }
104
105                } catch (ClassCastException e) {
106                }
107
108                return map;
109        }
110
111        /**
112         * Returns the value of an attribute.
113         * 
114         * @see #setAttribute
115         * @param attribute
116         *            the name of the attribute
117         * @return the value of the attribute
118         */
119        public String getAttribute(String attribute) {
120
121                try {
122                        Element element = (Element) node;
123                        return element.getAttribute(attribute);
124                } catch (ClassCastException e) {
125                }
126
127                return null;
128
129        }
130
131        /**
132         * Returns {@code true} if the attribute is existent.
133         * 
134         * @param attribute
135         *            the name of the attribute
136         * @return the value of the attribute
137         */
138        public boolean hasAttribute(String attribute) {
139
140                try {
141                        Element element = (Element) node;
142                        if (element.hasAttribute(attribute)) {
143                                return true;
144                        }
145                } catch (ClassCastException e) {
146                }
147                return false;
148
149        }
150
151        /**
152         * Returns the first child.
153         * 
154         * @return the first child
155         */
156        public JNode getChild() {
157                return new JNode(node.getFirstChild());
158        }
159
160        /**
161         * Returns the first child with a specified name.
162         * 
163         * @param name
164         *            the name of the child
165         * @return the first child with the specified name
166         */
167        public JNode getChild(String name) {
168                List<JNode> list = getChildren();
169
170                for (JNode node : list) {
171                        if (node.getName().equalsIgnoreCase(name)) {
172                                return node;
173                        }
174                }
175                return null;
176        }
177
178        /**
179         * Returns all children.
180         * 
181         * @return a list of all children.
182         */
183        public List<JNode> getChildren() {
184
185                List<JNode> list = new LinkedList<JNode>();
186                NodeList children = node.getChildNodes();
187
188                for (int i = 0; i < children.getLength(); i++) {
189                        Node child = children.item(i);
190                        list.add(new JNode(child));
191                }
192
193                return list;
194        }
195
196        /**
197         * Returns all children with a specified name.
198         * 
199         * @param name
200         *            the name of the children
201         * @return a list of all children with the specified name
202         */
203        public List<JNode> getChildren(String name) {
204
205                List<JNode> list = getChildren();
206                List<JNode> namedList = new LinkedList<JNode>();
207
208                for (JNode node : list) {
209                        if (node.getName().equalsIgnoreCase(name)) {
210                                namedList.add(node);
211                        }
212                }
213                return namedList;
214        }
215
216        /**
217         * Returns the strings content.
218         * 
219         * @see #setText
220         * @return the strings content
221         */
222        public String getText() {
223                return node.getTextContent();
224        }
225
226        /**
227         * Returns {@code true} if this node has a strings content.
228         * 
229         * @return {@code true} if this node has a strings content
230         */
231        public boolean hasText() {
232                return (node.getTextContent() != null && !node.getTextContent().equals(""));
233        }
234
235        /**
236         * Sets the strings content.
237         * 
238         * @see #getText
239         * @param text
240         *            the strings content to set
241         */
242        public void setText(String text) {
243                node.setTextContent(text);
244        }
245
246        /**
247         * Sets an attribute to a value.
248         * 
249         * @see #getAttribute
250         * @param attribute
251         *            the name of the attribute
252         * @param value
253         *            the value
254         */
255        public void setAttribute(String attribute, String value) {
256                ((Element) node).setAttribute(attribute, value);
257        }
258
259        /**
260         * Append a new {@link JNode} with a given name.
261         * 
262         * @param name
263         *            the specified name
264         * @return the appended JNode
265         */
266        public JNode appendChild(String name) {
267                Document document = node.getOwnerDocument();
268                Node n = document.createElement(name);
269                return appendChild(n);
270        }
271
272        /**
273         * Append an XML {@link Node}.
274         * 
275         * @param node
276         *            the XML node
277         * @return the appended JNode
278         */
279        public JNode appendChild(Node node) {
280                return appendChild(new JNode(node));
281        }
282
283        /**
284         * Append a {@link JNode}.
285         * 
286         * @param jNode
287         *            the node to append
288         * @return the appended JNode
289         */
290        public JNode appendChild(JNode jNode) {
291                this.node.appendChild(jNode.getNode());
292                return jNode;
293        }
294
295        /**
296         * Returns the owner {@link Document}.
297         * 
298         * @return the owner XML document
299         */
300        public Document getDocument() {
301                return node.getOwnerDocument();
302        }
303
304        /*
305         * (non-Javadoc)
306         * 
307         * @see java.lang.Object#toString()
308         */
309        @Override
310        public String toString() {
311                return node.toString();
312        }
313}
