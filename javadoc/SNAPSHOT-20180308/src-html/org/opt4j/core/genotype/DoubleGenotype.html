024package org.opt4j.core.genotype;
025
026import java.lang.reflect.Constructor;
027import java.util.ArrayList;
028import java.util.Random;
029
030import org.opt4j.core.Genotype;
031
032/**
033 * <p>
034 * The {@link DoubleGenotype} consists of double values that can be used as a
035 * {@link Genotype}.
036 * </p>
037 * <p>
038 * Example problem: Select filling level of five bottles<br/>
039 * Example usage: <blockquote>
040 * 
041 * <pre>
042 * DoubleGenotype&lt;Switch&gt; genotype = new DoubleGenotype();
043 * genotype.init(new Random(), 5);
044 * </pre>
045 * 
046 * </blockquote> Example instance: [0.5035947840006195, 0.9693492473483428,
047 * 0.12786372316728167, 0.5299369900029843, 0.8055193291478467]<br/>
048 * Example search space size: [0;1]<sup>5</sup>
049 * </p>
050 * 
051 * @author lukasiewycz
052 * 
053 */
054@SuppressWarnings("serial")
055public class DoubleGenotype extends ArrayList<Double> implements ListGenotype<Double> {
056
057        protected final Bounds<Double> bounds;
058
059        /**
060         * Constructs a {@link DoubleGenotype} with lower bounds {@code 0.0} and
061         * upper bounds {@code 1.0}.
062         */
063        public DoubleGenotype() {
064                this(0, 1);
065        }
066
067        /**
068         * Constructs a {@link DoubleGenotype} with a specified lower and upper
069         * bound for all values.
070         * 
071         * @param lowerBound
072         *            the lower bound
073         * @param upperBound
074         *            the upper bound
075         */
076        public DoubleGenotype(double lowerBound, double upperBound) {
077                this(new FixedBounds<Double>(lowerBound, upperBound));
078        }
079
080        /**
081         * Constructs a {@link DoubleGenotype} with the given {@link Bounds}.
082         * 
083         * @param bounds
084         *            the bounds
085         */
086        public DoubleGenotype(Bounds<Double> bounds) {
087                this.bounds = bounds;
088        }
089
090        /**
091         * Returns the lower bound for the {@code i}-th element.
092         * 
093         * @param index
094         *            the {@code i}-th element
095         * @return the lower bound of the {@code i}-th element
096         */
097        public double getLowerBound(int index) {
098                return bounds.getLowerBound(index);
099        }
100
101        /**
102         * Returns the upper bound for the {@code i}-th element.
103         * 
104         * @param index
105         *            the {@code i}-th element
106         * @return the upper bound of the {@code i}-th element
107         */
108        public double getUpperBound(int index) {
109                return bounds.getUpperBound(index);
110        }
111
112        /*
113         * (non-Javadoc)
114         * 
115         * @see org.opt4j.core.Genotype#newInstance()
116         */
117        @Override
118        @SuppressWarnings("unchecked")
119        public <G extends Genotype> G newInstance() {
120                try {
121                        Constructor<? extends DoubleGenotype> cstr = this.getClass().getConstructor(Bounds.class);
122                        return (G) cstr.newInstance(bounds);
123                } catch (Exception e) {
124                        throw new RuntimeException(e);
125                }
126        }
127
128        /**
129         * Initialize this genotype with {@code n} random values.
130         * 
131         * @param random
132         *            the random number generator
133         * @param n
134         *            the number of elements in the resulting genotype
135         */
136        public void init(Random random, int n) {
137                try {
138                        getLowerBound(n - 1);
139                        getUpperBound(n - 1);
140                } catch (IndexOutOfBoundsException outOfBoundException) {
141                        String message = outOfBoundException.getMessage() == null ? "" : outOfBoundException.getMessage() + "\n";
142                        message += "Can not initialize a genotype with " + n + " entries with the specified bounds";
143                        throw new IndexOutOfBoundsException(message);
144                }
145                for (int i = 0; i < n; i++) {
146                        double lo = getLowerBound(i);
147                        double hi = getUpperBound(i);
148                        double value = lo + random.nextDouble() * (hi - lo);
149                        if (i >= size()) {
150                                add(value);
151                        } else {
152                                set(i, value);
153                        }
154                }
155        }
156}
