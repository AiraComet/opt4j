023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.ArrayList;
027import java.util.Random;
028
029import org.opt4j.core.Genotype;
030
031/**
032 * <p>
033 * The {@link IntegerGenotype} is a {@link Genotype} that consists of
034 * {@link Integer} values.
035 * </p>
036 * <p>
037 * Example problem: Select the outcome of throwing five dice<br/>
038 * Example usage: <blockquote>
039 * 
040 * <pre>
041 * IntegerGenotype genotype = new IntegerGenotype(1, 6);
042 * genotype.init(new Random(), 5);
043 * </pre>
044 * 
045 * </blockquote> Example instance: [3, 5, 6, 1, 3]<br/>
046 * Example search space size: 6<sup>5</sup>
047 * </p>
048 * 
049 * @author lukasiewycz
050 * 
051 */
052public class IntegerGenotype extends ArrayList<Integer> implements ListGenotype<Integer> {
053
054        protected final Bounds<Integer> bounds;
055
056        /**
057         * Constructs a {@link IntegerGenotype} with a specified lower and upper
058         * bound for all values.
059         * 
060         * @param lowerBound
061         *            the lower bound
062         * @param upperBound
063         *            the upper bound
064         */
065        public IntegerGenotype(int lowerBound, int upperBound) {
066                this(new FixedBounds<Integer>(lowerBound, upperBound));
067        }
068
069        /**
070         * Constructs a {@link IntegerGenotype} with the given {@link Bounds}.
071         * 
072         * @param bounds
073         *            the bounds
074         */
075        public IntegerGenotype(Bounds<Integer> bounds) {
076                this.bounds = bounds;
077        }
078
079        /**
080         * Returns the lower bound for the {@code i}-th element.
081         * 
082         * @param index
083         *            the {@code i}-th element
084         * @return the lower bound of the {@code i}-th element
085         */
086        public int getLowerBound(int index) {
087                return bounds.getLowerBound(index);
088        }
089
090        /**
091         * Returns the upper bound for the {@code i}-th element.
092         * 
093         * @param index
094         *            the {@code i}-th element
095         * @return the upper bound of the {@code i}-th element
096         */
097        public int getUpperBound(int index) {
098                return bounds.getUpperBound(index);
099        }
100
101        /*
102         * (non-Javadoc)
103         * 
104         * @see org.opt4j.core.Genotype#newInstance()
105         */
106        @SuppressWarnings("unchecked")
107        @Override
108        public <G extends Genotype> G newInstance() {
109                try {
110                        Constructor<? extends IntegerGenotype> cstr = this.getClass().getConstructor(Bounds.class);
111                        return (G) cstr.newInstance(bounds);
112                } catch (Exception e) {
113                        throw new RuntimeException(e);
114                }
115        }
116
117        /**
118         * Initialize this genotype with {@code n} random values.
119         * 
120         * @param random
121         *            the random number generator
122         * @param n
123         *            the number of elements in the resulting genotype
124         */
125        public void init(Random random, int n) {
126                try {
127                        getLowerBound(n - 1);
128                        getUpperBound(n - 1);
129                } catch (IndexOutOfBoundsException outOfBoundException) {
130                        String message = outOfBoundException.getMessage() == null ? "" : outOfBoundException.getMessage() + "\n";
131                        message += "Can not initialize a genotype with " + n + " entries with the specified bounds";
132                        throw new IndexOutOfBoundsException(message);
133                }
134
135                for (int i = 0; i < n; i++) {
136                        int lo = getLowerBound(i);
137                        int hi = getUpperBound(i);
138                        int value = lo + random.nextInt(hi - lo + 1);
139                        if (i >= size()) {
140                                add(value);
141                        } else {
142                                set(i, value);
143                        }
144                }
145        }
146
147        private static final long serialVersionUID = 1L;
148}
