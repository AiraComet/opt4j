057/**
058 * The {@link ConvergencePlotWidget} plots the convergence for each
059 * {@link Objective}.
060 * 
061 * @author lukasiewycz
062 * 
063 */
064@WidgetParameters(title = "Convergence Plot", icon = Icons.CONVERGENCE)
065public class ConvergencePlotWidget implements Widget, OptimizerIterationListener, ObjectivesListener {
066
067        protected final DelayTask task = new DelayTask(40);
068
069        protected final ConvergencePlotData data;
070        protected final Selection selection;
071
072        protected final JPanel panel;
073        protected final Plot plot;
074
075        /**
076         * Constructs a {@link ConvergencePlotWidget}.
077         * 
078         * @param optimizer
079         *            the optimizer
080         * @param data
081         *            the data
082         * @param objectivesMonitor
083         *            the objective monitor that determine the objective of the
084         *            optimization problem
085         */
086        @Inject
087        public ConvergencePlotWidget(Optimizer optimizer, ConvergencePlotData data, ObjectivesMonitor objectivesMonitor,
088                        AutoZoomButton autoZoom) {
089                super();
090                this.data = data;
091                selection = new Selection();
092
093                panel = new JPanel();
094                panel.setLayout(new BorderLayout());
095
096                plot = new Plot();
097                plot.addLegend(0, "Max");
098                plot.addLegend(1, "Mean");
099                plot.addLegend(2, "Min");
100                Color[] colors = new Color[3];
101                colors[0] = Color.RED;
102                colors[1] = Color.LIGHT_GRAY;
103                colors[2] = Color.BLUE;
104                plot.setColors(colors);
105
106                panel.add(plot);
107
108                JToolBar menu = new JToolBar();
109                menu.setFloatable(false);
110                menu.add(selection);
111                menu.addSeparator();
112                autoZoom.setPlotBox(plot);
113                menu.add(autoZoom);
114
115                Border border = BorderFactory.createMatteBorder(0, 0, 1, 0, menu.getBackground().darker());
116                menu.setBorder(border);
117
118                panel.add(menu, BorderLayout.NORTH);
119
120                optimizer.addOptimizerIterationListener(this);
121
122                objectivesMonitor.addListener(this);
123                doPaint();
124        }
125
126        /**
127         * The {@link ObjectiveDropDown} is a combo box for objectives.
128         * 
129         * @author lukasiewycz
130         * 
131         */
132        static class ObjectiveDropDown extends JComboBox<Objective> {
133
134                private static final long serialVersionUID = 1L;
135
136                public ObjectiveDropDown() {
137                        super();
138
139                        setRenderer(new ListCellRenderer<Objective>() {
140                                protected DefaultListCellRenderer renderer = new DefaultListCellRenderer();
141
142                                @Override
143                                public Component getListCellRendererComponent(JList<? extends Objective> list, Objective value,
144                                                int index, boolean isSelected, boolean cellHasFocus) {
145                                        JLabel cell = (JLabel) renderer.getListCellRendererComponent(list, value, index, isSelected,
146                                                        cellHasFocus);
147                                        Objective objective = value;
148                                        cell.setText((objective != null ? "objective: " + objective.getName() : ""));
149                                        return cell;
150                                }
151                        });
152
153                        setMaximumSize(getPreferredSize());
154                }
155
156                @Override
157                public void addItem(Objective objective) {
158                        super.addItem(objective);
159                        setMaximumSize(getPreferredSize());
160                }
161
162                public Objective getSelected() {
163                        return (Objective) getSelectedItem();
164                }
165
166                public void setSelected(Objective objective) {
167                        setSelectedItem(objective);
168                }
169        }
170
171        /**
172         * The selection box for the current objective.
173         * 
174         * @author lukasiewycz
175         * 
176         */
177        class Selection extends ObjectiveDropDown implements ActionListener {
178
179                private static final long serialVersionUID = 1L;
180
181                public Selection() {
182                        super();
183                        addActionListener(this);
184                }
185
186                @Override
187                public Objective getSelected() {
188                        return super.getSelected();
189                }
190
191                @Override
192                public void actionPerformed(ActionEvent e) {
193                        ConvergencePlotWidget.this.doPaint();
194                }
195        }
196

198         * (non-Javadoc)
199         * 
200         * @see org.opt4j.viewer.Widget#getPanel()
201         */
202        @Override
203        public JPanel getPanel() {
204                return panel;
205        }
206

208         * (non-Javadoc)
209         * 
210         * @see org.opt4j.viewer.Widget#init(org.opt4j.viewer.Viewport)

212        @Override
213        public void init(Viewport viewport) {
214
215        }

217        /*
218         * (non-Javadoc)
219         * 
220         * @see
221         * org.opt4j.core.optimizer.OptimizerIterationListener#iterationComplete
222         * (org.opt4j.core.optimizer.Optimizer, int)
223         */
224        @Override
225        public void iterationComplete(final int iteration) {
226                doPaint();
227        }
228
229        /**
230         * Force a repaint of the plot.
231         */
232        protected void doPaint() {
233                task.execute(new Runnable() {
234                        @Override
235                        public void run() {
236                                paint();
237                        }
238                });
239        }
240
241        /**
242         * Repaints the plot. Do not call this method directly, call
243         * {@link #doPaint()} instead.
244         */
245        protected void paint() {
246                final Objective objective = selection.getSelected();
247
248                SwingUtilities.invokeLater(new Runnable() {
249                        @Override
250                        public void run() {
251                                plot.clear(false);
252
253                                plot.setXLabel("iteration");
254
255                                if (objective != null) {
256                                        plot.setYLabel(objective.getName());
257
258                                        paintList(data.getMaxPoints(objective), 0);
259                                        paintList(data.getMeanPoints(objective), 1);
260                                        paintList(data.getMinPoints(objective), 2);
261                                } else {
262                                        plot.setYLabel("");
263                                }
264
265                                plot.revalidate();
266                                plot.repaint();
267                        }
268                });
269        }
270
271        private void paintList(List<Double> list, int dataset) {
272                final int iteration = data.getIteration();
273
274                for (int i = 0; i < list.size(); i++) {
275                        Point2D.Double p1 = list.get(i);
276
277                        double x = p1.getX();
278                        double y = p1.getY();
279
280                        plot.addPoint(dataset, x, y, i != 0);
281                }
282
283                if (!list.isEmpty() && iteration > 0) {
284                        Point2D.Double p1 = list.get(list.size() - 1);
285                        double x = iteration;
286                        double y = p1.getY();
287
288                        plot.addPoint(dataset, x, y, true);
289                }
290        }
291
292        /*
293         * (non-Javadoc)
294         * 
295         * @see
296         * org.opt4j.viewer.ObjectivesMonitor.ObjectivesListener#objectives(java
297         * .util.Collection)
298         */
299        @Override
300        public void objectives(Collection<Objective> objectives) {
301                for (Objective objective : objectives) {
302                        selection.addItem(objective);
303                        selection.setSelectedIndex(0);
304                        selection.repaint();
305                }
306        }
