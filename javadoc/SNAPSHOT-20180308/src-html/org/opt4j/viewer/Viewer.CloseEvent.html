024package org.opt4j.viewer;
025
026import java.awt.BorderLayout;
027import java.awt.Dimension;
028import java.awt.event.WindowAdapter;
029import java.awt.event.WindowEvent;
030import java.awt.event.WindowListener;
031
032import javax.swing.ImageIcon;
033import javax.swing.JFrame;
034import javax.swing.UIManager;
035import javax.swing.WindowConstants;
036
037import org.opt4j.core.config.Icons;
038import org.opt4j.core.config.annotations.Info;
039import org.opt4j.core.optimizer.Control;
040import org.opt4j.core.optimizer.Optimizer;
041import org.opt4j.core.optimizer.OptimizerStateListener;
042import org.opt4j.core.start.Constant;
043
044import com.google.inject.Inject;
045
046/**
047 * The {@link Viewer}.
048 * 
049 * @author lukasiewycz
050 * 
051 */
052public class Viewer implements OptimizerStateListener {
053
054        /**
055         * The {@link CloseEvent} that is triggered if this viewer is closed.
056         * 
057         * @author lukasiewycz
058         * 
059         */
060        public enum CloseEvent {
061                /**
062                 * Optimization continues.
063                 */
064                @Info("Opimization is not influenced")
065                NONE,
066                /**
067                 * Optimization is stopped.
068                 */
069                @Info("Opimization id stopped")
070                STOP,
071                /**
072                 * Optimization is terminated.
073                 */
074                @Info("Opimization is terminated")
075                TERMINATE;
076
077        }
078
079        protected final Control control;
080
081        protected final CloseEvent closeEvent;
082
083        protected final boolean closeOnStop;
084
085        protected final String title;
086
087        protected final Viewport viewport;
088
089        protected final ToolBar toolBar;
090
091        protected final StatusBar statusBar;
092
093        protected JFrame frame = null;
094
095        /**
096         * Constructs a {@link Viewer}.
097         * 
098         * @param viewport
099         *            the viewport
100         * @param toolBar
101         *            the toolBar
102         * @param statusBar
103         *            the statusBar
104         * @param control
105         *            the control
106         * @param title
107         *            the title of the frame
108         * @param closeEvent
109         *            event on closing the GUI window
110         * @param closeOnStop
111         *            close window at the end of the optimization run
112         */
113        @Inject
114        public Viewer(Viewport viewport, ToolBar toolBar, StatusBar statusBar, Control control,
115                        @Constant(value = "title", namespace = Viewer.class) String title,
116                        @Constant(value = "closeEvent", namespace = Viewer.class) CloseEvent closeEvent,
117                        @Constant(value = "closeOnStop", namespace = Viewer.class) boolean closeOnStop) {
118                this.viewport = viewport;
119                this.toolBar = toolBar;
120                this.statusBar = statusBar;
121                this.title = title;
122                this.control = control;
123                this.closeEvent = closeEvent;
124                this.closeOnStop = closeOnStop;
125        }
126
127        /*
128         * (non-Javadoc)
129         * 
130         * @see
131         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStarted(org
132         * .opt4j.core.optimizer.Optimizer)
133         */
134        @Override
135        public void optimizationStarted(Optimizer optimizer) {
136
137                try {
138                        UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
139                } catch (Exception e) {
140                        e.printStackTrace();
141                }
142
143                Viewer.this.frame = new JFrame();
144                final JFrame frame = Viewer.this.frame;
145
146                ImageIcon logo = Icons.getIcon(Icons.OPT4J);
147                frame.setIconImage(logo.getImage());
148                frame.setTitle(title);
149                frame.setLayout(new BorderLayout());
150                frame.setPreferredSize(new Dimension(800, 600));
151
152                toolBar.init();
153                viewport.init();
154                statusBar.init();
155
156                frame.add(toolBar.get(), BorderLayout.NORTH);
157                frame.add(viewport.get(), BorderLayout.CENTER);
158                frame.add(statusBar.get(), BorderLayout.SOUTH);
159
160                WindowListener windowListener = new WindowAdapter() {
161                        @Override
162                        public void windowClosing(WindowEvent e) {
163                                switch (closeEvent) {
164                                case NONE:
165                                        break;
166                                case TERMINATE:
167                                        control.doTerminate();
168                                        break;
169                                default:
170                                        control.doStop();
171                                }
172                                frame.dispose();
173                        }
174                };
175
176                frame.addWindowListener(windowListener);
177                frame.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
178
179                frame.pack();
180                frame.setVisible(true);
181        }
182
183        /*
184         * (non-Javadoc)
185         * 
186         * @see
187         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStopped(org
188         * .opt4j.core.optimizer.Optimizer)
189         */
190        @Override
191        public void optimizationStopped(Optimizer optimizer) {
192                if (closeOnStop && frame != null) {
193                        frame.dispose();
194                }
195        }
196}
