065/**
066 * A widget that monitors the {@link Population}.
067 * 
068 * @author lukasiewycz
069 * 
070 */
071@WidgetParameters(title = "Population Monitor", icon = Icons.POPULATION)
072public class PopulationWidget implements IndividualStateListener, IndividualSetListener, Widget, ObjectivesListener {
073
074        protected final static int OFFSET = 3;
075
076        protected final Archive archive;
077
078        protected final SynchronizedIndividualList population;
079        protected List<Individual> swtIndividuals = Collections.emptyList();
080
081        protected final List<Objective> objectives = new ArrayList<Objective>();
082
083        protected final JPanel panel = new JPanel();
084
085        protected final JTable table;
086
087        protected int size = 0;
088
089        protected final DelayTask task = new DelayTask(40);
090
091        protected final List<IndividualMouseListener> individualMouseListeners = new CopyOnWriteArrayList<IndividualMouseListener>();
092
093        /**
094         * The {@link Table}.
095         * 
096         * @author lukasiewycz
097         * 
098         */
099        protected class Table extends QTable {
100                private static final long serialVersionUID = 1L;
101
102                public Table(TableModel tableModel) {
103                        super(tableModel);
104                }
105
106                @Override
107                public Component prepareRenderer(TableCellRenderer renderer, int row, int column) {
108                        Component c = super.prepareRenderer(renderer, row, column);
109
110                        if (row < swtIndividuals.size()) {
111                                try {
112                                        Individual individual = swtIndividuals.get(row);
113
114                                        if (individual != null) {
115                                                if (archive.contains(individual)) {
116                                                        String family = c.getFont().getFamily();
117                                                        int size = c.getFont().getSize();
118                                                        c.setFont(new Font(family, Font.BOLD, size));
119                                                }
120                                                if (individual.getState().isProcessing()) {
121                                                        c.setForeground(Color.RED);
122                                                } else {
123                                                        c.setForeground(Color.BLACK);
124                                                }
125
126                                        }
127                                } catch (IndexOutOfBoundsException e) {
128                                }
129                        }
130                        return c;
131                }
132        }
133
134        /**
135         * The {@link Model}.
136         * 
137         * @author lukasiewycz
138         * 
139         */
140        protected class Model extends AbstractTableModel {
141                private static final long serialVersionUID = 1L;
142
143                @Override
144                public int getColumnCount() {
145                        return OFFSET + objectives.size();
146                }
147
148                @Override
149                public int getRowCount() {
150                        size = Math.max(size, swtIndividuals.size());
151                        return size;
152                }
153
154                @Override
155                public String getColumnName(int col) {
156                        if (col == 0) {
157                                return "#";
158                        } else if (col == 1) {
159                                return "Individual";
160                        } else if (col == 2) {
161                                return "State";
162                        } else {
163                                final int index = col - OFFSET;
164                                final Objective objective = objectives.get(index);
165
166                                return objective.getName() + " (" + objective.getSign() + ")";
167                        }
168                }
169
170                @Override
171                public Object getValueAt(final int row, final int col) {
172
173                        try {
174                                final Individual individual = swtIndividuals.get(row);
175                                final State state = individual.getState();
176
177                                if (individual != null) {
178                                        if (col == 0) {
179                                                return row + 1;
180                                        } else if (col == 1) {
181                                                return individual.getPhenotype();
182                                        } else if (col == 2) {
183                                                return state;
184                                        } else {
185                                                if (state == Individual.State.EVALUATED) {
186
187                                                        final int index = col - OFFSET;
188                                                        final Objectives o = individual.getObjectives();
189                                                        final Objective objective = objectives.get(index);
190
191                                                        return o.get(objective).getValue();
192                                                }
193                                        }
194                                }
195                        } catch (IndexOutOfBoundsException e) {
196                        } catch (NullPointerException e) {
197                        }
198
199                        return null;
200                }
201        }
202
203        /**
204         * Constructs a {@link PopulationWidget}.
205         * 
206         * @param population
207         *            the population
208         * @param archive
209         *            the archive
210         * @param individualFactory
211         *            the individual creator
212         * @param objectivesMonitor
213         *            the objective monitor that determine the objective of the
214         *            optimization problem
215         */
216        @Inject
217        public PopulationWidget(Population population, Archive archive, IndividualFactory individualFactory,
218                        ObjectivesMonitor objectivesMonitor) {
219                this.archive = archive;
220
221                this.population = new SynchronizedIndividualList(population);
222                this.population.addListener(this);
223
224                table = getTable();
225
226                JScrollPane scroll = new JScrollPane(table);
227
228                panel.setLayout(new BorderLayout());
229                panel.add(scroll, BorderLayout.CENTER);
230
231                individualFactory.addIndividualStateListener(this);
232                objectivesMonitor.addListener(this);
233
234                paint();
235        }
236
237        protected final JTable getTable() {
238                Model model = new Model();
239                JTable table = new Table(model);
240                table.addMouseListener(new TableMouseListener());
241
242                final TableColumnModel columnModel = table.getColumnModel();
243                columnModel.getColumn(0).setPreferredWidth(25);
244                columnModel.getColumn(1).setPreferredWidth(140);
245                return table;
246        }

293        protected void paint() {
294                task.execute(new Runnable() {
295                        @Override
296                        public void run() {
297                                final List<Individual> temp = new ArrayList<Individual>();
298                                temp.addAll(population);
299
300                                SwingUtilities.invokeLater(new Runnable() {
301                                        @Override
302                                        public void run() {
303                                                swtIndividuals = temp;
304                                                if (table != null) {
305                                                        if (table.getColumnCount() != table.getModel().getColumnCount()) {
306                                                                table.createDefaultColumnsFromModel();
307                                                        }
308                                                        table.revalidate();
309                                                        table.repaint();
310                                                }
311                                        }
312                                });
313                        }
314                });
315        }

345        /**
346         * Set the {@link IndividualMouseListener}s to inform.
347         * 
348         * @param mouseListeners
349         *            the mouse listeners
350         */
351        @Inject(optional = true)
352        public void addMouseListeners(Set<IndividualMouseListener> mouseListeners) {
353                this.individualMouseListeners.addAll(mouseListeners);
354                sortIndividiualMouseListeners();
355        }
356
357        protected void sortIndividiualMouseListeners() {
358                List<IndividualMouseListener> list = new ArrayList<IndividualMouseListener>();
359                list.addAll(individualMouseListeners);
360                Collections.sort(list, new ToolBarOrderComparator<IndividualMouseListener>());
361                individualMouseListeners.clear();
362                individualMouseListeners.addAll(list);
363        }
364
365        /**
366         * The {@link TableMouseListener} that listens to right and double click of
367         * the table item.
368         * 
369         * @author lukasiewycz, reimann
370         * 
371         */
372        class TableMouseListener extends MouseAdapter {
373                private void reservTableShow(MouseEvent e, boolean released) {
374                        if (table.isEnabled()) {
375                                Point p = new Point(e.getX(), e.getY());
376                                int row = table.rowAtPoint(p);
377
378                                if (row >= 0 && row < table.getRowCount()) {
379                                        Individual individual = swtIndividuals.get(row);
380
381                                        if (individual.isEvaluated()) {
382                                                if (e.isPopupTrigger()) {
383                                                        table.getSelectionModel().setSelectionInterval(row, row);
384                                                        JPopupMenu menu = new JPopupMenu();
385                                                        for (IndividualMouseListener listener : individualMouseListeners) {
386                                                                listener.onPopup(individual, table, p, menu);
387                                                        }
388                                                        if (menu.getComponentCount() > 0) {
389                                                                menu.show(table, p.x, p.y);
390                                                        }
391                                                } else if (e.getClickCount() == 2 && released) {
392                                                        for (IndividualMouseListener listener : individualMouseListeners) {
393                                                                listener.onDoubleClick(individual, table, p);
394                                                        }
395                                                }
396                                        }
397                                }
398                        }
399                }
400
401                @Override
402                public void mouseReleased(MouseEvent e) {
403                        reservTableShow(e, true);
404                }
405
406                @Override
407                public void mousePressed(MouseEvent e) {
408                        reservTableShow(e, false);
409                }
410        }
