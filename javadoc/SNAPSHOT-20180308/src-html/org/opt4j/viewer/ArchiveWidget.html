106        /**
107         * The {@link TableMouseListener} that listens to right and double click of
108         * the table item.
109         * 
110         * @author lukasiewycz
111         * 
112         */
113        class TableMouseListener extends MouseAdapter {
114                private void reservTableShow(MouseEvent e, boolean released) {
115                        if (table.isEnabled()) {
116                                Point p = new Point(e.getX(), e.getY());
117                                int row = table.rowAtPoint(p);
118
119                                if (row >= 0 && row < table.getRowCount()) {
120                                        Individual individual = swtIndividuals.get(row);
121
122                                        if (e.isPopupTrigger()) {
123                                                table.getSelectionModel().setSelectionInterval(row, row);
124                                                JPopupMenu menu = new JPopupMenu();
125                                                for (IndividualMouseListener listener : individualMouseListeners) {
126                                                        listener.onPopup(individual, table, p, menu);
127                                                }
128                                                if (menu.getComponentCount() > 0) {
129                                                        menu.show(table, p.x, p.y);
130                                                }
131                                        } else if (e.getClickCount() == 2 && released) {
132                                                for (IndividualMouseListener listener : individualMouseListeners) {
133                                                        listener.onDoubleClick(individual, table, p);
134                                                }
135                                        }
136                                }
137                        }
138                }
139
140                @Override
141                public void mouseReleased(MouseEvent e) {
142                        reservTableShow(e, true);
143                }
144
145                @Override
146                public void mousePressed(MouseEvent e) {
147                        reservTableShow(e, false);
148                }
149        }

151        /**
152         * The model of the table.
153         * 
154         * @author lukasiewycz

157        protected class Model extends AbstractTableModel {
158                private static final long serialVersionUID = 1L;
159
160                @Override
161                public String getColumnName(int col) {
162                        if (col == 0) {
163                                return INDEX;
164                        } else if (col == 1) {
165                                return "Individual";
166                        }
167                        final int index = col - OFFSET;
168                        final Objective objective = objectives.get(index);
169
170                        return objective.getName() + " (" + objective.getSign() + ")";
171                }
172
173                @Override
174                public int getColumnCount() {
175                        return OFFSET + objectives.size();
176                }
177
178                @Override
179                public int getRowCount() {
180                        return swtIndividuals.size();
181                }
182
183                @Override
184                public Object getValueAt(final int row, final int col) {
185
186                        try {
187                                final Individual individual = swtIndividuals.get(row);
188
189                                if (individual != null) {
190                                        if (col == 0) {
191                                                return row + 1;
192                                        } else if (col == 1) {
193                                                return individual.getPhenotype();
194                                        } else if (individual.getState() == Individual.State.EVALUATED) {
195
196                                                final int index = col - OFFSET;
197                                                final Objectives o = individual.getObjectives();
198                                                final Objective objective = objectives.get(index);
199
200                                                return o.get(objective).getValue();
201                                        }
202                                }
203                        } catch (IndexOutOfBoundsException e) {
204                        } catch (NullPointerException e) {
205
206                        }
207                        return null;
208                }
209        }

330        protected void sortIndividiualMouseListeners() {
331                List<IndividualMouseListener> list = new ArrayList<IndividualMouseListener>();
332                list.addAll(individualMouseListeners);
333                Collections.sort(list, new ToolBarOrderComparator<IndividualMouseListener>());
334                individualMouseListeners.clear();
335                individualMouseListeners.addAll(list);
336        }

348        protected boolean updateOrder() {
349                if (table.getColumnCount() != objectives.size() + OFFSET) {
350                        return false;
351                }
352
353                List<Objective> list = new ArrayList<Objective>();
354                TableColumnModel model = table.getColumnModel();
355
356                for (int i = 0; i < model.getColumnCount(); i++) {
357                        int index = model.getColumn(i).getModelIndex();
358                        if (index >= OFFSET) {
359                                Objective o = objectives.get(index - OFFSET);
360                                list.add(o);
361                        }
362                }
363
364                if (!list.equals(order)) {
365                        order.clear();
366                        order.addAll(list);
367                        return true;
368                }
369                return false;
370        }

372        /**
373         * Repaints the table.
374         */
375        protected void paint() {
376                task.execute(new Runnable() {
377                        @Override
378                        public void run() {
379                                final List<Individual> temp = new ArrayList<Individual>();
380                                temp.addAll(archive);
381                                Collections.sort(temp, new Comp());
382
383                                SwingUtilities.invokeLater(new Runnable() {
384                                        @Override
385                                        public void run() {
386                                                swtIndividuals = temp;
387                                                sizeLabel.setText(Integer.toString(swtIndividuals.size()));
388
389                                                if (table != null) {
390                                                        if (table.getColumnCount() != table.getModel().getColumnCount()) {
391                                                                table.createDefaultColumnsFromModel();
392                                                        }
393                                                        table.revalidate();
394                                                        table.repaint();
395                                                }
396                                        }
397                                });
398                        }
399                });
400        }

402        protected class Comp implements Comparator<Individual> {
403
404                @Override
405                public int compare(Individual arg0, Individual arg1) {
406                        if (arg0 == null) {
407                                if (arg1 == null) {
408                                        return 0;
409                                }
410                                return 1;
411                        } else if (arg1 == null) {
412                                return -1;
413                        }
414
415                        Objectives obj0 = arg0.getObjectives();
416                        Objectives obj1 = arg1.getObjectives();
417
418                        for (Objective o : order) {
419                                assert obj0.get(o) != null;
420                                assert obj1.get(o) != null;
421
422                                Double v0 = obj0.get(o).getDouble();
423                                Double v1 = obj1.get(o).getDouble();
424
425                                int c = 0;
426
427                                if (v0 == null) {
428                                        if (v1 == null) {
429                                                continue;
430                                        }
431                                        c = 1;
432                                } else if (v1 == null) {
433                                        c = -1;
434                                } else {
435                                        c = v0.compareTo(v1);
436                                }
437
438                                if (c != 0) {
439                                        if (o.getSign() == Sign.MIN) {
440                                                return c;
441                                        }
442                                        return -c;
443                                }
444                        }
445                        return 0;
446                }
447
448        }
