023package org.opt4j.viewer;
024
025import java.awt.BorderLayout;
026import java.awt.Dimension;
027import java.awt.FlowLayout;
028import java.text.DecimalFormat;
029
030import javax.swing.BorderFactory;
031import javax.swing.JComponent;
032import javax.swing.JLabel;
033import javax.swing.JPanel;
034import javax.swing.JProgressBar;
035import javax.swing.SwingConstants;
036import javax.swing.SwingUtilities;
037import javax.swing.border.Border;
038
039import org.opt4j.core.config.visualization.DelayTask;
040import org.opt4j.core.optimizer.Optimizer;
041import org.opt4j.core.optimizer.OptimizerIterationListener;
042import org.opt4j.core.optimizer.OptimizerStateListener;
043import org.opt4j.core.start.Progress;
044
045import com.google.inject.Inject;
046
047/**
048 * The {@link StatusBar} contains informations about the optimization progress
049 * and time per iteration.
050 * 
051 * @author lukasiewycz
052 * 
053 */
054public class StatusBar implements OptimizerIterationListener, OptimizerStateListener {
055
056        protected final Progress progress;
057
058        protected final DelayTask task = new DelayTask(40);
059
060        protected final JLabel label = new JLabel();
061
062        protected final JLabel timeLabel = new JLabel();
063
064        protected final JProgressBar bar = new JProgressBar();
065
066        protected final JPanel panel = new JPanel();
067
068        protected long time;
069
070        /**
071         * Constructs a {@link StatusBar}.
072         * 
073         * @param progress
074         *            the progress
075         */
076        @Inject
077        public StatusBar(Progress progress) {
078                this.progress = progress;
079        }
080
081        /**
082         * Initialization. This method has to called once after construction.
083         */
084        public void init() {
085                panel.setLayout(new BorderLayout());
086                Border border = BorderFactory.createMatteBorder(1, 0, 0, 0, label.getBackground().darker());
087                panel.setBorder(border);
088
089                JPanel containerText = new JPanel(new FlowLayout(0, 5, 0));
090                timeLabel.setText("[time per iteration in ms]");
091                timeLabel.setHorizontalAlignment(SwingConstants.LEFT);
092                label.setPreferredSize(new Dimension(50, 18));
093                label.setHorizontalAlignment(SwingConstants.RIGHT);
094                label.setPreferredSize(new Dimension(50, 18));
095
096                containerText.add(label);
097                containerText.add(timeLabel);
098                panel.add(containerText, BorderLayout.WEST);
099
100                JPanel container = new JPanel(new BorderLayout());
101                container.setBorder(BorderFactory.createEmptyBorder(1, 1, 1, 4));
102                bar.setMinimum(0);
103                bar.setMaximum(progress.getMaxIterations());
104
105                bar.setPreferredSize(new Dimension(200, 16));
106                bar.setStringPainted(true);
107
108                container.add(bar);
109                panel.add(container, BorderLayout.EAST);
110
111                time = System.nanoTime();
112
113                update("", "init");
114        }
115
116        protected void update(final String message, final String progressMessage) {
117                task.execute(new Runnable() {
118                        @Override
119                        public void run() {
120                                SwingUtilities.invokeLater(new Runnable() {
121                                        @Override
122                                        public void run() {
123                                                String pm = progressMessage;
124                                                if (pm == null) {
125                                                        pm = "" + progress.getCurrentIteration() + "/" + progress.getMaxIterations();
126                                                }
127                                                label.setText(" " + message);
128                                                bar.setValue(progress.getCurrentIteration());
129                                                bar.setString(pm);
130                                        }
131                                });
132                        }
133                });
134        }
135
136        /**
137         * Returns the component.
138         * 
139         * @return the component
140         */
141        public JComponent get() {
142                return panel;
143        }
144
145        /*
146         * (non-Javadoc)
147         * 
148         * @see
149         * org.opt4j.core.optimizer.OptimizerIterationListener#iterationComplete
150         * (org.opt4j.core.optimizer.Optimizer, int)
151         */
152        @Override
153        public void iterationComplete(int iteration) {
154                final long diff = Math.round(((System.nanoTime() - time)) / 1000000.0);
155                time = System.nanoTime();
156
157                DecimalFormat format = new DecimalFormat();
158
159                update(format.format(diff), null);
160        }
161
162        /*
163         * (non-Javadoc)
164         * 
165         * @see
166         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStarted(org
167         * .opt4j.core.optimizer.Optimizer)
168         */
169        @Override
170        public void optimizationStarted(Optimizer optimizer) {
171        }
172
173        /*
174         * (non-Javadoc)
175         * 
176         * @see
177         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStopped(org
178         * .opt4j.core.optimizer.Optimizer)
179         */
180        @Override
181        public void optimizationStopped(Optimizer optimizer) {
182                update("", "done");
183                timeLabel.setText("");
184        }
185}
