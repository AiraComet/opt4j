023package org.opt4j.viewer;
024
025import java.awt.event.ActionEvent;
026import java.awt.event.ActionListener;
027
028import javax.swing.JButton;
029import javax.swing.JToolBar;
030
031import org.opt4j.core.config.Icons;
032
033import com.google.inject.Inject;
034import com.google.inject.Provider;
035
036/**
037 * The {@link ViewsToolBarService} can open a {@link ArchiveWidget},
038 * {@link PopulationWidget}, or {@link ParetoPlotWidget}. This service has a
039 * {@link ToolBarOrder} with {@code -50}.
040 * 
041 * @see ArchiveWidget
042 * @see PopulationWidget
043 * @see ParetoPlotWidget
044 * @author lukasiewycz
045 * 
046 */
047@ToolBarOrder(-50)
048public class ViewsToolBarService implements ToolBarService {
049
050        protected final Viewport viewport;
051        protected final Provider<ArchiveWidget> archiveWidgetProvider;
052        protected final Provider<PopulationWidget> populationWidgetProvider;
053        protected final Provider<ParetoPlotWidget> plotWidgetProvider;
054        protected final Provider<ConvergencePlotWidget> objectivesPlotWidgetProvider;
055        protected ArchiveWidget archiveWidget = null;
056        protected PopulationWidget populationWidget = null;
057
058        /**
059         * Constructs a {@link ViewsToolBarService}.
060         * 
061         * @param viewport
062         *            the viewport
063         * @param archiveWidgetProvider
064         *            the archiveWidgetProvider
065         * @param populationWidgetProvider
066         *            the populationWidgetProvider
067         * @param plotWidgetProvider
068         *            the plotWidgetProvider
069         * @param objectivesPlotWidgetProvider
070         *            the objectivesPlotWidgetProvider
071         */
072        @Inject
073        public ViewsToolBarService(Viewport viewport, Provider<ArchiveWidget> archiveWidgetProvider,
074                        Provider<PopulationWidget> populationWidgetProvider, Provider<ParetoPlotWidget> plotWidgetProvider,
075                        Provider<ConvergencePlotWidget> objectivesPlotWidgetProvider) {
076                super();
077                this.viewport = viewport;
078                this.archiveWidgetProvider = archiveWidgetProvider;
079                this.populationWidgetProvider = populationWidgetProvider;
080                this.plotWidgetProvider = plotWidgetProvider;
081                this.objectivesPlotWidgetProvider = objectivesPlotWidgetProvider;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see org.opt4j.viewer.ToolBarService#getToolBar()
088         */
089        @Override
090        public JToolBar getToolBar() {
091                JToolBar toolbar = new JToolBar("Views");
092
093                JButton buttonArchive = new JButton(Icons.getIcon(Icons.ARCHIVE));
094                buttonArchive.addActionListener(new ActionListener() {
095                        @Override
096                        public void actionPerformed(ActionEvent arg0) {
097                                Widget widget = getArchiveWidget();
098                                viewport.addWidget(widget);
099                        }
100                });
101                buttonArchive.setToolTipText("Archive");
102                buttonArchive.setFocusable(false);
103
104                JButton buttonPopulation = new JButton(Icons.getIcon(Icons.POPULATION));
105                buttonPopulation.addActionListener(new ActionListener() {
106                        @Override
107                        public void actionPerformed(ActionEvent arg0) {
108                                Widget widget = getPopulationWidget();
109                                viewport.addWidget(widget);
110                        }
111                });
112                buttonPopulation.setToolTipText("Population");
113                buttonPopulation.setFocusable(false);
114
115                JButton buttonPlot = new JButton(Icons.getIcon(Icons.PARETO));
116                buttonPlot.addActionListener(new ActionListener() {
117                        @Override
118                        public void actionPerformed(ActionEvent arg0) {
119                                Widget widget = getPlotWidget();
120                                viewport.addWidget(widget);
121                        }
122                });
123                buttonPlot.setToolTipText("Pareto Plot");
124                buttonPlot.setFocusable(false);
125
126                JButton buttonObjectivesPlot = new JButton(Icons.getIcon(Icons.CONVERGENCE));
127                buttonObjectivesPlot.addActionListener(new ActionListener() {
128                        @Override
129                        public void actionPerformed(ActionEvent arg0) {
130                                Widget widget = objectivesPlotWidgetProvider.get();
131                                viewport.addWidget(widget);
132                        }
133                });
134                buttonObjectivesPlot.setToolTipText("Objectives Plot");
135                buttonObjectivesPlot.setFocusable(false);
136
137                toolbar.add(buttonArchive);
138                toolbar.add(buttonPopulation);
139                toolbar.add(buttonPlot);
140                toolbar.add(buttonObjectivesPlot);
141
142                return toolbar;
143        }
144
145        protected ArchiveWidget getArchiveWidget() {
146                if (archiveWidget == null) {
147                        archiveWidget = archiveWidgetProvider.get();
148                }
149                return archiveWidget;
150        }
151
152        protected PopulationWidget getPopulationWidget() {
153                if (populationWidget == null) {
154                        populationWidget = populationWidgetProvider.get();
155                }
156                return populationWidget;
157        }
158
159        protected ParetoPlotWidget getPlotWidget() {
160                return plotWidgetProvider.get();
161        }
162
163}
