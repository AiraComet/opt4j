023package org.opt4j.viewer;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.List;
029import java.util.Set;
030import java.util.concurrent.CopyOnWriteArraySet;
031
032import org.opt4j.core.Individual;
033import org.opt4j.core.Individual.State;
034import org.opt4j.core.IndividualFactory;
035import org.opt4j.core.IndividualStateListener;
036import org.opt4j.core.Objective;
037import org.opt4j.core.optimizer.Archive;
038
039import com.google.inject.Inject;
040import com.google.inject.Singleton;
041
042/**
043 * The {@link ObjectivesMonitor} informs its listeners about the
044 * {@link Objective}s of the optimization problem as soon as it is available.
045 * 
046 * @author lukasiewycz
047 * 
048 */
049@Singleton
050public class ObjectivesMonitor implements IndividualStateListener {
051
052        protected boolean done = false;
053
054        protected List<Objective> objectives = null;
055
056        protected Set<ObjectivesListener> listeners = new CopyOnWriteArraySet<ObjectivesListener>();
057
058        /**
059         * The {@link ObjectivesListener} is an interface for classes that need the
060         * objectives.
061         * 
062         * @author lukasiewycz
063         * 
064         */
065        public interface ObjectivesListener {
066
067                /**
068                 * Callback method that passes the objectives of the optimization
069                 * problem.
070                 * 
071                 * @param objectives
072                 *            the objectives
073                 */
074                public void objectives(Collection<Objective> objectives);
075        }
076
077        /**
078         * Constructs the {@link ObjectivesMonitor}.
079         * 
080         * @param individualFactory
081         *            the individual factory
082         * @param archive
083         *            the archive
084         */
085        @Inject
086        public ObjectivesMonitor(IndividualFactory individualFactory, Archive archive) {
087                super();
088                individualFactory.addIndividualStateListener(this);
089
090                if (!archive.isEmpty()) {
091                        publishObjectives(archive.iterator().next());
092                }
093        }
094
095        /**
096         * Adds an {@link ObjectivesListener}.
097         * 
098         * @param listener
099         *            the listener
100         */
101        public void addListener(ObjectivesListener listener) {
102                synchronized (this) {
103                        if (done) {
104                                listener.objectives(objectives);
105                        } else {
106                                listeners.add(listener);
107                        }
108                }
109        }
110
111        /**
112         * Remove an {@link ObjectivesListener}.
113         * 
114         * @param listener
115         *            the listener
116         */
117        public void removeListener(ObjectivesListener listener) {
118                listeners.remove(listener);
119        }
120
121        /*
122         * (non-Javadoc)
123         * 
124         * @see
125         * org.opt4j.core.IndividualStateListener#inidividualStateChanged(org.opt4j
126         * .core.Individual)
127         */
128        @Override
129        public void inidividualStateChanged(Individual individual) {
130                if (!done && individual.getState() == State.EVALUATED) {
131                        publishObjectives(individual);
132                }
133        }
134
135        protected synchronized void publishObjectives(Individual individual) {
136                if (!done) {
137                        objectives = new ArrayList<Objective>(individual.getObjectives().getKeys());
138                        for (ObjectivesListener listener : listeners) {
139                                listener.objectives(objectives);
140                        }
141                        Collections.sort(objectives);
142                        done = true;
143                }
144        }
145}
