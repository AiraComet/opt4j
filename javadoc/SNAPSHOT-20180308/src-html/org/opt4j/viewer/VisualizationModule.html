023package org.opt4j.viewer;
024
025import org.opt4j.core.common.logger.OutputModule;
026
027import com.google.inject.Binder;
028import com.google.inject.multibindings.Multibinder;
029
030/**
031 * The {@link VisualizationModule}.
032 * 
033 * @author lukasiewycz
034 * 
035 */
036public abstract class VisualizationModule extends OutputModule {
037
038        /**
039         * Add a {@link ToolBarService}.
040         * 
041         * @param toolBarService
042         *            the tool bar service to be added
043         */
044        public void addToolBarService(Class<? extends ToolBarService> toolBarService) {
045                Multibinder<ToolBarService> multibinder = Multibinder.newSetBinder(binder(), ToolBarService.class);
046                multibinder.addBinding().to(toolBarService);
047        }
048
049        /**
050         * Add a {@link ToolBarService}.
051         * 
052         * @param binder
053         *            the binder
054         * @param toolBarService
055         *            the tool bar service to be added
056         */
057        public static void addToolBarService(Binder binder, Class<? extends ToolBarService> toolBarService) {
058                Multibinder<ToolBarService> multibinder = Multibinder.newSetBinder(binder, ToolBarService.class);
059                multibinder.addBinding().to(toolBarService);
060        }
061
062        /**
063         * Add a {@link IndividualMouseListener}.
064         * 
065         * @param individualMouseListener
066         *            the individual mouse listener to be added
067         */
068        public void addIndividualMouseListener(Class<? extends IndividualMouseListener> individualMouseListener) {
069                Multibinder<IndividualMouseListener> multibinder = Multibinder.newSetBinder(binder(),
070                                IndividualMouseListener.class);
071                multibinder.addBinding().to(individualMouseListener);
072        }
073
074        /**
075         * Add a {@link IndividualMouseListener}.
076         * 
077         * @param binder
078         *            the binder
079         * @param individualMouseListener
080         *            the individual mouse listener to be added
081         */
082        public static final void addIndividualMouseListener(Binder binder,
083                        Class<? extends IndividualMouseListener> individualMouseListener) {
084                Multibinder<IndividualMouseListener> multibinder = Multibinder.newSetBinder(binder,
085                                IndividualMouseListener.class);
086                multibinder.addBinding().to(individualMouseListener);
087        }
088
089        /*
090         * (non-Javadoc)
091         * 
092         * @see com.google.inject.AbstractModule#configure()
093         */
094        @Override
095        protected void configure() {
096                super.configure();
097
098                multi(ToolBarService.class);
099                multi(IndividualMouseListener.class);
100        }
101
102}
