023package org.opt4j.viewer;
024
025import java.awt.BorderLayout;
026import java.awt.Color;
027import java.beans.PropertyVetoException;
028import java.util.HashMap;
029import java.util.Map;
030
031import javax.swing.Icon;
032import javax.swing.JComponent;
033import javax.swing.JDesktopPane;
034import javax.swing.JInternalFrame;
035import javax.swing.JPanel;
036import javax.swing.SwingUtilities;
037import javax.swing.event.InternalFrameAdapter;
038import javax.swing.event.InternalFrameEvent;
039
040import org.opt4j.core.config.Icons;
041
042/**
043 * The {@link Viewport} is a desktop for {@link Widget}s.
044 * 
045 * @see Widget
046 * @author lukasiewycz
047 * 
048 */
049public class Viewport {
050
051        protected final JDesktopPane desktop = new JDesktopPane();
052
053        protected final JPanel panel = new JPanel();
054
055        protected final static int OFFSET = 30;
056
057        protected final Map<Widget, JInternalFrame> widgets = new HashMap<Widget, JInternalFrame>();
058
059        /**
060         * Constructs a {@link Viewport} for {@link Widget}s.
061         * 
062         */
063        public Viewport() {
064
065        }
066
067        /**
068         * Initialization. This method has to be called once after construction.
069         */
070        public void init() {
071                desktop.setDragMode(JDesktopPane.OUTLINE_DRAG_MODE);
072                panel.setLayout(new BorderLayout());
073                panel.add(desktop, BorderLayout.CENTER);
074
075                desktop.setBackground(Color.WHITE);
076        }
077
078        /**
079         * Adds a widget.
080         * 
081         * @param widget
082         *            the widget to be added
083         */
084        public void addWidget(final Widget widget) {
085                final JInternalFrame frame;
086
087                if (!widgets.keySet().contains(widget)) {
088                        frame = createInternalFrame(widget);
089                        addToDesktop(frame);
090                } else {
091                        frame = widgets.get(widget);
092                }
093                SwingUtilities.invokeLater(new Runnable() {
094                        @Override
095                        public void run() {
096                                if (frame.isIcon()) {
097                                        try {
098                                                frame.setIcon(false);
099                                        } catch (PropertyVetoException e) {
100                                                // ignore
101                                        }
102                                }
103                                frame.setVisible(true);
104                                frame.moveToFront();
105                        }
106                });
107        }
108
109        /**
110         * Creates a widget as a {@link JInternalFrame}.
111         * 
112         * @param widget
113         *            the widget
114         * @return the constructed JInternalFrame
115         */
116        protected JInternalFrame createInternalFrame(final Widget widget) {
117                final JPanel panel = widget.getPanel();
118
119                final String frameTitle;
120                final Icon frameIcon;
121
122                boolean resizable = true;
123                boolean closable = true;
124                boolean maximizable = true;
125                boolean iconifiable = true;
126
127                WidgetParameters parameters = widget.getClass().getAnnotation(WidgetParameters.class);
128                if (parameters != null) {
129                        frameTitle = parameters.title();
130                        if (!parameters.icon().equals("")) {
131                                frameIcon = Icons.getIcon(parameters.icon());
132                        } else {
133                                frameIcon = null;
134                        }
135                        resizable = parameters.resizable();
136                        closable = parameters.closable();
137                        maximizable = parameters.maximizable();
138                        iconifiable = parameters.iconifiable();
139                } else {
140                        frameTitle = null;
141                        frameIcon = null;
142                }
143
144                final JInternalFrame inFrame = new JInternalFrame(frameTitle, resizable, closable, maximizable, iconifiable);
145                widget.init(this);
146
147                SwingUtilities.invokeLater(new Runnable() {
148                        @Override
149                        public void run() {
150                                inFrame.addInternalFrameListener(new InternalFrameAdapter() {
151                                        @Override
152                                        public void internalFrameClosed(InternalFrameEvent arg0) {
153                                                widgets.remove(widget);
154                                        }
155                                });
156                                inFrame.setFrameIcon(frameIcon);
157                                inFrame.getContentPane().setLayout(new BorderLayout());
158                                inFrame.getContentPane().add(panel, BorderLayout.CENTER);
159
160                                int windows = desktop.getAllFrames().length;
161                                windows = windows % 6;
162                                inFrame.setLocation(OFFSET * windows, OFFSET * windows);
163                        }
164                });
165
166                widgets.put(widget, inFrame);
167
168                return inFrame;
169        }
170
171        /**
172         * Adds a {@link JInternalFrame}.
173         * 
174         * @param inFrame
175         *            the JInternalFrame to be added
176         */
177        public void addToDesktop(final JInternalFrame inFrame) {
178                SwingUtilities.invokeLater(new Runnable() {
179                        @Override
180                        public void run() {
181                                inFrame.pack();
182                                desktop.add(inFrame);
183                        }
184                });
185        }
186
187        /**
188         * Returns the {@link JInternalFrame} of a {@link Widget} if it exists and
189         * {@code null} otherwise.
190         * 
191         * @param widget
192         *            the widget
193         * @return the internal frame
194         */
195        public JInternalFrame getInternalFrame(Widget widget) {
196                return widgets.get(widget);
197        }
198
199        /**
200         * Returns the component.
201         * 
202         * @return the component
203         */
204        public JComponent get() {
205                return panel;
206        }
207
208}
