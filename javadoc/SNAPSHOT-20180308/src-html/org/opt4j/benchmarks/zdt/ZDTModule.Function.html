024package org.opt4j.benchmarks.zdt;
025
026import org.opt4j.benchmarks.BinaryCopyDecoder;
027import org.opt4j.benchmarks.BinaryCreator;
028import org.opt4j.benchmarks.BinaryToDoubleDecoder;
029import org.opt4j.benchmarks.Bits;
030import org.opt4j.benchmarks.DoubleCopyDecoder;
031import org.opt4j.benchmarks.DoubleCreator;
032import org.opt4j.benchmarks.N;
033import org.opt4j.core.config.annotations.Info;
034import org.opt4j.core.config.annotations.Order;
035import org.opt4j.core.config.annotations.Required;
036import org.opt4j.core.problem.Creator;
037import org.opt4j.core.problem.Decoder;
038import org.opt4j.core.problem.Evaluator;
039import org.opt4j.core.problem.ProblemModule;
040
041/**
042 * The {@link ZDTModule} configures the ZDT benchmarks, see "Comparison of
043 * Multiobjective Evolutionary Algorithms: Empirical Results, Eckart Zitzler,
044 * Kalyanmoy Deb, Lothar Thiele, Evolutionary Computation Volume 8 , Issue 2
045 * (June 2000)".
046 * 
047 * @author lukasiewycz
048 * 
049 */
050@Info("ZDT Problem Suite.")
051public class ZDTModule extends ProblemModule {
052
053        @Order(0)
054        @Info("The ZDT function.")
055        protected Function function = Function.ZDT1;
056
057        @Order(1)
058        @Info("The size of the search space.")
059        protected int n = 30;
060
061        @Order(2)
062        @Required(property = "function", elements = { "ZDT1", "ZDT2", "ZDT3", "ZDT4", "ZDT6" })
063        @Info("The encoding of the optimization problem.")
064        protected Encoding encoding = Encoding.BINARY;
065
066        @Order(3)
067        @Required(property = "encoding", elements = { "BINARY" })
068        @Info("The number of bits per double value.")
069        protected int bits = 30;
070
071        /**
072         * The used ZDT function.
073         * 
074         * @author lukasiewycz
075         */
076        public enum Function {
077                /**
078                 * Use the ZDT1 function.
079                 */
080                ZDT1,
081                /**
082                 * Use the ZDT2 function.
083                 */
084                ZDT2,
085                /**
086                 * Use the ZDT3 function.
087                 */
088                ZDT3,
089                /**
090                 * Use the ZDT4 function.
091                 */
092                ZDT4,
093                /**
094                 * Use the ZDT5 function.
095                 */
096                ZDT5,
097                /**
098                 * Use the ZDT6 function.
099                 */
100                ZDT6;
101        }
102
103        /**
104         * The used encoding type.
105         * 
106         * @author lukasiewycz
107         */
108        public enum Encoding {
109                /**
110                 * Use double values.
111                 */
112                DOUBLE,
113                /**
114                 * Use binary values.
115                 */
116                BINARY;
117        }
118
119        /**
120         * Returns the number of bits per double value.
121         * 
122         * @return the bits
123         */
124        public int getBits() {
125                return bits;
126        }
127
128        /**
129         * Sets the number of bits per double value.
130         * 
131         * @param bits
132         *            the bits to set
133         */
134        public void setBits(int bits) {
135                this.bits = bits;
136        }
137
138        /**
139         * Returns the encoding (double or binary).
140         * 
141         * @return the encoding
142         */
143        public Encoding getEncoding() {
144                return encoding;
145        }
146
147        /**
148         * Set the encoding (double or binary).
149         * 
150         * @param encoding
151         *            the encoding
152         */
153        public void setEncoding(Encoding encoding) {
154                this.encoding = encoding;
155                validate();
156        }
157
158        /**
159         * Returns the ZDT function.
160         * 
161         * @return the ZDT function
162         */
163        public Function getFunction() {
164                return function;
165        }
166
167        /**
168         * Sets the ZDT function.
169         * 
170         * @param function
171         *            the ZDT function
172         */
173        public void setFunction(Function function) {
174                this.function = function;
175                validate();
176
177                switch (function) {
178                case ZDT1:
179                        n = 30;
180                        break;
181                case ZDT2:
182                        n = 30;
183                        break;
184                case ZDT3:
185                        n = 30;
186                        break;
187                case ZDT4:
188                        n = 10;
189                        break;
190                case ZDT5:
191                        n = 11;
192                        break;
193                case ZDT6:
194                        n = 10;
195                        break;
196                default:
197                        n = 30;
198                }
199
200        }
201
202        /**
203         * Returns the n value.
204         * 
205         * @return the n value
206         */
207        public int getN() {
208                return n;
209        }
210
211        /**
212         * Sets the n value.
213         * 
214         * @param n
215         *            the n value
216         */
217        public void setN(int n) {
218                this.n = n;
219        }
220
221        /**
222         * Validates all settings. In particular, ZDT5 can be encoded binary only.
223         */
224        protected void validate() {
225                if (function == Function.ZDT5 && encoding == Encoding.DOUBLE) {
226                        encoding = Encoding.BINARY;
227                }
228        }
229
230        /*
231         * (non-Javadoc)
232         * 
233         * @see org.opt4j.start.Opt4JModule#config()
234         */
235        @Override
236        public void config() {
237                validate();
238
239                Class<? extends Creator<?>> creator = null;
240                Class<? extends Decoder<?, ?>> decoder = null;
241                Class<? extends Evaluator<?>> evaluator = null;
242
243                switch (encoding) {
244                case BINARY:
245                        if (function == Function.ZDT5) {
246                                creator = ZDT5BinaryCreator.class;
247                                decoder = BinaryCopyDecoder.class;
248                        } else {
249                                creator = BinaryCreator.class;
250                                decoder = BinaryToDoubleDecoder.class;
251                        }
252                        break;
253                case DOUBLE:
254                        // only for ZDT1,ZDT2,ZDT3,ZDT4,ZDT6
255                        creator = DoubleCreator.class;
256                        decoder = DoubleCopyDecoder.class;
257                        break;
258                }
259
260                switch (function) {
261                case ZDT1:
262                        evaluator = ZDT1.class;
263                        break;
264                case ZDT2:
265                        evaluator = ZDT2.class;
266                        break;
267                case ZDT3:
268                        evaluator = ZDT3.class;
269                        break;
270                case ZDT4:
271                        evaluator = ZDT4.class;
272                        break;
273                case ZDT5:
274                        evaluator = ZDT5.class;
275                        break;
276                case ZDT6:
277                        evaluator = ZDT6.class;
278                        break;
279                default:
280                }
281
282                bindConstant(N.class).to(n);
283                bindConstant(Bits.class).to(bits);
284                bindProblem(creator, decoder, evaluator);
285
286        }
287
288}
