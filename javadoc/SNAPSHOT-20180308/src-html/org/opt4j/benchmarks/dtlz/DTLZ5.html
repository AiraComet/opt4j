024package org.opt4j.benchmarks.dtlz;
025
026import static java.lang.Math.PI;
027import static java.lang.Math.cos;
028import static java.lang.Math.sin;
029
030import java.util.ArrayList;
031import java.util.Collections;
032import java.util.List;
033
034import org.opt4j.benchmarks.M;
035import org.opt4j.benchmarks.N;
036
037import com.google.inject.Inject;
038
039/**
040 * The {@link DTLZ5} function.
041 * 
042 * @author lukasiewycz
043 * 
044 */
045public class DTLZ5 extends DTLZ2 {
046
047        /**
048         * Constructs a {@link DTLZ5} function.
049         * 
050         * @param m
051         *            the number of objective functions.
052         * @param n
053         *            the size of the search space (length of the x vector).
054         */
055        @Inject
056        public DTLZ5(@M int m, @N int n) {
057                super(m, n);
058        }
059
060        /*
061         * (non-Javadoc)
062         * 
063         * @see org.opt4j.benchmark.dtlz.DTLZEvaluator#f(java.util.List, double)
064         */
065        @Override
066        protected List<Double> f(List<Double> x, double g) {
067                List<Double> f = new ArrayList<Double>();
068                List<Double> theta = theta(x, g);
069
070                double a = (1.0 + g);
071
072                for (int i = 0; i < m; i++) {
073                        if (i > 0) {
074                                a *= cos(theta.get(i - 1));
075                        }
076                        double value = a;
077                        if (i < x.size()) {
078                                value *= sin(theta.get(i));
079                        }
080                        f.add(value);
081                }
082
083                Collections.reverse(f);
084                return f;
085        }
086
087        /**
088         * The theta function.
089         * 
090         * @param x
091         *            the sublist
092         * @param g
093         *            the result of g(x)
094         * @return theta_i
095         */
096        protected List<Double> theta(List<Double> x, double g) {
097                List<Double> theta = new ArrayList<Double>();
098
099                theta.add(x.get(0) * PI / 2);
100
101                double t = PI / (4 * (1 + g));
102                for (int i = 1; i < x.size(); i++) {
103                        theta.add(t * (1 + 2 * g * x.get(i)));
104                }
105                return theta;
106        }
107}
