024package org.opt4j.benchmarks.dtlz;
025
026import org.opt4j.benchmarks.BinaryCreator;
027import org.opt4j.benchmarks.BinaryToDoubleDecoder;
028import org.opt4j.benchmarks.Bits;
029import org.opt4j.benchmarks.DoubleCopyDecoder;
030import org.opt4j.benchmarks.DoubleCreator;
031import org.opt4j.benchmarks.M;
032import org.opt4j.benchmarks.N;
033import org.opt4j.core.config.annotations.Info;
034import org.opt4j.core.config.annotations.Order;
035import org.opt4j.core.config.annotations.Required;
036import org.opt4j.core.problem.Creator;
037import org.opt4j.core.problem.Decoder;
038import org.opt4j.core.problem.Evaluator;
039import org.opt4j.core.problem.ProblemModule;
040import org.opt4j.core.start.Constant;
041
042/**
043 * Module for the DTLZ benchmarks as proposed in "Scalable multi-objective
044 * optimization test problems, Kalyanmoy Deb,Lothar Thiele, Marco Laumanns,
045 * Eckart Zitzler, Proc. Congress Evolutionary Computation Volume 1 (May 2002)".
046 * 
047 * @author lukasiewycz
048 * 
049 */
050@Info("DTLZ Problem Suite.  (n = m + k - 1)")
051public class DTLZModule extends ProblemModule {
052
053        @Order(0)
054        @Info("The DTLZ function")
055        protected Function function = Function.DTLZ1;
056
057        @Required(property = "function", elements = { "DTLZ4" })
058        @Order(1)
059        @Constant(value = "alpha", namespace = DTLZ4.class)
060        protected double alpha = 100.0;
061
062        @Order(2)
063        @Info("The number of objectives.")
064        protected int m = 3;
065
066        @Order(3)
067        @Info("The size of the search space.")
068        protected int n = 7;
069
070        @Order(4)
071        @Info("The k value.")
072        protected int k = 5;
073
074        @Order(10)
075        @Info("The encoding of the optimization problem.")
076        protected Encoding encoding = Encoding.DOUBLE;
077
078        @Required(property = "encoding", elements = { "BINARY" })
079        @Info("The number of bits per double value.")
080        protected int bits = 30;
081
082        /**
083         * The used DTZL1 function.
084         * 
085         * @author lukasiewycz
086         */
087        public enum Function {
088                /**
089                 * Use the DTZL1 function.
090                 */
091                DTLZ1,
092                /**
093                 * Use the DTZL2 function.
094                 */
095                DTLZ2,
096                /**
097                 * Use the DTZL3 function.
098                 */
099                DTLZ3,
100                /**
101                 * Use the DTZL4 function.
102                 */
103                DTLZ4,
104                /**
105                 * Use the DTZL5 function.
106                 */
107                DTLZ5,
108                /**
109                 * Use the DTZL6 function.
110                 */
111                DTLZ6,
112                /**
113                 * Use the DTZL7 function.
114                 */
115                DTLZ7;
116        }
117
118        /**
119         * The used encoding type.
120         * 
121         * @author lukasiewycz
122         */
123        public enum Encoding {
124                /**
125                 * Use double values.
126                 */
127                DOUBLE,
128                /**
129                 * Use binary values.
130                 */
131                BINARY;
132        }
133
134        /**
135         * Returns the number of bits per double value.
136         * 
137         * @return the bits
138         */
139        public int getBits() {
140                return bits;
141        }
142
143        /**
144         * Sets the number of bits per double value.
145         * 
146         * @param bits
147         *            the bits to set
148         */
149        public void setBits(int bits) {
150                this.bits = bits;
151        }
152
153        /**
154         * Returns the encoding (double or binary).
155         * 
156         * @return the encoding
157         */
158        public Encoding getEncoding() {
159                return encoding;
160        }
161
162        /**
163         * Set the encoding (double or binary).
164         * 
165         * @param encoding
166         *            the encoding
167         */
168        public void setEncoding(Encoding encoding) {
169                this.encoding = encoding;
170        }
171
172        /**
173         * Returns the DTZL function.
174         * 
175         * @return the DTZL function
176         */
177        public Function getFunction() {
178                return function;
179        }
180
181        /**
182         * Sets the ZDT function.
183         * 
184         * @param function
185         *            the ZDT function
186         */
187        public void setFunction(Function function) {
188                this.function = function;
189
190                switch (function) {
191                default: // DTLZ1
192                        n = 7;
193                        m = 3;
194                        k = 5;
195                        break;
196                case DTLZ2:
197                case DTLZ3:
198                case DTLZ4:
199                case DTLZ5:
200                case DTLZ6:
201                        n = 12;
202                        m = 3;
203                        k = 10;
204                        break;
205                case DTLZ7:
206                        n = 22;
207                        m = 3;
208                        k = 20;
209                        break;
210                }
211
212        }
213
214        /**
215         * Returns the n value.
216         * 
217         * @return the n value
218         */
219        public int getN() {
220                return n;
221        }
222
223        /**
224         * Sets the n value.
225         * 
226         * @param n
227         *            the n value
228         */
229        public void setN(int n) {
230                this.n = n;
231                this.k = n - m + 1;
232        }
233
234        /**
235         * Returns the m value.
236         * 
237         * @return the m
238         */
239        public int getM() {
240                return m;
241        }
242
243        /**
244         * Sets the m value.
245         * 
246         * @param m
247         *            the m to set
248         */
249        public void setM(int m) {
250                this.m = m;
251                this.k = n - m + 1;
252        }
253
254        /**
255         * Returns the k value.
256         * 
257         * @return the k
258         */
259        public int getK() {
260                return k;
261        }
262
263        /**
264         * Sets the k value.
265         * 
266         * @param k
267         *            the k to set
268         */
269        public void setK(int k) {
270                this.k = k;
271                this.n = m + k - 1;
272        }
273
274        /**
275         * Returns the alpha.
276         * 
277         * @return the alpha
278         */
279        public double getAlpha() {
280                return alpha;
281        }
282
283        /**
284         * Sets the alpha.
285         * 
286         * @param alpha
287         *            the alpha to set
288         */
289        public void setAlpha(double alpha) {
290                this.alpha = alpha;
291        }
292
293        /*
294         * (non-Javadoc)
295         * 
296         * @see org.opt4j.start.Opt4JModule#config()
297         */
298        @Override
299        public void config() {
300                n = m + k - 1;
301
302                Class<? extends Creator<?>> creator = null;
303                Class<? extends Decoder<?, ?>> decoder = null;
304                Class<? extends Evaluator<?>> evaluator = null;
305
306                switch (encoding) {
307                case BINARY:
308                        creator = BinaryCreator.class;
309                        decoder = BinaryToDoubleDecoder.class;
310                        break;
311                case DOUBLE:
312                        creator = DoubleCreator.class;
313                        decoder = DoubleCopyDecoder.class;
314                        break;
315                }
316
317                switch (function) {
318                case DTLZ1:
319                        evaluator = DTLZ1.class;
320                        break;
321                case DTLZ2:
322                        evaluator = DTLZ2.class;
323                        break;
324                case DTLZ3:
325                        evaluator = DTLZ3.class;
326                        break;
327                case DTLZ4:
328                        evaluator = DTLZ4.class;
329                        break;
330                case DTLZ5:
331                        evaluator = DTLZ5.class;
332                        break;
333                case DTLZ6:
334                        evaluator = DTLZ6.class;
335                        break;
336                case DTLZ7:
337                        evaluator = DTLZ7.class;
338                        break;
339                default:
340                        evaluator = DTLZ1.class;
341                }
342
343                bindConstant(N.class).to(n);
344                bindConstant(M.class).to(m);
345                bindConstant(Bits.class).to(bits);
346                bindProblem(creator, decoder, evaluator);
347
348        }
349
350}
