024package org.opt4j.benchmarks.dtlz;
025
026import static java.lang.Math.PI;
027import static java.lang.Math.cos;
028import static java.lang.Math.sin;
029
030import java.util.ArrayList;
031import java.util.Collections;
032import java.util.List;
033
034import org.opt4j.benchmarks.M;
035import org.opt4j.benchmarks.N;
036
037import com.google.inject.Inject;
038
039/**
040 * The {@link DTLZ2} function.
041 * 
042 * @author lukasiewycz
043 * 
044 */
045public class DTLZ2 extends DTLZEvaluator {
046
047        /**
048         * Constructs a {@link DTLZ2} function.
049         * 
050         * @param m
051         *            the number of objective functions.
052         * @param n
053         *            the size of the search space (length of the x vector).
054         */
055        @Inject
056        public DTLZ2(@M int m, @N int n) {
057                super(m, n);
058        }
059
060        /*
061         * (non-Javadoc)
062         * 
063         * @see org.opt4j.benchmark.dtlz.DTLZEvaluator#f(java.util.List, double)
064         */
065        @Override
066        protected List<Double> f(List<Double> x, double g) {
067                List<Double> f = new ArrayList<Double>();
068
069                double a = (1.0 + g);
070
071                for (int i = 0; i < m; i++) {
072                        if (i > 0) {
073                                a *= cos(PI * x.get(i - 1) / 2);
074                        }
075                        double value = a;
076                        if (i < x.size()) {
077                                value *= sin(PI * x.get(i) / 2);
078                        }
079                        f.add(value);
080                }
081
082                Collections.reverse(f);
083                return f;
084        }
085
086        /*
087         * (non-Javadoc)
088         * 
089         * @see org.opt4j.benchmark.dtlz.DTLZEvaluator#g(java.util.List)
090         */
091        @Override
092        protected double g(List<Double> x) {
093                double g = 0;
094                for (double e : x) {
095                        g += (e - 0.5) * (e - 0.5);
096                }
097                return g;
098        }
099
100}
