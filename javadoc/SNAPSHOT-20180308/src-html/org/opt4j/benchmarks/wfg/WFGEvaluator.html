024package org.opt4j.benchmarks.wfg;
025
026import static java.lang.Math.max;
027
028import java.util.ArrayList;
029import java.util.List;
030
031import org.opt4j.benchmarks.DoubleString;
032import org.opt4j.benchmarks.K;
033import org.opt4j.benchmarks.M;
034import org.opt4j.core.Objective;
035import org.opt4j.core.Objectives;
036import org.opt4j.core.problem.Evaluator;
037
038import com.google.inject.Inject;
039
040/**
041 * The {@link WFGEvaluator}. All WFG test functions are derived from this class.
042 * 
043 * @author lukasiewycz
044 * 
045 */
046public abstract class WFGEvaluator implements Evaluator<DoubleString> {
047
048        protected final int M;
049        protected final int k;
050
051        protected final List<Objective> keys = new ArrayList<Objective>();
052
053        /**
054         * Constructs a {@link WFGEvaluator}.
055         * 
056         * @param k
057         *            the position parameters
058         * @param M
059         *            the number of objectives
060         */
061        @Inject
062        public WFGEvaluator(@K int k, @M int M) {
063                this.M = M;
064                this.k = k;
065
066                for (int i = 0; i < M; i++) {
067                        keys.add(new Objective("f" + (i + 1)));
068                }
069        }
070
071        public abstract List<Double> f(List<Double> y);
072
073        /*
074         * (non-Javadoc)
075         * 
076         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
077         */
078        @Override
079        public Objectives evaluate(DoubleString phenotype) {
080                List<Double> f = f(phenotype);
081
082                assert (f.size() == keys.size());
083
084                Objectives objectives = new Objectives();
085                for (int i = 0; i < M; i++) {
086                        objectives.add(keys.get(i), f.get(i));
087                }
088                return objectives;
089        }
090
091        protected static List<Boolean> createA(final int M, final boolean degenerate) {
092                assert (M >= 2);
093
094                List<Boolean> A = new ArrayList<Boolean>();
095                for (int i = 0; i < M - 1; i++) {
096                        A.add(!degenerate || i == 0);
097                }
098                return A;
099        }
100
101        protected static List<Double> calculateF(final List<Double> x, final List<Double> h) {
102
103                assert (x.size() == h.size());
104
105                final int M = h.size();
106
107                List<Double> S = new ArrayList<Double>();
108                for (int m = 1; m <= M; m++) {
109                        S.add(m * 2.0);
110                }
111
112                return caculateF(x, h, S);
113        }
114
115        protected static List<Double> normalizeZ(final List<Double> z, final List<Double> z_max) {
116                final List<Double> result = new ArrayList<Double>();
117
118                for (int i = 0; i < z.size(); i++) {
119                        assert (z.get(i) >= 0.0);
120                        assert (z.get(i) <= z_max.get(i));
121                        assert (z_max.get(i) > 0.0);
122
123                        result.add(z.get(i) / z_max.get(i));
124                }
125
126                return result;
127        }
128
129        protected static List<Double> calculateX(final List<Double> t_p, final List<Boolean> A) {
130                final int size = t_p.size();
131
132                assert (size != 0);
133                assert (A.size() == t_p.size() - 1);
134
135                final List<Double> result = new ArrayList<Double>();
136
137                for (int i = 0; i < t_p.size() - 1; i++) {
138                        int A_i = A.get(i) ? 1 : 0;
139
140                        final double tmp1 = max(t_p.get(size - 1), A_i);
141                        result.add(tmp1 * (t_p.get(i) - 0.5) + 0.5);
142                }
143
144                result.add(t_p.get(size - 1));
145
146                return result;
147        }
148
149        protected static List<Double> caculateF(final List<Double> x, final List<Double> h, final List<Double> S) {
150                assert (x.size() == h.size());
151                assert (h.size() == S.size());
152
153                final List<Double> result = new ArrayList<Double>();
154
155                for (int i = 0; i < h.size(); i++) {
156                        assert (S.get(i) > 0.0);
157
158                        result.add(x.get(x.size() - 1) + S.get(i) * h.get(i));
159                }
160
161                return result;
162        }
163
164}
