024package org.opt4j.benchmarks.wfg;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.opt4j.benchmarks.K;
030import org.opt4j.benchmarks.M;
031
032import com.google.inject.Inject;
033
034/**
035 * The {@link WFG8} benchmark function.
036 * 
037 * @author lukasiewycz
038 * 
039 */
040public class WFG8 extends WFGEvaluator {
041
042        /**
043         * Constructs a {@link WFG8} benchmark function.
044         * 
045         * @param k
046         *            the position parameters
047         * @param M
048         *            the number of objectives
049         */
050        @Inject
051        public WFG8(@K int k, @M int M) {
052                super(k, M);
053        }
054
055        protected static List<Double> t1(final List<Double> y, final int k) {
056                final int n = y.size();
057
058                assert (k >= 1);
059                assert (k < n);
060
061                List<Double> w = new ArrayList<Double>();
062                for (int i = 0; i < n; i++) {
063                        w.add(1.0);
064                }
065
066                List<Double> t = new ArrayList<Double>();
067
068                for (int i = 0; i < k; i++) {
069                        t.add(y.get(i));
070                }
071
072                for (int i = k; i < n; i++) {
073                        final List<Double> y_sub = y.subList(0, i);
074                        final List<Double> w_sub = w.subList(0, i);
075
076                        final double u = WFGTransFunctions.r_sum(y_sub, w_sub);
077
078                        t.add(WFGTransFunctions.b_param(y.get(i), u, 0.98 / 49.98, 0.02, 50));
079                }
080
081                return t;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see org.opt4j.benchmark.wfg.WFGEvaluator#f(java.util.List)
088         */
089        @Override
090        public List<Double> f(List<Double> y) {
091                y = WFG8.t1(y, k);
092                y = WFG1.t1(y, k);
093                y = WFG2.t3(y, k, M);
094
095                return WFG4.shape(y);
096        }
097}
