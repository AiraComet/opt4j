024package org.opt4j.benchmarks.wfg;
025
026import org.opt4j.benchmarks.BinaryCreator;
027import org.opt4j.benchmarks.BinaryToDoubleDecoder;
028import org.opt4j.benchmarks.Bits;
029import org.opt4j.benchmarks.DoubleCopyDecoder;
030import org.opt4j.benchmarks.DoubleCreator;
031import org.opt4j.benchmarks.K;
032import org.opt4j.benchmarks.M;
033import org.opt4j.benchmarks.N;
034import org.opt4j.core.config.annotations.Info;
035import org.opt4j.core.config.annotations.Order;
036import org.opt4j.core.config.annotations.Required;
037import org.opt4j.core.problem.Creator;
038import org.opt4j.core.problem.Decoder;
039import org.opt4j.core.problem.Evaluator;
040import org.opt4j.core.problem.ProblemModule;
041
042/**
043 * Module for the WFG (Walking Fish Group) benchmarks, see "A Scalable
044 * Multi-objective Test Problem Toolkit, Simon Huband, Luigi Barone, R. Lyndon
045 * While, and Philip Hingston (EMO 2005)".
046 * 
047 * @author lukasiewycz
048 * 
049 */
050@Info("WFG Problem Suite. The number of search variables is n=k+l.")
051public class WFGModule extends ProblemModule {
052
053        @Order(0)
054        @Info("The WFG function.")
055        protected Function function = Function.WFG1;
056
057        @Order(2)
058        @Info("The number of objectives.")
059        protected int m = 2;
060
061        @Order(3)
062        @Info("The number of distance parameters. (suggested: l=4 for m=2, l=2*(m-1) for m>2)")
063        protected int k = 6;
064
065        @Order(4)
066        @Info("The number of position parameters. (k%(m-1)=0)")
067        protected int l = 4;
068
069        @Required(property = "encoding", elements = { "BINARY" })
070        @Info("The number of bits per double value.")
071        protected int bits = 30;
072
073        @Order(10)
074        @Info("The encoding of the optimization problem.")
075        protected Encoding encoding = Encoding.DOUBLE;
076
077        /**
078         * The used WFG function.
079         * 
080         * @author lukasiewycz
081         */
082        public enum Function {
083                /**
084                 * Use the WFG1 function.
085                 */
086                WFG1,
087                /**
088                 * Use the WFG2 function.
089                 */
090                WFG2,
091                /**
092                 * Use the WFG3 function.
093                 */
094                WFG3,
095                /**
096                 * Use the WFG4 function.
097                 */
098                WFG4,
099                /**
100                 * Use the WFG5 function.
101                 */
102                WFG5,
103                /**
104                 * Use the WFG6 function.
105                 */
106                WFG6,
107                /**
108                 * Use the WFG7 function.
109                 */
110                WFG7,
111                /**
112                 * Use the WFG8 function.
113                 */
114                WFG8,
115                /**
116                 * Use the WFG9 function.
117                 */
118                WFG9,
119                /**
120                 * Use the I1 function.
121                 */
122                I1,
123                /**
124                 * Use the I2 function.
125                 */
126                I2,
127                /**
128                 * Use the I3 function.
129                 */
130                I3,
131                /**
132                 * Use the I4 function.
133                 */
134                I4,
135                /**
136                 * Use the I5 function.
137                 */
138                I5;
139        }
140
141        /**
142         * The used encoding type.
143         * 
144         * @author lukasiewycz
145         */
146        public enum Encoding {
147                /**
148                 * Use double values.
149                 */
150                DOUBLE,
151                /**
152                 * Use binary values.
153                 */
154                BINARY;
155        }
156
157        /**
158         * Returns the number of bits per double value.
159         * 
160         * @return the bits
161         */
162        public int getBits() {
163                return bits;
164        }
165
166        /**
167         * Sets the number of bits per double value.
168         * 
169         * @param bits
170         *            the bits to set
171         */
172        public void setBits(int bits) {
173                this.bits = bits;
174        }
175
176        /**
177         * Returns the encoding (double or binary).
178         * 
179         * @return the encoding
180         */
181        public Encoding getEncoding() {
182                return encoding;
183        }
184
185        /**
186         * Set the encoding (double or binary).
187         * 
188         * @param encoding
189         *            the encoding
190         */
191        public void setEncoding(Encoding encoding) {
192                this.encoding = encoding;
193        }
194
195        /**
196         * Returns the WFG function.
197         * 
198         * @return the WFG function
199         */
200        public Function getFunction() {
201                return function;
202        }
203
204        /**
205         * Sets the WFG function.
206         * 
207         * @param function
208         *            the WFG function
209         */
210        public void setFunction(Function function) {
211                this.function = function;
212
213        }
214
215        /**
216         * Returns the l value.
217         * 
218         * @return the l value
219         */
220        public int getL() {
221                return l;
222        }
223
224        /**
225         * Sets the l value.
226         * 
227         * @param l
228         *            the l value
229         */
230        public void setL(int l) {
231                assert (l > 0);
232
233                if (function == Function.WFG2 || function == Function.WFG3) {
234                        this.l = (int) Math.ceil((double) l / 2) * 2;
235                } else {
236                        this.l = l;
237                }
238
239        }
240
241        /**
242         * Returns the m value.
243         * 
244         * @return the m
245         */
246        public int getM() {
247                return m;
248        }
249
250        /**
251         * Sets the m value.
252         * 
253         * @param m
254         *            the m to set
255         */
256        public void setM(int m) {
257                assert (m >= 2);
258
259                this.m = m;
260                if (m == 2) {
261                        setL(4);
262                } else {
263                        setL(2 * (m - 1));
264                }
265                setK(k);
266        }
267
268        /**
269         * Returns the k value.
270         * 
271         * @return the k
272         */
273        public int getK() {
274                return k;
275        }
276
277        /**
278         * Sets the k value.
279         * 
280         * @param k
281         *            the k to set
282         */
283        public void setK(int k) {
284                assert (k > 0);
285
286                if ((k % (m - 1)) == 0) {
287                        this.k = k;
288                } else {
289                        double div = (double) k / (m - 1);
290                        this.k = (m - 1) * (int) Math.ceil(div);
291                }
292        }
293
294        /*
295         * (non-Javadoc)
296         * 
297         * @see org.opt4j.start.Opt4JModule#config()
298         */
299        @Override
300        public void config() {
301
302                Class<? extends Creator<?>> creator = null;
303                Class<? extends Decoder<?, ?>> decoder = null;
304                Class<? extends Evaluator<?>> evaluator = null;
305
306                switch (encoding) {
307                case BINARY:
308                        creator = BinaryCreator.class;
309                        decoder = BinaryToDoubleDecoder.class;
310                        break;
311                case DOUBLE:
312                        creator = DoubleCreator.class;
313                        decoder = DoubleCopyDecoder.class;
314                        break;
315                }
316
317                switch (function) {
318                default: // WFG1
319                        evaluator = WFG1.class;
320                        break;
321                case WFG2:
322                        evaluator = WFG2.class;
323                        break;
324                case WFG3:
325                        evaluator = WFG3.class;
326                        break;
327                case WFG4:
328                        evaluator = WFG4.class;
329                        break;
330                case WFG5:
331                        evaluator = WFG5.class;
332                        break;
333                case WFG6:
334                        evaluator = WFG6.class;
335                        break;
336                case WFG7:
337                        evaluator = WFG7.class;
338                        break;
339                case WFG8:
340                        evaluator = WFG8.class;
341                        break;
342                case WFG9:
343                        evaluator = WFG9.class;
344                        break;
345                case I1:
346                        evaluator = WFGI1.class;
347                        break;
348                case I2:
349                        evaluator = WFGI2.class;
350                        break;
351                case I3:
352                        evaluator = WFGI3.class;
353                        break;
354                case I4:
355                        evaluator = WFGI4.class;
356                        break;
357                case I5:
358                        evaluator = WFGI5.class;
359                        break;
360                }
361
362                bindConstant(N.class).to(k + l);
363                bindConstant(M.class).to(m);
364                bindConstant(K.class).to(k);
365                bindConstant(Bits.class).to(bits);
366                bindProblem(creator, decoder, evaluator);
367
368        }
369
370}
