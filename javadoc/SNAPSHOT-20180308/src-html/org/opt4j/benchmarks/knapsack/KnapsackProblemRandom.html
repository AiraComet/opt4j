023package org.opt4j.benchmarks.knapsack;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.List;
028import java.util.Random;
029
030import org.opt4j.core.common.random.RandomMersenneTwister;
031import org.opt4j.core.start.Constant;
032
033import com.google.inject.Inject;
034import com.google.inject.Singleton;
035
036/**
037 * The {@link KnapsackProblemRandom}, see "E. Zitzler and L. Thiele:
038 * Multiobjective evolutionary algorithms: A comparative case study and the
039 * strength Pareto approach. IEEE Transactions on Evolutionary Computation, vol.
040 * 3, no. 4, pp. 257-271, Nov. 1999.", defines an instance of a
041 * {@link KnapsackProblem} that is initialized randomly. It initializes profit
042 * and weight values which are randomly created according to Zitzler and Thiele
043 * 1999. This means, each value/knapsack pair is uniformly distributed in
044 * [10,100].
045 * 
046 * @author reimann, lukasiewycz
047 * 
048 */
049@Singleton
050public class KnapsackProblemRandom implements KnapsackProblem {
051
052        protected final List<Knapsack> knapsacks = new ArrayList<Knapsack>();
053        protected final List<Item> items = new ArrayList<Item>();
054
055        /**
056         * Creates a new {@link KnapsackProblemRandom}.
057         * 
058         * @param items
059         *            the number of items
060         * @param knapsacks
061         *            the number of knapsacks
062         * @param seed
063         *            the seed for the initialization
064         */
065        @Inject
066        public KnapsackProblemRandom(@Constant(value = "items", namespace = KnapsackProblemRandom.class) int items,
067                        @Constant(value = "knapsacks", namespace = KnapsackProblemRandom.class) int knapsacks,
068                        @Constant(value = "seed", namespace = KnapsackProblemRandom.class) int seed) {
069
070                for (int i = 0; i < items; i++) {
071                        this.items.add(new Item(i + 1));
072                }
073
074                Random random = new RandomMersenneTwister(seed);
075
076                for (int k = 0; k < knapsacks; k++) {
077                        Knapsack knapsack = new Knapsack(k + 1);
078
079                        double capacity = 0;
080
081                        for (Item item : this.items) {
082                                int profit = random.nextInt(91) + 10;
083                                int weight = random.nextInt(91) + 10;
084
085                                knapsack.setProfit(item, profit);
086                                knapsack.setWeight(item, weight);
087
088                                capacity += 0.5 * weight;
089                        }
090                        knapsack.setCapacity(capacity);
091
092                        this.knapsacks.add(knapsack);
093                }
094        }
095
096        /*
097         * (non-Javadoc)
098         * 
099         * @see org.opt4j.benchmark.knapsack.KnapsackProblem#getKnapsacks()
100         */
101        @Override
102        public Collection<Knapsack> getKnapsacks() {
103                return knapsacks;
104        }
105
106        /*
107         * (non-Javadoc)
108         * 
109         * @see org.opt4j.benchmark.knapsack.KnapsackProblem#getItems()
110         */
111        @Override
112        public Collection<Item> getItems() {
113                return items;
114        }
115
116}
