023package org.opt4j.benchmarks.knapsack;
024
025import java.util.HashMap;
026import java.util.Map;
027
028/**
029 * A knapsack has a given capacity.
030 * 
031 * @author reimann, lukasiewycz
032 * 
033 */
034public class Knapsack {
035
036        protected final int id;
037
038        protected Map<Item, Integer> profit = new HashMap<Item, Integer>();
039        protected Map<Item, Integer> weight = new HashMap<Item, Integer>();
040        protected double capacity = 0;
041
042        /**
043         * Construct a knapsack
044         * 
045         * @param id
046         *            the id of the knapsack
047         */
048        public Knapsack(int id) {
049                super();
050                this.id = id;
051        }
052
053        /**
054         * Sets the capacity of this knapsack.
055         * 
056         * @param capacity
057         *            the capacity
058         */
059        public void setCapacity(double capacity) {
060                this.capacity = capacity;
061        }
062
063        /**
064         * Returns the capacity of this knapsack.
065         * 
066         * @return the capacity
067         */
068        public double getCapacity() {
069                return capacity;
070        }
071
072        /**
073         * Set the profit of an item.
074         * 
075         * @param item
076         *            the item
077         * @param value
078         *            the profit
079         */
080        public void setProfit(Item item, int value) {
081                profit.put(item, value);
082        }
083
084        /**
085         * Get the profit of an item.
086         * 
087         * @param item
088         *            the item
089         * @return the profit
090         */
091        public int getProfit(Item item) {
092                return profit.get(item);
093        }
094
095        /**
096         * Set the weight of an item.
097         * 
098         * @param item
099         *            the item
100         * @param value
101         *            the weight
102         */
103        public void setWeight(Item item, int value) {
104                weight.put(item, value);
105        }
106
107        /**
108         * Get the weight of an item.
109         * 
110         * @param item
111         *            the item
112         * @return the weight
113         */
114        public int getWeight(Item item) {
115                return weight.get(item);
116        }
117
118        /*
119         * (non-Javadoc)
120         * 
121         * @see java.lang.Object#toString()
122         */
123        @Override
124        public String toString() {
125                return "Knapsack" + id;
126        }
127
128}
