023package org.opt4j.benchmarks.knapsack;
024
025import static org.opt4j.core.Objective.INFEASIBLE;
026import static org.opt4j.core.Objective.Sign.MAX;
027
028import java.util.Collection;
029import java.util.HashMap;
030import java.util.Map;
031
032import org.opt4j.core.Objective;
033import org.opt4j.core.Objectives;
034import org.opt4j.core.problem.Evaluator;
035import org.opt4j.core.problem.Priority;
036
037import com.google.inject.Inject;
038
039/**
040 * The {@link KnapsackProfitEvaluator} evaluates a given {@link ItemSelection}
041 * using one {@link Objective} to sum up all knapsack overloads and one
042 * {@link Objective} for the profit of each knapsack.
043 * 
044 * @author reimann, lukasiewycz
045 * 
046 */
047@Priority(0)
048public class KnapsackProfitEvaluator implements Evaluator<ItemSelection> {
049        protected final Map<Knapsack, Objective> profitObjectives = new HashMap<Knapsack, Objective>();
050        protected KnapsackProblem problem;
051
052        /**
053         * Creates a new {@link KnapsackProfitEvaluator}.
054         * 
055         * @param problem
056         *            the problem instance
057         */
058        @Inject
059        public KnapsackProfitEvaluator(KnapsackProblem problem) {
060                this.problem = problem;
061                int i = 0;
062                for (Knapsack knapsack : problem.getKnapsacks()) {
063                        profitObjectives.put(knapsack, new Objective("profit" + i++, MAX));
064                }
065        }
066
067        /*
068         * (non-Javadoc)
069         * 
070         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
071         */
072        @Override
073        public Objectives evaluate(ItemSelection itemSelection) {
074                // evaluate profit
075                Objectives objectives = new Objectives();
076                for (Knapsack knapsack : problem.getKnapsacks()) {
077                        if (itemSelection.isFeasible()) {
078                                objectives.add(profitObjectives.get(knapsack), getProfit(knapsack, itemSelection));
079                        } else {
080                                objectives.add(profitObjectives.get(knapsack), INFEASIBLE);
081                        }
082                }
083                return objectives;
084        }
085
086        /**
087         * Computes the profit for the given items for this knapsack.
088         * 
089         * @param knapsack
090         *            the knapsack
091         * @param items
092         *            the items
093         * @return the profit
094         */
095        protected int getProfit(Knapsack knapsack, Collection<Item> items) {
096                int profit = 0;
097                for (Item i : items) {
098                        profit += knapsack.getProfit(i);
099                }
100                return profit;
101        }
102}
