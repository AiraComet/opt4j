018package org.opt4j.benchmarks.knapsack;
019
020import java.io.BufferedReader;
021import java.io.FileNotFoundException;
022import java.io.FileReader;
023import java.io.IOException;
024import java.util.ArrayList;
025import java.util.Collection;
026import java.util.List;
027import java.util.regex.Matcher;
028import java.util.regex.Pattern;
029
030import org.opt4j.core.start.Constant;
031
032import com.google.inject.Inject;
033
034/**
035 * 
036 * The {@link KnapsackProblemFile} is a {@link KnapsackProblem} that is initialized by a file. The format can be found
037 * on the referenced website.
038 * 
039 * @see <a href=
040 *      "http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/">http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/</a>
041 * 
042 * @author lukasiewycz
043 * 
044 */
045public class KnapsackProblemFile implements KnapsackProblem {
046
047        protected static final String INT = "[+-]?\\d+";
048        protected static final String DOUBLE = "[+-]?\\d+(\\.\\d*)?";
049        protected static final String BLANK = "[\\p{Blank}]*";
050        protected static final String POSINT = "[-]?\\d+"; // skip +
051        protected static final String POSDOUBLE = "[-]?\\d+(\\.\\d*)?"; // skip +
052
053        protected final List<Knapsack> knapsacks = new ArrayList<Knapsack>();
054        protected final List<Item> items = new ArrayList<Item>();
055
056        /**
057         * Constructs a {@link KnapsackProblemFile} with a given file name.
058         * 
059         * @param filename
060         *            the file name
061         */
062        @Inject
063        public KnapsackProblemFile(@Constant(value = "filename", namespace = KnapsackProblemFile.class) String filename) {
064                FileReader fileReader;
065                try {
066                        fileReader = new FileReader(filename);
067                } catch (FileNotFoundException e1) {
068                        throw new IllegalArgumentException("Knapsack problem file not found " + filename, e1);
069                }
070
071                BufferedReader reader = new BufferedReader(fileReader);
072                try {
073
074                        String line;
075
076                        Knapsack knapsack = null;
077                        Item item = null;
078
079                        while ((line = reader.readLine()) != null) {
080
081                                if (line.matches(BLANK + "knapsack" + BLANK + INT + BLANK + ":" + BLANK)) {
082                                        knapsack = new Knapsack(getInt(line));
083                                        knapsacks.add(knapsack);
084                                } else if (knapsack == null) {
085                                        throw new IllegalStateException("knapsack not initialized." + filename);
086                                } else if (line.matches(BLANK + "capacity:" + BLANK + DOUBLE + BLANK)) {
087                                        double capacity = getDouble(line);
088                                        knapsack.setCapacity(capacity);
089                                } else if (line.matches(BLANK + "item" + BLANK + INT + BLANK + ":" + BLANK)) {
090                                        int i = getInt(line);
091                                        if (items.size() < i) {
092                                                item = new Item(getInt(line));
093                                                items.add(i - 1, item);
094                                        } else {
095                                                item = items.get(i - 1);
096                                        }
097                                } else if (line.matches(BLANK + "weight:" + BLANK + DOUBLE + BLANK)) {
098                                        int weight = getInt(line);
099                                        knapsack.setWeight(item, weight);
100                                } else if (line.matches(BLANK + "profit:" + BLANK + DOUBLE + BLANK)) {
101                                        int profit = getInt(line);
102                                        knapsack.setProfit(item, profit);
103                                }
104                        }
105                } catch (IOException e) {
106                        throw new IllegalArgumentException(e);
107                } finally {
108                        try {
109                                reader.close();
110                        } catch (IOException e) {
111                        }
112                }
113        }
114
115        /**
116         * Get the integer value in the line.
117         * 
118         * @param line
119         *            the line
120         * @return the integer value
121         */
122        protected int getInt(String line) {
123                Matcher matcher = Pattern.compile(POSINT).matcher(line);
124                matcher.find();
125                return Integer.parseInt(matcher.group());
126        }
127
128        /**
129         * Get the double value in the line.
130         * 
131         * @param line
132         *            the line
133         * @return the double value
134         */
135        protected double getDouble(String line) {
136                Matcher matcher = Pattern.compile(POSDOUBLE).matcher(line);
137                matcher.find();
138                return Double.parseDouble(matcher.group());
139        }
140
141        /*
142         * (non-Javadoc)
143         * 
144         * @see org.opt4j.benchmark.knapsack.KnapsackProblem#getKnapsacks()
145         */
146        @Override
147        public Collection<Knapsack> getKnapsacks() {
148                return knapsacks;
149        }
150
151        /*
152         * (non-Javadoc)
153         * 
154         * @see org.opt4j.benchmark.knapsack.KnapsackProblem#getItems()
155         */
156        @Override
157        public Collection<Item> getItems() {
158                return items;
159        }
160
161}
