023package org.opt4j.benchmarks.knapsack;
024
025import org.opt4j.core.config.annotations.File;
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Required;
028import org.opt4j.core.problem.ProblemModule;
029import org.opt4j.core.start.Constant;
030
031/**
032 * The multiobjective 0/1 ILP knapsack problem as proposed in "E. Zitzler and L.
033 * Thiele: Multiobjective evolutionary algorithms: A comparative case study and
034 * the strength Pareto approach. IEEE Transactions on Evolutionary Computation,
035 * vol. 3, no. 4, pp. 257-271, Nov. 1999.". Either one of the nine benchmark
036 * problems from Zitzler and Thiele 1999 can be selected or the number of
037 * knapsacks and items can be set manually.
038 * 
039 * @see <a
040 *      href="http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/">http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/</a>
041 * 
042 * @author reimann, lukasiewycz
043 * 
044 */
045public class KnapsackModule extends ProblemModule {
046
047        protected int knapsacks = 5;
048
049        protected int items = 250;
050
051        @Required(property = "testCase", elements = { "MANUAL", "ZT1", "ZT2", "ZT3", "ZT4", "ZT5", "ZT6", "ZT7", "ZT8",
052                        "ZT9" })
053        @Constant(value = "seed", namespace = KnapsackProblemRandom.class)
054        protected int seed = 0;
055
056        @File
057        @Required(property = "testCase", elements = { "FILE" })
058        @Constant(value = "filename", namespace = KnapsackProblemFile.class)
059        protected String filename = "";
060
061        protected Representation representation = Representation.BITSTRING;
062
063        protected TestCase testCase = TestCase.MANUAL;
064
065        public enum TestCase {
066                FILE, @Info("use test data from an input file")
067                MANUAL, @Info("2 knapsacks, 250 items")
068                ZT1, @Info("3 knapsacks, 250 items")
069                ZT2, @Info("5 knapsacks, 250 items")
070                ZT3, @Info("2 knapsacks, 500 items")
071                ZT4, @Info("3 knapsacks, 500 items")
072                ZT5, @Info("5 knapsacks, 500 items")
073                ZT6, @Info("2 knapsacks, 750 items")
074                ZT7, @Info("3 knapsacks, 750 items")
075                ZT8, @Info("5 knapsacks, 750 items")
076                ZT9;
077        }
078
079        public enum Representation {
080                @Info("Bitstring genotype")
081                BITSTRING, @Info("SAT decoding")
082                SAT;
083        }
084
085        @Override
086        protected void config() {
087                switch (testCase) {
088                case ZT1:
089                        knapsacks = 2;
090                        items = 250;
091                        break;
092                case ZT2:
093                        knapsacks = 3;
094                        items = 250;
095                        break;
096                case ZT3:
097                        knapsacks = 5;
098                        items = 250;
099                        break;
100                case ZT4:
101                        knapsacks = 2;
102                        items = 500;
103                        break;
104                case ZT5:
105                        knapsacks = 3;
106                        items = 500;
107                        break;
108                case ZT6:
109                        knapsacks = 5;
110                        items = 500;
111                        break;
112                case ZT7:
113                        knapsacks = 2;
114                        items = 750;
115                        break;
116                case ZT8:
117                        knapsacks = 3;
118                        items = 750;
119                        break;
120                case ZT9:
121                        knapsacks = 5;
122                        items = 750;
123                        break;
124                default:
125                        break;
126                }
127
128                bindConstant("items", KnapsackProblemRandom.class).to(items);
129                bindConstant("knapsacks", KnapsackProblemRandom.class).to(knapsacks);
130
131                if (testCase == TestCase.FILE) {
132                        bind(KnapsackProblem.class).to(KnapsackProblemFile.class).in(SINGLETON);
133                } else {
134                        bind(KnapsackProblem.class).to(KnapsackProblemRandom.class).in(SINGLETON);
135                }
136
137                switch (representation) {
138                case BITSTRING:
139                        bindProblem(KnapsackBinaryCreatorDecoder.class, KnapsackBinaryCreatorDecoder.class,
140                                        KnapsackProfitEvaluator.class);
141                        break;
142                case SAT:
143                        bindProblem(KnapsackSATCreatorDecoder.class, KnapsackSATCreatorDecoder.class, KnapsackProfitEvaluator.class);
144                        break;
145                }
146                addEvaluator(KnapsackOverloadEvaluator.class);
147        }
148
149        @Required(property = "testCase", elements = { "MANUAL" })
150        public int getKnapsacks() {
151                return knapsacks;
152        }
153
154        public void setKnapsacks(int knapsacks) {
155                this.knapsacks = knapsacks;
156        }
157
158        @Required(property = "testCase", elements = { "MANUAL" })
159        public int getItems() {
160                return items;
161        }
162
163        public void setItems(int items) {
164                this.items = items;
165        }
166
167        public Representation getRepresentation() {
168                return representation;
169        }
170
171        public void setRepresentation(Representation representation) {
172                this.representation = representation;
173        }
174
175        public TestCase getTestCase() {
176                return testCase;
177        }
178
179        public void setTestCase(TestCase testCase) {
180                this.testCase = testCase;
181        }
182
183        public int getSeed() {
184                return seed;
185        }
186
187        public void setSeed(int seed) {
188                this.seed = seed;
189        }
190
191        public String getFilename() {
192                return filename;
193        }
194
195        public void setFilename(String filename) {
196                this.filename = filename;
197        }
198
199}
