024package org.opt4j.benchmarks.queens;
025
026import static org.opt4j.core.Objective.Sign.MIN;
027
028import org.opt4j.core.Objective;
029import org.opt4j.core.Objectives;
030import org.opt4j.core.problem.Evaluator;
031import org.opt4j.core.problem.Priority;
032
033import com.google.inject.Inject;
034
035/**
036 * The evaluator for the {@link QueensProblem}.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041@Priority(-10)
042public class QueensErrorEvaluator implements Evaluator<QueensBoard> {
043
044        protected final QueensProblem problem;
045
046        protected final Objective error = new Objective("error", MIN);
047
048        /**
049         * Constructs the evaluator for the {@link QueensProblem}.
050         * 
051         * @param problem
052         *            the problem to set
053         * 
054         */
055        @Inject
056        public QueensErrorEvaluator(QueensProblem problem) {
057                this.problem = problem;
058        }
059
060        /*
061         * (non-Javadoc)
062         * 
063         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
064         */
065        @Override
066        public Objectives evaluate(QueensBoard queensBoard) {
067                int errors = countErrors(queensBoard);
068
069                Objectives objectives = new Objectives();
070                objectives.add(error, errors);
071                queensBoard.setFeasible(errors == 0);
072                return objectives;
073        }
074
075        /**
076         * Counts the number of errors for a given {@link QueensBoard}. That means
077         * the number of attacking queens.
078         * 
079         * @param queensBoard
080         *            the board
081         * @return the error count
082         */
083        private int countErrors(QueensBoard queensBoard) {
084                int errorcount = 0;
085                int size = problem.size();
086
087                for (int i = 0; i < size; i++) {
088                        // horizontal
089                        int sum = 0;
090                        for (int j = 0; j < size; j++) {
091                                if (queensBoard.isQueen(i, j)) {
092                                        sum++;
093                                }
094                        }
095                        sum = Math.abs(1 - sum);
096                        errorcount += sum;
097
098                        // vertikal
099                        sum = 0;
100                        for (int j = 0; j < size; j++) {
101                                if (queensBoard.isQueen(j, i)) {
102                                        sum++;
103                                }
104                        }
105                        sum = Math.abs(1 - sum);
106                        errorcount += sum;
107                }
108
109                for (int k = -size + 1; k < size; k++) {
110                        // diagonal 1
111                        int sum = 0;
112                        for (int j = 0; j < size; j++) {
113                                int i = k + j;
114                                if (0 <= i && i < size && queensBoard.isQueen(j, i)) {
115                                        sum++;
116                                }
117                        }
118                        sum = Math.max(0, sum - 1);
119                        errorcount += sum;
120                }
121
122                for (int k = 0; k < 2 * size - 1; k++) {
123                        // diagonal 2
124                        int sum = 0;
125                        for (int j = 0; j < size; j++) {
126                                int i = k - j;
127                                if (0 <= i && i < size && queensBoard.isQueen(j, i)) {
128                                        sum++;
129                                }
130                        }
131                        sum = Math.max(0, sum - 1);
132                        errorcount += sum;
133                }
134
135                return errorcount;
136        }
137
138}
