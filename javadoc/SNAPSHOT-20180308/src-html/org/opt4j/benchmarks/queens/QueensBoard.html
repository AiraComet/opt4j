024package org.opt4j.benchmarks.queens;
025
026/**
027 * A {@link QueensBoard} represents a chessboard with a given side length on
028 * which with queens can be placed.
029 * 
030 * @author lukasiewycz
031 * 
032 */
033public class QueensBoard {
034
035        protected final boolean[][] board;
036
037        protected final int size;
038
039        private boolean feasible;
040
041        /**
042         * Constructs a board with the side length {@code size}.
043         * 
044         * @param size
045         *            the size of the board
046         */
047        public QueensBoard(int size) {
048                this.size = size;
049                board = new boolean[size][size];
050
051                for (int i = 0; i < size; i++) {
052                        for (int j = 0; j < size; j++) {
053                                board[i][j] = false;
054                        }
055                }
056
057        }
058
059        /**
060         * Adds or removes queen from a field.
061         * 
062         * @param i
063         *            the row
064         * @param j
065         *            the column
066         * @param value
067         *            {@code true} if a queen is on the field, else {@code false}
068         */
069        public void setQueen(int i, int j, boolean value) {
070                board[i][j] = value;
071        }
072
073        /**
074         * Returns {@code true} if a queen is on this field.
075         * 
076         * @param i
077         *            the row
078         * @param j
079         *            the column
080         * @return {@code true} if a queen is on the field, else {@code false}
081         */
082        public boolean isQueen(int i, int j) {
083                return board[i][j];
084        }
085
086        /**
087         * Returns the size of the board.
088         * 
089         * @return the size of the board
090         */
091        public int size() {
092                return size;
093        }
094
095        /**
096         * The {@link QueensBoard} is feasible if no queen can attack another.
097         * 
098         * @return true if the queens board is feasible
099         */
100        public boolean isFeasible() {
101                return feasible;
102        }
103
104        /**
105         * Sets the feasibility of the {@link QueensBoard}.
106         * 
107         * @param feasible
108         *            true if the queens board is feasible, false otherwise
109         */
110        public void setFeasible(boolean feasible) {
111                this.feasible = feasible;
112        }
113
114        /*
115         * (non-Javadoc)
116         * 
117         * @see java.lang.Object#toString()
118         */
119        @Override
120        public String toString() {
121                String s = "";
122
123                for (int i = 0; i < size; i++) {
124                        for (int j = 0; j < size; j++) {
125                                if (board[i][j]) {
126                                        s += "[" + i + "," + j + "]";
127                                }
128                        }
129                }
130                return s;
131        }
132}
