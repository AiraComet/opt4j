024package org.opt4j.benchmarks.queens;
025
026import java.util.Random;
027
028import org.opt4j.core.start.Constant;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link QueensProblem} information. Contains the size of the board and the
034 * costs of the fields.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039public class QueensProblem {
040
041        protected final int[][][] board;
042
043        protected final int size;
044
045        protected final int dim;
046
047        protected final Random random;
048
049        /**
050         * Constructs a new {@link QueensProblem}.
051         * 
052         * @param size
053         *            the size of the board
054         * @param dim
055         *            the number of objective functions
056         * @param seed
057         *            the seed for the random number generator
058         */
059        @Inject
060        public QueensProblem(@Constant(value = "size", namespace = QueensProblem.class) int size,
061                        @Constant(value = "dim", namespace = QueensProblem.class) int dim,
062                        @Constant(value = "seed", namespace = QueensProblem.class) int seed) {
063                random = new Random(seed);
064
065                this.size = size;
066                this.dim = dim;
067                board = new int[dim][size][size];
068
069                for (int d = 0; d < dim; d++) {
070                        for (int i = 0; i < size; i++) {
071                                for (int j = 0; j < size; j++) {
072                                        board[d][i][j] = random.nextInt(100);
073                                }
074                        }
075                }
076
077        }
078
079        /**
080         * Returns the costs for one field of a specified dimension.
081         * 
082         * @param d
083         *            the dimension
084         * @param i
085         *            the row
086         * @param j
087         *            the column
088         * @return the costs of this field in the dimension
089         */
090        public int costs(int d, int i, int j) {
091                return board[d][i][j];
092        }
093
094        /**
095         * Returns the size of the board.
096         * 
097         * @return the size of the board
098         */
099        public int size() {
100                return size;
101        }
102
103        /**
104         * The number of objective functions.
105         * 
106         * @return the number of objective functions
107         */
108        public int dim() {
109                return dim;
110        }
111
112}
