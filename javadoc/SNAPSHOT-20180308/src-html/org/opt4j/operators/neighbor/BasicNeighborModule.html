024package org.opt4j.operators.neighbor;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.genotype.PermutationGenotype;
028
029/**
030 * The basic {@link NeighborModule}.
031 * 
032 * @author lukasiewycz
033 * 
034 */
035@Info("Setting for the basic neighbor classOperators for genotype variation.")
036public class BasicNeighborModule extends NeighborModule {
037
038        @Info("The type of the neighbor operator for the Permutation genotype.")
039        protected PermutationType permutationType = PermutationType.MIXED;
040
041        /**
042         * Type of {@link Neighbor} operator for the {@link PermutationGenotype}.
043         * 
044         * @author lukasiewycz
045         * 
046         */
047        public enum PermutationType {
048                /**
049                 * Use the {@link NeighborPermutationMixed}.
050                 */
051                @Info("Use randomly SWAP,INSERT, or REVERT")
052                MIXED,
053                /**
054                 * Use the {@link NeighborPermutationSwap}.
055                 */
056                @Info("Swaps two elements")
057                SWAP,
058                /**
059                 * Use the {@link NeighborPermutationInsert}.
060                 */
061                @Info("Moves one element to another position")
062                INSERT,
063                /**
064                 * Use the {@link NeighborPermutationRevert}.
065                 */
066                @Info("Reverts a part for the genotype")
067                REVERT;
068        }
069
070        /**
071         * Returns the permutation mode.
072         * 
073         * @return the permutation mode
074         */
075        public PermutationType getPermutationType() {
076                return permutationType;
077        }
078
079        /**
080         * Sets the permutation mode.
081         * 
082         * @param permutationMode
083         *            the permutation mode
084         */
085        public void setPermutationType(PermutationType permutationMode) {
086                this.permutationType = permutationMode;
087        }
088
089        /**
090         * Constructs a {@link BasicNeighborModule}.
091         */
092        public BasicNeighborModule() {
093                super();
094        }
095
096        /*
097         * (non-Javadoc)
098         * 
099         * @see org.opt4j.start.Opt4JModule#config()
100         */
101        @Override
102        public void config() {
103
104                Class<? extends NeighborPermutation> permutation = NeighborPermutationMixed.class;
105                switch (permutationType) {
106                case MIXED:
107                        permutation = NeighborPermutationMixed.class;
108                        break;
109                case SWAP:
110                        permutation = NeighborPermutationSwap.class;
111                        break;
112                case INSERT:
113                        permutation = NeighborPermutationInsert.class;
114                        break;
115                case REVERT:
116                        permutation = NeighborPermutationRevert.class;
117                        break;
118                }
119                bind(NeighborPermutation.class).to(permutation).in(SINGLETON);
120        }
121}
