048/**
049 * Superclass for {@link GenericOperator}s.
050 * 
051 * @author lukasiewycz
052 * 
053 * @param <O>
054 *            The specified {@link Operator}.
055 * @param <Q>
056 *            The specified {@link Operator} with a wildcard (?).
057 */
058public abstract class AbstractGenericOperator<O extends Operator<?>, Q extends Operator<?>> implements
059                GenericOperator<O> {
060
061        /**
062         * Comparator for a specific order: Superclasses always are sorted after
063         * subclasses.
064         * 
065         * @author lukasiewycz
066         * 
067         */
068        protected static class ClassComparator implements Comparator<Class<? extends Genotype>> {
069
070                /*
071                 * (non-Javadoc)
072                 * 
073                 * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
074                 */
075                @Override
076                public int compare(Class<? extends Genotype> arg0, Class<? extends Genotype> arg1) {
077                        if (arg0.equals(arg1)) {
078                                return 0;
079                        } else if (arg0.isAssignableFrom(arg1)) {
080                                // arg0 is superclass of arg1
081                                return 1;
082                        } else if (arg1.isAssignableFrom(arg0)) {
083                                // arg1 is superclass of arg0
084                                return -1;
085                        } else {
086                                return arg0.getCanonicalName().compareTo(arg1.getCanonicalName());
087                        }
088                }
089
090        }
091
092        protected SortedMap<Class<? extends Genotype>, O> classOperators = new TreeMap<Class<? extends Genotype>, O>(
093                        new ClassComparator());
094        protected Map<OperatorPredicate, O> genericOperators = new HashMap<OperatorPredicate, O>();
095
096        protected List<Class<? extends Q>> cldef = new ArrayList<Class<? extends Q>>();
097
098        /**
099         * Constructs an {@link AbstractGenericOperator} class with the given
100         * clazzes of default operators.
101         * 
102         * @param clazzes
103         *            the default operators
104         */
105        public AbstractGenericOperator(Class<? extends Q>... clazzes) {
106                for (Class<? extends Q> cl : clazzes) {
107                        cldef.add(cl);
108                }
109        }
110
111        /**
112         * Inject and organize the operators.
113         * 
114         * @param holder
115         *            the operator holder
116         */
117        @SuppressWarnings("unchecked")
118        @Inject
119        protected synchronized void inject(OperatorHolder<Q> holder) {
120                if (classOperators.isEmpty()) {
121                        classOperators.put(CompositeGenotype.class, null);
122                        holder.add(cldef);
123
124                        for (Entry<OperatorPredicate, Q> entry : holder.getMap().entrySet()) {
125                                addOperator(entry.getKey(), (O) entry.getValue());
126                        }
127
128                }
129        }
130
131        /*
132         * (non-Javadoc)
133         * 
134         * @see org.opt4j.operator.GenericOperator#addOperator(org.opt4j.operator.
135         * AbstractGenericOperator.OperatorPredicate,
136         * org.opt4j.core.optimizer.Operator)
137         */
138        @Override
139        public void addOperator(OperatorPredicate predicate, O operator) {
140                if (predicate instanceof OperatorClassPredicate) {
141                        Class<? extends Genotype> clazz = ((OperatorClassPredicate) predicate).getClazz();
142                        classOperators.put(clazz, operator);
143                } else {
144                        genericOperators.put(predicate, operator);
145                }
146        }
147
148        /*
149         * (non-Javadoc)
150         * 
151         * @see
152         * org.opt4j.operator.GenericOperator#getOperator(org.opt4j.core.problem
153         * .Genotype)
154         */
155        @Override
156        public O getOperator(Genotype genotype) {
157                if (classOperators.containsKey(genotype.getClass())) {
158                        return classOperators.get(genotype.getClass());
159                } else {
160                        // Search for a predicate that satisfies the genotype.
161                        for (Entry<OperatorPredicate, O> predicate : genericOperators.entrySet()) {
162                                if (predicate.getKey().evaluate(genotype)) {
163                                        return predicate.getValue();
164                                }
165                        }
166
167                        // Searches for a superclass that is registered as an operator.
168                        for (Entry<Class<? extends Genotype>, O> entry : classOperators.entrySet()) {
169                                Class<? extends Genotype> c = entry.getKey();
170                                if (c.isAssignableFrom(genotype.getClass())) {
171                                        O operator = entry.getValue();
172                                        addOperator(new OperatorClassPredicate(genotype.getClass()), operator);
173                                        return operator;
174                                }
175                        }
176                        throw new IncompatibilityException("No handler found for " + genotype.getClass() + " in " + this.getClass());
177                }
178
179        }
180
181        /*
182         * (non-Javadoc)
183         * 
184         * @see org.opt4j.operator.common.GenericOperator#getHandlers()
185         */
186        @Override
187        public Collection<O> getOperators() {
188                Set<O> set = new HashSet<O>();
189                set.addAll(classOperators.values());
190                set.addAll(genericOperators.values());
191                return set;
192        }
193
194        /**
195         * Returns the target {@link Genotype} for an operator based on the
196         * {@link Apply} annotation.
197         * 
198         * @param <O>
199         *            the type of operator
200         * @param operator
201         *            the operator
202         * @return the target genotype
203         */
204        protected static <O> Class<? extends Genotype> getTarget(O operator) {
205                Apply apply = operator.getClass().getAnnotation(Apply.class);
206
207                if (apply != null) {
208                        return apply.value();
209                }
210
211                Type type = Parameters.getType(Operator.class, operator, "G");
212                if (type != null) {
213                        Class<? extends Genotype> target = Parameters.getClass(type).asSubclass(Genotype.class);
214                        return target;
215                }
216
217                throw new IllegalArgumentException("No target specified for the operator " + operator.getClass().getName()
218                                + ". Either parameterize the Operator or use the " + Apply.class.getName()
219                                + " annotation to specify a target.");
220        }
221
222        protected static class OperatorHolder<P> {
223
224                @Inject(optional = true)
225                protected Map<OperatorPredicate, P> map = new HashMap<OperatorPredicate, P>();
226
227                @Inject
228                protected Injector injector;
229
230                protected Collection<Class<? extends P>> clazzes = new ArrayList<Class<? extends P>>();
231
232                public void add(Collection<Class<? extends P>> clazzes) {
233                        this.clazzes.addAll(clazzes);
234                }
235
236                public Map<OperatorPredicate, P> getMap() {
237                        Map<OperatorPredicate, P> map = new HashMap<OperatorPredicate, P>();
238
239                        for (Class<? extends P> clazz : clazzes) {
240                                P p = injector.getInstance(clazz);
241                                map.put(new OperatorClassPredicate(getTarget((Operator<?>) p)), p);
242                        }
243
244                        for (Entry<OperatorPredicate, P> entry : this.map.entrySet()) {
245                                OperatorPredicate predicate = entry.getKey();
246                                if (predicate instanceof OperatorVoidPredicate) {
247                                        predicate = new OperatorClassPredicate(getTarget((Operator<?>) entry.getValue()));
248                                }
249                                map.put(predicate, entry.getValue());
250                        }
251
252                        return map;
253                }
254        }
255
256        /**
257         * The {@link OperatorPredicate} interface.
258         * 
259         * @author lukasiewycz
260         * 
261         */
262        public interface OperatorPredicate {
263
264                /**
265                 * Checks whether a {@link Genotype} satisfies the predicate.
266                 * 
267                 * @param genotype
268                 *            the genotype
269                 * @return {@code true} if the predicate is satisfied
270                 */
271                public boolean evaluate(Genotype genotype);
272        }
273
274        /**
275         * The {@link OperatorVoidPredicate} interface is used as marker for
276         * {@link Operator}s for which the predicate is not explicitly defined.
277         * 
278         * @author lukasiewycz
279         * 
280         */
281        public static class OperatorVoidPredicate implements OperatorPredicate {
282                /*
283                 * (non-Javadoc)
284                 * 
285                 * @see
286                 * org.opt4j.operator.AbstractGenericOperator.OperatorPredicate#evaluate
287                 * (org.opt4j.core.problem.Genotype)
288                 */
289                @Override
290                public boolean evaluate(Genotype genotype) {
291                        return false;
292                }
293
294                /*
295                 * (non-Javadoc)
296                 * 
297                 * @see java.lang.Object#toString()
298                 */
299                @Override
300                public String toString() {
301                        return "Predicate Void";
302                }
303
304        }
305
306        /**
307         * The {@link OperatorClassPredicate} returns {@code true} for a given
308         * specific class.
309         * 
310         * @author lukasiewycz
311         * 
312         */
313        public static class OperatorClassPredicate implements OperatorPredicate {
314
315                protected final Class<? extends Genotype> clazz;
316
317                /**
318                 * Creates a new {@link OperatorClassPredicate} for the given
319                 * {@link Genotype} class.
320                 * 
321                 * @param clazz
322                 *            the class of the genotype
323                 */
324                public OperatorClassPredicate(Class<? extends Genotype> clazz) {
325                        this.clazz = clazz;
326                }
327
328                /*
329                 * (non-Javadoc)
330                 * 
331                 * @see
332                 * org.opt4j.operator.AbstractGenericOperator.OperatorPredicate#evaluate
333                 * (org.opt4j.core.problem.Genotype)
334                 */
335                @Override
336                public boolean evaluate(Genotype genotype) {
337                        return clazz.equals(genotype.getClass());
338                }
339
340                /**
341                 * Returns the genotype class for the operator.
342                 * 
343                 * @return the genotype class
344                 */
345                public Class<? extends Genotype> getClazz() {
346                        return clazz;
347                }
348
349                /*
350                 * (non-Javadoc)
351                 * 
352                 * @see java.lang.Object#toString()
353                 */
354                @Override
355                public String toString() {
356                        return "Predicate [clazz=" + clazz.getSimpleName() + "]";
357                }
358
359                /*
360                 * (non-Javadoc)
361                 * 
362                 * @see java.lang.Object#hashCode()
363                 */
364                @Override
365                public int hashCode() {
366                        final int prime = 31;
367                        int result = 1;
368                        result = prime * result + ((clazz == null) ? 0 : clazz.hashCode());
369                        return result;
370                }
371
372                /*
373                 * (non-Javadoc)
374                 * 
375                 * @see java.lang.Object#equals(java.lang.Object)
376                 */
377                @Override
378                public boolean equals(Object obj) {
379                        if (this == obj) {
380                                return true;
381                        }
382                        if (obj == null) {
383                                return false;
384                        }
385                        if (getClass() != obj.getClass()) {
386                                return false;
387                        }
388                        OperatorClassPredicate other = (OperatorClassPredicate) obj;
389                        if (clazz == null) {
390                                if (other.clazz != null) {
391                                        return false;
392                                }
393                        } else if (!clazz.equals(other.clazz)) {
394                                return false;
395                        }
396                        return true;
397                }
398        }
