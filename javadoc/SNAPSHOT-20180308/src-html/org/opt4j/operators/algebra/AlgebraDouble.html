024package org.opt4j.operators.algebra;
025
026import org.opt4j.core.Genotype;
027import org.opt4j.core.genotype.DoubleGenotype;
028import org.opt4j.operators.normalize.NormalizeDouble;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link AlgebraDouble} for {@link DoubleGenotype}.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038public class AlgebraDouble implements Algebra<DoubleGenotype> {
039
040        protected final NormalizeDouble normalize;
041
042        /**
043         * Constructs a {@link AlgebraDouble} .
044         * 
045         * @param normalize
046         *            the normalize operator for double values
047         */
048        @Inject
049        public AlgebraDouble(final NormalizeDouble normalize) {
050                super();
051                this.normalize = normalize;
052        }
053
054        /*
055         * (non-Javadoc)
056         * 
057         * @see
058         * org.opt4j.operator.algebra.Algebra#algebra(org.opt4j.operator.algebra
059         * .Term, org.opt4j.core.Genotype[])
060         */
061        @Override
062        public DoubleGenotype algebra(Term term, Genotype... genotypes) {
063                int n = genotypes.length;
064                assert (n > 0);
065
066                DoubleGenotype[] list = new DoubleGenotype[n];
067
068                for (int i = 0; i < n; i++) {
069                        list[i] = (DoubleGenotype) genotypes[i];
070                }
071
072                DoubleGenotype offspring = list[0].newInstance();
073                offspring.clear();
074
075                int size = list[0].size();
076
077                double[] values = new double[n];
078
079                for (int j = 0; j < size; j++) {
080                        for (int i = 0; i < n; i++) {
081                                if (list[i] == null) {
082                                        values[i] = 0;
083                                } else {
084                                        values[i] = list[i].get(j);
085                                }
086                        }
087                        double result = term.calculate(values);
088                        offspring.add(result);
089                }
090
091                normalize.normalize(offspring);
092
093                return offspring;
094        }
095
096}
