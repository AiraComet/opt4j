024package org.opt4j.operators.crossover;
025
026import java.util.List;
027
028import org.opt4j.core.common.random.Rand;
029import org.opt4j.core.start.Constant;
030import org.opt4j.operators.normalize.NormalizeDouble;
031
032import com.google.inject.Inject;
033
034/**
035 * The {@link CrossoverDoubleUnfairAverage} implements the unfair average
036 * crossover proposed by Nomura and Miyhoshi, 1996.
037 * 
038 * <p>
039 * The unfair average crossover is applied to the whole genotype, i.e., the used
040 * vector of double values. In the original work, two offspring created using
041 * </p>
042 * <p>
043 * {@code z_1 = (1 + alpha)x - alpha * y} for {@code i = 1, ..., j} and<br>
044 * {@code z_1 = -alpha * x + (1 + alpha)y} for {@code i = j + 1, ..., n}<br>
045 * {@code z_2 = (1 - alpha)x + alpha * y} for {@code i = 1, ..., j} and<br>
046 * {@code z_2 = alpha * x + (1 - alpha)y} for {@code i = j + 1, ..., n}
047 * </p>
048 * <p>
049 * for each double value in the double vector. In this implementation, one
050 * offspring is created following {@code z_1} with a probability of {@code 0.5}
051 * and following {@code z_2} with a probability of {@code 0.5}, respectively. At
052 * this juncture, {@code n} is the number of variables in the vector and
053 * {@code j} a randomly chosen integer between {@code 1} and {@code n}. The
054 * value {@code alpha} is within the range (0, 0.5).
055 * </p>
056 * <p>
057 * Unlike the BLX and SBX operator, the unfair average will create offspring
058 * towards one of the parent solutions.
059 * </p>
060 * 
061 * @author glass
062 * 
063 */
064public class CrossoverDoubleUnfairAverage extends CrossoverDouble {
065
066        protected final double alpha;
067
068        /**
069         * Constructs an {@link CrossoverDoubleUnfairAverage} with an alpha value
070         * and a random number generator.
071         * 
072         * @param alpha
073         *            the alpha value
074         * @param normalize
075         *            a normalize operator
076         * @param random
077         *            the random number generator
078         */
079        @Inject
080        public CrossoverDoubleUnfairAverage(
081                        @Constant(value = "alpha", namespace = CrossoverDoubleUnfairAverage.class) double alpha,
082                        NormalizeDouble normalize, Rand random) {
083                super(normalize, random);
084                this.alpha = alpha;
085        }
086
087        /*
088         * (non-Javadoc)
089         * 
090         * @see
091         * org.opt4j.operator.crossover.CrossoverDouble#crossover(java.util.List,
092         * java.util.List, java.util.List)
093         */
094        @Override
095        protected void crossover(List<Double> p1, List<Double> p2, List<Double> o1, List<Double> o2) {
096                double z1, z2, x, y;
097
098                int size = p1.size();
099                int j = random.nextInt(size);
100
101                for (int i = 0; i < size; i++) {
102                        x = p1.get(i);
103                        y = p2.get(i);
104
105                        if (i <= j) {
106                                z1 = (1 + alpha) * x - alpha * y;
107                                z2 = (1 - alpha) * x + alpha * y;
108                        } else {
109                                z1 = -alpha * x + (1 + alpha) * y;
110                                z2 = alpha * x + (1 - alpha) * y;
111                        }
112                        o1.add(z1);
113                        o2.add(z2);
114                }
115        }
116
117}
