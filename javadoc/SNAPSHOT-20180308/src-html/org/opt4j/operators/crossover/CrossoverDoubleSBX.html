024package org.opt4j.operators.crossover;
025
026import static java.lang.annotation.RetentionPolicy.RUNTIME;
027
028import java.lang.annotation.Retention;
029
030import org.opt4j.core.common.random.Rand;
031import org.opt4j.core.start.Constant;
032import org.opt4j.operators.normalize.NormalizeDouble;
033
034import com.google.inject.BindingAnnotation;
035import com.google.inject.Inject;
036
037/**
038 * The {@link CrossoverDoubleSBX} is an implementation of the simulated binary
039 * crossover operator proposed by Deb and Kumar, 1995.
040 * 
041 * <p>
042 * The SBX crossover is applied bitwise. In the original work, two offspring are
043 * created using
044 * </p>
045 * <p>
046 * {@code z_1 = 0.5 * [(1 + beta)x + (1 - beta)y]}<br>
047 * {@code z_2 = 0.5 * [(1 - beta)x + (1 + beta)y]}.
048 * </p>
049 * <p>
050 * In this implementation, one offspring is created while the actual bit is set
051 * to {@code z_1} with a probability of {@code 0.5} and to {@code z_2} with a
052 * probability of {@code 0.5}, respectively.<br>
053 * The value {@code beta} is defined as<br>
054 * </p>
055 * <p>
056 * {@code beta = (2u)^(1 / (nu + 1))} if {@code u <= 0.5} and<br>
057 * {@code beta = (1 / (2 (1 - u)))^(1 / (nu + 1))} otherwise
058 * </p>
059 * <p>
060 * At this juncture, {@code u} is a random number between {@code (0, 1)}. The
061 * variable {@code nu} influences the crossover with a high value leading to a
062 * higher probability of 'near-parent' solution while a small value leads to a
063 * higher probability of more distant solutions, respectively.
064 * </p>
065 * 
066 * @author lukasiewycz
067 * 
068 */
069public class CrossoverDoubleSBX extends CrossoverDoubleElementwise {
070
071        @Retention(RUNTIME)
072        @BindingAnnotation
073        protected @interface Nu {
074        }
075
076        protected double nu;
077
078        /**
079         * Constructs a {@link CrossoverDoubleSBX} with a nu value and a random
080         * generator.
081         * 
082         * @param nu
083         *            the {@code nu} value
084         * @param normalize
085         *            the normalize operator
086         * @param random
087         *            the random number generator
088         */
089        @Inject
090        public CrossoverDoubleSBX(@Constant(value = "nu", namespace = CrossoverDoubleSBX.class) double nu,
091                        NormalizeDouble normalize, Rand random) {
092                super(normalize, random);
093                this.nu = nu;
094        }
095
096        /*
097         * (non-Javadoc)
098         * 
099         * @see
100         * org.opt4j.operator.crossover.CrossoverDoubleElementwise#crossover(double,
101         * double)
102         */
103        @Override
104        public Pair<Double> crossover(double x, double y) {
105                double u = random.nextDouble(); // [0,1)
106                double beta;
107                if (0.5 <= u) {
108                        beta = Math.pow(2 * u, 1 / (nu + 1));
109                } else {
110                        beta = Math.pow(1 / (2 - 2 * u), 1 / (nu + 1));
111                }
112
113                double o1 = 0.5 * ((1 + beta) * x + (1 - beta) * y);
114                double o2 = 0.5 * ((1 - beta) * x + (1 + beta) * y);
115
116                if (random.nextDouble() < 0.5) {
117                        double tmp = o1;
118                        o1 = o2;
119                        o2 = tmp;
120                }
121
122                return new Pair<Double>(o1, o2);
123        }
124
125}
