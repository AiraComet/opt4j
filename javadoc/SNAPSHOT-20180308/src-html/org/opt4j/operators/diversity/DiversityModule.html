024package org.opt4j.operators.diversity;
025
026import org.opt4j.core.config.Icons;
027import org.opt4j.core.config.annotations.Icon;
028import org.opt4j.operators.OperatorModule;
029
030import com.google.inject.TypeLiteral;
031
032/**
033 * The {@link DiversityModule} is used for modules for the {@link Diversity}
034 * operator.
035 * 
036 * A new {@link Diversity} operator can be registered by implementing this
037 * module and adding the operator with {@link #addOperator}.
038 * 
039 * @author glass, lukasiewycz
040 * @see Diversity
041 * 
042 */
043@Icon(Icons.OPERATOR)
044public abstract class DiversityModule extends OperatorModule<Diversity<?>> {
045
046        /*
047         * (non-Javadoc)
048         * 
049         * @see org.opt4j.operator.OperatorModule#getOperatorTypeLiteral()
050         */
051        @Override
052        protected TypeLiteral<Diversity<?>> getOperatorTypeLiteral() {
053                return new TypeLiteral<Diversity<?>>() {
054                };
055        }
056
057}
