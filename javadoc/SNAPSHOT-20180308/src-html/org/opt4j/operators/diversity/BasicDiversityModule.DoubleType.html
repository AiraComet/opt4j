024package org.opt4j.operators.diversity;
025
026import org.opt4j.core.config.PropertyModule;
027import org.opt4j.core.config.annotations.Ignore;
028import org.opt4j.core.config.annotations.Info;
029import org.opt4j.core.genotype.BooleanGenotype;
030import org.opt4j.core.genotype.DoubleGenotype;
031
032/**
033 * The {@link BasicDiversityModule} is the basic {@link PropertyModule} for the
034 * {@link Diversity} operator.
035 * 
036 * @author glass
037 * 
038 */
039@Info("Setting for the basic diversity classOperators for genotypes.")
040public class BasicDiversityModule extends DiversityModule {
041
042        @Ignore
043        @Info("The type of the diversity operator for the Boolean genotype.")
044        BooleanType booleanType = BooleanType.FRACTION;
045
046        @Info("The type of the diversity operator for the Double genotype.")
047        DoubleType doubleType = DoubleType.ABSOLUTE;
048
049        /**
050         * Type of {@link Diversity} operator for the {@link BooleanGenotype}.
051         * 
052         * @author glass
053         */
054        public enum BooleanType {
055                /**
056                 * Use the {@link DiversityBooleanFraction} operator.
057                 */
058                FRACTION;
059        }
060
061        /**
062         * Type of {@link Diversity} operator for the {@link DoubleGenotype}.
063         * 
064         * @author glass
065         */
066        public enum DoubleType {
067                /**
068                 * Use the {@link DiversityDoubleAbsolute} operator.
069                 */
070                ABSOLUTE,
071                /**
072                 * Use the {@link DiversityDoubleEuclidean} operator.
073                 */
074                EUCLIDEAN;
075        }
076
077        /**
078         * Constructs a {@link BasicDiversityModule}.
079         * 
080         */
081        public BasicDiversityModule() {
082                super();
083        }
084
085        /**
086         * Returns the Type of {@link Diversity} operator for
087         * {@link BooleanGenotype}.
088         * 
089         * @return the type of diversity operator for boolean genotypes
090         */
091        public BooleanType getBooleanType() {
092                return booleanType;
093        }
094
095        /**
096         * Sets the Type of {@link Diversity} operator for {@link BooleanGenotype}.
097         * 
098         * @param booleanType
099         *            the booleanType to set
100         */
101        public void setBooleanType(BooleanType booleanType) {
102                this.booleanType = booleanType;
103        }
104
105        /**
106         * Returns the Type of {@link Diversity} operator for {@link DoubleGenotype}
107         * .
108         * 
109         * @return the type of diversity operator for double genotypes
110         */
111        public DoubleType getDoubleType() {
112                return doubleType;
113        }
114
115        /**
116         * Sets the the Type of {@link Diversity} operator for
117         * {@link DoubleGenotype}.
118         * 
119         * @param doubleType
120         *            the doubleType to set
121         */
122        public void setDoubleType(DoubleType doubleType) {
123                this.doubleType = doubleType;
124        }
125
126        /*
127         * (non-Javadoc)
128         * 
129         * @see org.opt4j.start.Opt4JModule#config()
130         */
131        @Override
132        public void config() {
133                switch (booleanType) {
134                case FRACTION:
135                        bind(DiversityBoolean.class).to(DiversityBooleanFraction.class).in(SINGLETON);
136                        break;
137                }
138
139                switch (doubleType) {
140                case ABSOLUTE:
141                        bind(DiversityDouble.class).to(DiversityDoubleAbsolute.class).in(SINGLETON);
142                        break;
143                case EUCLIDEAN:
144                        bind(DiversityDouble.class).to(DiversityDoubleEuclidean.class).in(SINGLETON);
145                        break;
146                }
147        }
148}
