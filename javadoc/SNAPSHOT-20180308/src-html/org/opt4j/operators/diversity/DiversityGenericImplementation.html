024package org.opt4j.operators.diversity;
025
026import org.opt4j.core.Genotype;
027import org.opt4j.core.genotype.CompositeGenotype;
028import org.opt4j.operators.AbstractGenericOperator;
029
030import com.google.inject.Inject;
031import com.google.inject.Singleton;
032
033/**
034 * The {@link DiversityGenericImplementation} is a standard implementation of
035 * the {@link Diversity} interface.
036 * 
037 * @author glass, lukasiewycz
038 * 
039 */
040@Singleton
041public class DiversityGenericImplementation extends AbstractGenericOperator<Diversity<Genotype>, Diversity<?>>
042                implements Diversity<Genotype> {
043
044        /**
045         * Constructs the {@link DiversityGenericImplementation}.
046         */
047        @SuppressWarnings("unchecked")
048        @Inject
049        protected DiversityGenericImplementation() {
050                super(DiversityBoolean.class, DiversityDouble.class, DiversityInteger.class, DiversityPermutation.class);
051        }
052
053        /*
054         * (non-Javadoc)
055         * 
056         * @see
057         * org.opt4j.operator.diversity.Diversity#diversity(org.opt4j.core.Genotype,
058         * org.opt4j.core.Genotype)
059         */
060        @Override
061        public double diversity(Genotype a, Genotype b) {
062                Diversity<Genotype> diversity = getOperator(a);
063
064                if (diversity == null) {
065                        return diversityComposite((CompositeGenotype<?, ?>) a, (CompositeGenotype<?, ?>) b);
066                } else {
067                        return diversity.diversity(a, b);
068                }
069        }
070
071        protected double diversityComposite(CompositeGenotype<?, ?> a, CompositeGenotype<?, ?> b) {
072
073                double diversity = 0;
074                for (Object key : a.keySet()) {
075                        Genotype childA = a.get(key);
076                        Genotype childB = b.get(key);
077                        int size = childA.size();
078
079                        diversity += diversity(childA, childB) * size;
080                }
081                return diversity / a.size();
082
083        }
084
085}
