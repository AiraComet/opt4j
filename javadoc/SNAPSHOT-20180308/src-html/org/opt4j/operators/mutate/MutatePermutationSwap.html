024package org.opt4j.operators.mutate;
025
026import java.util.Collections;
027import java.util.Random;
028
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.core.genotype.PermutationGenotype;
031
032import com.google.inject.Inject;
033
034/**
035 * <p>
036 * Mutate operator for the {@link PermutationGenotype}. With a given mutation
037 * rate two elements are selected and swapped.
038 * </p>
039 * 
040 * <p>
041 * Given a permutation {@code 1 2 3 4 5 6 7 8}, this might result in {@code 1 2
042 * 7 4 5 6 3 8}.
043 * </p>
044 * 
045 * @author lukasiewycz
046 * 
047 */
048public class MutatePermutationSwap implements MutatePermutation {
049
050        protected final Random random;
051
052        /**
053         * Constructs a new {@link MutatePermutation} with the given mutation rate.
054         * 
055         * @param random
056         *            the random number generator
057         */
058        @Inject
059        public MutatePermutationSwap(Rand random) {
060                this.random = random;
061        }
062
063        /*
064         * (non-Javadoc)
065         * 
066         * @see
067         * org.opt4j.operator.mutate.Mutate#mutate(org.opt4j.core.problem.Genotype,
068         * double)
069         */
070        @Override
071        public void mutate(PermutationGenotype<?> genotype, double p) {
072                int size = genotype.size();
073
074                if (size > 1) {
075                        for (int i = 0; i < size; i++) {
076                                if (random.nextDouble() < p) {
077                                        int j;
078                                        do {
079                                                j = random.nextInt(size);
080                                        } while (j == i);
081
082                                        Collections.swap(genotype, i, j);
083                                }
084                        }
085                }
086        }
087
088}
