024package org.opt4j.operators.mutate;
025
026import org.opt4j.core.common.random.Rand;
027import org.opt4j.core.genotype.DoubleGenotype;
028import org.opt4j.operators.normalize.NormalizeDouble;
029
030/**
031 * The {@link MutateDoubleElementwise}.
032 * 
033 * @author lukasiewycz
034 * 
035 */
036public abstract class MutateDoubleElementwise extends MutateDouble {
037
038        /**
039         * Constructs a {@link MutateDoubleElementwise}.
040         * 
041         * @param random
042         *            the random number generator
043         * @param normalize
044         *            the normalizer
045         */
046        public MutateDoubleElementwise(Rand random, NormalizeDouble normalize) {
047                super(random, normalize);
048        }
049
050        /*
051         * (non-Javadoc)
052         * 
053         * @see
054         * org.opt4j.operator.mutate.MutateDouble#mutateList(org.opt4j.genotype.
055         * DoubleGenotype, double)
056         */
057        @Override
058        protected void mutateList(DoubleGenotype vector, double p) {
059                for (int i = 0; i < vector.size(); i++) {
060                        double x = vector.get(i);
061                        double y = mutateElement(x, vector.getLowerBound(i), vector.getUpperBound(i), p);
062
063                        if (x != y) {
064                                vector.set(i, y);
065                        }
066                }
067        }
068
069        /**
070         * Mutate one element of the list.
071         * 
072         * @param element
073         *            the element
074         * @param lb
075         *            the lower bound
076         * @param ub
077         *            the upper bound
078         * @param p
079         *            the mutation rate
080         * @return the mutated element
081         */
082        protected abstract double mutateElement(double element, double lb, double ub, double p);
083
084}
